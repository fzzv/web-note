# Python 装饰器

装饰器（Decorator）是 Python 中一种在**不修改原函数代码**的前提下，为函数添加额外功能的机制。本质上就是一个接收函数并返回新函数的高阶函数。

## 1. 理解装饰器的前置知识

### 1.1 函数是一等公民

在 Python 中，函数可以像普通变量一样传递、赋值和返回。

```python
# 函数可以赋值给变量
def greet(name):
    return f"Hello, {name}!"

say_hello = greet  # 赋值给另一个变量
print(say_hello("Alice"))  # Hello, Alice!
print(type(say_hello))     # <class 'function'>

# 函数可以作为参数传递
def apply(func, value):
    return func(value)

print(apply(len, "hello"))      # 5
print(apply(str.upper, "hi"))   # HI

# 函数可以作为返回值
def make_greeter(greeting):
    def greeter(name):
        return f"{greeting}, {name}!"
    return greeter

hi = make_greeter("Hi")
hey = make_greeter("Hey")
print(hi("Alice"))   # Hi, Alice!
print(hey("Bob"))    # Hey, Bob!
```

### 1.2 闭包回顾

装饰器的核心机制就是闭包：内层函数引用了外层函数的变量，外层函数返回后变量仍然存活。

```python
def counter(start=0):
    count = start
    def increment():
        nonlocal count
        count += 1
        return count
    return increment

c = counter(10)
print(c())  # 11
print(c())  # 12
print(c())  # 13
```

## 2. 装饰器基础

### 2.1 最简单的装饰器

```python
def my_decorator(func):
    def wrapper():
        print("--- 函数执行前 ---")
        func()
        print("--- 函数执行后 ---")
    return wrapper

# 手动装饰
def say_hello():
    print("Hello!")

say_hello = my_decorator(say_hello)  # 用新函数替换原函数
say_hello()
# --- 函数执行前 ---
# Hello!
# --- 函数执行后 ---
```

### 2.2 @ 语法糖

`@decorator` 等价于 `func = decorator(func)`，只是更简洁。

```python
def my_decorator(func):
    def wrapper():
        print("--- 函数执行前 ---")
        func()
        print("--- 函数执行后 ---")
    return wrapper

@my_decorator        # 等价于 say_hello = my_decorator(say_hello)
def say_hello():
    print("Hello!")

say_hello()
# --- 函数执行前 ---
# Hello!
# --- 函数执行后 ---
```

### 2.3 处理参数和返回值

装饰器需要透传原函数的参数和返回值。

```python
def my_decorator(func):
    def wrapper(*args, **kwargs):       # 接收任意参数
        print(f"调用 {func.__name__}，参数：args={args}, kwargs={kwargs}")
        result = func(*args, **kwargs)  # 透传给原函数
        print(f"返回值：{result}")
        return result                   # 返回原函数的结果
    return wrapper

@my_decorator
def add(a, b):
    return a + b

@my_decorator
def greet(name, greeting="Hello"):
    return f"{greeting}, {name}!"

print(add(3, 5))
# 调用 add，参数：args=(3, 5), kwargs={}
# 返回值：8
# 8

print(greet("Alice", greeting="Hi"))
# 调用 greet，参数：args=('Alice',), kwargs={'greeting': 'Hi'}
# 返回值：Hi, Alice!
# Hi, Alice!
```

### 2.4 使用 functools.wraps 保留元信息

装饰后，函数的 `__name__`、`__doc__` 等属性会变成 wrapper 的。用 `@wraps` 修复。

```python
from functools import wraps

# ❌ 没有 @wraps
def bad_decorator(func):
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@bad_decorator
def hello():
    """这是 hello 的文档"""
    pass

print(hello.__name__)  # wrapper   ← 名字丢了
print(hello.__doc__)   # None      ← 文档丢了

# ✅ 使用 @wraps
def good_decorator(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        return func(*args, **kwargs)
    return wrapper

@good_decorator
def hello():
    """这是 hello 的文档"""
    pass

print(hello.__name__)  # hello     ← 保留了
print(hello.__doc__)   # 这是 hello 的文档  ← 保留了

# 还可以通过 __wrapped__ 访问原始函数
print(hello.__wrapped__)  # <function hello at 0x...>
```

> **规则：写装饰器时永远加上 `@wraps(func)`。**

## 3. 常用装饰器实战

### 3.1 计时装饰器

```python
import time
from functools import wraps

def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        elapsed = time.perf_counter() - start
        print(f"[{func.__name__}] 耗时 {elapsed:.4f}s")
        return result
    return wrapper

@timer
def slow_sum(n):
    return sum(range(n))

@timer
def slow_sleep():
    time.sleep(0.5)

print(slow_sum(1_000_000))  # [slow_sum] 耗时 0.0312s  →  499999500000
slow_sleep()                 # [slow_sleep] 耗时 0.5003s
```

### 3.2 日志装饰器

```python
from functools import wraps
from datetime import datetime

def log(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        timestamp = datetime.now().strftime("%H:%M:%S")
        args_str = ", ".join(
            [repr(a) for a in args] +
            [f"{k}={v!r}" for k, v in kwargs.items()]
        )
        print(f"[{timestamp}] 调用 {func.__name__}({args_str})")
        try:
            result = func(*args, **kwargs)
            print(f"[{timestamp}] {func.__name__} → {result!r}")
            return result
        except Exception as e:
            print(f"[{timestamp}] {func.__name__} 异常：{e}")
            raise
    return wrapper

@log
def divide(a, b):
    return a / b

divide(10, 3)
# [14:30:25] 调用 divide(10, 3)
# [14:30:25] divide → 3.3333333333333335

try:
    divide(10, 0)
except ZeroDivisionError:
    pass
# [14:30:25] 调用 divide(10, 0)
# [14:30:25] divide 异常：division by zero
```

### 3.3 缓存装饰器

```python
from functools import wraps

def cache(func):
    """简易缓存：相同参数返回缓存结果"""
    _cache = {}

    @wraps(func)
    def wrapper(*args):
        if args in _cache:
            print(f"  [缓存命中] {func.__name__}{args}")
            return _cache[args]
        result = func(*args)
        _cache[args] = result
        return result

    wrapper.cache_clear = _cache.clear  # 提供清除缓存的方法
    return wrapper

@cache
def fibonacci(n):
    if n <= 1:
        return n
    return fibonacci(n - 1) + fibonacci(n - 2)

print(fibonacci(10))  # 55
print(fibonacci(10))  # [缓存命中] fibonacci(10,)  → 55

# 实际项目中直接用 functools.lru_cache
from functools import lru_cache

@lru_cache(maxsize=128)
def fib(n):
    if n <= 1:
        return n
    return fib(n - 1) + fib(n - 2)

print(fib(100))  # 354224848179261915075
print(fib.cache_info())
# CacheInfo(hits=98, misses=101, maxsize=128, currsize=101)
```

### 3.4 重试装饰器

```python
import time
import random
from functools import wraps

def retry(max_attempts=3, delay=1):
    """失败后自动重试"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_exception = None
            for attempt in range(1, max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except Exception as e:
                    last_exception = e
                    print(f"  [{func.__name__}] 第{attempt}次失败：{e}")
                    if attempt < max_attempts:
                        time.sleep(delay)
            raise last_exception
        return wrapper
    return decorator

@retry(max_attempts=3, delay=0.1)
def unstable_api():
    """模拟不稳定的网络请求"""
    if random.random() < 0.7:
        raise ConnectionError("网络超时")
    return {"status": "ok"}

try:
    result = unstable_api()
    print(f"成功：{result}")
except ConnectionError:
    print("多次重试后仍失败")
```

### 3.5 权限校验装饰器

```python
from functools import wraps

# 模拟当前用户
current_user = {"name": "Alice", "role": "admin"}

def require_role(*roles):
    """检查用户角色"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            user_role = current_user.get("role")
            if user_role not in roles:
                raise PermissionError(
                    f"需要 {roles} 权限，当前角色：{user_role}"
                )
            return func(*args, **kwargs)
        return wrapper
    return decorator

@require_role("admin")
def delete_user(user_id):
    print(f"删除用户 {user_id}")

@require_role("admin", "editor")
def edit_post(post_id):
    print(f"编辑文章 {post_id}")

# admin 可以执行
delete_user(1)   # 删除用户 1
edit_post(42)    # 编辑文章 42

# 切换为普通用户
current_user["role"] = "viewer"
try:
    delete_user(2)
except PermissionError as e:
    print(e)  # 需要 ('admin',) 权限，当前角色：viewer
```

### 3.6 类型检查装饰器

```python
from functools import wraps
import inspect

def type_check(func):
    """根据类型注解自动检查参数类型"""
    hints = func.__annotations__
    sig = inspect.signature(func)

    @wraps(func)
    def wrapper(*args, **kwargs):
        bound = sig.bind(*args, **kwargs)
        bound.apply_defaults()
        for param_name, value in bound.arguments.items():
            expected = hints.get(param_name)
            if expected and not isinstance(value, expected):
                raise TypeError(
                    f"参数 {param_name} 期望 {expected.__name__}，"
                    f"实际 {type(value).__name__}"
                )
        return func(*args, **kwargs)
    return wrapper

@type_check
def repeat(text: str, times: int) -> str:
    return text * times

print(repeat("hi", 3))  # hihihi

try:
    repeat("hi", "3")
except TypeError as e:
    print(e)  # 参数 times 期望 int，实际 str
```

## 4. 带参数的装饰器

### 4.1 三层嵌套结构

带参数的装饰器需要三层函数：外层接收参数，中层接收函数，内层执行逻辑。

```python
from functools import wraps

# @decorator(arg)  等价于  func = decorator(arg)(func)
# 所以 decorator(arg) 需要返回一个"真正的装饰器"

def repeat(n=2):
    """让函数重复执行 n 次"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            result = None
            for i in range(n):
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeat(n=3)
def say(msg):
    print(msg)

say("Hello!")
# Hello!
# Hello!
# Hello!

@repeat()       # 使用默认参数也需要加括号
def hi():
    print("Hi!")

hi()
# Hi!
# Hi!
```

### 4.2 同时支持 @deco 和 @deco(args)

```python
from functools import wraps

def timer(func=None, *, label=""):
    """
    灵活用法：
      @timer         → 无参数
      @timer()       → 无参数
      @timer(label="xxx") → 有参数
    """
    def decorator(fn):
        @wraps(fn)
        def wrapper(*args, **kwargs):
            import time
            start = time.perf_counter()
            result = fn(*args, **kwargs)
            elapsed = time.perf_counter() - start
            tag = label or fn.__name__
            print(f"[{tag}] {elapsed:.4f}s")
            return result
        return wrapper

    if func is not None:
        # @timer 无括号调用，func 就是被装饰的函数
        return decorator(func)
    # @timer() 或 @timer(label="xxx")，返回装饰器
    return decorator

@timer
def task_a():
    return sum(range(100_000))

@timer(label="任务B")
def task_b():
    return sum(range(100_000))

task_a()  # [task_a] 0.0023s
task_b()  # [任务B] 0.0022s
```

## 5. 多层装饰器

### 5.1 装饰器的执行顺序

多层装饰器**从下往上**装饰，**从上往下**执行。

```python
from functools import wraps

def decorator_a(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("A - 前")
        result = func(*args, **kwargs)
        print("A - 后")
        return result
    return wrapper

def decorator_b(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print("B - 前")
        result = func(*args, **kwargs)
        print("B - 后")
        return result
    return wrapper

@decorator_a     # 第 2 步：hello = decorator_a(decorator_b(hello))
@decorator_b     # 第 1 步：hello = decorator_b(hello)
def hello():
    print("Hello!")

hello()
# A - 前      ← 最外层的装饰器先执行"前"
# B - 前
# Hello!      ← 原函数
# B - 后
# A - 后      ← 最外层的装饰器最后执行"后"

# 想象成洋葱模型：A 包裹 B，B 包裹原函数
# A前 → B前 → 原函数 → B后 → A后
```

### 5.2 实际示例

```python
from functools import wraps
import time

def timer(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        start = time.perf_counter()
        result = func(*args, **kwargs)
        print(f"  ⏱ {func.__name__} 耗时 {time.perf_counter() - start:.4f}s")
        return result
    return wrapper

def log(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        print(f"  📝 调用 {func.__name__}")
        result = func(*args, **kwargs)
        print(f"  📝 {func.__name__} 返回 {result!r}")
        return result
    return wrapper

def cache(func):
    _cache = {}
    @wraps(func)
    def wrapper(*args):
        if args not in _cache:
            _cache[args] = func(*args)
        return _cache[args]
    return wrapper

@timer      # 最外层：计时（包含了日志和缓存的时间）
@log        # 中间层：日志
@cache      # 最内层：缓存
def fib(n):
    if n <= 1:
        return n
    return fib(n - 1) + fib(n - 2)

print(fib(5))
#   📝 调用 fib
#   📝 fib 返回 5
#   ⏱ fib 耗时 0.0001s
# 5
```

## 6. 类装饰器

### 6.1 用类实现装饰器

利用 `__call__` 方法让类的实例像函数一样可调用。

```python
from functools import wraps

class CountCalls:
    """统计函数被调用的次数"""
    def __init__(self, func):
        wraps(func)(self)    # 等价于用 @wraps 保留元信息
        self.func = func
        self.count = 0

    def __call__(self, *args, **kwargs):
        self.count += 1
        print(f"  [{self.func.__name__}] 第 {self.count} 次调用")
        return self.func(*args, **kwargs)

@CountCalls
def say_hello(name):
    """打招呼"""
    print(f"  Hello, {name}!")

say_hello("Alice")   #   [say_hello] 第 1 次调用  →  Hello, Alice!
say_hello("Bob")     #   [say_hello] 第 2 次调用  →  Hello, Bob!
say_hello("Charlie") #   [say_hello] 第 3 次调用  →  Hello, Charlie!

print(f"总调用次数：{say_hello.count}")  # 总调用次数：3
print(say_hello.__name__)  # say_hello
print(say_hello.__doc__)   # 打招呼
```

### 6.2 带参数的类装饰器

```python
class Retry:
    """带参数的重试类装饰器"""
    def __init__(self, max_attempts=3, exceptions=(Exception,)):
        self.max_attempts = max_attempts
        self.exceptions = exceptions

    def __call__(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            last_err = None
            for attempt in range(1, self.max_attempts + 1):
                try:
                    return func(*args, **kwargs)
                except self.exceptions as e:
                    last_err = e
                    print(f"  [{func.__name__}] 第{attempt}次失败：{e}")
            raise last_err
        return wrapper

import random

@Retry(max_attempts=5, exceptions=(ValueError,))
def parse_data():
    if random.random() < 0.7:
        raise ValueError("数据格式错误")
    return {"data": 42}

try:
    result = parse_data()
    print(f"成功：{result}")
except ValueError:
    print("多次重试后仍失败")
```

### 6.3 用装饰器装饰类

装饰器不仅可以装饰函数，也可以装饰类。

```python
def singleton(cls):
    """单例模式装饰器：确保一个类只有一个实例"""
    instances = {}
    @wraps(cls, updated=[])  # 类没有 __wrapped__ 等属性，用 updated=[]
    def get_instance(*args, **kwargs):
        if cls not in instances:
            instances[cls] = cls(*args, **kwargs)
        return instances[cls]
    return get_instance

@singleton
class Database:
    def __init__(self):
        print("  创建数据库连接...")
        self.connected = True

db1 = Database()   #   创建数据库连接...
db2 = Database()   #  （没有打印，使用了同一个实例）
print(db1 is db2)  # True

# ---

def add_repr(cls):
    """为类自动添加 __repr__ 方法"""
    def __repr__(self):
        attrs = ", ".join(f"{k}={v!r}" for k, v in self.__dict__.items())
        return f"{cls.__name__}({attrs})"
    cls.__repr__ = __repr__
    return cls

@add_repr
class User:
    def __init__(self, name, age):
        self.name = name
        self.age = age

print(User("Alice", 25))  # User(name='Alice', age=25)
```

## 7. 内置装饰器

### 7.1 @property

```python
class Temperature:
    def __init__(self, celsius):
        self._celsius = celsius

    @property
    def celsius(self):
        return self._celsius

    @celsius.setter
    def celsius(self, value):
        if value < -273.15:
            raise ValueError("温度不能低于绝对零度")
        self._celsius = value

    @property
    def fahrenheit(self):
        return self._celsius * 9 / 5 + 32

t = Temperature(100)
print(t.celsius)      # 100
print(t.fahrenheit)   # 212.0

t.celsius = 0
print(t.fahrenheit)   # 32.0

try:
    t.celsius = -300
except ValueError as e:
    print(e)  # 温度不能低于绝对零度
```

class MathHelper:

print(MathHelper.add(3, 5))          # 8

## 8. 高级技巧

### 8.1 装饰器工厂

```python
from functools import wraps

def tag(tag_name):
    """HTML 标签包裹装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            result = func(*args, **kwargs)
            return f"<{tag_name}>{result}</{tag_name}>"
        return wrapper
    return decorator

@tag("b")
@tag("i")
def greet(name):
    return f"Hello, {name}"

print(greet("Alice"))  # <b><i>Hello, Alice</i></b>
```

### 8.2 保留装饰器可选参数的通用模板

```python
from functools import wraps

def decorator(func=None, *, param1="default1", param2="default2"):
    """
    通用模板：同时支持以下写法：
      @decorator
      @decorator()
      @decorator(param1="xxx")
    """
    def actual_decorator(fn):
        @wraps(fn)
        def wrapper(*args, **kwargs):
            print(f"  param1={param1}, param2={param2}")
            return fn(*args, **kwargs)
        return wrapper

    if func is not None:
        return actual_decorator(func)
    return actual_decorator

@decorator
def task_a():
    print("  执行 A")

@decorator()
def task_b():
    print("  执行 B")

@decorator(param1="自定义", param2="参数")
def task_c():
    print("  执行 C")

task_a()
#   param1=default1, param2=default2
#   执行 A

task_b()
#   param1=default1, param2=default2
#   执行 B

task_c()
#   param1=自定义, param2=参数
#   执行 C
```

### 8.3 基于描述符的方法装饰器

当装饰器装饰类的方法时，普通闭包装饰器可能丢失对 `self` 的感知。使用描述符协议可以解决。

```python
from functools import wraps

class MethodLogger:
    """作为描述符的方法装饰器"""
    def __init__(self, func):
        wraps(func)(self)
        self.func = func

    def __set_name__(self, owner, name):
        self.owner = owner
        self.name = name

    def __get__(self, obj, objtype=None):
        if obj is None:
            return self
        # 返回绑定了 obj 的 wrapper
        @wraps(self.func)
        def wrapper(*args, **kwargs):
            print(f"  [{self.owner.__name__}.{self.name}] 被调用")
            return self.func(obj, *args, **kwargs)
        return wrapper

class MyService:
    @MethodLogger
    def process(self, data):
        return f"处理了 {data}"

    @MethodLogger
    def save(self, data):
        return f"保存了 {data}"

svc = MyService()
print(svc.process("hello"))
#   [MyService.process] 被调用
# 处理了 hello

print(svc.save("world"))
#   [MyService.save] 被调用
# 保存了 world
```

### 示例 1：API 接口模拟框架

```python
from functools import wraps

# 简易路由注册
_routes = {}

def route(path, method="GET"):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            return func(*args, **kwargs)
        _routes[(method, path)] = wrapper
        return wrapper
    return decorator

def require_auth(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        token = kwargs.get("token")
        if not token:
            return {"error": "未授权", "code": 401}
        return func(*args, **kwargs)
    return wrapper

@route("/users", method="GET")
def list_users():
    return {"users": ["Alice", "Bob", "Charlie"]}

@route("/users", method="POST")
@require_auth
def create_user(name="", **kwargs):
    return {"message": f"创建用户 {name}", "code": 200}

# 模拟请求
print(list_users())
# {'users': ['Alice', 'Bob', 'Charlie']}

print(create_user(name="David"))
# {'error': '未授权', 'code': 401}

print(create_user(name="David", token="abc123"))
# {'message': '创建用户 David', 'code': 200}

# 查看注册的路由
for (method, path), handler in _routes.items():
    print(f"  {method:>6s} {path} → {handler.__name__}")
#     GET /users → list_users
#    POST /users → create_user
```

### 示例 2：性能监控装饰器

```python
import time
from functools import wraps
from collections import defaultdict

class PerfMonitor:
    """收集所有被装饰函数的性能数据"""
    def __init__(self):
        self.stats = defaultdict(lambda: {"calls": 0, "total_time": 0, "errors": 0})

    def monitor(self, func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            start = time.perf_counter()
            try:
                result = func(*args, **kwargs)
                return result
            except Exception:
                self.stats[func.__name__]["errors"] += 1
                raise
            finally:
                elapsed = time.perf_counter() - start
                self.stats[func.__name__]["calls"] += 1
                self.stats[func.__name__]["total_time"] += elapsed
        return wrapper

    def report(self):
        print(f"{'函数':<15s} {'调用次数':>8s} {'总耗时':>10s} {'平均耗时':>10s} {'错误':>5s}")
        print("-" * 52)
        for name, data in self.stats.items():
            avg = data["total_time"] / data["calls"] if data["calls"] else 0
            print(f"{name:<15s} {data['calls']:>8d} {data['total_time']:>10.4f}s {avg:>10.4f}s {data['errors']:>5d}")

perf = PerfMonitor()

@perf.monitor
def fast_task():
    return sum(range(1000))

@perf.monitor
def slow_task():
    time.sleep(0.1)
    return "done"

for _ in range(5):
    fast_task()
for _ in range(3):
    slow_task()

perf.report()
# 函数                调用次数       总耗时       平均耗时    错误
# ----------------------------------------------------
# fast_task              5     0.0002s     0.0000s     0
# slow_task              3     0.3015s     0.1005s     0
```

## 10. 总结

### 装饰器模板

```python
# === 无参数装饰器 ===
from functools import wraps

def my_decorator(func):
    @wraps(func)
    def wrapper(*args, **kwargs):
        # 前置逻辑
        result = func(*args, **kwargs)
        # 后置逻辑
        return result
    return wrapper

# === 带参数装饰器 ===
def my_decorator(param1="default"):
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 使用 param1
            result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator
```

### 内置 / 标准库装饰器速查

| 装饰器 | 来源 | 用途 |
|--------|------|------|
| `@property` | 内置 | 将方法变为属性访问 |
| `@staticmethod` | 内置 | 静态方法 |
| `@classmethod` | 内置 | 类方法 |
| `@abstractmethod` | `abc` | 抽象方法 |
| `@wraps` | `functools` | 保留被装饰函数的元信息 |
| `@lru_cache` | `functools` | 缓存函数结果 |
| `@cache` | `functools` (3.9+) | 无限缓存 |
| `@total_ordering` | `functools` | 自动生成比较方法 |
| `@dataclass` | `dataclasses` | 自动生成类的样板方法 |

### 使用原则

1. **永远加 `@wraps(func)`**，保留原函数的元信息
2. **`*args, **kwargs` 透传参数**，确保装饰器通用
3. **返回原函数的返回值**，不要把 `return` 吞掉
4. 简单需求用**函数装饰器**，需要维护状态时考虑**类装饰器**
5. 多个装饰器从下往上装饰、从上往下执行（洋葱模型）
6. 能用标准库的就不要自己写（`lru_cache`、`dataclass` 等）
