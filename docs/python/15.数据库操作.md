# Python 数据库操作

数据库是应用程序持久化数据的核心。本章涵盖 SQLite、MySQL、PostgreSQL 的原生操作，以及 SQLAlchemy ORM 和 Redis 的使用。

## 1. SQLite —— 轻量级内嵌数据库

SQLite 是 Python 标准库自带的数据库，无需安装服务，数据存储在单个文件中。

### 1.1 基本操作

```python
import sqlite3

# 连接数据库（文件不存在会自动创建）
conn = sqlite3.connect("demo.db")
# conn = sqlite3.connect(":memory:")  # 内存数据库（临时使用）

# 创建游标
cursor = conn.cursor()

# 创建表
cursor.execute("""
    CREATE TABLE IF NOT EXISTS users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        email TEXT UNIQUE,
        age INTEGER DEFAULT 0,
        created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
    )
""")

# 插入数据
cursor.execute(
    "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
    ("Alice", "alice@test.com", 25)
)

# 批量插入
users = [
    ("Bob", "bob@test.com", 30),
    ("Charlie", "charlie@test.com", 28),
    ("David", "david@test.com", 35),
]
cursor.executemany(
    "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
    users
)

# 提交事务
conn.commit()

# 查询数据
cursor.execute("SELECT id, name, email, age FROM users")
rows = cursor.fetchall()
for row in rows:
    print(f"  {row[0]}: {row[1]} <{row[2]}>, {row[3]}岁")
# 1: Alice <alice@test.com>, 25岁
# 2: Bob <bob@test.com>, 30岁
# 3: Charlie <charlie@test.com>, 28岁
# 4: David <david@test.com>, 35岁

# 关闭连接
cursor.close()
conn.close()

import os
os.remove("demo.db")
```

### 1.2 参数化查询与防注入

```python
import sqlite3

conn = sqlite3.connect(":memory:")
cursor = conn.cursor()
cursor.execute("CREATE TABLE users (id INTEGER PRIMARY KEY, name TEXT, age INTEGER)")
cursor.executemany("INSERT INTO users (name, age) VALUES (?, ?)",
                   [("Alice", 25), ("Bob", 30), ("Charlie", 28)])
conn.commit()

# ✅ 参数化查询（安全，防止 SQL 注入）
name = "Alice"
cursor.execute("SELECT * FROM users WHERE name = ?", (name,))
print(cursor.fetchone())  # (1, 'Alice', 25)

# 命名参数
cursor.execute("SELECT * FROM users WHERE age > :min_age", {"min_age": 26})
print(cursor.fetchall())  # [(2, 'Bob', 30), (3, 'Charlie', 28)]

# ❌ 永远不要用字符串拼接（SQL 注入风险）
# cursor.execute(f"SELECT * FROM users WHERE name = '{user_input}'")  # 危险！

conn.close()
```

### 1.3 使用 Row 工厂（按列名访问）

```python
import sqlite3

conn = sqlite3.connect(":memory:")
conn.row_factory = sqlite3.Row  # 设置行工厂

conn.commit()

cursor.execute("SELECT * FROM users WHERE name = ?", ("Alice",))
row = cursor.fetchone()

# 可以按列名访问（而不只是索引）
print(row["name"])     # Alice
print(row["age"])      # 25

print(row.keys())      # ['id', 'name', 'age']

conn.close()
```

### 1.4 上下文管理器与事务

```python
import sqlite3

# with 语句自动管理事务：成功时 commit，异常时 rollback
conn = sqlite3.connect(":memory:")
cursor = conn.cursor()
cursor.execute("CREATE TABLE accounts (id INTEGER PRIMARY KEY, name TEXT, balance REAL)")
cursor.executemany("INSERT INTO accounts (name, balance) VALUES (?, ?)",
                   [("Alice", 1000), ("Bob", 500)])
conn.commit()

def transfer(conn, from_name, to_name, amount):
    """转账（事务保证原子性）"""
    try:
        with conn:  # 自动 commit 或 rollback
            cursor = conn.cursor()
            cursor.execute("UPDATE accounts SET balance = balance - ? WHERE name = ?",
                           (amount, from_name))
            cursor.execute("UPDATE accounts SET balance = balance + ? WHERE name = ?",
                           (amount, to_name))

            # 检查余额
            cursor.execute("SELECT balance FROM accounts WHERE name = ?", (from_name,))
            balance = cursor.fetchone()[0]
            if balance < 0:
                raise ValueError(f"{from_name} 余额不足")

        print(f"转账成功：{from_name} → {to_name}，金额 {amount}")
    except ValueError as e:
        print(f"转账失败：{e}")

# 正常转账
transfer(conn, "Alice", "Bob", 300)
# 转账成功：Alice → Bob，金额 300

# 余额不足
transfer(conn, "Alice", "Bob", 9999)
# 转账失败：Alice 余额不足

# 查看最终余额
cursor = conn.cursor()
cursor.execute("SELECT name, balance FROM accounts")
for row in cursor.fetchall():
    print(f"  {row[0]}: {row[1]}")
# Alice: 700.0
# Bob: 800.0

conn.close()
```

```python
import pymysql

# 连接 MySQL
conn = pymysql.connect(
    host="localhost",
    port=3306,
    user="root",
    password="your_password",
    database="testdb",
    charset="utf8mb4",
    cursorclass=pymysql.cursors.DictCursor,  # 返回字典格式
)

try:
    with conn.cursor() as cursor:
        # 创建表
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS products (
                id INT AUTO_INCREMENT PRIMARY KEY,
                name VARCHAR(100) NOT NULL,
                price DECIMAL(10, 2) NOT NULL,
                stock INT DEFAULT 0,
                created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            ) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4
        """)

        # 插入数据
        cursor.execute(
            "INSERT INTO products (name, price, stock) VALUES (%s, %s, %s)",
            ("Python 编程", 59.90, 100)
        )

        # 批量插入
        products = [
            ("数据结构", 49.90, 80),
            ("算法导论", 89.00, 50),
            ("深度学习", 79.50, 60),
        ]
        cursor.executemany(
            "INSERT INTO products (name, price, stock) VALUES (%s, %s, %s)",
            products
        )

    conn.commit()  # 提交事务

    with conn.cursor() as cursor:
        # 查询
        cursor.execute("SELECT * FROM products WHERE price < %s ORDER BY price", (70,))
        rows = cursor.fetchall()
        for row in rows:
            print(f"  {row['name']}: ¥{row['price']}, 库存 {row['stock']}")
        # 数据结构: ¥49.90, 库存 80
        # Python 编程: ¥59.90, 库存 100

        # 聚合查询
        cursor.execute("SELECT COUNT(*) as total, AVG(price) as avg_price FROM products")
        stats = cursor.fetchone()
        print(f"  共 {stats['total']} 种商品，均价 ¥{stats['avg_price']:.2f}")
        # 共 4 种商品，均价 ¥69.58

finally:
    conn.close()
```

```python
import psycopg2
from psycopg2.extras import RealDictCursor

# 连接 PostgreSQL
conn = psycopg2.connect(
    host="localhost",
    port=5432,
    user="postgres",
    password="your_password",
    dbname="testdb",
)
conn.autocommit = False  # 手动管理事务（默认行为）

try:
    with conn.cursor(cursor_factory=RealDictCursor) as cursor:
        # 创建表（PostgreSQL 支持更丰富的数据类型）
        cursor.execute("""
            CREATE TABLE IF NOT EXISTS articles (
                id SERIAL PRIMARY KEY,
                title VARCHAR(200) NOT NULL,
                content TEXT,
                tags TEXT[] DEFAULT '{}',
                metadata JSONB DEFAULT '{}',
                views INTEGER DEFAULT 0,
                published BOOLEAN DEFAULT FALSE,
                created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
            )
        """)

        # 插入数据（PostgreSQL 使用 %s 占位符）
        cursor.execute("""
            INSERT INTO articles (title, content, tags, metadata, published)
            VALUES (%s, %s, %s, %s, %s)
            RETURNING id, title
        """, (
            "Python 入门",
            "Python 是一门优雅的语言...",
            ["python", "教程", "入门"],      # 数组类型
            '{"author": "Alice", "level": "beginner"}',  # JSONB
            True,
        ))

        new_article = cursor.fetchone()
        print(f"  创建文章：[{new_article['id']}] {new_article['title']}")

        # 批量插入
        from psycopg2.extras import execute_values
        articles = [
            ("数据库设计", "关系型数据库设计原则...", ["database", "sql"], '{"author": "Bob"}', True),
            ("算法基础", "常见排序算法...", ["algorithm"], '{"author": "Charlie"}', False),
        ]
        execute_values(
            cursor,
            "INSERT INTO articles (title, content, tags, metadata, published) VALUES %s",
            articles
        )

    conn.commit()

    with conn.cursor(cursor_factory=RealDictCursor) as cursor:
        # 查询
        cursor.execute("SELECT id, title, tags, published FROM articles ORDER BY id")
        for row in cursor.fetchall():
            status = "已发布" if row["published"] else "草稿"
            print(f"  [{row['id']}] {row['title']} | 标签：{row['tags']} | {status}")
        # [1] Python 入门 | 标签：['python', '教程', '入门'] | 已发布
        # [2] 数据库设计 | 标签：['database', 'sql'] | 已发布
        # [3] 算法基础 | 标签：['algorithm'] | 草稿

except Exception:
    conn.rollback()
    raise
finally:
    conn.close()
```

### 3.2 PostgreSQL 特色查询

```python
import psycopg2
from psycopg2.extras import RealDictCursor

conn = psycopg2.connect(host="localhost", dbname="testdb", user="postgres", password="your_password")

with conn.cursor(cursor_factory=RealDictCursor) as cursor:

    # ===== 数组查询 =====
    # 包含指定标签的文章
    cursor.execute("SELECT title FROM articles WHERE %s = ANY(tags)", ("python",))
    print("标签含 python：", [r["title"] for r in cursor.fetchall()])
    # 标签含 python：['Python 入门']

    # 标签数组重叠
    cursor.execute("SELECT title FROM articles WHERE tags && %s", (["python", "database"],))
    print("标签含 python 或 database：", [r["title"] for r in cursor.fetchall()])

    # ===== JSONB 查询 =====
    # 按 JSON 字段查询
    cursor.execute("""SELECT title FROM articles WHERE metadata->>'author' = %s""", ("Alice",))
    print("Alice 的文章：", [r["title"] for r in cursor.fetchall()])
    # Alice 的文章：['Python 入门']

    # JSON 包含查询
    cursor.execute("""SELECT title FROM articles WHERE metadata @> '{"level": "beginner"}'""")
    print("初级文章：", [r["title"] for r in cursor.fetchall()])

    # ===== UPSERT（插入或更新）=====
    cursor.execute("""
        INSERT INTO articles (title, content, tags, published)
        VALUES (%s, %s, %s, %s)
        ON CONFLICT (title)
        DO UPDATE SET content = EXCLUDED.content, published = EXCLUDED.published
    """, ("Python 入门", "更新后的内容...", ["python"], True))

    # ===== CTE（公共表表达式）=====
    cursor.execute("""
        WITH published AS (
            SELECT * FROM articles WHERE published = TRUE
        )
        SELECT title, array_length(tags, 1) as tag_count
        FROM published
        ORDER BY tag_count DESC
    """)
    for row in cursor.fetchall():
        print(f"  {row['title']}: {row['tag_count']} 个标签")

conn.commit()
conn.close()
```

### 4.2 定义模型

```python
from datetime import datetime
from typing import Optional
from sqlalchemy import String, Integer, Float, Boolean, Text, ForeignKey, func
from sqlalchemy.orm import (
    DeclarativeBase, Mapped, mapped_column, relationship, Session
)
from sqlalchemy import create_engine

# 基类
class Base(DeclarativeBase):
    pass

# 用户表
class User(Base):
    __tablename__ = "users"

    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str] = mapped_column(String(50))
    email: Mapped[str] = mapped_column(String(100), unique=True)
    age: Mapped[Optional[int]] = mapped_column(Integer, default=0)
    active: Mapped[bool] = mapped_column(Boolean, default=True)
    created_at: Mapped[datetime] = mapped_column(default=func.now())

    # 关系：一个用户有多篇文章
    posts: Mapped[list["Post"]] = relationship(back_populates="author", cascade="all, delete-orphan")

    def __repr__(self):
        return f"User(id={self.id}, name={self.name!r})"

# 文章表
class Post(Base):
    __tablename__ = "posts"

    id: Mapped[int] = mapped_column(primary_key=True)
    title: Mapped[str] = mapped_column(String(200))
    content: Mapped[Optional[str]] = mapped_column(Text)
    views: Mapped[int] = mapped_column(Integer, default=0)
    published: Mapped[bool] = mapped_column(Boolean, default=False)
    author_id: Mapped[int] = mapped_column(ForeignKey("users.id"))
    created_at: Mapped[datetime] = mapped_column(default=func.now())

    # 关系：多对一
    author: Mapped["User"] = relationship(back_populates="posts")

    def __repr__(self):
        return f"Post(id={self.id}, title={self.title!r})"

# 创建表
engine = create_engine("sqlite:///blog.db", echo=False)
Base.metadata.create_all(engine)
```

### 4.3 CRUD 操作

```python
from sqlalchemy.orm import Session

# ===== 创建（Create）=====
with Session(engine) as session:
    # 创建用户
    alice = User(name="Alice", email="alice@test.com", age=25)
    bob = User(name="Bob", email="bob@test.com", age=30)
    session.add_all([alice, bob])
    session.flush()  # 刷新以获取自动生成的 ID

    # 创建文章（自动关联外键）
    post1 = Post(title="Python 入门", content="学习 Python...", author=alice, published=True)
    post2 = Post(title="SQLAlchemy 教程", content="ORM 基础...", author=alice, published=True)
    post3 = Post(title="草稿文章", content="待完善...", author=bob, published=False)
    session.add_all([post1, post2, post3])
    session.commit()

    print(f"Alice ID: {alice.id}, Bob ID: {bob.id}")
    # Alice ID: 1, Bob ID: 2

# ===== 查询（Read）=====
from sqlalchemy import select

with Session(engine) as session:
    # 查询单个
    stmt = select(User).where(User.name == "Alice")
    alice = session.execute(stmt).scalar_one()
    print(alice)  # User(id=1, name='Alice')

    # 查询所有
    stmt = select(User).order_by(User.name)
    users = session.execute(stmt).scalars().all()
    print(users)  # [User(id=1, name='Alice'), User(id=2, name='Bob')]

    # 条件查询
    stmt = select(Post).where(Post.published == True).order_by(Post.views.desc())
    published_posts = session.execute(stmt).scalars().all()
    for post in published_posts:
        print(f"  {post.title} (作者: {post.author.name})")
    # Python 入门 (作者: Alice)
    # SQLAlchemy 教程 (作者: Alice)

    # 关联查询
    stmt = select(User).where(User.name == "Alice")
    alice = session.execute(stmt).scalar_one()
    print(f"Alice 的文章：{[p.title for p in alice.posts]}")
    # Alice 的文章：['Python 入门', 'SQLAlchemy 教程']

# ===== 更新（Update）=====
with Session(engine) as session:
    stmt = select(Post).where(Post.title == "Python 入门")
    post = session.execute(stmt).scalar_one()

    post.views = 100
    post.title = "Python 入门指南"
    session.commit()

    print(f"更新后：{post.title}, 浏览量 {post.views}")
    # 更新后：Python 入门指南, 浏览量 100

# ===== 删除（Delete）=====
with Session(engine) as session:
    stmt = select(Post).where(Post.title == "草稿文章")
    post = session.execute(stmt).scalar_one()
    session.delete(post)
    session.commit()
    print("草稿文章已删除")
```

```python
from sqlalchemy import select, func, and_, or_, desc

with Session(engine) as session:
    # 模糊查询
    stmt = select(Post).where(Post.title.like("%Python%"))
    results = session.execute(stmt).scalars().all()
    print(f"含 Python 的文章：{[r.title for r in results]}")

    # IN 查询
    stmt = select(User).where(User.name.in_(["Alice", "Bob", "Charlie"]))
    users = session.execute(stmt).scalars().all()
    print(f"匹配的用户：{[u.name for u in users]}")

    # AND / OR
    stmt = select(Post).where(
        and_(Post.published == True, Post.views >= 50)
    )

    stmt = select(Post).where(
        or_(Post.views > 100, Post.published == False)
    )

    # 聚合查询
    stmt = select(
        User.name,
        func.count(Post.id).label("post_count"),
        func.coalesce(func.sum(Post.views), 0).label("total_views"),
    ).join(Post, isouter=True).group_by(User.id)

    for row in session.execute(stmt):
        print(f"  {row.name}: {row.post_count} 篇文章, {row.total_views} 浏览")
    # Alice: 2 篇文章, 100 浏览
    # Bob: 0 篇文章, 0 浏览

    # 分页
    page, per_page = 1, 10
    stmt = (
        select(Post)
        .order_by(desc(Post.created_at))
        .offset((page - 1) * per_page)
        .limit(per_page)
    )
    posts = session.execute(stmt).scalars().all()

    # 子查询
    subq = (
        select(func.count(Post.id))
        .where(Post.author_id == User.id)
        .correlate(User)
        .scalar_subquery()
    )
    stmt = select(User.name, subq.label("post_count"))
    for row in session.execute(stmt):
```

### 6.2 数据结构操作

```python
import redis

r = redis.Redis(host="localhost", port=6379, db=0, decode_responses=True)

# ===== 哈希（Hash）—— 存储对象 =====
r.hset("user:1", mapping={"name": "Alice", "age": "25", "city": "Beijing"})

print(r.hget("user:1", "name"))     # Alice
print(r.hgetall("user:1"))          # {'name': 'Alice', 'age': '25', 'city': 'Beijing'}

r.hincrby("user:1", "age", 1)
print(r.hget("user:1", "age"))      # 26

# ===== 列表（List）—— 队列/栈 =====
r.rpush("tasks", "task1", "task2", "task3")   # 右侧入队
print(r.lrange("tasks", 0, -1))                # ['task1', 'task2', 'task3']
print(r.lpop("tasks"))                          # task1  （左侧出队 → 队列）
print(r.llen("tasks"))                          # 2

# ===== 集合（Set）—— 去重/交并集 =====
r.sadd("skills:alice", "python", "sql", "docker")
r.sadd("skills:bob", "python", "java", "docker")

print(r.smembers("skills:alice"))                # {'python', 'sql', 'docker'}
print(r.sinter("skills:alice", "skills:bob"))    # {'python', 'docker'} 交集
print(r.sunion("skills:alice", "skills:bob"))    # {'python', 'sql', 'docker', 'java'} 并集
print(r.sdiff("skills:alice", "skills:bob"))     # {'sql'} 差集

# ===== 有序集合（Sorted Set）—— 排行榜 =====
r.zadd("leaderboard", {"Alice": 95, "Bob": 87, "Charlie": 92, "David": 88})

# 按分数从高到低

# [('Alice', 95.0), ('Charlie', 92.0), ('David', 88.0), ('Bob', 87.0)]

# 获取排名（从 0 开始，从高到低）

print(r.zscore("leaderboard", "Bob"))       # 87.0

# 加分
r.zincrby("leaderboard", 10, "Bob")
print(r.zscore("leaderboard", "Bob"))       # 97.0

# 清理
r.delete("user:1", "tasks", "skills:alice", "skills:bob", "leaderboard",
         "name", "counter", "token")
```

### 6.3 缓存模式

```python
import redis
import json
import time
from functools import wraps

r = redis.Redis(host="localhost", port=6379, db=0, decode_responses=True)

def cache(prefix="cache", ttl=300):
    """Redis 缓存装饰器"""
    def decorator(func):
        @wraps(func)
        def wrapper(*args, **kwargs):
            # 生成缓存键
            key = f"{prefix}:{func.__name__}:{args}:{kwargs}"

            # 尝试从缓存获取
            cached = r.get(key)
            if cached is not None:
                print(f"  [缓存命中] {key}")
                return json.loads(cached)

            # 缓存未命中，执行函数
            print(f"  [缓存未命中] {key}")
            result = func(*args, **kwargs)

            # 存入缓存
            r.set(key, json.dumps(result, ensure_ascii=False), ex=ttl)
            return result
        return wrapper
    return decorator

@cache(prefix="api", ttl=60)
def get_user_profile(user_id):
    """模拟耗时的数据库查询"""
    time.sleep(0.5)  # 模拟查询延迟
    return {"id": user_id, "name": f"User_{user_id}", "score": 95}

# 第一次调用：缓存未命中
result = get_user_profile(42)
print(result)
#   [缓存未命中] api:get_user_profile:(42,):{}
# {'id': 42, 'name': 'User_42', 'score': 95}

# 第二次调用：缓存命中
result = get_user_profile(42)
print(result)
#   [缓存命中] api:get_user_profile:(42,):{}
# {'id': 42, 'name': 'User_42', 'score': 95}
```

## 7. 实战项目：博客系统数据层

```python
"""
简易博客系统的数据访问层
演示 SQLAlchemy ORM 的完整用法
"""
from datetime import datetime
from typing import Optional
from sqlalchemy import create_engine, select, func, String, Integer, Text, Boolean, ForeignKey
from sqlalchemy.orm import (
    DeclarativeBase, Mapped, mapped_column, relationship,
    Session, sessionmaker,
)

# ===== 模型定义 =====

class Base(DeclarativeBase):
    pass

class Author(Base):
    __tablename__ = "authors"
    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str] = mapped_column(String(50))
    email: Mapped[str] = mapped_column(String(100), unique=True)
    posts: Mapped[list["BlogPost"]] = relationship(back_populates="author")

    def __repr__(self):
        return f"Author({self.name!r})"

class BlogPost(Base):
    __tablename__ = "blog_posts"
    id: Mapped[int] = mapped_column(primary_key=True)
    title: Mapped[str] = mapped_column(String(200))
    content: Mapped[str] = mapped_column(Text, default="")
    views: Mapped[int] = mapped_column(Integer, default=0)
    published: Mapped[bool] = mapped_column(Boolean, default=False)
    author_id: Mapped[int] = mapped_column(ForeignKey("authors.id"))
    created_at: Mapped[datetime] = mapped_column(default=func.now())
    author: Mapped["Author"] = relationship(back_populates="posts")

    def __repr__(self):
        return f"Post({self.title!r})"

# ===== 数据访问层（Repository）=====

class BlogRepository:
    def __init__(self, session: Session):
        self.session = session

    # --- 作者 ---
    def create_author(self, name: str, email: str) -> Author:
        author = Author(name=name, email=email)
        self.session.add(author)
        self.session.flush()
        return author

    def get_author(self, author_id: int) -> Optional[Author]:
        return self.session.get(Author, author_id)

    # --- 文章 ---
    def create_post(self, title: str, content: str, author_id: int) -> BlogPost:
        post = BlogPost(title=title, content=content, author_id=author_id)
        self.session.add(post)
        self.session.flush()
        return post

    def get_post(self, post_id: int) -> Optional[BlogPost]:
        return self.session.get(BlogPost, post_id)

    def list_posts(self, published_only: bool = True,
                   page: int = 1, per_page: int = 10) -> list[BlogPost]:
        stmt = select(BlogPost).order_by(BlogPost.created_at.desc())
        if published_only:
            stmt = stmt.where(BlogPost.published == True)
        stmt = stmt.offset((page - 1) * per_page).limit(per_page)
        return list(self.session.execute(stmt).scalars())

    def publish_post(self, post_id: int) -> bool:
        post = self.get_post(post_id)
        if post:
            post.published = True
            return True
        return False

    def increment_views(self, post_id: int) -> None:
        post = self.get_post(post_id)
        if post:
            post.views += 1

    def search_posts(self, keyword: str) -> list[BlogPost]:
        stmt = select(BlogPost).where(
            BlogPost.title.contains(keyword) | BlogPost.content.contains(keyword)
        )
        return list(self.session.execute(stmt).scalars())

    def get_stats(self) -> dict:
        total = self.session.execute(select(func.count(BlogPost.id))).scalar()
        published = self.session.execute(
            select(func.count(BlogPost.id)).where(BlogPost.published == True)
        ).scalar()
        total_views = self.session.execute(select(func.sum(BlogPost.views))).scalar() or 0
        return {"total": total, "published": published, "total_views": total_views}


# ===== 使用示例 =====

engine = create_engine("sqlite:///blog_demo.db", echo=False)
Base.metadata.create_all(engine)
SessionLocal = sessionmaker(bind=engine)

with SessionLocal() as session:
    repo = BlogRepository(session)

    # 创建作者
    alice = repo.create_author("Alice", "alice@blog.com")
    bob = repo.create_author("Bob", "bob@blog.com")

    # 创建文章
    p1 = repo.create_post("Python ORM 实战", "使用 SQLAlchemy...", alice.id)
    p2 = repo.create_post("数据库设计原则", "三大范式...", alice.id)
    p3 = repo.create_post("Redis 缓存策略", "缓存穿透与雪崩...", bob.id)
    p4 = repo.create_post("草稿文章", "待完善...", bob.id)

    # 发布文章
    repo.publish_post(p1.id)
    repo.publish_post(p2.id)
    repo.publish_post(p3.id)

    # 模拟浏览
    for _ in range(50):
        repo.increment_views(p1.id)
    for _ in range(30):
        repo.increment_views(p2.id)

    session.commit()

    # 查询已发布文章
    print("已发布文章：")
    for post in repo.list_posts(published_only=True):
        print(f"  [{post.id}] {post.title} | 👁 {post.views} | 作者: {post.author.name}")
    # [1] Python ORM 实战 | 👁 50 | 作者: Alice
    # [2] 数据库设计原则 | 👁 30 | 作者: Alice
    # [3] Redis 缓存策略 | 👁 0 | 作者: Bob

    # 搜索
    print("\n搜索 'Python'：")
    for post in repo.search_posts("Python"):
        print(f"  {post.title}")
    # Python ORM 实战

    # 统计
    stats = repo.get_stats()
    print(f"\n统计：共 {stats['total']} 篇文章，"
          f"{stats['published']} 篇已发布，总浏览 {stats['total_views']} 次")
    # 统计：共 4 篇文章，3 篇已发布，总浏览 80 次

import os
os.remove("blog_demo.db")
```

## 8. 总结

### 数据库选择

| 数据库 | 类型 | 适用场景 | Python 驱动 |
|--------|------|----------|-------------|
| **SQLite** | 嵌入式 | 小项目、原型、移动端、测试 | `sqlite3`（标准库） |
| **MySQL** | 关系型 | Web 应用、中大型项目 | `pymysql` / `mysqlclient` |
| **PostgreSQL** | 关系型 | 复杂查询、JSONB、地理信息 | `psycopg2` / `psycopg` |
| **Redis** | 键值存储 | 缓存、会话、排行榜、消息队列 | `redis` |

### ORM vs 原生 SQL

| 维度 | ORM（SQLAlchemy） | 原生 SQL |
|------|-------------------|----------|
| 开发效率 | 高（Python 对象操作） | 低（手写 SQL） |
| 性能 | 略低（有转换开销） | 高（直接执行） |
| 可维护性 | 高（类型安全、迁移工具） | 低（SQL 散落各处） |
| 灵活性 | 中（复杂查询较麻烦） | 高（任意 SQL） |
| 推荐 | 业务逻辑、CRUD | 复杂报表、性能瓶颈 |

### 最佳实践
