## 1. Web 框架概览

| 框架 | 类型 | 异步 | 特点 | 适用场景 |
|------|------|------|------|----------|
| **Flask** | 微框架 | 有限 | 简单灵活，生态丰富 | 小中型项目、快速原型 |
| **FastAPI** | 现代框架 | 原生 | 类型驱动、自动文档、高性能 | API 服务、中大型项目 |
| Django | 全栈框架 | 有限 | 内置 ORM/Admin/Auth，开箱即用 | 内容管理、全栈应用 |

```python
from flask import Flask, request, jsonify

app = Flask(__name__)

# ===== 路由参数 =====

# 路径参数 + 类型转换
@app.route("/user/<int:user_id>")
def get_user(user_id):
    return jsonify({"id": user_id, "name": f"User_{user_id}"})

# 多种类型：string（默认）, int, float, path, uuid
@app.route("/file/<path:filepath>")
def get_file(filepath):
    return f"请求文件：{filepath}"

# ===== HTTP 方法 =====

@app.route("/api/items", methods=["GET", "POST"])
def items():
    if request.method == "GET":
        # 查询参数：/api/items?page=1&limit=10
        page = request.args.get("page", 1, type=int)
        limit = request.args.get("limit", 10, type=int)
        return jsonify({"page": page, "limit": limit, "items": []})

    elif request.method == "POST":
        # JSON 请求体
        data = request.get_json()
        return jsonify({"message": "创建成功", "data": data}), 201

# ===== 请求对象 =====

@app.route("/debug", methods=["POST"])
def debug_request():
    info = {
        "method": request.method,
        "url": request.url,
        "headers": dict(request.headers),
        "args": dict(request.args),           # 查询参数
        "form": dict(request.form),           # 表单数据
        "json": request.get_json(silent=True), # JSON 数据
        "remote_addr": request.remote_addr,    # 客户端 IP
    }
    return jsonify(info)

if __name__ == "__main__":
    app.run(debug=True)

# 测试：
# curl http://localhost:5000/user/42
```

### 2.5 错误处理与中间件

```python
from flask import Flask, jsonify, request
import time

app = Flask(__name__)

# ===== 自定义错误处理 =====

@app.errorhandler(404)
def not_found(error):
    return jsonify({"error": "资源不存在", "code": 404}), 404

@app.errorhandler(500)
def server_error(error):
    return jsonify({"error": "服务器内部错误", "code": 500}), 500

@app.errorhandler(Exception)
def handle_exception(error):
    return jsonify({"error": str(error), "code": 500}), 500

# ===== 请求钩子（类似中间件）=====

@app.before_request
def before():
    """每个请求之前执行"""
    request.start_time = time.time()

@app.after_request
def after(response):
    """每个请求之后执行"""
    elapsed = time.time() - getattr(request, "start_time", time.time())
    response.headers["X-Response-Time"] = f"{elapsed:.4f}s"
    print(f"  {request.method} {request.path} → {response.status_code} ({elapsed:.4f}s)")
    return response

@app.route("/")
def index():
    return jsonify({"message": "ok"})

@app.route("/error")
def trigger_error():
    raise ValueError("手动触发的错误")

if __name__ == "__main__":
    app.run(debug=True)
```

```python
from fastapi import FastAPI, Query, Path, Body, Header
from pydantic import BaseModel, Field, EmailStr
from typing import Optional
from enum import Enum

app = FastAPI()

# ===== 路径参数（自动类型校验）=====

class SortOrder(str, Enum):
    asc = "asc"
    desc = "desc"

@app.get("/users/{user_id}")
async def get_user(
    user_id: int = Path(..., gt=0, description="用户 ID，必须大于 0")
):
    return {"user_id": user_id}

# GET /users/abc → 422 自动报错：value is not a valid integer
# GET /users/-1  → 422 自动报错：ensure this value is greater than 0
# GET /users/42  → {"user_id": 42}

# ===== 查询参数 =====

@app.get("/items")
async def list_items(
    page: int = Query(1, ge=1, description="页码"),
    limit: int = Query(10, ge=1, le=100, description="每页数量"),
    keyword: Optional[str] = Query(None, min_length=1, description="搜索关键词"),
    order: SortOrder = Query(SortOrder.desc, description="排序方式"),
):
    return {
        "page": page,
        "limit": limit,
        "keyword": keyword,
        "order": order,
    }

# GET /items?page=2&limit=20&keyword=python&order=asc
# → {"page": 2, "limit": 20, "keyword": "python", "order": "asc"}

# ===== 请求体（Pydantic 模型）=====

class CreateUser(BaseModel):

    email: str = Field(..., examples=["alice@test.com"])

        }
    }

@app.post("/users", status_code=201)

# POST /users {"name": "Alice", "email": "alice@test.com", "age": 25}
```

```python
from fastapi import FastAPI, Depends, HTTPException, Header
from typing import Optional

app = FastAPI()

# ===== 简单依赖 =====

def get_pagination(page: int = 1, limit: int = 10):
    """分页参数（可复用）"""
    return {"page": max(1, page), "limit": min(100, limit), "offset": (max(1, page) - 1) * limit}

@app.get("/posts")
async def list_posts(pagination: dict = Depends(get_pagination)):
    return {"pagination": pagination, "data": []}

@app.get("/comments")
async def list_comments(pagination: dict = Depends(get_pagination)):
    return {"pagination": pagination, "data": []}

# ===== 认证依赖 =====

FAKE_TOKENS = {"abc123": {"user": "Alice", "role": "admin"}}

async def get_current_user(authorization: Optional[str] = Header(None)):
    if not authorization or not authorization.startswith("Bearer "):
        raise HTTPException(status_code=401, detail="未提供有效的认证信息")
    token = authorization.split(" ")[1]
    user = FAKE_TOKENS.get(token)
    if not user:
        raise HTTPException(status_code=401, detail="无效的 Token")
    return user

async def require_admin(user: dict = Depends(get_current_user)):
    if user.get("role") != "admin":
        raise HTTPException(status_code=403, detail="需要管理员权限")
    return user

# 需要登录
@app.get("/me")
async def me(user: dict = Depends(get_current_user)):
    return user

# 需要管理员
@app.delete("/users/{user_id}")
async def delete_user(user_id: int, admin: dict = Depends(require_admin)):
    return {"message": f"用户 {user_id} 已删除", "deleted_by": admin["user"]}

# → {"user": "Alice", "role": "admin"}
```

### 3.5 数据库集成

```python
from fastapi import FastAPI, Depends, HTTPException
from sqlalchemy import create_engine, select, String, Integer
from sqlalchemy.orm import (
    DeclarativeBase, Mapped, mapped_column, Session, sessionmaker
)
from pydantic import BaseModel
from typing import Optional

# ===== 数据库配置 =====

engine = create_engine("sqlite:///fastapi_demo.db")
SessionLocal = sessionmaker(bind=engine)

class Base(DeclarativeBase):
    pass

class UserDB(Base):
    __tablename__ = "users"
    id: Mapped[int] = mapped_column(primary_key=True)
    name: Mapped[str] = mapped_column(String(50))
    email: Mapped[str] = mapped_column(String(100), unique=True)

Base.metadata.create_all(engine)

# ===== 依赖注入数据库会话 =====

def get_db():
    db = SessionLocal()
    try:
        yield db
    finally:
        db.close()

# ===== Pydantic 模型 =====

class UserCreate(BaseModel):
    name: str
    email: str

class UserResponse(BaseModel):
    id: int
    name: str
    email: str

    model_config = {"from_attributes": True}

# ===== 路由 =====

app = FastAPI()

@app.post("/users", response_model=UserResponse, status_code=201)
def create_user(user: UserCreate, db: Session = Depends(get_db)):
    db_user = UserDB(name=user.name, email=user.email)
    db.add(db_user)
    db.commit()
    db.refresh(db_user)
    return db_user

@app.get("/users", response_model=list[UserResponse])
def list_users(skip: int = 0, limit: int = 10, db: Session = Depends(get_db)):
    stmt = select(UserDB).offset(skip).limit(limit)
    return list(db.execute(stmt).scalars())

@app.get("/users/{user_id}", response_model=UserResponse)
def get_user(user_id: int, db: Session = Depends(get_db)):
    user = db.get(UserDB, user_id)
    if not user:
        raise HTTPException(status_code=404, detail="用户不存在")
    return user

# 启动：fastapi dev main.py
# 文档：http://localhost:8000/docs
```

```python
    response = await call_next(request)
    elapsed = time.time() - start
    response.headers["X-Response-Time"] = f"{elapsed:.4f}s"
```

### 4.1 模型定义与校验

```python
from pydantic import BaseModel, Field, field_validator, model_validator
from typing import Optional
from datetime import datetime
from enum import Enum

class Role(str, Enum):
    admin = "admin"
    user = "user"
    guest = "guest"

class Address(BaseModel):
    city: str
    street: str
    zip_code: str = Field(..., pattern=r"^\d{6}$")  # 正则校验

class UserProfile(BaseModel):
    name: str = Field(..., min_length=2, max_length=50)
    email: str
    age: int = Field(0, ge=0, le=150)
    role: Role = Role.user
    tags: list[str] = []
    address: Optional[Address] = None
    created_at: datetime = Field(default_factory=datetime.now)

    # 字段级别校验
    @field_validator("email")
    @classmethod
    def validate_email(cls, v):
        if "@" not in v:
            raise ValueError("邮箱格式不正确")
        return v.lower()

    # 模型级别校验（多字段联合校验）
    @model_validator(mode="after")
    def check_admin_age(self):
        if self.role == Role.admin and self.age < 18:
            raise ValueError("管理员必须年满 18 岁")
        return self

# 正常创建
user = UserProfile(
    name="Alice",
    email="ALICE@Test.com",
    age=25,
    role="admin",
    tags=["python", "developer"],
    address={"city": "Beijing", "street": "中关村", "zip_code": "100080"},
)
print(user.model_dump())
# {'name': 'Alice', 'email': 'alice@test.com', 'age': 25, 'role': 'admin',
#  'tags': ['python', 'developer'], 'address': {'city': 'Beijing', ...}, ...}

# 自动类型转换
user2 = UserProfile(name="Bob", email="bob@test.com", age="30")  # "30" → 30
print(user2.age, type(user2.age))  # 30 <class 'int'>

# 校验失败
from pydantic import ValidationError

try:
    UserProfile(name="A", email="bad", age=200)
except ValidationError as e:
    print(e)
    # 3 validation errors for UserProfile
    # name: String should have at least 2 characters
    # email: 邮箱格式不正确
    # age: Input should be less than or equal to 150

# JSON 序列化与反序列化
json_str = user.model_dump_json(indent=2)
print(json_str)

user_from_json = UserProfile.model_validate_json(json_str)
print(user_from_json.name)  # Alice
```

## 5. 认证与鉴权

### 5.1 JWT 认证（FastAPI）

```python
from fastapi import FastAPI, Depends, HTTPException
from fastapi.security import HTTPBearer, HTTPAuthorizationCredentials
from pydantic import BaseModel
from datetime import datetime, timedelta
import jwt  # pip install PyJWT

app = FastAPI()
security = HTTPBearer()

SECRET_KEY = "your-secret-key"
ALGORITHM = "HS256"

# 模拟用户数据库
USERS_DB = {
    "alice": {"password": "secret123", "role": "admin"},
    "bob": {"password": "pass456", "role": "user"},
}

class LoginRequest(BaseModel):
    username: str
    password: str

class TokenResponse(BaseModel):
    access_token: str
    token_type: str = "bearer"
    expires_in: int

def create_token(username: str, role: str, expires_minutes: int = 60) -> str:
    payload = {
        "sub": username,
        "role": role,
        "exp": datetime.utcnow() + timedelta(minutes=expires_minutes),
        "iat": datetime.utcnow(),
    }
    return jwt.encode(payload, SECRET_KEY, algorithm=ALGORITHM)

def verify_token(credentials: HTTPAuthorizationCredentials = Depends(security)) -> dict:
    try:
        payload = jwt.decode(credentials.credentials, SECRET_KEY, algorithms=[ALGORITHM])
        return payload
    except jwt.ExpiredSignatureError:
        raise HTTPException(status_code=401, detail="Token 已过期")
    except jwt.InvalidTokenError:
        raise HTTPException(status_code=401, detail="无效的 Token")

# ===== 路由 =====

@app.post("/login", response_model=TokenResponse)
async def login(req: LoginRequest):
    user = USERS_DB.get(req.username)
    if not user or user["password"] != req.password:
        raise HTTPException(status_code=401, detail="用户名或密码错误")
    token = create_token(req.username, user["role"])
    return TokenResponse(access_token=token, expires_in=3600)

@app.get("/protected")
async def protected(payload: dict = Depends(verify_token)):
    return {"message": f"你好 {payload['sub']}，角色：{payload['role']}"}

@app.get("/admin")
async def admin_only(payload: dict = Depends(verify_token)):
    if payload["role"] != "admin":
        raise HTTPException(status_code=403, detail="需要管理员权限")
    return {"message": "欢迎管理员"}

# 测试流程：
```

```python
# app/__init__.py
from flask import Flask

def create_app(config_name="development"):
    app = Flask(__name__)
    app.config.from_object(f"app.config.{config_name}")

    # 注册扩展
    from app.extensions import db, migrate
    db.init_app(app)
    migrate.init_app(app, db)

    # 注册蓝图
    from app.blueprints.auth import auth_bp
    from app.blueprints.api import api_bp
    app.register_blueprint(auth_bp)
    app.register_blueprint(api_bp)

    return app
```

```python
# app/routers/users.py
from fastapi import APIRouter, Depends
from sqlalchemy.orm import Session
from app.database import get_db
from app.schemas.user import UserCreate, UserResponse
from app.services.user_service import UserService

router = APIRouter()

@router.get("/", response_model=list[UserResponse])

    service = UserService(db)
    return service.list_users(skip, limit)

@router.post("/", response_model=UserResponse, status_code=201)
```

## 8. 总结

### Flask vs FastAPI 对比

| 维度 | Flask | FastAPI |
|------|-------|---------|
| 类型系统 | 无（手动校验） | Pydantic 自动校验 |
| 异步支持 | 有限（需 async 视图） | 原生 async/await |
| API 文档 | 需要扩展（flask-restx） | 自动生成 Swagger/ReDoc |
| 性能 | 中等 | 高（接近 Node.js/Go） |
| 学习曲线 | 低 | 中（需了解类型提示） |
| 模板渲染 | 内置 Jinja2 | 需手动集成 |
| 生态 | 非常成熟 | 快速增长 |
| 适合场景 | 全栈 Web、快速原型 | API 服务、微服务 |

### 选择建议

```
纯 API 服务        → FastAPI（自动文档、类型安全、高性能）
需要模板渲染       → Flask（Jinja2 内置支持好）
全栈 + 后台管理    → Django（内置 Admin、ORM、Auth）
微服务/高并发      → FastAPI + uvicorn
学习 Web 开发入门  → Flask（最简单直观）
```
