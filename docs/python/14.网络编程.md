### 1.3 超时、重试与异常处理

```python
import requests
from requests.adapters import HTTPAdapter
from urllib3.util.retry import Retry

# 设置超时
try:
    response = requests.get(
        "https://httpbin.org/delay/5",
        timeout=3  # 3 秒超时（也可以传元组 (连接超时, 读取超时)）
    )
except requests.Timeout:
    print("请求超时！")

# 完善的异常处理
def safe_request(url, **kwargs):
    try:
        response = requests.get(url, timeout=10, **kwargs)
        response.raise_for_status()  # 4xx/5xx 会抛出 HTTPError
        return response
    except requests.ConnectionError:
        print(f"连接失败：{url}")
    except requests.Timeout:
        print(f"请求超时：{url}")
    except requests.HTTPError as e:
        print(f"HTTP 错误：{e.response.status_code}")
    except requests.RequestException as e:
        print(f"请求异常：{e}")
    return None

result = safe_request("https://httpbin.org/status/404")
# HTTP 错误：404

# 自动重试
session = requests.Session()
retry_strategy = Retry(
    total=3,              # 最多重试 3 次
    backoff_factor=1,     # 重试间隔：1s, 2s, 4s
    status_forcelist=[429, 500, 502, 503, 504],
)
adapter = HTTPAdapter(max_retries=retry_strategy)
session.mount("https://", adapter)
session.mount("http://", adapter)

# 之后用 session 发请求就会自动重试
# response = session.get("https://example.com/api")
```

```python
# ===== 文件下载 =====
def download_file(url, filepath, chunk_size=8192):
    """流式下载大文件"""
    with requests.get(url, stream=True) as r:
        r.raise_for_status()
        total = int(r.headers.get("Content-Length", 0))
        downloaded = 0
        with open(filepath, "wb") as f:
            for chunk in r.iter_content(chunk_size=chunk_size):
                f.write(chunk)
                downloaded += len(chunk)
                if total:
                    percent = downloaded / total * 100
                    print(f"\r下载进度：{percent:.1f}%", end="", flush=True)
        print("\n下载完成")
```

### 2.2 封装 API 客户端

```python
import requests
from typing import Optional, Any

class APIClient:
    """通用 REST API 客户端"""

    def __init__(self, base_url: str, token: Optional[str] = None):
        self.base_url = base_url.rstrip("/")
        self.session = requests.Session()
        self.session.headers["Accept"] = "application/json"
        if token:
            self.session.headers["Authorization"] = f"Bearer {token}"

    def _request(self, method: str, path: str, **kwargs) -> dict:
        url = f"{self.base_url}/{path.lstrip('/')}"
        response = self.session.request(method, url, timeout=30, **kwargs)
        response.raise_for_status()
        return response.json() if response.text else {}

    def get(self, path: str, params: dict = None) -> Any:
        return self._request("GET", path, params=params)

    def post(self, path: str, data: dict = None) -> Any:
        return self._request("POST", path, json=data)

    def put(self, path: str, data: dict = None) -> Any:
        return self._request("PUT", path, json=data)

    def patch(self, path: str, data: dict = None) -> Any:
        return self._request("PATCH", path, json=data)

    def delete(self, path: str) -> Any:
        return self._request("DELETE", path)

    def close(self):
        self.session.close()

    def __enter__(self):
        return self

    def __exit__(self, *args):
        self.close()

# 使用
with APIClient("https://jsonplaceholder.typicode.com") as api:
    # 获取用户列表
    users = api.get("/users", params={"_limit": 3})
    for user in users:
        print(f"  {user['name']} <{user['email']}>")
    #   Leanne Graham <Sincere@april.biz>
    #   Ervin Howell <Shanna@melissa.tv>
    #   Clementine Bauch <Nathan@yesenia.net>

    # 创建新帖子
    post = api.post("/posts", data={"title": "Test", "body": "Hello", "userId": 1})
    print(f"  创建帖子 ID：{post['id']}")  # 创建帖子 ID：101
```

## 4. Socket 编程基础

Socket 是网络通信的底层接口，HTTP 库内部也是基于 Socket 实现的。

### 4.1 TCP 服务端与客户端

```python
# ===== server.py =====
import socket

def run_server(host="127.0.0.1", port=9000):
    # 创建 TCP Socket
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as server:
        server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        server.bind((host, port))
        server.listen(5)  # 最大等待连接数
        print(f"服务器启动：{host}:{port}")

        while True:
            conn, addr = server.accept()  # 阻塞等待连接
            with conn:
                print(f"客户端连接：{addr}")
                while True:
                    data = conn.recv(1024)  # 接收数据
                    if not data:
                        break
                    message = data.decode("utf-8")
                    print(f"  收到：{message}")

                    # 回复客户端
                    response = f"收到你的消息：{message}"
                    conn.sendall(response.encode("utf-8"))
                print(f"客户端断开：{addr}")

# run_server()
```

### 4.2 多线程 TCP 服务器

```python
import socket
import threading

def handle_client(conn, addr):
    """处理单个客户端连接"""
    print(f"[+] 新连接：{addr}")
    with conn:
        while True:
            data = conn.recv(1024)
            if not data:
                break
            message = data.decode("utf-8")
            print(f"  [{addr}] {message}")
            conn.sendall(f"ECHO: {message}".encode("utf-8"))
    print(f"[-] 断开：{addr}")

def run_threaded_server(host="127.0.0.1", port=9000):
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as server:
        server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
        server.bind((host, port))
        server.listen(5)
        print(f"多线程服务器启动：{host}:{port}")

        while True:
            conn, addr = server.accept()
            # 每个连接启动一个线程
            thread = threading.Thread(target=handle_client, args=(conn, addr))
            thread.daemon = True
            thread.start()

# run_threaded_server()
```

```python
import socket

# UDP 服务端
def udp_server(host="127.0.0.1", port=9001):
    with socket.socket(socket.AF_INET, socket.SOCK_DGRAM) as server:
        server.bind((host, port))
        print(f"UDP 服务器启动：{host}:{port}")
        while True:
            data, addr = server.recvfrom(1024)
            message = data.decode("utf-8")
            print(f"  来自 {addr}：{message}")
            server.sendto(f"ACK: {message}".encode("utf-8"), addr)

# UDP 客户端
def udp_client(host="127.0.0.1", port=9001):
    with socket.socket(socket.AF_INET, socket.SOCK_DGRAM) as client:
        for msg in ["Hello", "UDP", "Test"]:
            client.sendto(msg.encode("utf-8"), (host, port))
            data, _ = client.recvfrom(1024)
            print(f"  发送：{msg} → 回复：{data.decode('utf-8')}")

# TCP vs UDP
# TCP：可靠、有序、面向连接 → 网页、API、文件传输
# UDP：快速、无连接、可能丢包 → 视频流、游戏、DNS
```

```python
import asyncio

async def handle_client(reader: asyncio.StreamReader, writer: asyncio.StreamWriter):
    addr = writer.get_extra_info("peername")
    print(f"[+] 新连接：{addr}")

    try:
        while True:
            data = await reader.read(1024)
            if not data:
                break
            message = data.decode("utf-8")
            print(f"  [{addr}] {message}")

            response = f"ECHO: {message}"
            writer.write(response.encode("utf-8"))
            await writer.drain()  # 确保数据发送完毕
    except asyncio.CancelledError:
        pass
    finally:
        writer.close()
        await writer.wait_closed()
        print(f"[-] 断开：{addr}")

async def main():
    server = await asyncio.start_server(handle_client, "127.0.0.1", 9000)
    addr = server.sockets[0].getsockname()
    print(f"异步服务器启动：{addr}")

    async with server:
        await server.serve_forever()
```

### 6.1 WebSocket 服务端

```python
import asyncio
import websockets
import json
from datetime import datetime

# 存储所有连接的客户端
connected_clients = set()

async def handler(websocket):
    # 注册客户端
    connected_clients.add(websocket)
    addr = websocket.remote_address
    print(f"[+] 客户端连接：{addr}，当前在线：{len(connected_clients)}")

    try:
        async for message in websocket:
            data = json.loads(message)
            print(f"  [{addr}] {data}")

            # 广播消息给所有客户端
            broadcast = json.dumps({
                "from": str(addr),
                "message": data.get("message", ""),
                "time": datetime.now().strftime("%H:%M:%S"),
            })
            await asyncio.gather(
                *[client.send(broadcast) for client in connected_clients]
            )
    except websockets.ConnectionClosed:
        pass
    finally:
        connected_clients.discard(websocket)
        print(f"[-] 客户端断开：{addr}，当前在线：{len(connected_clients)}")

async def main():
    async with websockets.serve(handler, "localhost", 8765):
        print("WebSocket 服务器启动：ws://localhost:8765")
        await asyncio.Future()  # 永远运行

# asyncio.run(main())
```

## 7. 实战项目

### 项目 1：GitHub API 信息聚合工具

```python
"""
GitHub 用户信息查询工具
使用 GitHub REST API 获取用户和仓库信息
"""
import httpx
import asyncio
from dataclasses import dataclass
from typing import Optional

GITHUB_API = "https://api.github.com"

@dataclass
class RepoInfo:
    name: str
    stars: int
    language: Optional[str]
    description: Optional[str]

@dataclass
class UserInfo:
    login: str
    name: Optional[str]
    bio: Optional[str]
    public_repos: int
    followers: int
    top_repos: list[RepoInfo]

async def fetch_user(client: httpx.AsyncClient, username: str) -> dict:
    """获取用户基本信息"""
    response = await client.get(f"{GITHUB_API}/users/{username}")
    response.raise_for_status()
    return response.json()

async def fetch_repos(client: httpx.AsyncClient, username: str) -> list[dict]:
    """获取用户仓库列表"""
    response = await client.get(
        f"{GITHUB_API}/users/{username}/repos",
        params={"sort": "stars", "per_page": 5}
    )
    response.raise_for_status()
    return response.json()

async def get_user_info(username: str) -> UserInfo:
    """聚合用户信息（并发获取）"""
    async with httpx.AsyncClient() as client:
        # 并发获取用户信息和仓库列表
        user_data, repos_data = await asyncio.gather(
            fetch_user(client, username),
            fetch_repos(client, username),
        )

    top_repos = [
        RepoInfo(
            name=r["name"],
            stars=r["stargazers_count"],
            language=r["language"],
            description=r["description"],
        )
        for r in repos_data
    ]

    return UserInfo(
        login=user_data["login"],
        name=user_data.get("name"),
        bio=user_data.get("bio"),
        public_repos=user_data["public_repos"],
        followers=user_data["followers"],
        top_repos=top_repos,
    )

def display_user(user: UserInfo):
    print(f"\n{'=' * 50}")
    print(f"  用户：{user.login}" + (f" ({user.name})" if user.name else ""))
    if user.bio:
        print(f"  简介：{user.bio[:60]}")
    print(f"  公开仓库：{user.public_repos}  |  粉丝：{user.followers}")
    print(f"\n  Top 仓库（按 Star 排序）：")
    for repo in user.top_repos:
        lang = repo.language or "N/A"
        desc = (repo.description or "")[:40]
        print(f"    ⭐ {repo.stars:<6d} [{lang:<12s}] {repo.name}")
        if desc:
```

```python
"""
网站健康检查工具
并发检测多个 URL 的状态和响应时间
"""
import httpx
import asyncio
import time
from dataclasses import dataclass
from enum import Enum

class Status(Enum):
    OK = "✅ 正常"
    SLOW = "🐌 缓慢"
    ERROR = "❌ 异常"
    TIMEOUT = "⏱️ 超时"

@dataclass
class CheckResult:
    url: str
    status: Status
    status_code: int
    response_time: float
    error: str = ""

async def check_url(
    client: httpx.AsyncClient,
    url: str,
    slow_threshold: float = 2.0,
) -> CheckResult:
    """检查单个 URL"""
    start = time.time()
    try:
        response = await client.get(url, follow_redirects=True)
        elapsed = time.time() - start

        if response.status_code >= 400:
            status = Status.ERROR
        elif elapsed > slow_threshold:
            status = Status.SLOW
        else:
            status = Status.OK

        return CheckResult(
            url=url,
            status=status,
            status_code=response.status_code,
            response_time=elapsed,
        )
    except httpx.TimeoutException:
        return CheckResult(url=url, status=Status.TIMEOUT,
                           status_code=0, response_time=time.time() - start,
                           error="请求超时")
    except httpx.ConnectError:
        return CheckResult(url=url, status=Status.ERROR,
                           status_code=0, response_time=time.time() - start,
                           error="连接失败")

async def health_check(urls: list[str]) -> list[CheckResult]:
    """并发检查所有 URL"""
    async with httpx.AsyncClient(timeout=10.0) as client:
        tasks = [check_url(client, url) for url in urls]
        return await asyncio.gather(*tasks)

def print_report(results: list[CheckResult]):
    print(f"\n{'URL':<45} {'状态':<12} {'HTTP':<6} {'耗时':<8}")
    print("-" * 75)
    for r in sorted(results, key=lambda x: x.response_time):
        url_short = r.url[:43] + ".." if len(r.url) > 45 else r.url
        code = str(r.status_code) if r.status_code else "N/A"
        print(f"{url_short:<45} {r.status.value:<10} {code:<6} {r.response_time:.2f}s")
        if r.error:
            print(f"{'':>45} └ {r.error}")

    ok = sum(1 for r in results if r.status == Status.OK)
    print(f"\n汇总：{ok}/{len(results)} 正常，"
          f"平均响应 {sum(r.response_time for r in results) / len(results):.2f}s")

async def main():
    urls = [
        "https://www.google.com",
        "https://www.github.com",
        "https://httpbin.org/get",
        "https://httpbin.org/delay/3",
        "https://httpbin.org/status/500",
        "https://nonexistent.example.com",
    ]

    print("开始健康检查...")
    start = time.time()
    results = await health_check(urls)
    print(f"检查完成，总耗时：{time.time() - start:.2f}s")
    print_report(results)

asyncio.run(main())
```

### 项目 3：简易聊天室（Socket + 多线程）

```python
"""
简易命令行聊天室
- 服务端：接受多客户端连接，广播消息
- 客户端：发送和接收消息
"""
# ===== chat_server.py =====
import socket
import threading
from datetime import datetime

class ChatServer:
    def __init__(self, host="127.0.0.1", port=9000):
        self.host = host
        self.port = port
        self.clients: dict[socket.socket, str] = {}  # socket → 昵称
        self.lock = threading.Lock()

    def broadcast(self, message: str, sender: socket.socket = None):
        """广播消息给所有客户端"""
        timestamp = datetime.now().strftime("%H:%M:%S")
        full_msg = f"[{timestamp}] {message}"
        print(f"  {full_msg}")

        with self.lock:
            for client in list(self.clients):
                if client != sender:
                    try:
                        client.sendall(full_msg.encode("utf-8"))
                    except OSError:
                        self.remove_client(client)

    def remove_client(self, client: socket.socket):
        with self.lock:
            nickname = self.clients.pop(client, "未知")
        try:
            client.close()
        except OSError:
            pass
        self.broadcast(f"📤 {nickname} 离开了聊天室")

    def handle_client(self, client: socket.socket, addr):
        """处理单个客户端"""
        try:
            # 接收昵称
            nickname = client.recv(1024).decode("utf-8").strip()
            with self.lock:
                self.clients[client] = nickname

            self.broadcast(f"📥 {nickname} 加入了聊天室")
            client.sendall(f"欢迎 {nickname}！当前在线 {len(self.clients)} 人\n".encode("utf-8"))

            while True:
                data = client.recv(1024)
                if not data:
                    break
                message = data.decode("utf-8").strip()
                if message == "/quit":
                    break
                self.broadcast(f"{nickname}：{message}", sender=client)

        except (OSError, ConnectionResetError):
            pass
        finally:
            self.remove_client(client)

    def start(self):
        with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as server:
            server.setsockopt(socket.SOL_SOCKET, socket.SO_REUSEADDR, 1)
            server.bind((self.host, self.port))
            server.listen(10)
            print(f"聊天服务器启动：{self.host}:{self.port}")
            print("等待连接...\n")

            while True:
                client, addr = server.accept()
                thread = threading.Thread(
                    target=self.handle_client, args=(client, addr), daemon=True
                )
                thread.start()
```

```python
import socket
import threading

def receive_messages(sock: socket.socket):
    """接收消息的线程"""
    while True:
        try:
            data = sock.recv(1024)
            if not data:
                print("\n连接已断开")
                break
            print(f"\r{data.decode('utf-8')}\n> ", end="", flush=True)
        except OSError:
            break

def start_client(host="127.0.0.1", port=9000):
    with socket.socket(socket.AF_INET, socket.SOCK_STREAM) as client:
        client.connect((host, port))

        # 发送昵称
        nickname = input("输入昵称：").strip() or "匿名"
        client.sendall(nickname.encode("utf-8"))

        # 接收欢迎消息
        welcome = client.recv(1024).decode("utf-8")
        print(welcome)

        # 启动接收线程
        recv_thread = threading.Thread(target=receive_messages, args=(client,), daemon=True)
        recv_thread.start()

        # 主线程负责发送
        print("输入消息（/quit 退出）：")
        while True:
            message = input("> ")
            if message == "/quit":
                client.sendall(b"/quit")
                break
            if message:
                client.sendall(message.encode("utf-8"))

    print("已退出聊天室")
```

## 8. 总结

### HTTP 库选择

| 库 | 同步 | 异步 | 特点 |
|-----|------|------|------|
| **requests** | ✅ | ❌ | 最流行，文档丰富，社区生态好 |
| **httpx** | ✅ | ✅ | 现代化，同步异步统一 API，支持 HTTP/2 |
| **aiohttp** | ❌ | ✅ | 纯异步，性能优秀，也可做 Web 服务器 |
| **urllib3** | ✅ | ❌ | requests 的底层库，更低层控制 |

### Socket vs HTTP

| 维度 | Socket | HTTP |
|------|--------|------|
| 层级 | 传输层 | 应用层 |
| 灵活性 | 高（自定义协议） | 低（遵循 HTTP 规范） |
| 开发难度 | 高 | 低 |
| 适用场景 | 即时通讯、游戏、IoT | Web API、爬虫 |
| 推荐库 | `socket` / `asyncio` | `requests` / `httpx` |
