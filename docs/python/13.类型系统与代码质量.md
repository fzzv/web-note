# Python 类型系统与代码质量

Python 是动态类型语言，但从 3.5 开始引入了**类型提示（Type Hints）**，配合静态检查工具和测试框架，可以在不牺牲灵活性的前提下大幅提升代码质量。

## 1. 类型提示基础

### 1.1 变量与函数注解

```python
# 变量注解
name: str = "Alice"
age: int = 25
score: float = 95.5
active: bool = True

# 函数注解
def greet(name: str, times: int = 1) -> str:
    return (f"Hello, {name}! " * times).strip()

print(greet("Alice", 2))  # Hello, Alice! Hello, Alice!

# 类型提示不会在运行时强制检查，写错类型不会报错
result: int = "这其实是字符串"  # 运行正常，不报错
print(result)  # 这其实是字符串

# 查看注解
print(greet.__annotations__)
# {'name': <class 'str'>, 'times': <class 'int'>, 'return': <class 'str'>}
```

### 1.2 内置容器类型

```python
# Python 3.9+ 可以直接用内置类型
names: list[str] = ["Alice", "Bob"]
scores: dict[str, int] = {"Alice": 90, "Bob": 85}
point: tuple[int, int] = (3, 4)
unique: set[str] = {"a", "b", "c"}

# 嵌套类型
matrix: list[list[int]] = [[1, 2], [3, 4]]
users: dict[str, dict[str, int]] = {
    "Alice": {"age": 25, "score": 90}
}

# 不定长元组
values: tuple[int, ...] = (1, 2, 3, 4, 5)  # 任意长度的 int 元组

# Python 3.8 及以下需要从 typing 导入
# from typing import List, Dict, Tuple, Set
# names: List[str] = ["Alice", "Bob"]
```

### 1.3 None 和返回值

```python
# 函数无返回值
def log(message: str) -> None:
    print(f"[LOG] {message}")

log("启动")  # [LOG] 启动

# NoReturn：函数永远不会正常返回（抛异常或无限循环）
from typing import NoReturn

def fatal_error(message: str) -> NoReturn:
    raise SystemExit(message)
```

## 2. typing 模块核心类型

### 2.1 Optional —— 可选类型

值可以是指定类型或 `None`。

```python
from typing import Optional

def find_user(user_id: int) -> Optional[str]:
    """返回用户名，找不到返回 None"""
    users = {1: "Alice", 2: "Bob"}
    return users.get(user_id)

result = find_user(1)
print(result)  # Alice

result = find_user(99)
print(result)  # None

# Optional[X] 等价于 X | None（Python 3.10+）
def find_user_v2(user_id: int) -> str | None:
    users = {1: "Alice", 2: "Bob"}
    return users.get(user_id)
```

### 2.2 Union —— 联合类型

值可以是多种类型之一。

```python
from typing import Union

def process(value: Union[int, str]) -> str:
    if isinstance(value, int):
        return f"数字：{value}"
    return f"字符串：{value}"

print(process(42))       # 数字：42
print(process("hello"))  # 字符串：hello

# Python 3.10+ 可以用 | 语法
def process_v2(value: int | str | float) -> str:
    return str(value)
```

### 2.3 Any —— 任意类型

```python
from typing import Any

def debug_print(value: Any) -> None:
    """接受任意类型的值"""
    print(f"[DEBUG] {type(value).__name__}: {value}")

debug_print(42)        # [DEBUG] int: 42
debug_print("hello")   # [DEBUG] str: hello
debug_print([1, 2])    # [DEBUG] list: [1, 2]

# Any 跳过类型检查，尽量少用
# 大多数场景用 object 代替（object 是所有类型的基类，但更严格）
```

### 2.4 Literal —— 字面量类型

限制值只能是特定的几个字面值。

```python
from typing import Literal

def set_color(color: Literal["red", "green", "blue"]) -> str:
    return f"颜色设为 {color}"

print(set_color("red"))    # 颜色设为 red
# set_color("yellow")      # mypy 报错：不在允许的值中（运行时不报错）

# 常用场景
def sort_data(
    data: list[int],
    order: Literal["asc", "desc"] = "asc"
) -> list[int]:
    return sorted(data, reverse=(order == "desc"))

print(sort_data([3, 1, 2], "asc"))   # [1, 2, 3]
print(sort_data([3, 1, 2], "desc"))  # [3, 2, 1]

# 字面量联合
Mode = Literal["r", "w", "a", "rb", "wb"]
```

### 2.5 TypeAlias —— 类型别名

```python
from typing import TypeAlias

# 简单别名
UserId: TypeAlias = int
Username: TypeAlias = str

# 复杂类型起个简短名字
JSON: TypeAlias = dict[str, "JSON"] | list["JSON"] | str | int | float | bool | None
Headers: TypeAlias = dict[str, str]
Callback: TypeAlias = callable  # 简写

# 使用别名
def get_user(uid: UserId) -> Username:
    return f"User_{uid}"

def fetch(url: str, headers: Headers = {}) -> JSON:
    return {"status": "ok"}

print(get_user(42))  # User_42

# Python 3.12+ 使用 type 语句
# type UserId = int
# type JSON = dict[str, JSON] | list[JSON] | str | int | float | bool | None
```

## 3. 泛型与高级类型

### 3.1 TypeVar —— 类型变量

```python
from typing import TypeVar

T = TypeVar("T")

def first(items: list[T]) -> T:
    """返回列表的第一个元素，返回类型与列表元素类型一致"""
    return items[0]

# 类型检查器能推断出返回类型
result1 = first([1, 2, 3])       # 推断为 int
result2 = first(["a", "b"])      # 推断为 str
print(result1)  # 1
print(result2)  # a

# 带约束的 TypeVar
Numeric = TypeVar("Numeric", int, float)

def add(a: Numeric, b: Numeric) -> Numeric:
    return a + b

print(add(1, 2))      # 3
print(add(1.5, 2.5))  # 4.0
# add("a", "b")        # mypy 报错：str 不在约束中

# 带上界的 TypeVar
from typing import Sized
ST = TypeVar("ST", bound=Sized)  # T 必须是 Sized 的子类型

def longest(a: ST, b: ST) -> ST:
    return a if len(a) >= len(b) else b

print(longest([1, 2, 3], [4, 5]))       # [1, 2, 3]
print(longest("hello", "hi"))           # hello
```

### 3.2 Generic —— 泛型类

```python
from typing import TypeVar, Generic

T = TypeVar("T")

class Stack(Generic[T]):
    """泛型栈"""
    def __init__(self) -> None:
        self._items: list[T] = []

    def push(self, item: T) -> None:
        self._items.append(item)

    def pop(self) -> T:
        if not self._items:
            raise IndexError("栈为空")
        return self._items.pop()

    def peek(self) -> T:
        if not self._items:
            raise IndexError("栈为空")
        return self._items[-1]

    def __len__(self) -> int:
        return len(self._items)

    def __repr__(self) -> str:
        return f"Stack({self._items})"

# 使用时指定类型
int_stack: Stack[int] = Stack()
int_stack.push(1)
int_stack.push(2)
int_stack.push(3)
print(int_stack)       # Stack([1, 2, 3])
print(int_stack.pop())  # 3

str_stack: Stack[str] = Stack()
str_stack.push("hello")
str_stack.push("world")
print(str_stack.peek())  # world
```

### 3.3 Protocol —— 结构化子类型（鸭子类型的形式化）

`Protocol` 定义了一组方法签名，任何实现了这些方法的类都被视为该协议的子类型——无需显式继承。

```python
from typing import Protocol, runtime_checkable

@runtime_checkable   # 加上才能用 isinstance 检查
class Drawable(Protocol):
    def draw(self) -> str:
        ...

class Circle:
    def draw(self) -> str:
        return "画一个圆"

class Square:
    def draw(self) -> str:
        return "画一个方"

class Text:
    def display(self) -> str:   # 没有 draw 方法
        return "显示文本"

# Circle 和 Square 没有继承 Drawable，但实现了 draw 方法
def render(shape: Drawable) -> None:
    print(shape.draw())

render(Circle())   # 画一个圆
render(Square())   # 画一个方
# render(Text())   # mypy 报错：Text 没有 draw 方法（运行时不报错）

# runtime_checkable 允许用 isinstance 检查
print(isinstance(Circle(), Drawable))  # True
print(isinstance(Text(), Drawable))    # False

# 更复杂的协议
class Serializable(Protocol):
    def to_dict(self) -> dict:
        ...

    def to_json(self) -> str:
        ...

# 任何实现了 to_dict 和 to_json 的类都满足 Serializable 协议
```

### 3.4 Callable —— 可调用类型

```python
from typing import Callable

# Callable[[参数类型列表], 返回类型]
def apply(func: Callable[[int, int], int], a: int, b: int) -> int:
    return func(a, b)

print(apply(lambda x, y: x + y, 3, 5))  # 8
print(apply(lambda x, y: x * y, 3, 5))  # 15

# 无参数的回调
def on_complete(callback: Callable[[], None]) -> None:
    print("任务完成")
    callback()

on_complete(lambda: print("  回调执行了"))
# 任务完成
#   回调执行了

# 接受任意参数的可调用对象
from typing import Any
Logger: TypeAlias = Callable[..., None]  # ... 表示任意参数
```

### 3.5 TypedDict —— 类型化字典

```python
from typing import TypedDict, NotRequired

class UserDict(TypedDict):
    name: str
    age: int
    email: NotRequired[str]  # 可选字段（Python 3.11+）

# 使用
user: UserDict = {"name": "Alice", "age": 25}
print(user["name"])  # Alice

user2: UserDict = {"name": "Bob", "age": 30, "email": "bob@test.com"}
print(user2)  # {'name': 'Bob', 'age': 30, 'email': 'bob@test.com'}

# TypedDict 在运行时就是普通 dict
print(type(user))  # <class 'dict'>

# 函数参数中使用
def greet_user(user: UserDict) -> str:
    return f"Hello, {user['name']}! (age: {user['age']})"

print(greet_user(user))  # Hello, Alice! (age: 25)
```

### 3.6 其他常用类型

```python
from typing import (
    Final, ClassVar, Annotated,
    Iterator, Generator, Iterable, Sequence,
    overload,
)

# Final —— 常量（不允许重新赋值）
MAX_SIZE: Final[int] = 100
# MAX_SIZE = 200  # mypy 报错

# ClassVar —— 类变量（不是实例变量）
class Config:
    DEBUG: ClassVar[bool] = False   # 类变量
    name: str = ""                  # 实例变量

# Annotated —— 附加元数据
from typing import Annotated
Age = Annotated[int, "必须大于 0"]
Name = Annotated[str, "最长 50 个字符"]

def create_user(name: Name, age: Age) -> None:
    print(f"{name}, {age}")

create_user("Alice", 25)  # Alice, 25

# 迭代相关类型
def count_up(n: int) -> Iterator[int]:
    for i in range(n):
        yield i

def process(items: Iterable[str]) -> list[str]:
    return [item.upper() for item in items]

# Sequence 表示有序序列（list、tuple、str 等都满足）
def first_item(seq: Sequence[int]) -> int:
    return seq[0]

print(first_item([10, 20]))   # 10
print(first_item((30, 40)))   # 30
```

### 3.7 函数重载

```python
from typing import overload

@overload
def parse(value: str) -> str: ...
@overload
def parse(value: int) -> int: ...
@overload
def parse(value: list) -> list: ...

def parse(value):
    """实际实现"""
    if isinstance(value, str):
        return value.strip().lower()
    elif isinstance(value, int):
        return value * 2
    elif isinstance(value, list):
        return sorted(value)
    raise TypeError(f"不支持的类型：{type(value)}")

# overload 装饰器帮助类型检查器推断返回类型
print(parse("  HELLO  "))  # hello
print(parse(21))           # 42
print(parse([3, 1, 2]))   # [1, 2, 3]
```

## 5. 单元测试

### 5.1 unittest —— 标准库测试框架

```python
# test_calculator.py
import unittest

def add(a, b):
    return a + b

def divide(a, b):
    if b == 0:
        raise ValueError("除数不能为零")
    return a / b

class TestCalculator(unittest.TestCase):

    def test_add_integers(self):
        self.assertEqual(add(2, 3), 5)

    def test_add_floats(self):
        self.assertAlmostEqual(add(0.1, 0.2), 0.3, places=7)

    def test_add_strings(self):
        self.assertEqual(add("hello", " world"), "hello world")

    def test_divide(self):
        self.assertEqual(divide(10, 2), 5.0)
        self.assertAlmostEqual(divide(1, 3), 0.3333, places=3)

    def test_divide_by_zero(self):
        with self.assertRaises(ValueError) as ctx:
            divide(10, 0)
        self.assertEqual(str(ctx.exception), "除数不能为零")

    def test_types(self):
        self.assertIsInstance(add(1, 2), int)
        self.assertIsInstance(divide(1, 2), float)
        self.assertIsNone(None)
        self.assertTrue(add(1, 1) == 2)
        self.assertIn(3, [1, 2, 3])

# 运行测试
if __name__ == "__main__":
    unittest.main()

# 命令行运行：
# python -m unittest test_calculator.py
# python -m unittest test_calculator.TestCalculator.test_add_integers
```

### 5.2 unittest 的 setUp / tearDown

```python
import unittest
from pathlib import Path

class TestFileProcessor(unittest.TestCase):

    def setUp(self):
        """每个测试方法执行前调用"""
        self.test_dir = Path("test_output")
        self.test_dir.mkdir(exist_ok=True)
        self.test_file = self.test_dir / "test.txt"
        self.test_file.write_text("hello world", encoding="utf-8")

    def tearDown(self):
        """每个测试方法执行后调用"""
        if self.test_file.exists():
            self.test_file.unlink()
        if self.test_dir.exists():
            self.test_dir.rmdir()

    def test_file_exists(self):
        self.assertTrue(self.test_file.exists())

    def test_file_content(self):
        content = self.test_file.read_text(encoding="utf-8")
        self.assertEqual(content, "hello world")

    def test_file_size(self):
        self.assertGreater(self.test_file.stat().st_size, 0)

    @classmethod
    def setUpClass(cls):
        """整个测试类只调用一次（所有测试方法之前）"""
        print(">>> 测试类开始")

    @classmethod
    def tearDownClass(cls):
        """整个测试类只调用一次（所有测试方法之后）"""
        print(">>> 测试类结束")
```

### 5.6 pytest mock

```python
from unittest.mock import patch, MagicMock
import pytest

class UserService:
    def __init__(self, db):
        self.db = db

    def get_user(self, user_id: int) -> dict:
        """从数据库获取用户"""
        row = self.db.query(f"SELECT * FROM users WHERE id = {user_id}")
        if row is None:
            raise ValueError(f"用户 {user_id} 不存在")
        return {"id": user_id, "name": row["name"]}

def test_get_user():
    # 创建模拟的数据库对象
    mock_db = MagicMock()
    mock_db.query.return_value = {"name": "Alice"}

    service = UserService(mock_db)
    user = service.get_user(1)

    assert user == {"id": 1, "name": "Alice"}
    mock_db.query.assert_called_once()  # 确认调用了一次

def test_get_user_not_found():
    mock_db = MagicMock()
    mock_db.query.return_value = None

    service = UserService(mock_db)

    with pytest.raises(ValueError, match="用户 99 不存在"):
        service.get_user(99)

# patch 装饰器：临时替换模块中的对象
# @patch("module_name.function_name")
# def test_something(mock_func):
#     mock_func.return_value = "mocked"
#     ...
```

### 示例 1：类型安全的配置系统

```python
from dataclasses import dataclass, field
from typing import TypeVar, Generic, Any, get_type_hints

T = TypeVar("T")

class ConfigField(Generic[T]):
    """类型安全的配置字段描述符"""
    def __init__(self, default: T, description: str = ""):
        self.default = default
        self.description = description
        self.expected_type = type(default)

    def __set_name__(self, owner, name):
        self.name = name

    def __get__(self, obj, objtype=None) -> T:
        if obj is None:
            return self  # type: ignore
        return obj.__dict__.get(self.name, self.default)

    def __set__(self, obj, value: T) -> None:
        if not isinstance(value, self.expected_type):
            raise TypeError(
                f"{self.name}: 期望 {self.expected_type.__name__}，"
                f"实际 {type(value).__name__}"
            )
        obj.__dict__[self.name] = value

class AppConfig:
    debug = ConfigField(False, "是否开启调试模式")
    port = ConfigField(8080, "服务端口")
    host = ConfigField("localhost", "服务地址")
    max_connections = ConfigField(100, "最大连接数")

    def show(self):
        print("当前配置：")
        for name in ["debug", "port", "host", "max_connections"]:
            field_obj = type(self).__dict__[name]
            value = getattr(self, name)
            print(f"  {name} = {value!r:>20s}  # {field_obj.description}")

config = AppConfig()
config.show()
# 当前配置：
#   debug =                False  # 是否开启调试模式
#   port =                 8080  # 服务端口
#   host =          'localhost'  # 服务地址
#   max_connections =        100  # 最大连接数

config.port = 3000
config.debug = True
print(f"端口：{config.port}")    # 端口：3000

try:
    config.port = "3000"
except TypeError as e:
    print(e)  # port: 期望 int，实际 str
```

# user_service.py
from dataclasses import dataclass
from typing import Optional, Protocol

class Database(Protocol):
    def find(self, table: str, id: int) -> Optional[dict]: ...
    def insert(self, table: str, data: dict) -> int: ...

@dataclass
class User:
    id: int
    name: str
    email: str

class UserService:
    def __init__(self, db: Database) -> None:
        self.db = db

    def get_user(self, user_id: int) -> User:
        row = self.db.find("users", user_id)
        if row is None:
            raise ValueError(f"用户 {user_id} 不存在")
        return User(id=row["id"], name=row["name"], email=row["email"])

    def create_user(self, name: str, email: str) -> User:
        if not name.strip():
            raise ValueError("名字不能为空")
        if "@" not in email:
            raise ValueError("邮箱格式不正确")
        new_id = self.db.insert("users", {"name": name, "email": email})
        return User(id=new_id, name=name, email=email)

## 9. 总结

### 类型提示速查

| 类型 | 用途 | 示例 |
|------|------|------|
| `int`, `str`, `float`, `bool` | 基础类型 | `x: int = 42` |
| `list[T]`, `dict[K, V]` | 容器类型 | `items: list[str]` |
| `tuple[T, ...]` | 不定长元组 | `nums: tuple[int, ...]` |
| `Optional[T]` | 可选（T 或 None） | `Optional[str]` |
| `Union[A, B]` / `A \| B` | 联合类型 | `int \| str` |
| `Any` | 任意类型 | 尽量少用 |
| `Literal["a", "b"]` | 字面量 | 限制具体值 |
| `TypeVar("T")` | 泛型变量 | 泛型函数/类 |
| `Generic[T]` | 泛型基类 | `class Stack(Generic[T])` |
| `Protocol` | 结构化子类型 | 鸭子类型形式化 |
| `Callable[[A], R]` | 可调用 | 函数/回调类型 |
| `TypedDict` | 类型化字典 | 固定结构的字典 |
| `Final` | 常量 | 不允许重新赋值 |
| `ClassVar` | 类变量 | 标记为非实例变量 |
| `Annotated` | 附加元数据 | 框架利用的额外信息 |
