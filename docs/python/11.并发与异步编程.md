# Python 并发与异步编程

并发编程让程序同时处理多个任务，提高效率。Python 提供了**多线程**、**多进程**和**异步 IO**三种并发模型。

## 1. 并发编程的基本概念

### 1.1 并发 vs 并行

```
并发（Concurrency）：多个任务交替执行（一个 CPU 在任务间快速切换）
  任务A：██  ██  ██
  任务B：  ██  ██  ██
  ----> 时间轴

并行（Parallelism）：多个任务同时执行（多个 CPU 各自执行一个任务）
  CPU1 任务A：██████████
  CPU2 任务B：██████████
  ----> 时间轴
```

### 1.2 适用场景

| 场景 | 任务类型 | 推荐方案 | 原因 |
|------|----------|----------|------|
| 网络请求、文件 IO | IO 密集型 | 多线程 / 异步 | IO 等待时可以切换到其他任务 |
| 数据计算、图像处理 | CPU 密集型 | 多进程 | 利用多核 CPU 真正并行 |
| 高并发网络服务 | IO 密集型 | asyncio | 单线程高效处理数万连接 |

### 1.3 GIL（全局解释器锁）

CPython 的 GIL 限制同一时刻只有一个线程执行 Python 字节码。

```python
# GIL 的影响：
# - 多线程无法利用多核 CPU 做并行计算
# - 但 IO 操作（网络、文件、sleep）会释放 GIL，所以多线程对 IO 密集型任务仍有效
# - CPU 密集型任务必须用多进程绕过 GIL

# 简单验证
import threading
import time

counter = 0

def increment(n):
    global counter
    for _ in range(n):
        counter += 1   # ⚠️ 非线程安全，counter += 1 不是原子操作

# 单线程
counter = 0
increment(1_000_000)
print(f"单线程：{counter}")  # 单线程：1000000  ← 结果正确

# 多线程
counter = 0
t1 = threading.Thread(target=increment, args=(500_000,))
t2 = threading.Thread(target=increment, args=(500_000,))
t1.start(); t2.start()
t1.join(); t2.join()
print(f"多线程：{counter}")  # 多线程：876543  ← 结果不确定，可能小于 1000000
```

## 2. 多线程（threading）

### 2.1 创建线程

```python
import threading
import time

def download(url, duration):
    print(f"[{threading.current_thread().name}] 开始下载 {url}")
    time.sleep(duration)  # 模拟 IO 操作
    print(f"[{threading.current_thread().name}] 完成下载 {url}")

# ===== 方式 1：直接创建 Thread =====
t1 = threading.Thread(target=download, args=("page1.html", 2))
t2 = threading.Thread(target=download, args=("page2.html", 3))

start = time.time()
t1.start()   # 启动线程（非阻塞）
t2.start()

t1.join()    # 等待线程完成
t2.join()

print(f"总耗时：{time.time() - start:.1f}s")
# [Thread-1 (download)] 开始下载 page1.html
# [Thread-2 (download)] 开始下载 page2.html
# [Thread-1 (download)] 完成下载 page1.html
# [Thread-2 (download)] 完成下载 page2.html
# 总耗时：3.0s  ← 并发执行，不是 2+3=5s

# ===== 方式 2：继承 Thread =====
class DownloadThread(threading.Thread):
    def __init__(self, url, duration):
        super().__init__()
        self.url = url
        self.duration = duration
        self.result = None

    def run(self):
        time.sleep(self.duration)
        self.result = f"{self.url} 的内容"

t = DownloadThread("page3.html", 1)
t.start()
t.join()
print(t.result)  # page3.html 的内容
```

### 2.2 线程同步 —— Lock

```python
import threading

# ===== 不加锁：结果不确定 =====
counter = 0

def unsafe_increment(n):
    global counter
    for _ in range(n):
        counter += 1

# ===== 加锁：结果正确 =====
lock = threading.Lock()
safe_counter = 0

def safe_increment(n):
    global safe_counter
    for _ in range(n):
        lock.acquire()     # 获取锁
        safe_counter += 1
        lock.release()     # 释放锁

# 推荐用 with 语句（自动获取和释放锁）
def safe_increment_v2(n):
    global safe_counter
    for _ in range(n):
        with lock:          # 更简洁
            safe_counter += 1

threads = []
for _ in range(4):
    t = threading.Thread(target=safe_increment, args=(250_000,))
    threads.append(t)
    t.start()

for t in threads:
    t.join()

print(f"安全计数：{safe_counter}")  # 安全计数：1000000  ← 结果正确
```

### 2.3 其他同步工具

```python
import threading
import time

# ===== RLock（可重入锁）—— 同一线程可以多次获取 =====
rlock = threading.RLock()

def recursive_func(n):
    with rlock:
        if n > 0:
            print(f"  递归层级 {n}")
            recursive_func(n - 1)  # 同一线程再次获取锁，Lock 会死锁，RLock 不会

recursive_func(3)
#   递归层级 3
#   递归层级 2
#   递归层级 1

# ===== Event —— 线程间通知 =====
event = threading.Event()

def waiter():
    print("  等待信号...")
    event.wait()          # 阻塞直到 event 被 set
    print("  收到信号，继续执行")

def sender():
    time.sleep(1)
    print("  发送信号")
    event.set()           # 通知所有等待的线程

t1 = threading.Thread(target=waiter)
t2 = threading.Thread(target=sender)
t1.start(); t2.start()
t1.join(); t2.join()
#   等待信号...
#   发送信号
#   收到信号，继续执行

# ===== Semaphore —— 限制并发数量 =====
semaphore = threading.Semaphore(3)  # 最多 3 个线程同时访问

def limited_task(task_id):
    with semaphore:
        print(f"  任务 {task_id} 开始")
        time.sleep(1)
        print(f"  任务 {task_id} 完成")

threads = [threading.Thread(target=limited_task, args=(i,)) for i in range(6)]
for t in threads: t.start()
for t in threads: t.join()
# 每次最多 3 个任务同时运行
```

### 2.4 线程安全的数据结构

```python
import threading
import queue
import time

# queue.Queue 是线程安全的队列
q = queue.Queue(maxsize=5)

def producer():
    for i in range(5):
        item = f"产品-{i}"
        q.put(item)     # 队列满时会阻塞
        print(f"  [生产者] 生产了 {item}")
        time.sleep(0.3)
    q.put(None)          # 发送结束信号

def consumer():
    while True:
        item = q.get()   # 队列空时会阻塞
        if item is None:
            break
        print(f"  [消费者] 消费了 {item}")
        time.sleep(0.5)

t1 = threading.Thread(target=producer)
t2 = threading.Thread(target=consumer)
t1.start(); t2.start()
t1.join(); t2.join()
#   [生产者] 生产了 产品-0
#   [消费者] 消费了 产品-0
#   [生产者] 生产了 产品-1
#   [消费者] 消费了 产品-1
#   ...

# 其他队列类型
# queue.LifoQueue()      后进先出（栈）
# queue.PriorityQueue()  优先级队列
```

## 3. 多进程（multiprocessing）

多进程可以绑过 GIL，利用多核 CPU 实现真正的并行。

### 3.1 创建进程

```python
import multiprocessing
import os
import time

def compute(task_id, n):
    """CPU 密集型任务"""
    print(f"  进程 {os.getpid()} 执行任务 {task_id}")
    total = sum(i * i for i in range(n))
    return total

if __name__ == "__main__":
    # 创建进程
    p1 = multiprocessing.Process(target=compute, args=(1, 5_000_000))
    p2 = multiprocessing.Process(target=compute, args=(2, 5_000_000))

    start = time.time()
    p1.start()
    p2.start()
    p1.join()
    p2.join()
    print(f"多进程耗时：{time.time() - start:.2f}s")

    # 对比单进程
    start = time.time()
    compute(1, 5_000_000)
    compute(2, 5_000_000)
    print(f"单进程耗时：{time.time() - start:.2f}s")

    # 多进程耗时约为单进程的一半（双核情况下）
```

### 3.2 进程间通信

```python
import multiprocessing

if __name__ == "__main__":
    # ===== Queue：进程安全队列 =====
    q = multiprocessing.Queue()

    def producer(queue):
        for i in range(5):
            queue.put(f"数据-{i}")
        queue.put(None)

    def consumer(queue):
        while True:
            item = queue.get()
            if item is None:
                break
            print(f"  收到：{item}")

    p1 = multiprocessing.Process(target=producer, args=(q,))
    p2 = multiprocessing.Process(target=consumer, args=(q,))
    p1.start(); p2.start()
    p1.join(); p2.join()
    #   收到：数据-0
    #   收到：数据-1
    #   ...

    # ===== Value / Array：共享内存 =====
    shared_counter = multiprocessing.Value("i", 0)  # 'i' = int
    lock = multiprocessing.Lock()

    def increment(counter, lock, n):
        for _ in range(n):
            with lock:
                counter.value += 1

    procs = []
    for _ in range(4):
        p = multiprocessing.Process(
            target=increment, args=(shared_counter, lock, 250_000)
        )
        procs.append(p)
        p.start()

    for p in procs:
        p.join()

    print(f"共享计数器：{shared_counter.value}")  # 共享计数器：1000000
```

## 4. 线程池与进程池（concurrent.futures）

`concurrent.futures` 提供了统一的高层接口，是实际开发中最常用的方式。

### 4.1 ThreadPoolExecutor

```python
from concurrent.futures import ThreadPoolExecutor, as_completed
import time

def fetch_url(url):
    """模拟网络请求"""
    time.sleep(1)  # 模拟 IO
    return f"{url} 的内容（{len(url)}字节）"

urls = [
    "https://api.example.com/users",
    "https://api.example.com/posts",
    "https://api.example.com/comments",
    "https://api.example.com/albums",
    "https://api.example.com/photos",
]

# ===== 串行执行 =====
start = time.time()
for url in urls:
    fetch_url(url)
print(f"串行耗时：{time.time() - start:.1f}s")
# 串行耗时：5.0s

# ===== 线程池并发 =====
start = time.time()
with ThreadPoolExecutor(max_workers=5) as executor:
    # 方式 1：map（按顺序返回结果）
    results = executor.map(fetch_url, urls)
    for result in results:
        print(f"  {result}")

print(f"并发耗时：{time.time() - start:.1f}s")
# 并发耗时：1.0s  ← 快了 5 倍
```

### 4.2 submit + as_completed

```python
from concurrent.futures import ThreadPoolExecutor, as_completed
import time
import random

def fetch_data(task_id):
    duration = random.uniform(0.5, 2.0)
    time.sleep(duration)
    return {"task": task_id, "time": round(duration, 2)}

with ThreadPoolExecutor(max_workers=3) as executor:
    # submit 返回 Future 对象
    futures = {
        executor.submit(fetch_data, i): i
        for i in range(5)
    }

    # as_completed：哪个先完成就先返回哪个（不按提交顺序）
    for future in as_completed(futures):
        task_id = futures[future]
        try:
            result = future.result()  # 获取返回值
            print(f"  任务 {task_id} 完成：耗时 {result['time']}s")
        except Exception as e:
            print(f"  任务 {task_id} 失败：{e}")

# 输出顺序不固定，先完成的先输出：
#   任务 2 完成：耗时 0.53s
#   任务 0 完成：耗时 0.87s
#   任务 4 完成：耗时 1.12s
#   ...
```

### 4.3 ProcessPoolExecutor

```python
from concurrent.futures import ProcessPoolExecutor
import time
import math

def is_prime(n):
    """判断素数（CPU 密集型）"""
    if n < 2:
        return False
    for i in range(2, int(math.sqrt(n)) + 1):
        if n % i == 0:
            return False
    return True

def count_primes(start, end):
    """统计范围内的素数数量"""
    return sum(1 for n in range(start, end) if is_prime(n))

if __name__ == "__main__":
    ranges = [(1, 250_000), (250_000, 500_000),
              (500_000, 750_000), (750_000, 1_000_000)]

    # 串行
    start = time.time()
    serial_result = sum(count_primes(s, e) for s, e in ranges)
    serial_time = time.time() - start
    print(f"串行：{serial_result} 个素数，耗时 {serial_time:.2f}s")

    # 多进程并行
    start = time.time()
    with ProcessPoolExecutor(max_workers=4) as executor:
        futures = [executor.submit(count_primes, s, e) for s, e in ranges]
        parallel_result = sum(f.result() for f in futures)
    parallel_time = time.time() - start
    print(f"并行：{parallel_result} 个素数，耗时 {parallel_time:.2f}s")
    print(f"加速比：{serial_time / parallel_time:.1f}x")

    # 串行：78498 个素数，耗时 2.85s
    # 并行：78498 个素数，耗时 0.95s
    # 加速比：3.0x
```

### 4.4 Future 对象

```python
from concurrent.futures import ThreadPoolExecutor
import time

def task(n):
    time.sleep(n)
    if n == 3:
        raise ValueError("任务出错了")
    return n * 10

with ThreadPoolExecutor(max_workers=3) as executor:
    future = executor.submit(task, 1)

    # 查询状态
    print(future.done())       # False（可能还在执行）
    print(future.running())    # True（正在执行）
    print(future.cancelled())  # False

    # 获取结果（会阻塞直到完成）
    result = future.result(timeout=5)  # 超时 5 秒
    print(f"结果：{result}")   # 结果：10

    # 异常处理
    future_err = executor.submit(task, 3)
    try:
        future_err.result()
    except ValueError as e:
        print(f"捕获异常：{e}")  # 捕获异常：任务出错了

    # 也可以用 exception() 获取异常对象
    print(future_err.exception())  # 任务出错了

    # 添加回调
    def on_complete(fut):
        if fut.exception():
            print(f"  回调：失败 - {fut.exception()}")
        else:
            print(f"  回调：成功 - {fut.result()}")

    f = executor.submit(task, 2)
    f.add_done_callback(on_complete)
    # 任务完成后自动调用：回调：成功 - 20
```

## 5. 异步编程（asyncio）

asyncio 是 Python 的异步 IO 框架，使用 `async`/`await` 语法在**单线程**内实现高效并发。

### 5.1 async/await 基础

```python
import asyncio
import time

# async def 定义协程函数
async def say_after(delay, message):
    await asyncio.sleep(delay)  # 非阻塞等待
    print(message)

async def main():
    start = time.time()

    # 串行执行（一个等完再执行下一个）
    await say_after(1, "你好")
    await say_after(2, "世界")

    print(f"串行耗时：{time.time() - start:.1f}s")
    # 你好
    # 世界
    # 串行耗时：3.0s

# 运行协程
asyncio.run(main())
```

### 5.2 并发执行任务

```python
import asyncio
import time

async def fetch(name, delay):
    print(f"  开始 {name}")
    await asyncio.sleep(delay)
    print(f"  完成 {name}")
    return f"{name} 的数据"

async def main():
    start = time.time()

    # ===== 方式 1：asyncio.gather（最常用）=====
    # 同时启动多个协程，等待全部完成
    results = await asyncio.gather(
        fetch("任务A", 2),
        fetch("任务B", 1),
        fetch("任务C", 3),
    )
    print(f"结果：{results}")
    print(f"耗时：{time.time() - start:.1f}s")

asyncio.run(main())
#   开始 任务A
#   开始 任务B
#   开始 任务C
#   完成 任务B       ← B 最先完成（1秒）
#   完成 任务A       ← A 第二完成（2秒）
#   完成 任务C       ← C 最后完成（3秒）
# 结果：['任务A 的数据', '任务B 的数据', '任务C 的数据']  ← 按提交顺序
# 耗时：3.0s          ← 而非 2+1+3=6s
```

### 5.3 创建任务（Task）

```python
import asyncio

async def fetch(name, delay):
    await asyncio.sleep(delay)
    return f"{name}: {delay}s"

async def main():
    # ===== 方式 2：asyncio.create_task =====
    # 创建任务后立即开始执行（不需要 await 才启动）
    task1 = asyncio.create_task(fetch("A", 2))
    task2 = asyncio.create_task(fetch("B", 1))
    task3 = asyncio.create_task(fetch("C", 3))

    # 可以先做其他事情...
    print("任务已提交，做点别的事...")
    await asyncio.sleep(0.5)
    print("其他事做完了，等待结果...")

    # 等待结果
    r1 = await task1
    r2 = await task2
    r3 = await task3
    print(f"结果：{r1}, {r2}, {r3}")

asyncio.run(main())
# 任务已提交，做点别的事...
# 其他事做完了，等待结果...
# 结果：A: 2s, B: 1s, C: 3s
```

### 5.4 TaskGroup（Python 3.11+）

```python
import asyncio

async def fetch(name, delay):
    await asyncio.sleep(delay)
    if name == "bad":
        raise ValueError(f"{name} 出错了")
    return f"{name}: ok"

async def main():
    # TaskGroup 提供结构化并发，任何任务失败时会取消其他任务
    try:
        async with asyncio.TaskGroup() as tg:
            task1 = tg.create_task(fetch("A", 1))
            task2 = tg.create_task(fetch("B", 2))
            task3 = tg.create_task(fetch("C", 1))
        # 所有任务完成后才会到这里
        print(task1.result(), task2.result(), task3.result())
    except* ValueError as eg:
        # ExceptionGroup 捕获（Python 3.11+）
        for e in eg.exceptions:
            print(f"捕获错误：{e}")

asyncio.run(main())
# A: ok B: ok C: ok
```

### 5.5 异步迭代器与异步生成器

```python
import asyncio

# 异步生成器
async def async_range(start, stop, delay=0.5):
    for i in range(start, stop):
        await asyncio.sleep(delay)
        yield i

async def main():
    # async for 遍历异步迭代器
    async for num in async_range(1, 6, 0.3):
        print(f"  收到：{num}")
    #   收到：1
    #   收到：2
    #   收到：3
    #   收到：4
    #   收到：5

    # 异步列表推导式
    result = [num async for num in async_range(1, 4, 0.1)]
    print(result)  # [1, 2, 3]

asyncio.run(main())
```

### 5.6 异步上下文管理器

```python
import asyncio

class AsyncTimer:
    def __init__(self, label):
        self.label = label

    async def __aenter__(self):
        self.start = asyncio.get_event_loop().time()
        print(f"  [{self.label}] 开始")
        return self

    async def __aexit__(self, exc_type, exc_val, exc_tb):
        elapsed = asyncio.get_event_loop().time() - self.start
        print(f"  [{self.label}] 结束，耗时 {elapsed:.2f}s")

async def main():
    async with AsyncTimer("任务"):
        await asyncio.sleep(1)
        print("  执行中...")

asyncio.run(main())
#   [任务] 开始
#   执行中...
#   [任务] 结束，耗时 1.00s
```

## 6. asyncio 进阶

### 6.1 超时控制

```python
import asyncio

async def slow_task():
    await asyncio.sleep(10)
    return "完成"

async def main():
    # wait_for：超时后取消任务
    try:
        result = await asyncio.wait_for(slow_task(), timeout=2.0)
        print(result)
    except asyncio.TimeoutError:
        print("任务超时了！")  # 任务超时了！

    # timeout（Python 3.11+）：上下文管理器方式
    try:
        async with asyncio.timeout(2.0):
            await slow_task()
    except TimeoutError:
        print("又超时了！")  # 又超时了！

asyncio.run(main())
```

### 6.2 asyncio.wait —— 更灵活的等待

```python
import asyncio
import random

async def task(name):
    delay = random.uniform(0.5, 2.0)
    await asyncio.sleep(delay)
    return f"{name}: {delay:.1f}s"

async def main():
    tasks = [asyncio.create_task(task(f"T{i}")) for i in range(5)]

    # FIRST_COMPLETED：任意一个完成就返回
    done, pending = await asyncio.wait(tasks, return_when=asyncio.FIRST_COMPLETED)
    first = done.pop()
    print(f"最先完成：{first.result()}")
    print(f"还剩 {len(pending)} 个任务")

    # 等待剩余任务
    done2, _ = await asyncio.wait(pending)
    for t in done2:
        print(f"  {t.result()}")

asyncio.run(main())
# 最先完成：T3: 0.5s
# 还剩 4 个任务
#   T0: 1.2s
#   ...
```

### 6.3 信号量限制并发数

```python
import asyncio

async def fetch(sem, url):
    async with sem:    # 限制同时执行的协程数量
        print(f"  开始请求 {url}")
        await asyncio.sleep(1)
        print(f"  完成请求 {url}")
        return f"{url} 的响应"

async def main():
    sem = asyncio.Semaphore(3)  # 最多 3 个并发

    urls = [f"url_{i}" for i in range(8)]
    tasks = [fetch(sem, url) for url in urls]
    results = await asyncio.gather(*tasks)
    print(f"\n完成 {len(results)} 个请求")

asyncio.run(main())
# 每次最多 3 个请求同时进行
```

### 6.4 在异步中运行同步代码

```python
import asyncio
import time

def blocking_io(n):
    """模拟阻塞的同步 IO（比如数据库查询）"""
    time.sleep(n)
    return f"结果（耗时 {n}s）"

def cpu_heavy(n):
    """CPU 密集型计算"""
    return sum(i * i for i in range(n))

async def main():
    loop = asyncio.get_event_loop()

    # run_in_executor：在线程池中运行同步函数（不阻塞事件循环）
    start = time.time()
    results = await asyncio.gather(
        loop.run_in_executor(None, blocking_io, 2),  # None = 默认线程池
        loop.run_in_executor(None, blocking_io, 3),
    )
    print(f"IO 结果：{results}，耗时 {time.time() - start:.1f}s")
    # IO 结果：['结果（耗时 2s）', '结果（耗时 3s）']，耗时 3.0s

    # CPU 密集型任务放到进程池中
    from concurrent.futures import ProcessPoolExecutor
    with ProcessPoolExecutor() as pool:
        result = await loop.run_in_executor(pool, cpu_heavy, 10_000_000)
        print(f"计算结果：{result}")

asyncio.run(main())
```

### 7.2 生产者-消费者模式（asyncio）

```python
import asyncio
import random

async def producer(queue, producer_id):
    for i in range(3):
        item = f"P{producer_id}-item{i}"
        await asyncio.sleep(random.uniform(0.1, 0.5))
        await queue.put(item)
        print(f"  [生产者{producer_id}] 生产 {item}")
    await queue.put(None)  # 结束信号

async def consumer(queue, consumer_id):
    while True:
        item = await queue.get()
        if item is None:
            break
        print(f"  [消费者{consumer_id}] 消费 {item}")
        await asyncio.sleep(random.uniform(0.1, 0.3))

async def main():
    queue = asyncio.Queue(maxsize=5)

    # 2 个生产者 + 2 个消费者
    producers = [producer(queue, i) for i in range(2)]
    consumers = [consumer(queue, i) for i in range(2)]

    # 先等生产者完成
    await asyncio.gather(*producers)
    # 发送额外的 None 给第二个消费者
    await queue.put(None)
    # 等消费者完成
    await asyncio.gather(*consumers)
    print("全部完成")

asyncio.run(main())
#   [生产者0] 生产 P0-item0
#   [生产者1] 生产 P1-item0
#   [消费者0] 消费 P0-item0
#   [消费者1] 消费 P1-item0
#   ...
# 全部完成
```

## 8. 三种方案对比

### 8.1 IO 密集型任务对比

```python
import time
import asyncio
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor
import threading

def io_task(task_id):
    """模拟 IO 操作"""
    time.sleep(1)
    return task_id

async def async_io_task(task_id):
    await asyncio.sleep(1)
    return task_id

N = 10

# ===== 串行 =====
start = time.time()
results = [io_task(i) for i in range(N)]
print(f"串行：     {time.time() - start:.2f}s")

# ===== 多线程 =====
start = time.time()
with ThreadPoolExecutor(max_workers=N) as executor:
    results = list(executor.map(io_task, range(N)))
print(f"多线程：   {time.time() - start:.2f}s")

# ===== asyncio =====
async def async_main():
    start = time.time()
    tasks = [async_io_task(i) for i in range(N)]
    results = await asyncio.gather(*tasks)
    print(f"asyncio：  {time.time() - start:.2f}s")

asyncio.run(async_main())

# 串行：     10.03s
# 多线程：   1.01s   ← IO 密集型，线程很高效
# asyncio：  1.00s   ← 同样高效，且更轻量
```

### 8.2 CPU 密集型任务对比

```python
import time
from concurrent.futures import ThreadPoolExecutor, ProcessPoolExecutor

def cpu_task(n):
    """CPU 密集型计算"""
    return sum(i * i for i in range(n))

N = 5_000_000

if __name__ == "__main__":
    # 串行
    start = time.time()
    results = [cpu_task(N) for _ in range(4)]
    print(f"串行：     {time.time() - start:.2f}s")

    # 多线程（受 GIL 限制，不会更快）
    start = time.time()
    with ThreadPoolExecutor(max_workers=4) as executor:
        results = list(executor.map(lambda _: cpu_task(N), range(4)))
    print(f"多线程：   {time.time() - start:.2f}s")

    # 多进程（真正并行）
    start = time.time()
    with ProcessPoolExecutor(max_workers=4) as executor:
        results = list(executor.map(lambda _: cpu_task(N), range(4)))
    print(f"多进程：   {time.time() - start:.2f}s")

    # 串行：     3.20s
    # 多线程：   3.15s  ← GIL 限制，几乎没有提升
    # 多进程：   1.05s  ← 真正并行，接近 4 倍加速
```

## 9. 综合示例

### 示例 1：并发文件处理器

```python
from concurrent.futures import ThreadPoolExecutor
from pathlib import Path
import hashlib

def compute_hash(filepath):
    """计算文件的 MD5 哈希"""
    h = hashlib.md5()
    with open(filepath, "rb") as f:
        for chunk in iter(lambda: f.read(8192), b""):
            h.update(chunk)
    return filepath, h.hexdigest()

def batch_hash(directory, pattern="*"):
    """批量计算目录中文件的哈希"""
    files = [f for f in Path(directory).rglob(pattern) if f.is_file()]
    if not files:
        print("没有找到文件")
        return {}

    results = {}
    with ThreadPoolExecutor(max_workers=8) as executor:
        futures = {executor.submit(compute_hash, f): f for f in files}
        for future in futures:
            filepath, hash_value = future.result()
            results[str(filepath)] = hash_value
            print(f"  {filepath.name}: {hash_value[:16]}...")

    return results

# 使用示例
# hashes = batch_hash(".", "*.py")

```

### 示例 2：异步任务调度器

```python
import asyncio
from datetime import datetime

class AsyncScheduler:
    def __init__(self):
        self.tasks = []

    def schedule(self, coro_func, *args, delay=0, name=None):
        """添加定时任务"""
        self.tasks.append({
            "func": coro_func,
            "args": args,
            "delay": delay,
            "name": name or coro_func.__name__,
        })

    async def _run_task(self, task_info):
        await asyncio.sleep(task_info["delay"])
        name = task_info["name"]
        now = datetime.now().strftime("%H:%M:%S")
        print(f"  [{now}] 执行 {name}")
        result = await task_info["func"](*task_info["args"])
        print(f"  [{now}] {name} 完成: {result}")
        return result

    async def run_all(self):
        """并发运行所有任务"""
        coros = [self._run_task(t) for t in self.tasks]
        return await asyncio.gather(*coros)

# 定义一些异步任务
async def fetch_data(source):
    await asyncio.sleep(1)
    return f"{source} 的数据"

async def process_data(data_type):
    await asyncio.sleep(0.5)
    return f"{data_type} 处理完毕"

# 使用调度器
scheduler = AsyncScheduler()
scheduler.schedule(fetch_data, "数据库", delay=0, name="查询数据库")
scheduler.schedule(fetch_data, "API", delay=0, name="调用API")
scheduler.schedule(process_data, "日志", delay=0.2, name="处理日志")

asyncio.run(scheduler.run_all())
#   [14:30:00] 执行 处理日志
#   [14:30:00] 执行 查询数据库
#   [14:30:00] 执行 调用API
#   [14:30:00] 处理日志 完成: 日志 处理完毕
#   [14:30:01] 查询数据库 完成: 数据库 的数据
#   [14:30:01] 调用API 完成: API 的数据
```

## 10. 总结

### 方案选择速查

| 场景 | 方案 | 模块 |
|------|------|------|
| IO 密集 + 简单 | 线程池 | `concurrent.futures.ThreadPoolExecutor` |
| IO 密集 + 高并发 | asyncio | `asyncio` + `aiohttp`/`httpx` |
| CPU 密集 | 进程池 | `concurrent.futures.ProcessPoolExecutor` |
| 混合任务 | asyncio + 执行器 | `loop.run_in_executor()` |

### 常用 API 速查

```python
# ========== threading ==========
t = threading.Thread(target=func, args=())
t.start()                    # 启动
t.join()                     # 等待完成
threading.Lock()             # 互斥锁
threading.Event()            # 事件通知
threading.Semaphore(n)       # 信号量

# ========== concurrent.futures ==========
with ThreadPoolExecutor(max_workers=N) as executor:
    results = executor.map(func, iterable)     # 按顺序返回
    future = executor.submit(func, *args)      # 返回 Future
    for f in as_completed(futures):             # 先完成先返回
        result = f.result()

# ========== asyncio ==========
asyncio.run(main())                            # 运行入口
await asyncio.gather(*coros)                   # 并发等待全部
task = asyncio.create_task(coro)               # 创建任务
await asyncio.wait_for(coro, timeout=N)        # 超时控制
await asyncio.sleep(N)                         # 非阻塞等待
asyncio.Semaphore(N)                           # 限制并发数
loop.run_in_executor(pool, sync_func, *args)   # 运行同步代码
```

### 注意事项

```python
# 1. CPU 密集型任务不要用多线程（GIL 限制），用多进程
# 2. 多进程代码要放在 if __name__ == "__main__": 中（Windows 必须）
# 3. asyncio 中不能使用 time.sleep()（会阻塞事件循环），要用 await asyncio.sleep()
# 4. 共享数据要加锁，或使用线程安全的数据结构（queue.Queue）
# 5. 线程池/进程池的 max_workers 不是越大越好：
#    - IO 密集型：可以设较大（如 20-100）
#    - CPU 密集型：设为 CPU 核心数（os.cpu_count()）
```
