# Go 语言核心语法

## 1. 函数

### 1.1 函数定义与调用

```go
package main

import "fmt"

// 基本函数：无参数无返回值
func greet() {
    fmt.Println("Hello!")
}

// 带参数的函数
func add(a int, b int) int {
    return a + b
}

// 相同类型参数可以合并类型声明
func multiply(a, b int) int {
    return a * b
}

// 多返回值
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, fmt.Errorf("除数不能为零")
    }
    return a / b, nil
}

// 命名返回值
func rectangle(width, height float64) (area, perimeter float64) {
    area = width * height
    perimeter = 2 * (width + height)
    return // 裸返回，自动返回命名返回值
}

func main() {
    greet()

    sum := add(3, 5)
    fmt.Println("3 + 5 =", sum)

    product := multiply(4, 6)
    fmt.Println("4 × 6 =", product)

    result, err := divide(10, 3)
    if err != nil {
        fmt.Println("错误:", err)
    } else {
        fmt.Printf("10 / 3 = %.2f\n", result)
    }

    // 用 _ 忽略不需要的返回值
    _, err2 := divide(10, 0)
    if err2 != nil {
        fmt.Println("错误:", err2)
    }

    area, perimeter := rectangle(5, 3)
    fmt.Printf("面积: %.1f, 周长: %.1f\n", area, perimeter)
}
```

```
// 输出:
Hello!
3 + 5 = 8
4 × 6 = 24
10 / 3 = 3.33
错误: 除数不能为零
面积: 15.0, 周长: 16.0
```

### 1.2 可变参数

```go
package main

import "fmt"

// 可变参数函数：参数个数不固定
func sum(nums ...int) int {
    total := 0
    for _, n := range nums {
        total += n
    }
    return total
}

// 可变参数必须放在最后
func printInfo(prefix string, values ...int) {
    fmt.Printf("%s: %v\n", prefix, values)
}

func main() {
    fmt.Println(sum(1, 2, 3))       // 6
    fmt.Println(sum(1, 2, 3, 4, 5)) // 15
    fmt.Println(sum())               // 0

    // 切片展开传递给可变参数
    nums := []int{10, 20, 30}
    fmt.Println(sum(nums...)) // 60

    printInfo("成绩", 90, 85, 78) // 成绩: [90 85 78]
}
```

```
// 输出:
6
15
0
60
成绩: [90 85 78]
```

### 1.3 匿名函数与闭包

```go
package main

import "fmt"

func main() {
    // 匿名函数：直接赋值给变量
    square := func(n int) int {
        return n * n
    }
    fmt.Println(square(5)) // 25

    // 匿名函数：立即执行（IIFE）
    result := func(a, b int) int {
        return a + b
    }(3, 4)
    fmt.Println(result) // 7

    // 闭包：函数引用外部变量
    counter := makeCounter()
    fmt.Println(counter()) // 1
    fmt.Println(counter()) // 2
    fmt.Println(counter()) // 3

    // 闭包的常见陷阱
    funcs := make([]func(), 3)
    for i := 0; i < 3; i++ {
        i := i // 重新声明局部变量，捕获当前值
        funcs[i] = func() {
            fmt.Print(i, " ")
        }
    }
    for _, f := range funcs {
        f()
    }
    fmt.Println()
    // 输出: 0 1 2（如果不重新声明 i，会全部输出 3）
}

// 闭包工厂：返回一个计数器函数
func makeCounter() func() int {
    count := 0
    return func() int {
        count++
        return count
    }
}
```

```
// 输出:
25
7
1
2
3
0 1 2
```

### 1.4 函数作为参数和返回值

```go
package main

import (
    "fmt"
    "strings"
)

// 函数作为参数（回调模式）
func apply(nums []int, fn func(int) int) []int {
    result := make([]int, len(nums))
    for i, v := range nums {
        result[i] = fn(v)
    }
    return result
}

// 函数作为返回值
func makeGreeter(greeting string) func(string) string {
    return func(name string) string {
        return greeting + ", " + name + "!"
    }
}

// 使用函数类型别名
type StringProcessor func(string) string

func process(s string, processors ...StringProcessor) string {
    for _, p := range processors {
        s = p(s)
    }
    return s
}

func main() {
    nums := []int{1, 2, 3, 4, 5}

    // 传入匿名函数：每个元素乘以2
    doubled := apply(nums, func(n int) int { return n * 2 })
    fmt.Println("翻倍:", doubled)

    // 传入匿名函数：每个元素平方
    squared := apply(nums, func(n int) int { return n * n })
    fmt.Println("平方:", squared)

    // 函数作为返回值
    hello := makeGreeter("Hello")
    hi := makeGreeter("Hi")
    fmt.Println(hello("Alice"))
    fmt.Println(hi("Bob"))

    // 链式处理
    result := process("  Hello, World!  ",
        strings.TrimSpace,
        strings.ToUpper,
    )
    fmt.Println(result)
}
```

```
// 输出:
翻倍: [2 4 6 8 10]
平方: [1 4 9 16 25]
Hello, Alice!
Hi, Bob!
HELLO, WORLD!
```

### 1.5 defer 语句

`defer` 延迟执行函数调用，在当前函数返回前执行，常用于资源释放。

```go
package main

import (
    "fmt"
    "os"
)

func main() {
    // 基本用法：在函数返回前执行
    fmt.Println("开始")
    defer fmt.Println("延迟执行 1")
    defer fmt.Println("延迟执行 2")
    defer fmt.Println("延迟执行 3")
    fmt.Println("结束")
    // defer 按 LIFO（后进先出）顺序执行

    fmt.Println("---")

    // 实际用途：确保文件关闭
    content, err := readFile("test.txt")
    if err != nil {
        fmt.Println("读取文件:", err)
    } else {
        fmt.Println(content)
    }

    // defer 与循环的注意事项
    fmt.Println("defer 参数在声明时求值:")
    x := 10
    defer fmt.Printf("defer 中 x = %d\n", x) // x 在此时求值为 10
    x = 20
    fmt.Printf("当前 x = %d\n", x) // 20
}

func readFile(filename string) (string, error) {
    file, err := os.Open(filename)
    if err != nil {
        return "", err
    }
    defer file.Close() // 确保函数退出时关闭文件

    buf := make([]byte, 1024)
    n, err := file.Read(buf)
    if err != nil {
        return "", err
    }
    return string(buf[:n]), nil
}
```

```
// 输出:
开始
结束
延迟执行 3
延迟执行 2
延迟执行 1
---
读取文件: open test.txt: The system cannot find the file specified.
defer 参数在声明时求值:
当前 x = 20
defer 中 x = 10
```

**defer 要点：**

- 多个 `defer` 按**后进先出（LIFO）**顺序执行
- `defer` 的参数在**声明时求值**，而非执行时
- 常用于关闭文件、释放锁、关闭连接等资源清理

### 1.6 panic 与 recover

```go
package main

import "fmt"

func main() {
    fmt.Println("程序开始")
    safeDiv(10, 0)
    fmt.Println("程序继续运行") // panic 被 recover 捕获后，程序继续执行

    fmt.Println("---")

    // 手动触发 panic
    safeDo(func() {
        panic("出了点问题!")
    })
    fmt.Println("safeDo 之后继续执行")
}

func safeDiv(a, b int) {
    defer func() {
        if r := recover(); r != nil {
            fmt.Println("捕获到 panic:", r)
        }
    }()

    fmt.Println(a / b) // 除以 0 会触发 panic
}

func safeDo(fn func()) {
    defer func() {
        if r := recover(); r != nil {
            fmt.Println("恢复:", r)
        }
    }()
    fn()
}
```

```
// 输出:
程序开始
捕获到 panic: runtime error: integer divide by zero
程序继续运行
---
恢复: 出了点问题!
safeDo 之后继续执行
```

**panic / recover 使用原则：**

- `panic` 用于**不可恢复的错误**（如程序初始化失败、不该到达的代码路径）
- `recover` 必须在 `defer` 函数中调用才有效
- 日常错误处理应使用 `error` 返回值，**不要**用 panic 替代正常的错误处理

## 2. 复合数据类型

### 2.1 数组

数组是**定长**、**值类型**的数据结构。

```go
package main

import "fmt"

func main() {
    // 声明并初始化
    var arr1 [3]int               // 零值: [0 0 0]
    arr2 := [3]int{1, 2, 3}      // 指定值
    arr3 := [...]int{10, 20, 30} // 编译器推断长度
    arr4 := [5]int{0: 1, 4: 5}   // 指定索引赋值

    fmt.Println("arr1:", arr1) // [0 0 0]
    fmt.Println("arr2:", arr2) // [1 2 3]
    fmt.Println("arr3:", arr3) // [10 20 30]
    fmt.Println("arr4:", arr4) // [1 0 0 0 5]

    // 访问与修改
    arr2[1] = 99
    fmt.Println("修改后 arr2:", arr2)     // [1 99 3]
    fmt.Println("长度:", len(arr2))        // 3

    // 遍历
    for i, v := range arr3 {
        fmt.Printf("  arr3[%d] = %d\n", i, v)
    }

    // 数组是值类型：赋值和传参会复制整个数组
    a := [3]int{1, 2, 3}
    b := a      // 复制
    b[0] = 100
    fmt.Println("a:", a) // [1 2 3]（a 不受影响）
    fmt.Println("b:", b) // [100 2 3]

    // 数组比较（相同长度和类型才能比较）
    x := [3]int{1, 2, 3}
    y := [3]int{1, 2, 3}
    fmt.Println("x == y:", x == y) // true
}
```

```
// 输出:
arr1: [0 0 0]
arr2: [1 2 3]
arr3: [10 20 30]
arr4: [1 0 0 0 5]
修改后 arr2: [1 99 3]
长度: 3
  arr3[0] = 10
  arr3[1] = 20
  arr3[2] = 30
a: [1 2 3]
b: [100 2 3]
x == y: true
```

> 实际开发中数组使用较少，更常用的是**切片（slice）**。

### 2.2 切片（Slice）

切片是 Go 中最常用的数据结构之一，是对底层数组的**引用**，长度可变。

```go
package main

import "fmt"

func main() {
    // ========== 创建切片 ==========

    // 方式一：从数组或切片截取
    arr := [5]int{1, 2, 3, 4, 5}
    s1 := arr[1:4] // 左闭右开 [2 3 4]
    fmt.Println("s1:", s1)

    // 方式二：直接声明
    s2 := []int{10, 20, 30}
    fmt.Println("s2:", s2)

    // 方式三：make 创建（指定长度和容量）
    s3 := make([]int, 3, 5) // 长度3，容量5
    fmt.Println("s3:", s3)
    fmt.Printf("s3 len=%d, cap=%d\n", len(s3), cap(s3))

    // 方式四：nil 切片
    var s4 []int
    fmt.Println("s4:", s4, "nil?", s4 == nil) // [] nil? true

    fmt.Println()

    // ========== 切片操作 ==========

    s := []int{1, 2, 3, 4, 5}

    // 截取（不复制数据，共享底层数组）
    fmt.Println("s[1:3]:", s[1:3]) // [2 3]
    fmt.Println("s[:3]:", s[:3])   // [1 2 3]
    fmt.Println("s[2:]:", s[2:])   // [3 4 5]
    fmt.Println("s[:]:", s[:])     // [1 2 3 4 5]

    // append 追加元素
    s = append(s, 6, 7)
    fmt.Println("append:", s) // [1 2 3 4 5 6 7]

    // append 追加切片
    extra := []int{8, 9}
    s = append(s, extra...)
    fmt.Println("append 切片:", s) // [1 2 3 4 5 6 7 8 9]

    // 删除元素（删除索引 2 的元素）
    s = append(s[:2], s[3:]...)
    fmt.Println("删除索引2:", s) // [1 2 4 5 6 7 8 9]

    fmt.Println()

    // ========== 切片是引用类型 ==========
    original := []int{1, 2, 3}
    copied := original   // 共享底层数组
    copied[0] = 100
    fmt.Println("original:", original) // [100 2 3]（受影响！）
    fmt.Println("copied:", copied)     // [100 2 3]

    // copy 函数：真正的复制
    src := []int{1, 2, 3}
    dst := make([]int, len(src))
    copy(dst, src)
    dst[0] = 100
    fmt.Println("src:", src) // [1 2 3]（不受影响）
    fmt.Println("dst:", dst) // [100 2 3]
}
```

```
// 输出:
s1: [2 3 4]
s2: [10 20 30]
s3: [0 0 0]
s3 len=3, cap=5
s4: [] nil? true

s[1:3]: [2 3]
s[:3]: [1 2 3]
s[2:]: [3 4 5]
s[:]: [1 2 3 4 5]
append: [1 2 3 4 5 6 7]
append 切片: [1 2 3 4 5 6 7 8 9]
删除索引2: [1 2 4 5 6 7 8 9]

original: [100 2 3]
copied: [100 2 3]
src: [1 2 3]
dst: [100 2 3]
```

### 2.3 切片扩容机制

```go
package main

import "fmt"

func main() {
    s := make([]int, 0)
    prevCap := cap(s)

    for i := 0; i < 20; i++ {
        s = append(s, i)
        if cap(s) != prevCap {
            fmt.Printf("len=%-2d  cap: %d -> %d\n", len(s), prevCap, cap(s))
            prevCap = cap(s)
        }
    }
}
```

```
// 输出:
len=1   cap: 0 -> 1
len=2   cap: 1 -> 2
len=3   cap: 2 -> 4
len=5   cap: 4 -> 8
len=9   cap: 8 -> 16
len=17  cap: 16 -> 32
```

**扩容规则（Go 1.18+）：**

- 容量小于 256 时，新容量 = 旧容量 × 2
- 容量大于等于 256 时，增长因子逐渐从 2 降到 1.25
- **性能建议**：如果预知切片大小，使用 `make([]T, 0, capacity)` 预分配容量，避免频繁扩容

### 2.4 Map（映射）

```go
package main

import "fmt"

func main() {
    // ========== 创建 map ==========

    // 方式一：make 创建
    m1 := make(map[string]int)
    m1["apple"] = 5
    m1["banana"] = 3
    fmt.Println("m1:", m1)

    // 方式二：字面量初始化
    m2 := map[string]int{
        "go":     2009,
        "python": 1991,
        "java":   1995,
    }
    fmt.Println("m2:", m2)

    // 方式三：nil map（只能读不能写）
    var m3 map[string]int
    fmt.Println("m3 == nil:", m3 == nil) // true
    // m3["key"] = 1 // panic: assignment to entry in nil map

    fmt.Println()

    // ========== 基本操作 ==========

    scores := map[string]int{
        "Alice": 90,
        "Bob":   85,
        "Carol": 92,
    }

    // 读取
    fmt.Println("Alice:", scores["Alice"]) // 90

    // 判断 key 是否存在（重要！）
    val, ok := scores["Dave"]
    if ok {
        fmt.Println("Dave:", val)
    } else {
        fmt.Println("Dave 不存在, 零值:", val) // 0
    }

    // 修改
    scores["Bob"] = 95
    fmt.Println("修改后 Bob:", scores["Bob"]) // 95

    // 添加
    scores["Dave"] = 88
    fmt.Println("添加 Dave:", scores["Dave"]) // 88

    // 删除
    delete(scores, "Carol")
    fmt.Println("删除 Carol 后:", scores)

    // 长度
    fmt.Println("元素个数:", len(scores))

    fmt.Println()

    // ========== 遍历 ==========
    // 注意：map 遍历顺序是随机的
    fmt.Println("遍历 scores:")
    for key, value := range scores {
        fmt.Printf("  %s: %d\n", key, value)
    }
}
```

```
// 输出:

m3 == nil: true

Alice: 90
Dave 不存在, 零值: 0
修改后 Bob: 95
添加 Dave: 88
删除 Carol 后: map[Alice:90 Bob:95 Dave:88]
元素个数: 3

遍历 scores:
  Alice: 90

  Dave: 88
```

### 2.5 Map 进阶用法

```go
package main

import (
    "fmt"
    "sort"
)

func main() {
    // map 作为集合（Set）使用
    set := make(map[string]bool)
    set["apple"] = true
    set["banana"] = true
    set["apple"] = true // 重复添加无效

    if set["apple"] {
        fmt.Println("集合中包含 apple")
    }
    fmt.Println("集合大小:", len(set)) // 2

    fmt.Println()

    // map 的 value 为切片
    graph := map[string][]string{
        "Alice": {"Bob", "Carol"},
        "Bob":   {"Alice", "Dave"},
        "Carol": {"Alice"},
    }
    fmt.Println("Alice 的好友:", graph["Alice"])
    graph["Alice"] = append(graph["Alice"], "Dave")
    fmt.Println("添加后:", graph["Alice"])

    fmt.Println()

    // 有序遍历 map（先排序 key）
    ages := map[string]int{
        "Charlie": 30,
        "Alice":   25,
        "Bob":     28,
    }

    keys := make([]string, 0, len(ages))
    for k := range ages {
        keys = append(keys, k)
    }
    sort.Strings(keys)

    fmt.Println("按 key 排序遍历:")
    for _, k := range keys {
        fmt.Printf("  %s: %d\n", k, ages[k])
    }

    fmt.Println()

    // 统计词频
    words := []string{"go", "is", "great", "go", "is", "fast", "go"}
    freq := make(map[string]int)
    for _, w := range words {
        freq[w]++
    }
    fmt.Println("词频统计:", freq)
}
```

```
// 输出:
集合中包含 apple
集合大小: 2

Alice 的好友: [Bob Carol]
添加后: [Bob Carol Dave]

按 key 排序遍历:
  Alice: 25
  Bob: 28
  Charlie: 30

词频统计: map[fast:1 go:3 great:1 is:2]
```

### 2.6 字符串

```go
package main

import (
    "fmt"
    "strings"
    "unicode/utf8"
)

func main() {
    // ========== 字符串基础 ==========

    s := "Hello, 世界"

    // 长度：字节数 vs 字符数
    fmt.Println("字节长度:", len(s))                    // 13
    fmt.Println("字符长度:", utf8.RuneCountInString(s)) // 8

    // 索引访问得到的是字节（byte），不是字符
    fmt.Printf("s[0] = %c (byte)\n", s[0]) // H

    // 正确遍历字符
    fmt.Print("字符遍历: ")
    for _, ch := range s {
        fmt.Printf("%c ", ch)
    }
    fmt.Println()
    // H e l l o ,   世 界

    // 字符串不可变
    // s[0] = 'h' // 编译错误

    // 修改字符串需要转换为 []byte 或 []rune
    bs := []byte(s)
    bs[0] = 'h'
    fmt.Println(string(bs)) // hello, 世界

    rs := []rune(s)
    rs[7] = '界' // 修改最后一个字符
    fmt.Println(string(rs))

    fmt.Println()

    // ========== 字符串拼接 ==========

    // + 拼接（少量拼接时使用）
    s1 := "Hello" + ", " + "World"
    fmt.Println(s1)

    // fmt.Sprintf 格式化拼接
    name, age := "Go", 15
    s2 := fmt.Sprintf("%s 已经 %d 岁了", name, age)
    fmt.Println(s2)

    // strings.Builder（大量拼接时高效）
    var builder strings.Builder
    for i := 0; i < 5; i++ {
        fmt.Fprintf(&builder, "item%d ", i)
    }
    fmt.Println(builder.String())

    // strings.Join
    parts := []string{"Go", "is", "awesome"}
    s3 := strings.Join(parts, " ")
    fmt.Println(s3)

    fmt.Println()

    // ========== strings 包常用函数 ==========

    str := "  Hello, Go World!  "

    fmt.Println(strings.TrimSpace(str))           // "Hello, Go World!"
    fmt.Println(strings.ToUpper("hello"))          // HELLO
    fmt.Println(strings.ToLower("HELLO"))          // hello
    fmt.Println(strings.Contains(str, "Go"))       // true
    fmt.Println(strings.HasPrefix("golang", "go")) // true
    fmt.Println(strings.HasSuffix("main.go", ".go")) // true
    fmt.Println(strings.Index(str, "Go"))          // 9
    fmt.Println(strings.Count("banana", "a"))      // 3

    fmt.Println(strings.Replace("aabbcc", "b", "x", 1))  // aaxbcc（替换1次）
    fmt.Println(strings.ReplaceAll("aabbcc", "b", "x"))   // aaxxcc（全部替换）

    // 分割与合并
    csv := "apple,banana,cherry"
    items := strings.Split(csv, ",")
    fmt.Println(items) // [apple banana cherry]

    joined := strings.Join(items, " | ")
    fmt.Println(joined) // apple | banana | cherry

    // 重复
    fmt.Println(strings.Repeat("Go! ", 3)) // Go! Go! Go!
}
```

字符长度: 8
s[0] = H (byte)
字符遍历: H e l l o ,   世 界
hello, 世界

### 2.7 指针

```go
package main

import "fmt"

func main() {
    // ========== 指针基础 ==========

    x := 42
    p := &x // 取地址

    fmt.Println("x 的值:", x)     // 42
    fmt.Println("x 的地址:", p)   // 0xc0000b4008（每次运行不同）
    fmt.Println("指针指向的值:", *p) // 42（解引用）

    // 通过指针修改值
    *p = 100
    fmt.Println("修改后 x:", x) // 100

    fmt.Println()

    // ========== 值传递 vs 指针传递 ==========

    a := 10
    fmt.Println("修改前 a:", a)

    modifyValue(a)
    fmt.Println("值传递后 a:", a) // 不变

    modifyPointer(&a)
    fmt.Println("指针传递后 a:", a) // 变了

    fmt.Println()

    // ========== new 函数 ==========
    // new 分配内存并返回指针，值为零值
    ptr := new(int)
    fmt.Println("new(int):", *ptr) // 0
    *ptr = 42
    fmt.Println("赋值后:", *ptr)    // 42

    // ========== Go 指针的限制 ==========
    // 1. 不支持指针运算（不能 p++ 或 p+1）
    // 2. 不同类型的指针不能互相转换
    // 3. 指针不能与非指针类型比较（可以与 nil 比较）

    var nilPtr *int
    fmt.Println("nil 指针:", nilPtr == nil) // true
}

func modifyValue(n int) {
    n = 999 // 修改的是副本
}

func modifyPointer(n *int) {
    *n = 999 // 修改的是原始值
}
```

```
// 输出:
x 的值: 42
x 的地址: 0xc0000b4008
指针指向的值: 42
修改后 x: 100

修改前 a: 10
值传递后 a: 10
指针传递后 a: 999

new(int): 0
赋值后: 42
nil 指针: true
```

## 3. 结构体与方法

### 3.1 结构体定义与初始化

```go
package main

import "fmt"

// 定义结构体
type Person struct {
    Name string
    Age  int
    Email string
}

// 嵌套结构体
type Address struct {
    City    string
    ZipCode string
}

type Employee struct {
    Person  // 匿名嵌入（继承字段）
    Address // 匿名嵌入
    Company string
    Salary  float64
}

func main() {
    // ========== 创建结构体 ==========

    // 方式一：按字段名初始化（推荐）
    p1 := Person{
        Name:  "Alice",
        Age:   25,
        Email: "alice@example.com",
    }
    fmt.Println("p1:", p1)

    // 方式二：按顺序初始化（不推荐，字段多时易出错）
    p2 := Person{"Bob", 30, "bob@example.com"}
    fmt.Println("p2:", p2)

    // 方式三：零值初始化
    var p3 Person
    fmt.Println("p3:", p3) // {"" 0 ""}

    // 方式四：new 创建（返回指针）
    p4 := new(Person)
    p4.Name = "Carol"
    p4.Age = 28
    fmt.Println("p4:", *p4)

    // 方式五：取地址创建（返回指针，最常用）
    p5 := &Person{
        Name: "Dave",
        Age:  35,
    }
    fmt.Println("p5:", *p5)

    fmt.Println()

    // ========== 访问与修改 ==========
    p1.Age = 26
    fmt.Println("修改后:", p1.Name, p1.Age)

    // 指针访问：Go 自动解引用
    ptr := &p1
    fmt.Println("指针访问:", ptr.Name) // 等同于 (*ptr).Name

    fmt.Println()

    // ========== 嵌套结构体 ==========
    emp := Employee{
        Person:  Person{Name: "Eve", Age: 30, Email: "eve@corp.com"},
        Address: Address{City: "Beijing", ZipCode: "100000"},
        Company: "Tech Corp",
        Salary:  15000,
    }
    // 可以直接访问嵌入结构体的字段
    fmt.Println("姓名:", emp.Name)     // 提升的字段
    fmt.Println("城市:", emp.City)     // 提升的字段
    fmt.Println("公司:", emp.Company)
    fmt.Println("完整信息:", emp)

    fmt.Println()

    // ========== 结构体比较 ==========
    a := Person{Name: "Alice", Age: 25}
    b := Person{Name: "Alice", Age: 25}
    fmt.Println("a == b:", a == b) // true（所有字段都可比较时）
}
```

```
// 输出:
p1: {Alice 25 alice@example.com}
p2: {Bob 30 bob@example.com}
p3: { 0 }
p4: {Carol 28 }
p5: {Dave 35 }

修改后: Alice 26
指针访问: Alice

姓名: Eve
城市: Beijing
公司: Tech Corp
完整信息: {{Eve 30 eve@corp.com} {Beijing 100000} Tech Corp 15000}

a == b: true
```

### 3.2 结构体标签（Tag）

```go
package main

import (
    "encoding/json"
    "fmt"
)

type User struct {
    ID        int    `json:"id"`
    Username  string `json:"username"`
    Email     string `json:"email,omitempty"` // 空值时省略
    Password  string `json:"-"`               // 永远不序列化
    Age       int    `json:"age,string"`      // 序列化为字符串
}

func main() {
    user := User{
        ID:       1,
        Username: "alice",
        Email:    "",
        Password: "secret123",
        Age:      25,
    }

    // 结构体 → JSON
    data, _ := json.MarshalIndent(user, "", "  ")
    fmt.Println("序列化:")
    fmt.Println(string(data))

    fmt.Println()

    // JSON → 结构体
    jsonStr := `{"id":2,"username":"bob","email":"bob@example.com","age":"30"}`
    var user2 User
    json.Unmarshal([]byte(jsonStr), &user2)
    fmt.Println("反序列化:")
    fmt.Printf("%+v\n", user2)
}
```

```
// 输出:
序列化:
{
  "id": 1,
  "username": "alice",
  "age": "25"
}

反序列化:
{ID:2 Username:bob Email:bob@example.com Password: Age:30}
```

**常用 JSON 标签：**

| 标签 | 说明 |
|------|------|
| `` `json:"name"` `` | 指定 JSON 字段名 |
| `` `json:"name,omitempty"` `` | 零值时省略该字段 |
| `` `json:"-"` `` | 忽略该字段 |
| `` `json:"name,string"` `` | 将数值序列化为字符串 |

### 3.3 方法

方法是绑定到特定类型上的函数。

```go
package main

import (
    "fmt"
    "math"
)

// ========== 值接收者 vs 指针接收者 ==========

type Circle struct {
    Radius float64
}

// 值接收者：不会修改原始结构体
func (c Circle) Area() float64 {
    return math.Pi * c.Radius * c.Radius
}

// 值接收者：不会修改原始结构体
func (c Circle) Perimeter() float64 {
    return 2 * math.Pi * c.Radius
}

// 指针接收者：可以修改原始结构体
func (c *Circle) Scale(factor float64) {
    c.Radius *= factor
}

// 值接收者实现 Stringer 接口（自定义打印格式）
func (c Circle) String() string {
    return fmt.Sprintf("Circle(r=%.2f)", c.Radius)
}

func main() {
    c := Circle{Radius: 5}

    fmt.Println(c)                       // Circle(r=5.00)
    fmt.Printf("面积: %.2f\n", c.Area())      // 78.54
    fmt.Printf("周长: %.2f\n", c.Perimeter()) // 31.42

    // 指针接收者方法修改原始值
    c.Scale(2)
    fmt.Println("放大2倍后:", c)                 // Circle(r=10.00)
    fmt.Printf("新面积: %.2f\n", c.Area())     // 314.16

    // 指针也可以调用值接收者方法（Go 自动处理）
    p := &Circle{Radius: 3}
    fmt.Printf("指针调用: %.2f\n", p.Area()) // 28.27
}
```

```
// 输出:
Circle(r=5.00)
面积: 78.54
周长: 31.42
放大2倍后: Circle(r=10.00)
新面积: 314.16
指针调用: 28.27
```

**值接收者 vs 指针接收者选择原则：**

| 场景 | 选择 |
|------|------|
| 需要修改接收者的值 | 指针接收者 `(t *Type)` |
| 结构体较大，避免复制开销 | 指针接收者 |
| 只读操作，结构体较小 | 值接收者 `(t Type)` |
| 需要实现某个接口 | 保持一致性，通常用指针接收者 |
| **经验法则** | **如果不确定，优先使用指针接收者** |

### 3.4 构造函数模式

Go 没有构造函数，使用**工厂函数**代替。

```go
package main

import (
    "errors"
    "fmt"
)

type Config struct {
    Host    string
    Port    int
    Timeout int
    Debug   bool
}

// 简单构造函数
func NewConfig(host string, port int) *Config {
    return &Config{
        Host:    host,
        Port:    port,
        Timeout: 30, // 默认值
        Debug:   false,
    }
}

// 带验证的构造函数
func NewConfigValidated(host string, port int) (*Config, error) {
    if host == "" {
        return nil, errors.New("host 不能为空")
    }
    if port < 1 || port > 65535 {
        return nil, errors.New("port 必须在 1-65535 之间")
    }
    return &Config{
        Host:    host,
        Port:    port,
        Timeout: 30,
    }, nil
}

// 选项模式（Functional Options）—— 灵活设置可选参数
type Option func(*Config)

func WithTimeout(t int) Option {
    return func(c *Config) { c.Timeout = t }
}

func WithDebug(d bool) Option {
    return func(c *Config) { c.Debug = d }
}

func NewConfigWithOptions(host string, port int, opts ...Option) *Config {
    c := &Config{
        Host:    host,
        Port:    port,
        Timeout: 30,
        Debug:   false,
    }
    for _, opt := range opts {
        opt(c)
    }
    return c
}

func main() {
    // 简单构造
    c1 := NewConfig("localhost", 8080)
    fmt.Printf("c1: %+v\n", c1)

    // 带验证
    c2, err := NewConfigValidated("", 8080)
    if err != nil {
        fmt.Println("错误:", err)
    }
    _ = c2

    // 选项模式
    c3 := NewConfigWithOptions("0.0.0.0", 3000,
        WithTimeout(60),
        WithDebug(true),
    )
    fmt.Printf("c3: %+v\n", c3)

    // 只设置部分选项
    c4 := NewConfigWithOptions("127.0.0.1", 9090, WithDebug(true))
    fmt.Printf("c4: %+v\n", c4)
}
```

```
// 输出:
c1: &{Host:localhost Port:8080 Timeout:30 Debug:false}
错误: host 不能为空
c3: &{Host:0.0.0.0 Port:3000 Timeout:60 Debug:true}
c4: &{Host:127.0.0.1 Port:9090 Timeout:30 Debug:true}
```

### 3.5 方法与嵌入结构体

```go
package main

import "fmt"

type Animal struct {
    Name string
}

func (a Animal) Speak() string {
    return a.Name + " makes a sound"
}

type Dog struct {
    Animal // 嵌入 Animal
    Breed  string
}

// Dog 重写（覆盖）Speak 方法
func (d Dog) Speak() string {
    return d.Name + " says: Woof!"
}

type Cat struct {
    Animal
}

// Cat 没有重写 Speak，继承 Animal 的方法

func main() {
    dog := Dog{
        Animal: Animal{Name: "Buddy"},
        Breed:  "Golden Retriever",
    }
    cat := Cat{
        Animal: Animal{Name: "Kitty"},
    }

    fmt.Println(dog.Speak())        // Buddy says: Woof!（使用 Dog 自己的方法）
    fmt.Println(dog.Animal.Speak()) // Buddy makes a sound（显式调用嵌入的方法）
    fmt.Println(cat.Speak())        // Kitty makes a sound（继承 Animal 的方法）
    fmt.Println(dog.Name)           // Buddy（提升的字段）
    fmt.Println(dog.Breed)          // Golden Retriever
}
```

```
// 输出:
Buddy says: Woof!
Buddy makes a sound
Kitty makes a sound
Buddy
Golden Retriever
```

### 练习 1：实现一个简单的栈

```go
package main

import (
    "errors"
    "fmt"
)

type Stack struct {
    items []int
}

func NewStack() *Stack {
    return &Stack{items: make([]int, 0)}
}

func (s *Stack) Push(val int) {
    s.items = append(s.items, val)
}

func (s *Stack) Pop() (int, error) {
    if s.IsEmpty() {
        return 0, errors.New("栈为空")
    }
    last := len(s.items) - 1
    val := s.items[last]
    s.items = s.items[:last]
    return val, nil
}

func (s *Stack) Peek() (int, error) {
    if s.IsEmpty() {
        return 0, errors.New("栈为空")
    }
    return s.items[len(s.items)-1], nil
}

func (s *Stack) IsEmpty() bool {
    return len(s.items) == 0
}

func (s *Stack) Size() int {
    return len(s.items)
}

func (s Stack) String() string {
    return fmt.Sprintf("Stack%v", s.items)
}

func main() {
    s := NewStack()

    s.Push(10)
    s.Push(20)
    s.Push(30)
    fmt.Println(s)          // Stack[10 20 30]
    fmt.Println("大小:", s.Size()) // 3

    top, _ := s.Peek()
    fmt.Println("栈顶:", top) // 30

    val, _ := s.Pop()
    fmt.Println("弹出:", val) // 30
    fmt.Println(s)           // Stack[10 20]

    s.Pop()
    s.Pop()
    _, err := s.Pop()
    fmt.Println("空栈弹出:", err) // 栈为空
}
```

```
// 输出:
Stack[10 20 30]
大小: 3
栈顶: 30
弹出: 30
Stack[10 20]
空栈弹出: 栈为空
```

### 练习 2：学生成绩管理

```go
package main

import (
    "fmt"
    "sort"
)

type Student struct {
    Name   string
    Scores map[string]float64
}

func NewStudent(name string) *Student {
    return &Student{
        Name:   name,
        Scores: make(map[string]float64),
    }
}

func (s *Student) AddScore(subject string, score float64) {
    s.Scores[subject] = score
}

func (s *Student) Average() float64 {
    if len(s.Scores) == 0 {
        return 0
    }
    total := 0.0
    for _, score := range s.Scores {
        total += score
    }
    return total / float64(len(s.Scores))
}

func (s *Student) Grade() string {
    avg := s.Average()
    switch {
    case avg >= 90:
        return "A"
    case avg >= 80:
        return "B"
    case avg >= 70:
        return "C"
    case avg >= 60:
        return "D"
    default:
        return "F"
    }
}

func main() {
    students := []*Student{
        NewStudent("Alice"),
        NewStudent("Bob"),
        NewStudent("Carol"),
    }

    // 添加成绩
    students[0].AddScore("数学", 92)
    students[0].AddScore("英语", 88)
    students[0].AddScore("物理", 95)

    students[1].AddScore("数学", 78)
    students[1].AddScore("英语", 82)
    students[1].AddScore("物理", 75)

    students[2].AddScore("数学", 95)
    students[2].AddScore("英语", 91)
    students[2].AddScore("物理", 89)

    // 按平均分排序（降序）
    sort.Slice(students, func(i, j int) bool {
        return students[i].Average() > students[j].Average()
    })

    // 输出成绩单
    fmt.Println("===== 成绩排名 =====")
    for rank, s := range students {
        fmt.Printf("第%d名: %-6s 平均分: %.1f  等级: %s\n",
            rank+1, s.Name, s.Average(), s.Grade())
    }
}
```

```
// 输出:
===== 成绩排名 =====
第1名: Carol  平均分: 91.7  等级: A
第2名: Alice  平均分: 91.7  等级: A
第3名: Bob    平均分: 78.3  等级: C
```

## 5. 本章小结

| 知识点 | 要点 |
|--------|------|
| 函数 | 多返回值、命名返回值、可变参数 `...` |
| 匿名函数 | 闭包捕获外部变量，注意循环中的闭包陷阱 |
| 高阶函数 | 函数作为参数和返回值，函数类型别名 |
| defer | LIFO 顺序执行，参数声明时求值，用于资源释放 |
| panic/recover | 仅用于不可恢复错误，recover 必须在 defer 中 |
| 数组 | 定长、值类型、赋值会复制 |
| 切片 | 引用类型、可变长、`append` 追加、`copy` 复制 |
| Map | 引用类型、无序遍历、用 `val, ok` 判断 key 存在 |
| 字符串 | 不可变字节序列、`range` 遍历字符、`strings` 包 |
| 指针 | `&` 取地址、`*` 解引用、不支持指针运算 |
| 结构体 | 字段、匿名嵌入（组合）、标签（Tag） |
| 方法 | 值接收者 vs 指针接收者、不确定时用指针接收者 |
| 构造模式 | 工厂函数、带验证的构造、Functional Options 模式 |
