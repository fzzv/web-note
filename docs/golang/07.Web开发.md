## 1. net/http 标准库深入

### 1.1 Handler 接口与 HandlerFunc

```go
package main

import (
    "encoding/json"
    "fmt"
    "log"
    "net/http"
)

// http.Handler 接口只有一个方法：
// type Handler interface {
//     ServeHTTP(ResponseWriter, *Request)
// }

// 方式一：实现 Handler 接口
type HelloHandler struct {
    Greeting string
}

func (h *HelloHandler) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "%s, World!", h.Greeting)
}

// 方式二：使用 http.HandlerFunc（函数适配器）
func healthCheck(w http.ResponseWriter, r *http.Request) {
    w.Header().Set("Content-Type", "application/json")
    json.NewEncoder(w).Encode(map[string]string{
        "status": "ok",
    })
}

// 方式三：返回 http.Handler 的工厂函数
func makeGreeter(name string) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintf(w, "Hello, %s!", name)
    })
}

func main() {
    mux := http.NewServeMux()

    mux.Handle("/hello", &HelloHandler{Greeting: "Hi"})
    mux.HandleFunc("/health", healthCheck)
    mux.Handle("/greet", makeGreeter("Gopher"))

    fmt.Println("服务启动: http://localhost:8080")
    log.Fatal(http.ListenAndServe(":8080", mux))
}
```

```
// 访问结果:
// GET /hello  → Hi, World!
// GET /health → {"status":"ok"}
// GET /greet  → Hello, Gopher!
```

### 1.2 Go 1.22+ 增强路由

```go
package main

import (
    "encoding/json"
    "fmt"
    "log"
    "net/http"
)

func JSON(w http.ResponseWriter, status int, data any) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(data)
}

func main() {
    mux := http.NewServeMux()

    // 精确匹配方法 + 路径参数
    mux.HandleFunc("GET /api/users", listUsers)
    mux.HandleFunc("POST /api/users", createUser)
    mux.HandleFunc("GET /api/users/{id}", getUser)
    mux.HandleFunc("PUT /api/users/{id}", updateUser)
    mux.HandleFunc("DELETE /api/users/{id}", deleteUser)

    // 通配符 {path...} 匹配剩余路径
    mux.HandleFunc("GET /files/{path...}", func(w http.ResponseWriter, r *http.Request) {
        path := r.PathValue("path")
        fmt.Fprintf(w, "请求文件: %s", path)
    })

    // 精确匹配 "GET /" 不会匹配 "/other"
    mux.HandleFunc("GET /{$}", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, "首页")
    })

    fmt.Println("服务启动: http://localhost:8080")
    log.Fatal(http.ListenAndServe(":8080", mux))
}

func listUsers(w http.ResponseWriter, r *http.Request) {
    // 查询参数
    page := r.URL.Query().Get("page")
    limit := r.URL.Query().Get("limit")
    JSON(w, 200, map[string]string{"action": "list", "page": page, "limit": limit})
}

func createUser(w http.ResponseWriter, r *http.Request) {
    var input map[string]string
    json.NewDecoder(r.Body).Decode(&input)
    JSON(w, 201, map[string]any{"action": "create", "data": input})
}

func getUser(w http.ResponseWriter, r *http.Request) {
    id := r.PathValue("id") // Go 1.22+ 路径参数
    JSON(w, 200, map[string]string{"action": "get", "id": id})
}

func updateUser(w http.ResponseWriter, r *http.Request) {
    id := r.PathValue("id")
    JSON(w, 200, map[string]string{"action": "update", "id": id})
}

func deleteUser(w http.ResponseWriter, r *http.Request) {
    id := r.PathValue("id")
    JSON(w, 200, map[string]string{"action": "delete", "id": id})
}
```

```

// GET    /api/users/42              → {"action":"get","id":"42"}

// DELETE /api/users/42              → {"action":"delete","id":"42"}
```

### 1.3 中间件链

```go
package main

import (
    "fmt"
    "log"
    "net/http"
    "time"
)

type Middleware func(http.Handler) http.Handler

// 日志中间件
func Logger(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        // 包装 ResponseWriter 以捕获状态码
        wrapped := &statusWriter{ResponseWriter: w, status: 200}
        next.ServeHTTP(wrapped, r)
        log.Printf("[%d] %s %s %v", wrapped.status, r.Method, r.URL.Path, time.Since(start))
    })
}

type statusWriter struct {
    http.ResponseWriter
    status int
}

func (w *statusWriter) WriteHeader(code int) {
    w.status = code
    w.ResponseWriter.WriteHeader(code)
}

// 恢复中间件
func Recovery(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        defer func() {
            if err := recover(); err != nil {
                log.Printf("PANIC: %v", err)
                http.Error(w, "Internal Server Error", 500)
            }
        }()
        next.ServeHTTP(w, r)
    })
}

// CORS 中间件
func CORS(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        w.Header().Set("Access-Control-Allow-Origin", "*")
        w.Header().Set("Access-Control-Allow-Methods", "GET,POST,PUT,DELETE,OPTIONS")
        w.Header().Set("Access-Control-Allow-Headers", "Content-Type,Authorization")
        if r.Method == "OPTIONS" {
            w.WriteHeader(204)
            return
        }
        next.ServeHTTP(w, r)
    })
}

// 链式组装
func Chain(h http.Handler, middlewares ...Middleware) http.Handler {
    for i := len(middlewares) - 1; i >= 0; i-- {
        h = middlewares[i](h)
    }
    return h
}

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("GET /", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, "OK")
    })
    mux.HandleFunc("GET /panic", func(w http.ResponseWriter, r *http.Request) {
        panic("boom!")
    })

    handler := Chain(mux, Recovery, Logger, CORS)

    fmt.Println("服务启动: http://localhost:8080")
    log.Fatal(http.ListenAndServe(":8080", handler))
}
```

```
// 日志输出:
// [200] GET / 15.2µs
// PANIC: boom!
// [500] GET /panic 42.5µs
```

### 1.4 优雅关闭

```go
package main

import (
    "context"
    "fmt"
    "log"
    "net/http"
    "os"
    "os/signal"
    "syscall"
    "time"
)

func main() {
    mux := http.NewServeMux()
    mux.HandleFunc("GET /", func(w http.ResponseWriter, r *http.Request) {
        time.Sleep(2 * time.Second) // 模拟慢请求
        fmt.Fprintln(w, "done")
    })

    server := &http.Server{
        Addr:         ":8080",
        Handler:      mux,
        ReadTimeout:  10 * time.Second,
        WriteTimeout: 15 * time.Second,
        IdleTimeout:  60 * time.Second,
    }

    // 在后台启动服务
    go func() {
        fmt.Println("服务启动: http://localhost:8080")
        if err := server.ListenAndServe(); err != http.ErrServerClosed {
            log.Fatalf("服务异常: %v", err)
        }
    }()

    // 等待中断信号
    quit := make(chan os.Signal, 1)
    signal.Notify(quit, syscall.SIGINT, syscall.SIGTERM)
    sig := <-quit
    fmt.Printf("\n收到信号: %v，开始优雅关闭...\n", sig)

    // 给正在处理的请求 10 秒时间完成
    ctx, cancel := context.WithTimeout(context.Background(), 10*time.Second)
    defer cancel()

    if err := server.Shutdown(ctx); err != nil {
        log.Fatalf("强制关闭: %v", err)
    }
    fmt.Println("服务已安全关闭")
}
```

```
// 输出:
服务启动: http://localhost:8080
^C
收到信号: interrupt，开始优雅关闭...
服务已安全关闭
```

```go
package main

import (
    "net/http"

    "github.com/gin-gonic/gin"
)

func main() {
    r := gin.Default() // 默认包含 Logger 和 Recovery 中间件

    // 基本路由
    r.GET("/", func(c *gin.Context) {
        c.JSON(http.StatusOK, gin.H{
            "message": "Hello, Gin!",
        })
    })

    // 路径参数
    r.GET("/users/:id", func(c *gin.Context) {
        id := c.Param("id")
        c.JSON(200, gin.H{"id": id})
    })

    // 查询参数
    r.GET("/search", func(c *gin.Context) {
        q := c.DefaultQuery("q", "")       // 带默认值
        page := c.DefaultQuery("page", "1")
        c.JSON(200, gin.H{"query": q, "page": page})
    })

    // POST JSON 请求
    r.POST("/users", func(c *gin.Context) {
        var input struct {
            Name  string `json:"name" binding:"required"`
            Email string `json:"email" binding:"required,email"`
        }
        if err := c.ShouldBindJSON(&input); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }
        c.JSON(201, gin.H{"user": input})
    })

    // 表单数据
    r.POST("/login", func(c *gin.Context) {
        username := c.PostForm("username")
        password := c.PostForm("password")
        c.JSON(200, gin.H{"username": username, "password": "***", "len": len(password)})
    })

    r.Run(":8080") // 默认 :8080
}
```

### 2.2 参数绑定与校验

```go
package main

import (
    "net/http"

    "github.com/gin-gonic/gin"
)

// URI 绑定
type UserURI struct {
    ID int `uri:"id" binding:"required,min=1"`
}

// 查询参数绑定
type ListQuery struct {
    Page   int    `form:"page" binding:"min=1"`
    Limit  int    `form:"limit" binding:"min=1,max=100"`
    Sort   string `form:"sort" binding:"omitempty,oneof=asc desc"`
    Search string `form:"search"`
}

// JSON Body 绑定
type CreateUserInput struct {
    Name     string `json:"name" binding:"required,min=2,max=50"`
    Email    string `json:"email" binding:"required,email"`
    Age      int    `json:"age" binding:"required,min=1,max=150"`
    Password string `json:"password" binding:"required,min=6"`
    Role     string `json:"role" binding:"required,oneof=admin user guest"`
}

// 更新可以部分字段（不加 required）
type UpdateUserInput struct {
    Name  *string `json:"name" binding:"omitempty,min=2,max=50"`
    Email *string `json:"email" binding:"omitempty,email"`
    Age   *int    `json:"age" binding:"omitempty,min=1,max=150"`
}

func main() {
    r := gin.Default()

    // URI 参数绑定
    r.GET("/users/:id", func(c *gin.Context) {
        var uri UserURI
        if err := c.ShouldBindUri(&uri); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }
        c.JSON(200, gin.H{"id": uri.ID})
    })

    // 查询参数绑定
    r.GET("/users", func(c *gin.Context) {
        query := ListQuery{Page: 1, Limit: 20} // 默认值
        if err := c.ShouldBindQuery(&query); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }
        c.JSON(200, gin.H{
            "page": query.Page, "limit": query.Limit,
            "sort": query.Sort, "search": query.Search,
        })
    })

    // JSON 绑定 + 验证
    r.POST("/users", func(c *gin.Context) {
        var input CreateUserInput
        if err := c.ShouldBindJSON(&input); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }
        c.JSON(201, gin.H{
            "message": "创建成功",
            "user":    gin.H{"name": input.Name, "email": input.Email, "role": input.Role},
        })
    })

    // PATCH 部分更新
    r.PATCH("/users/:id", func(c *gin.Context) {
        var input UpdateUserInput
        if err := c.ShouldBindJSON(&input); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }
        updated := gin.H{}
        if input.Name != nil {
            updated["name"] = *input.Name
        }
        if input.Email != nil {
            updated["email"] = *input.Email
        }
        if input.Age != nil {
            updated["age"] = *input.Age
        }
        c.JSON(200, gin.H{"updated_fields": updated})
    })

    r.Run(":8080")
}
```

```
// 访问结果:
// GET  /users/0           → {"error":"...min..."}
// GET  /users?page=2&limit=50&sort=desc → {"limit":50,"page":2,"search":"","sort":"desc"}
// POST /users {"name":"A","email":"bad","age":25,"password":"123","role":"admin"}
//   → {"error":"...email...,min..."}
// POST /users {"name":"Alice","email":"a@b.com","age":25,"password":"123456","role":"admin"}
//   → {"message":"创建成功","user":{"email":"a@b.com","name":"Alice","role":"admin"}}
```

**常用 binding 验证标签：**

| 标签 | 说明 | 示例 |
|------|------|------|
| `required` | 必填 | `binding:"required"` |
| `min` / `max` | 最小/最大值（数字）或长度（字符串） | `binding:"min=1,max=100"` |
| `email` | 邮箱格式 | `binding:"email"` |
| `oneof` | 枚举值 | `binding:"oneof=admin user"` |
| `omitempty` | 空值时跳过验证 | `binding:"omitempty,email"` |
| `url` | URL 格式 | `binding:"url"` |
| `len` | 精确长度 | `binding:"len=11"` |

### 2.3 分组路由与中间件

```go
package main

import (
    "net/http"
    "strings"
    "time"

    "github.com/gin-gonic/gin"
)

// 认证中间件
func AuthMiddleware() gin.HandlerFunc {
    return func(c *gin.Context) {
        token := c.GetHeader("Authorization")
        if !strings.HasPrefix(token, "Bearer ") {
            c.AbortWithStatusJSON(401, gin.H{"error": "未授权"})
            return
        }
        // 模拟解析 token
        userID := "user_123"
        c.Set("userID", userID) // 在上下文中存储数据
        c.Next()
    }
}

// 限流中间件（简易版）
func RateLimiter(maxRequests int, window time.Duration) gin.HandlerFunc {
    requests := make(map[string][]time.Time)
    return func(c *gin.Context) {
        ip := c.ClientIP()
        now := time.Now()

        // 清理过期记录
        var valid []time.Time
        for _, t := range requests[ip] {
            if now.Sub(t) < window {
                valid = append(valid, t)
            }
        }
        requests[ip] = valid

        if len(requests[ip]) >= maxRequests {
            c.AbortWithStatusJSON(429, gin.H{"error": "请求过于频繁"})
            return
        }

        requests[ip] = append(requests[ip], now)
        c.Next()
    }
}

func main() {
    r := gin.Default()

    // 公开接口
    r.GET("/ping", func(c *gin.Context) {
        c.JSON(200, gin.H{"message": "pong"})
    })

    // API v1 分组
    v1 := r.Group("/api/v1")
    v1.Use(RateLimiter(100, time.Minute)) // 组级中间件
    {
        // 公开路由
        v1.POST("/login", func(c *gin.Context) {
            c.JSON(200, gin.H{"token": "Bearer abc123"})
        })

        // 需要认证的路由
        auth := v1.Group("")
        auth.Use(AuthMiddleware())
        {
            auth.GET("/profile", func(c *gin.Context) {
                userID, _ := c.Get("userID")
                c.JSON(200, gin.H{"userID": userID})
            })

            auth.GET("/orders", func(c *gin.Context) {
                c.JSON(200, gin.H{"orders": []string{"order1", "order2"}})
            })

            // 管理员路由（嵌套分组）
            admin := auth.Group("/admin")
            {
                admin.GET("/users", func(c *gin.Context) {
                    c.JSON(200, gin.H{"admin": "用户列表"})
                })

                admin.DELETE("/users/:id", func(c *gin.Context) {
                    c.JSON(200, gin.H{"deleted": c.Param("id")})
                })
            }
        }
    }

    r.Run(":8080")
}
```

```
// 访问结果:
// GET  /ping                               → {"message":"pong"}
// POST /api/v1/login                       → {"token":"Bearer abc123"}

// GET  /api/v1/profile  (Authorization: Bearer abc123) → {"userID":"user_123"}
// GET  /api/v1/admin/users                 → {"admin":"用户列表"}
```

### 2.4 统一响应格式

```go
package main

import (
    "net/http"

    "github.com/gin-gonic/gin"
)

// 统一响应结构
type Response struct {
    Code    int    `json:"code"`
    Message string `json:"message"`
    Data    any    `json:"data,omitempty"`
}

func Success(c *gin.Context, data any) {
    c.JSON(http.StatusOK, Response{Code: 0, Message: "success", Data: data})
}

func Created(c *gin.Context, data any) {
    c.JSON(http.StatusCreated, Response{Code: 0, Message: "created", Data: data})
}

func Fail(c *gin.Context, httpCode int, code int, msg string) {
    c.JSON(httpCode, Response{Code: code, Message: msg})
}

// 分页响应
type PageData struct {
    List  any   `json:"list"`
    Total int64 `json:"total"`
    Page  int   `json:"page"`
    Size  int   `json:"size"`
}

func SuccessPage(c *gin.Context, list any, total int64, page, size int) {
    Success(c, PageData{List: list, Total: total, Page: page, Size: size})
}

// 业务错误码
const (
    ErrCodeValidation = 40001
    ErrCodeNotFound   = 40401
    ErrCodeForbidden  = 40301
    ErrCodeInternal   = 50001
)

func main() {
    r := gin.Default()

    r.GET("/api/user", func(c *gin.Context) {
        Success(c, gin.H{"name": "Alice", "age": 25})
    })

    r.GET("/api/users", func(c *gin.Context) {
        users := []gin.H{
            {"id": 1, "name": "Alice"},
            {"id": 2, "name": "Bob"},
        }
        SuccessPage(c, users, 100, 1, 20)
    })

    r.GET("/api/notfound", func(c *gin.Context) {
        Fail(c, 404, ErrCodeNotFound, "资源不存在")
    })

    r.Run(":8080")
}
```

## 3. 数据库操作

### 3.1 database/sql 标准库

```go
package main

import (
    "database/sql"
    "fmt"
    "log"

    _ "github.com/mattn/go-sqlite3" // SQLite 驱动
)

type User struct {
    ID    int
    Name  string
    Email string
    Age   int
}

func main() {
    // 打开数据库连接
    db, err := sql.Open("sqlite3", ":memory:")
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()

    // 配置连接池
    db.SetMaxOpenConns(25)
    db.SetMaxIdleConns(5)

    // 建表
    db.Exec(`CREATE TABLE users (
        id INTEGER PRIMARY KEY AUTOINCREMENT,
        name TEXT NOT NULL,
        email TEXT UNIQUE,
        age INTEGER DEFAULT 0
    )`)

    // ========== 插入 ==========
    result, _ := db.Exec(
        "INSERT INTO users (name, email, age) VALUES (?, ?, ?)",
        "Alice", "alice@test.com", 25,
    )
    id, _ := result.LastInsertId()
    fmt.Println("插入 ID:", id)

    // 批量插入
    tx, _ := db.Begin()
    stmt, _ := tx.Prepare("INSERT INTO users (name, email, age) VALUES (?, ?, ?)")
    defer stmt.Close()

    users := []User{
        {Name: "Bob", Email: "bob@test.com", Age: 30},
        {Name: "Carol", Email: "carol@test.com", Age: 28},
        {Name: "Dave", Email: "dave@test.com", Age: 35},
    }
    for _, u := range users {
        stmt.Exec(u.Name, u.Email, u.Age)
    }
    tx.Commit()
    fmt.Println("批量插入完成")

    // ========== 查询单行 ==========
    var user User
    err = db.QueryRow("SELECT id, name, email, age FROM users WHERE id = ?", 1).
        Scan(&user.ID, &user.Name, &user.Email, &user.Age)
    if err == sql.ErrNoRows {
        fmt.Println("未找到")
    } else {
        fmt.Printf("单行查询: %+v\n", user)
    }

    // ========== 查询多行 ==========
    rows, _ := db.Query("SELECT id, name, email, age FROM users ORDER BY age")
    defer rows.Close()

    fmt.Println("\n所有用户:")
    for rows.Next() {
        var u User
        rows.Scan(&u.ID, &u.Name, &u.Email, &u.Age)
        fmt.Printf("  %d: %-6s %-20s %d岁\n", u.ID, u.Name, u.Email, u.Age)
    }

    // ========== 更新 ==========
    result, _ = db.Exec("UPDATE users SET age = ? WHERE name = ?", 26, "Alice")
    affected, _ := result.RowsAffected()
    fmt.Printf("\n更新 %d 行\n", affected)

    // ========== 删除 ==========
    result, _ = db.Exec("DELETE FROM users WHERE name = ?", "Dave")
    affected, _ = result.RowsAffected()
    fmt.Printf("删除 %d 行\n", affected)

    // 验证
    var count int
    db.QueryRow("SELECT COUNT(*) FROM users").Scan(&count)
    fmt.Println("剩余用户数:", count)
}
```

```
// 输出:
插入 ID: 1
```

### 3.4 GORM 事务

```go
package main

import (
    "errors"
    "fmt"
    "log"

    "gorm.io/driver/sqlite"
    "gorm.io/gorm"
)

type Account struct {
    ID      uint    `gorm:"primaryKey"`
    Name    string  `gorm:"size:50"`
    Balance float64 `gorm:"not null"`
}

func Transfer(db *gorm.DB, fromID, toID uint, amount float64) error {
    // 手动事务
    return db.Transaction(func(tx *gorm.DB) error {
        var from, to Account

        // 查询并加锁（悲观锁模拟，SQLite 自动处理）
        if err := tx.First(&from, fromID).Error; err != nil {
            return fmt.Errorf("发送方不存在: %w", err)
        }
        if err := tx.First(&to, toID).Error; err != nil {
            return fmt.Errorf("接收方不存在: %w", err)
        }

        // 业务校验
        if from.Balance < amount {
            return errors.New("余额不足")
        }

        // 执行转账
        if err := tx.Model(&from).Update("balance", gorm.Expr("balance - ?", amount)).Error; err != nil {
            return err // 返回错误会自动回滚
        }
        if err := tx.Model(&to).Update("balance", gorm.Expr("balance + ?", amount)).Error; err != nil {
            return err
        }

        fmt.Printf("  转账成功: %s -> %s ¥%.2f\n", from.Name, to.Name, amount)
        return nil // 返回 nil 自动提交
    })
}

func main() {
    db, err := gorm.Open(sqlite.Open(":memory:"), &gorm.Config{})
    if err != nil {
        log.Fatal(err)
    }
    db.AutoMigrate(&Account{})

    // 初始数据
    db.Create(&Account{ID: 1, Name: "Alice", Balance: 1000})
    db.Create(&Account{ID: 2, Name: "Bob", Balance: 500})

    printBalances(db)

    // 正常转账
    fmt.Println("\n--- 转账 200 ---")
    if err := Transfer(db, 1, 2, 200); err != nil {
        fmt.Println("转账失败:", err)
    }
    printBalances(db)

    // 余额不足
    fmt.Println("\n--- 转账 5000（余额不足）---")
    if err := Transfer(db, 1, 2, 5000); err != nil {
        fmt.Println("转账失败:", err) // 自动回滚
    }
    printBalances(db) // 余额不变
}

func printBalances(db *gorm.DB) {
    var accounts []Account
    db.Order("id").Find(&accounts)
    for _, a := range accounts {
        fmt.Printf("  %s: ¥%.2f\n", a.Name, a.Balance)
    }
}
```

```go
    rdb := redis.NewClient(&redis.Options{
        Addr:     "localhost:6379",
        Password: "",
        DB:       0,
    })
    defer rdb.Close()

    // 测试连接
    if err := rdb.Ping(ctx).Err(); err != nil {
        fmt.Println("Redis 连接失败:", err)
        return
    }
    fmt.Println("Redis 已连接")

    // ========== 字符串 ==========
    rdb.Set(ctx, "name", "Go", 0)              // 永不过期
    rdb.Set(ctx, "token", "abc123", time.Hour)  // 1小时过期

    val, _ := rdb.Get(ctx, "name").Result()
    fmt.Println("name:", val)

    ttl, _ := rdb.TTL(ctx, "token").Result()
    fmt.Println("token TTL:", ttl)

    // 不存在时才设置（分布式锁常用）
    ok, _ := rdb.SetNX(ctx, "lock:order:1", "holder_1", 10*time.Second).Result()
    fmt.Println("获取锁:", ok)

    // ========== JSON 缓存 ==========
    user := User{ID: 1, Name: "Alice"}
    data, _ := json.Marshal(user)
    rdb.Set(ctx, "user:1", data, 5*time.Minute)

    cached, _ := rdb.Get(ctx, "user:1").Bytes()
    var cachedUser User
    json.Unmarshal(cached, &cachedUser)
    fmt.Printf("缓存用户: %+v\n", cachedUser)

    // ========== Hash ==========
    rdb.HSet(ctx, "user:2", map[string]any{
        "name":  "Bob",
        "email": "bob@test.com",
        "age":   "30",
    })

    name, _ := rdb.HGet(ctx, "user:2", "name").Result()
    fmt.Println("Hash name:", name)

    all, _ := rdb.HGetAll(ctx, "user:2").Result()
    fmt.Println("Hash all:", all)

    // ========== List（队列）==========
    rdb.RPush(ctx, "queue:tasks", "task1", "task2", "task3")
    task, _ := rdb.LPop(ctx, "queue:tasks").Result() // FIFO
    fmt.Println("出队:", task)

    length, _ := rdb.LLen(ctx, "queue:tasks").Result()
    fmt.Println("队列长度:", length)

    // ========== Set ==========
    rdb.SAdd(ctx, "tags:article:1", "Go", "教程", "后端")
    rdb.SAdd(ctx, "tags:article:2", "Go", "并发", "后端")

    // 交集
    common, _ := rdb.SInter(ctx, "tags:article:1", "tags:article:2").Result()
    fmt.Println("共同标签:", common)

    // ========== 自增（计数器/限流）==========
    rdb.Set(ctx, "views:page:1", 0, 0)
    rdb.Incr(ctx, "views:page:1")
    rdb.Incr(ctx, "views:page:1")
    rdb.IncrBy(ctx, "views:page:1", 10)
    views, _ := rdb.Get(ctx, "views:page:1").Int64()
    fmt.Println("页面浏览量:", views)
}
```

```go
package main

import (
    "fmt"
    "net/http"
    "strings"
    "time"

    "github.com/gin-gonic/gin"
    "github.com/golang-jwt/jwt/v5"
)

var jwtSecret = []byte("my-secret-key-change-in-production")

// 自定义 Claims
type Claims struct {
    UserID   uint   `json:"user_id"`
    Username string `json:"username"`
    Role     string `json:"role"`
    jwt.RegisteredClaims
}

// 生成 Token
func GenerateToken(userID uint, username, role string) (string, error) {
    claims := Claims{
        UserID:   userID,
        Username: username,
        Role:     role,
        RegisteredClaims: jwt.RegisteredClaims{
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(24 * time.Hour)),
            IssuedAt:  jwt.NewNumericDate(time.Now()),
            Issuer:    "myapp",
        },
    }

    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    return token.SignedString(jwtSecret)
}

// 解析 Token
func ParseToken(tokenString string) (*Claims, error) {
    token, err := jwt.ParseWithClaims(tokenString, &Claims{},
        func(token *jwt.Token) (any, error) {
            if _, ok := token.Method.(*jwt.SigningMethodHMAC); !ok {
                return nil, fmt.Errorf("unexpected signing method: %v", token.Header["alg"])
            }
            return jwtSecret, nil
        },
    )
    if err != nil {
        return nil, err
    }

    if claims, ok := token.Claims.(*Claims); ok && token.Valid {
        return claims, nil
    }
    return nil, fmt.Errorf("invalid token")
}

// JWT 中间件
func JWTAuth() gin.HandlerFunc {
    return func(c *gin.Context) {
        auth := c.GetHeader("Authorization")
        if !strings.HasPrefix(auth, "Bearer ") {
            c.AbortWithStatusJSON(401, gin.H{"error": "缺少 token"})
            return
        }

        tokenStr := strings.TrimPrefix(auth, "Bearer ")
        claims, err := ParseToken(tokenStr)
        if err != nil {
            c.AbortWithStatusJSON(401, gin.H{"error": "token 无效: " + err.Error()})
            return
        }

        // 将用户信息存入上下文
        c.Set("userID", claims.UserID)
        c.Set("username", claims.Username)
        c.Set("role", claims.Role)
        c.Next()
    }
}

// 角色检查中间件
func RequireRole(roles ...string) gin.HandlerFunc {
    return func(c *gin.Context) {
        role, _ := c.Get("role")
        for _, r := range roles {
            if role == r {
                c.Next()
                return
            }
        }
        c.AbortWithStatusJSON(403, gin.H{"error": "权限不足"})
    }
}

func main() {
    r := gin.Default()

    // 登录（获取 token）
    r.POST("/login", func(c *gin.Context) {
        var input struct {
            Username string `json:"username" binding:"required"`
            Password string `json:"password" binding:"required"`
        }
        if err := c.ShouldBindJSON(&input); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }

        // 模拟验证（实际应查数据库 + bcrypt 验证密码）
        if input.Username == "admin" && input.Password == "123456" {
            token, _ := GenerateToken(1, "admin", "admin")
            c.JSON(200, gin.H{"token": token})
            return
        }
        if input.Username == "user" && input.Password == "123456" {
            token, _ := GenerateToken(2, "user", "user")
            c.JSON(200, gin.H{"token": token})
            return
        }
        c.JSON(401, gin.H{"error": "用户名或密码错误"})
    })

    // 需要认证的路由
    auth := r.Group("/api")
    auth.Use(JWTAuth())
    {
        auth.GET("/profile", func(c *gin.Context) {
            c.JSON(200, gin.H{
                "user_id":  c.GetUint("userID"),
                "username": c.GetString("username"),
                "role":     c.GetString("role"),
            })
        })

        // 需要管理员权限
        admin := auth.Group("/admin")
        admin.Use(RequireRole("admin"))
        {
            admin.GET("/dashboard", func(c *gin.Context) {
                c.JSON(200, gin.H{"message": "管理员面板"})
            })
        }
    }

    r.Run(":8080")
}
```

```go
package main

import (
    "fmt"

    "golang.org/x/crypto/bcrypt"
)

// 哈希密码
func HashPassword(password string) (string, error) {
    bytes, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
    return string(bytes), err
}

// 验证密码
func CheckPassword(password, hash string) bool {
    err := bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))
    return err == nil
}

func main() {
    password := "mySecretPassword123"

    // 哈希
    hash, err := HashPassword(password)
    if err != nil {
        fmt.Println("哈希失败:", err)
        return
    }
    fmt.Println("原始密码:", password)
    fmt.Println("哈希结果:", hash)
    fmt.Println("哈希长度:", len(hash))

    // 验证
    fmt.Println("\n正确密码验证:", CheckPassword(password, hash))
    fmt.Println("错误密码验证:", CheckPassword("wrongPassword", hash))

    // 每次哈希结果不同（内含随机盐）
    hash2, _ := HashPassword(password)
    fmt.Println("\n第二次哈希:", hash2)
    fmt.Println("两次哈希相同:", hash == hash2)       // false
    fmt.Println("两次都能验证:", CheckPassword(password, hash2)) // true
}
```

```go
type UserServer struct {
    pb.UnimplementedUserServiceServer
    mu     sync.RWMutex
    users  map[int32]*pb.User
    nextID int32
}

func NewUserServer() *UserServer {
    return &UserServer{users: make(map[int32]*pb.User), nextID: 1}
}

func (s *UserServer) CreateUser(ctx context.Context, req *pb.CreateUserRequest) (*pb.User, error) {
    s.mu.Lock()
    defer s.mu.Unlock()

    if req.Name == "" {
        return nil, status.Error(codes.InvalidArgument, "name 不能为空")
    }

    user := &pb.User{
        Id:    s.nextID,
        Name:  req.Name,
        Email: req.Email,
        Age:   req.Age,
    }
    s.users[s.nextID] = user
    s.nextID++
    return user, nil
}

func (s *UserServer) GetUser(ctx context.Context, req *pb.GetUserRequest) (*pb.User, error) {
    s.mu.RLock()
    defer s.mu.RUnlock()

    user, ok := s.users[req.Id]
    if !ok {
        return nil, status.Errorf(codes.NotFound, "用户 %d 不存在", req.Id)
    }
    return user, nil
}

func (s *UserServer) ListUsers(ctx context.Context, req *pb.ListUsersRequest) (*pb.ListUsersResponse, error) {
    s.mu.RLock()
    defer s.mu.RUnlock()

    var users []*pb.User
    for _, u := range s.users {
        users = append(users, u)
    }

    return &pb.ListUsersResponse{
        Users: users,
        Total: int32(len(users)),
    }, nil
}

// 服务端流：实时推送新用户
func (s *UserServer) WatchUsers(req *pb.Empty, stream pb.UserService_WatchUsersServer) error {
    // 模拟：每秒推送一次当前用户列表
    for i := 0; i < 5; i++ {
        s.mu.RLock()
        for _, u := range s.users {
            if err := stream.Send(u); err != nil {
                s.mu.RUnlock()
                return err
            }
        }
        s.mu.RUnlock()
        time.Sleep(time.Second)
    }
    return nil
}

func main() {
    lis, err := net.Listen("tcp", ":50051")
    if err != nil {
        log.Fatalf("监听失败: %v", err)
    }

    server := grpc.NewServer()
    pb.RegisterUserServiceServer(server, NewUserServer())

    fmt.Println("gRPC 服务启动: :50051")
    log.Fatal(server.Serve(lis))
}
```

```go
package main

import (
    "context"

    "log"

    "google.golang.org/grpc/credentials/insecure"

    conn, err := grpc.NewClient("localhost:50051",
        grpc.WithTransportCredentials(insecure.NewCredentials()),
    )
    if err != nil {
        log.Fatalf("连接失败: %v", err)

    defer conn.Close()

    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)

        Name: "Bob", Email: "bob@test.com", Age: 30,
    })

    // 查询单个
    found, _ := client.GetUser(ctx, &pb.GetUserRequest{Id: 1})
    fmt.Printf("查询用户: %+v\n", found)

    // 查询列表
    list, _ := client.ListUsers(ctx, &pb.ListUsersRequest{Page: 1, Limit: 10})

    }

    // 服务端流
    fmt.Println("\n监听用户流:")
    stream, _ := client.WatchUsers(ctx, &pb.Empty{})
    for {

        if err == io.EOF {
            break
        }
        if err != nil {
            log.Fatalf("流错误: %v", err)
```

```go
package main

    "time"

// 一元拦截器：日志
func LoggingUnaryInterceptor(
    ctx context.Context,
    req any,
    info *grpc.UnaryServerInfo,
    handler grpc.UnaryHandler,
) (any, error) {
    start := time.Now()
    resp, err := handler(ctx, req)
    log.Printf("[gRPC] %s %v err=%v", info.FullMethod, time.Since(start), err)
    return resp, err
}

// 一元拦截器：认证
func AuthUnaryInterceptor(
    ctx context.Context,
    req any,
    info *grpc.UnaryServerInfo,
    handler grpc.UnaryHandler,
) (any, error) {
    // 从 metadata 获取 token 并验证
    // md, ok := metadata.FromIncomingContext(ctx)

    return handler(ctx, req)
}

// 使用拦截器创建 server
func newServer() *grpc.Server {
    return grpc.NewServer(
        grpc.ChainUnaryInterceptor(
            LoggingUnaryInterceptor,

        ),
    )
```

**HTTP vs gRPC 对比：**

| 特性 | HTTP/REST | gRPC |
|------|-----------|------|
| 协议 | HTTP/1.1 或 HTTP/2 | HTTP/2 |
| 序列化 | JSON（文本） | Protobuf（二进制） |
| 性能 | 较慢 | 快 2-10 倍 |
| 流式通信 | 需 WebSocket | 原生支持 |
| 代码生成 | 手动编写 | 自动生成客户端/服务端 |
| 浏览器支持 | 原生支持 | 需要 gRPC-Web |
| 适用场景 | 公开 API、前端对接 | 微服务内部通信 |

## 8. 综合练习：完整 REST API

```go
package main

import (
    "fmt"
    "net/http"
    "strconv"
    "sync"
    "sync/atomic"
    "time"

    "github.com/gin-gonic/gin"
)

// ==================== Model ====================

type Todo struct {
    ID        int       `json:"id"`
    Title     string    `json:"title"`
    Completed bool      `json:"completed"`
    CreatedAt time.Time `json:"created_at"`
}

type TodoStore struct {
    mu     sync.RWMutex
    items  map[int]*Todo
    nextID atomic.Int32
}

func NewTodoStore() *TodoStore {
    s := &TodoStore{items: make(map[int]*Todo)}
    s.nextID.Store(1)
    return s
}

func (s *TodoStore) Create(title string) *Todo {
    id := int(s.nextID.Add(1) - 1)
    todo := &Todo{ID: id, Title: title, CreatedAt: time.Now()}
    s.mu.Lock()
    s.items[id] = todo
    s.mu.Unlock()
    return todo
}

func (s *TodoStore) GetAll(completed *bool) []*Todo {
    s.mu.RLock()
    defer s.mu.RUnlock()
    var list []*Todo
    for _, t := range s.items {
        if completed != nil && t.Completed != *completed {
            continue
        }
        list = append(list, t)
    }
    return list
}

func (s *TodoStore) GetByID(id int) (*Todo, bool) {
    s.mu.RLock()
    defer s.mu.RUnlock()
    t, ok := s.items[id]
    return t, ok
}

func (s *TodoStore) Update(id int, title *string, completed *bool) (*Todo, bool) {
    s.mu.Lock()
    defer s.mu.Unlock()
    t, ok := s.items[id]
    if !ok {
        return nil, false
    }
    if title != nil {
        t.Title = *title
    }
    if completed != nil {
        t.Completed = *completed
    }
    return t, true
}

func (s *TodoStore) Delete(id int) bool {
    s.mu.Lock()
    defer s.mu.Unlock()
    if _, ok := s.items[id]; !ok {
        return false
    }
    delete(s.items, id)
    return true
}

// ==================== Handler ====================

type TodoHandler struct {
    store *TodoStore
}

func (h *TodoHandler) List(c *gin.Context) {
    var filter *bool
    if q := c.Query("completed"); q != "" {
        val := q == "true"
        filter = &val
    }
    todos := h.store.GetAll(filter)
    c.JSON(200, gin.H{"data": todos, "total": len(todos)})
}

func (h *TodoHandler) Get(c *gin.Context) {
    id, err := strconv.Atoi(c.Param("id"))
    if err != nil {
        c.JSON(400, gin.H{"error": "无效的 ID"})
        return
    }
    todo, ok := h.store.GetByID(id)
    if !ok {
        c.JSON(404, gin.H{"error": "未找到"})
        return
    }
    c.JSON(200, gin.H{"data": todo})
}

func (h *TodoHandler) Create(c *gin.Context) {
    var input struct {
        Title string `json:"title" binding:"required,min=1,max=200"`
    }
    if err := c.ShouldBindJSON(&input); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }
    todo := h.store.Create(input.Title)
    c.JSON(201, gin.H{"data": todo})
}

func (h *TodoHandler) Update(c *gin.Context) {
    id, _ := strconv.Atoi(c.Param("id"))
    var input struct {
        Title     *string `json:"title" binding:"omitempty,min=1,max=200"`
        Completed *bool   `json:"completed"`
    }
    if err := c.ShouldBindJSON(&input); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }
    todo, ok := h.store.Update(id, input.Title, input.Completed)
    if !ok {
        c.JSON(404, gin.H{"error": "未找到"})
        return
    }
    c.JSON(200, gin.H{"data": todo})
}

func (h *TodoHandler) Delete(c *gin.Context) {
    id, _ := strconv.Atoi(c.Param("id"))
    if !h.store.Delete(id) {
        c.JSON(404, gin.H{"error": "未找到"})
        return
    }
    c.JSON(200, gin.H{"message": "已删除"})
}

// ==================== Router ====================

func setupRouter(store *TodoStore) *gin.Engine {
    r := gin.Default()
    h := &TodoHandler{store: store}

    r.GET("/api/todos", h.List)
    r.POST("/api/todos", h.Create)
    r.GET("/api/todos/:id", h.Get)
    r.PATCH("/api/todos/:id", h.Update)
    r.DELETE("/api/todos/:id", h.Delete)

    return r
}

func main() {
    store := NewTodoStore()
    r := setupRouter(store)

    fmt.Println("Todo API: http://localhost:8080")
    fmt.Println("  GET    /api/todos              列表（?completed=true 过滤）")
    fmt.Println("  POST   /api/todos              创建 {\"title\":\"...\"}")
    fmt.Println("  GET    /api/todos/:id           详情")
    fmt.Println("  PATCH  /api/todos/:id           更新 {\"title\":\"..\",\"completed\":true}")
    fmt.Println("  DELETE /api/todos/:id           删除")
    r.Run(":8080")
}
```

## 9. 本章小结

| 知识点 | 要点 |
|--------|------|
| Handler 接口 | `ServeHTTP(w, r)` 唯一方法，`HandlerFunc` 函数适配器 |
| Go 1.22 路由 | `"GET /users/{id}"` 方法+路径参数，`{path...}` 通配符 |
| 中间件 | `func(http.Handler) http.Handler` 模式，链式组装 |
| 优雅关闭 | `server.Shutdown(ctx)` 等待请求完成后关闭 |
| Gin 路由 | `r.GET/POST`、`:id` 路径参数、分组路由 `r.Group` |
| Gin 绑定 | `ShouldBindJSON`/`ShouldBindQuery`/`ShouldBindUri` |
| 验证标签 | `required`、`min/max`、`email`、`oneof`、`omitempty` |
| 统一响应 | 封装 `Success`/`Fail` 函数，业务错误码 |
| database/sql | `Query`/`QueryRow`/`Exec`，连接池配置，预编译语句 |
| GORM | `AutoMigrate`、CRUD、`Preload` 预加载、软删除 |
| GORM 关联 | 一对多、多对多、`Preload` 嵌套预加载 |
| GORM 事务 | `db.Transaction(func(tx) error)` 自动提交/回滚 |
| Redis | 字符串/Hash/List/Set、`SetNX` 分布式锁、`Incr` 计数器 |
| JWT | `GenerateToken`/`ParseToken`、中间件验证、角色控制 |
| bcrypt | `GenerateFromPassword` 哈希、`CompareHashAndPassword` 验证 |
| gRPC | Protobuf 定义、一元 RPC、服务端流、拦截器 |
| HTTP vs gRPC | REST 适合公开 API，gRPC 适合微服务内部通信 |
