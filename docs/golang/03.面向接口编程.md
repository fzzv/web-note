# Go 语言面向接口编程

## 1. 接口基础

### 1.1 接口的定义与实现

Go 的接口是**隐式实现**的：只要类型实现了接口定义的所有方法，就自动满足该接口，无需显式声明。

```go
package main

import "fmt"

// 定义接口
type Speaker interface {
    Speak() string
}

// Dog 实现了 Speaker 接口（隐式）
type Dog struct {
    Name string
}

func (d Dog) Speak() string {
    return d.Name + ": 汪汪汪!"
}

// Cat 也实现了 Speaker 接口
type Cat struct {
    Name string
}

func (c Cat) Speak() string {
    return c.Name + ": 喵喵喵~"
}

// Robot 也实现了 Speaker 接口
type Robot struct {
    Model string
}

func (r Robot) Speak() string {
    return r.Model + ": 嘟嘟嘟..."
}

// 面向接口编程：函数接收接口类型
func MakeNoise(s Speaker) {
    fmt.Println(s.Speak())
}

func main() {
    dog := Dog{Name: "旺财"}
    cat := Cat{Name: "咪咪"}
    robot := Robot{Model: "R2D2"}

    // 不同类型都可以传给接口参数
    MakeNoise(dog)
    MakeNoise(cat)
    MakeNoise(robot)

    // 接口类型的切片：存放不同类型
    speakers := []Speaker{dog, cat, robot}
    fmt.Println("\n所有发声者:")
    for _, s := range speakers {
        fmt.Println(" ", s.Speak())
    }
}
```

```
// 输出:
旺财: 汪汪汪!
咪咪: 喵喵喵~
R2D2: 嘟嘟嘟...

所有发声者:
  旺财: 汪汪汪!
  咪咪: 喵喵喵~
  R2D2: 嘟嘟嘟...
```

### 1.2 多方法接口

```go
package main

import (
    "fmt"
    "math"
)

// 接口可以包含多个方法
type Shape interface {
    Area() float64
    Perimeter() float64
}

type Circle struct {
    Radius float64
}

func (c Circle) Area() float64 {
    return math.Pi * c.Radius * c.Radius
}

func (c Circle) Perimeter() float64 {
    return 2 * math.Pi * c.Radius
}

type Rectangle struct {
    Width, Height float64
}

func (r Rectangle) Area() float64 {
    return r.Width * r.Height
}

func (r Rectangle) Perimeter() float64 {
    return 2 * (r.Width + r.Height)
}

// 接收接口类型的函数
func PrintShapeInfo(name string, s Shape) {
    fmt.Printf("%s -> 面积: %.2f, 周长: %.2f\n", name, s.Area(), s.Perimeter())
}

// 计算所有图形的总面积
func TotalArea(shapes []Shape) float64 {
    total := 0.0
    for _, s := range shapes {
        total += s.Area()
    }
    return total
}

func main() {
    c := Circle{Radius: 5}
    r := Rectangle{Width: 4, Height: 6}

    PrintShapeInfo("圆形", c)
    PrintShapeInfo("矩形", r)

    shapes := []Shape{c, r, Circle{Radius: 3}}
    fmt.Printf("总面积: %.2f\n", TotalArea(shapes))
}
```

```
// 输出:
圆形 -> 面积: 78.54, 周长: 31.42
矩形 -> 面积: 24.00, 周长: 20.00
总面积: 130.81
```

### 1.3 接口组合（嵌入接口）

```go
package main

import "fmt"

// 小接口
type Reader interface {
    Read(p []byte) (n int, err error)
}

type Writer interface {
    Write(p []byte) (n int, err error)
}

type Closer interface {
    Close() error
}

// 接口组合：嵌入多个小接口
type ReadWriter interface {
    Reader
    Writer
}

type ReadWriteCloser interface {
    Reader
    Writer
    Closer
}

// 实现所有方法的类型
type MyFile struct {
    Name string
}

func (f *MyFile) Read(p []byte) (int, error) {
    fmt.Printf("[%s] 读取数据\n", f.Name)
    return len(p), nil
}

func (f *MyFile) Write(p []byte) (int, error) {
    fmt.Printf("[%s] 写入: %s\n", f.Name, string(p))
    return len(p), nil
}

func (f *MyFile) Close() error {
    fmt.Printf("[%s] 已关闭\n", f.Name)
    return nil
}

// 接受不同级别接口的函数
func ReadData(r Reader) {
    buf := make([]byte, 10)
    r.Read(buf)
}

func WriteData(w Writer, data string) {
    w.Write([]byte(data))
}

func ProcessFile(rwc ReadWriteCloser) {
    buf := make([]byte, 10)
    rwc.Read(buf)
    rwc.Write([]byte("hello"))
    rwc.Close()
}

func main() {
    file := &MyFile{Name: "data.txt"}

    // MyFile 同时满足 Reader、Writer、Closer、ReadWriter、ReadWriteCloser
    ReadData(file)
    WriteData(file, "Go 语言")

    fmt.Println("---")
    ProcessFile(file)
}
```

```
// 输出:
[data.txt] 读取数据
[data.txt] 写入: Go 语言
---
[data.txt] 读取数据
[data.txt] 写入: hello
data.txt] 已关闭
```

**接口设计原则：**

- **小接口优于大接口**：Go 标准库大量使用 1-2 个方法的小接口
- **面向接口编程**：函数参数尽量使用接口类型，而非具体类型
- **接口由使用方定义**：定义在使用接口的包中，而非实现方

## 2. 空接口与类型断言

### 2.1 空接口 interface{} 与 any

空接口没有任何方法，所有类型都隐式实现了空接口，因此可以接收**任意类型**的值。

```go
package main

import "fmt"

func main() {
    // 空接口可以存放任何值
    var anything interface{}

    anything = 42
    fmt.Println(anything) // 42

    anything = "hello"
    fmt.Println(anything) // hello

    anything = true
    fmt.Println(anything) // true

    anything = []int{1, 2, 3}
    fmt.Println(anything) // [1 2 3]

    fmt.Println()

    // any 是 interface{} 的别名（Go 1.18+）
    var val any = "Go 1.18+"
    fmt.Println(val)

    // 空接口切片：存放混合类型
    mixed := []any{42, "hello", true, 3.14, nil}
    for i, v := range mixed {
        fmt.Printf("  [%d] 值: %-8v 类型: %T\n", i, v, v)
    }

    fmt.Println()

    // 空接口作为函数参数
    PrintAny("字符串")
    PrintAny(100)
    PrintAny([]string{"a", "b", "c"})
}

// 接收任意类型参数
func PrintAny(v any) {
    fmt.Printf("值: %v, 类型: %T\n", v, v)
}
```

```
// 输出:
42
hello
true
[1 2 3]

Go 1.18+
  [0] 值: 42       类型: int
  [1] 值: hello    类型: string
  [2] 值: true     类型: bool
  [3] 值: 3.14     类型: float64
  [4] 值: <nil>    类型: <nil>

值: 字符串, 类型: string
值: 100, 类型: int
值: [a b c], 类型: []string
```

### 2.2 类型断言

类型断言用于从接口值中提取其底层具体类型。

```go
package main

import "fmt"

func main() {
    var val any = "hello"

    // 基本类型断言
    str := val.(string)
    fmt.Println("断言成功:", str) // hello

    // 如果断言错误，会触发 panic
    // num := val.(int) // panic: interface conversion

    // 安全的类型断言（带 ok 检查）
    str2, ok := val.(string)
    fmt.Println("string?", ok, str2) // true hello

    num, ok := val.(int)
    fmt.Println("int?", ok, num) // false 0

    fmt.Println()

    // 实际应用：处理不同类型的数据
    data := []any{42, "Go", 3.14, true, nil, []int{1, 2}}
    for _, item := range data {
        processItem(item)
    }
}

func processItem(v any) {
    if v == nil {
        fmt.Println("  nil 值")
        return
    }
    if str, ok := v.(string); ok {
        fmt.Printf("  字符串: %q (长度 %d)\n", str, len(str))
        return
    }
    if num, ok := v.(int); ok {
        fmt.Printf("  整数: %d (是否为偶数: %v)\n", num, num%2 == 0)
        return
    }
    fmt.Printf("  其他类型: %T = %v\n", v, v)
}
```

```
// 输出:
断言成功: hello
string? true hello
int? false 0

  整数: 42 (是否为偶数: true)
  字符串: "Go" (长度 2)
  其他类型: float64 = 3.14
  其他类型: bool = true
  nil 值
  其他类型: []int = [1 2]
```

### 2.3 类型选择（Type Switch）

`type switch` 是处理多种类型的优雅方式。

```go
package main

import "fmt"

func describe(val any) string {
    switch v := val.(type) {
    case nil:
        return "空值"
    case int:
        return fmt.Sprintf("整数 %d", v)
    case float64:
        return fmt.Sprintf("浮点数 %.2f", v)
    case string:
        return fmt.Sprintf("字符串 %q (长度 %d)", v, len(v))
    case bool:
        if v {
            return "布尔值: 真"
        }
        return "布尔值: 假"
    case []int:
        return fmt.Sprintf("整数切片 %v (长度 %d)", v, len(v))
    case []string:
        return fmt.Sprintf("字符串切片 %v", v)
    default:
        return fmt.Sprintf("未知类型 %T", v)
    }
}

// 多类型匹配
func isNumeric(val any) bool {
    switch val.(type) {
    case int, int8, int16, int32, int64,
        uint, uint8, uint16, uint32, uint64,
        float32, float64:
        return true
    default:
        return false
    }
}

func main() {
    values := []any{
        42, 3.14, "hello", true, nil,
        []int{1, 2, 3}, []string{"a", "b"},
        map[string]int{"x": 1},
    }

    for _, v := range values {
        fmt.Println(describe(v))
    }

    fmt.Println()

    // 数值类型判断
    fmt.Println("42 是数值?", isNumeric(42))         // true
    fmt.Println("3.14 是数值?", isNumeric(3.14))     // true
    fmt.Println("hello 是数值?", isNumeric("hello")) // false
}
```

```
// 输出:
整数 42
浮点数 3.14
字符串 "hello" (长度 5)
布尔值: 真
空值
整数切片 [1 2 3] (长度 3)
字符串切片 [a b]
未知类型 map[string]int

42 是数值? true
3.14 是数值? true
hello 是数值? false
```

### 3.2 sort.Interface 接口

实现 `Len()`、`Less()`、`Swap()` 三个方法，即可使用 `sort.Sort` 排序。

```go
package main

import (
    "fmt"
    "sort"
)

type Student struct {
    Name  string
    Score int
}

// 按分数排序的类型
type ByScore []Student

func (s ByScore) Len() int           { return len(s) }
func (s ByScore) Less(i, j int) bool { return s[i].Score > s[j].Score } // 降序
func (s ByScore) Swap(i, j int)      { s[i], s[j] = s[j], s[i] }

// 按姓名排序的类型
type ByName []Student

func (s ByName) Len() int           { return len(s) }
func (s ByName) Less(i, j int) bool { return s[i].Name < s[j].Name } // 升序
func (s ByName) Swap(i, j int)      { s[i], s[j] = s[j], s[i] }

func main() {
    students := []Student{
        {"Carol", 92},
        {"Alice", 88},
        {"Bob", 95},
        {"Dave", 78},
    }

    // 按分数降序排列
    sort.Sort(ByScore(students))
    fmt.Println("按分数排序:")
    for _, s := range students {
        fmt.Printf("  %-6s %d\n", s.Name, s.Score)
    }

    // 按姓名升序排列
    sort.Sort(ByName(students))
    fmt.Println("按姓名排序:")
    for _, s := range students {
        fmt.Printf("  %-6s %d\n", s.Name, s.Score)
    }

    // 更简洁的方式：sort.Slice（推荐）
    sort.Slice(students, func(i, j int) bool {
        return students[i].Score > students[j].Score
    })
    fmt.Println("sort.Slice 按分数:")
    for _, s := range students {
        fmt.Printf("  %-6s %d\n", s.Name, s.Score)
    }
}
```

```
// 输出:
按分数排序:
  Bob    95
  Carol  92
  Alice  88
  Dave   78
按姓名排序:
  Alice  88
  Bob    95
  Carol  92
  Dave   78
sort.Slice 按分数:
  Bob    95
  Carol  92
  Alice  88
  Dave   78
```

### 3.3 io.Reader 与 io.Writer 接口

这两个接口是 Go I/O 体系的核心。

```go
package main

import (
    "bytes"
    "fmt"
    "io"
    "strings"
)

func main() {
    // ========== io.Reader ==========
    // 接口定义：Read(p []byte) (n int, err error)

    // strings.Reader 实现了 io.Reader
    reader := strings.NewReader("Hello, Go!")
    buf := make([]byte, 5)

    fmt.Println("逐块读取:")
    for {
        n, err := reader.Read(buf)
        if n > 0 {
            fmt.Printf("  读取 %d 字节: %q\n", n, buf[:n])
        }
        if err == io.EOF {
            break
        }
    }

    fmt.Println()

    // ========== io.Writer ==========
    // 接口定义：Write(p []byte) (n int, err error)

    // bytes.Buffer 同时实现了 Reader 和 Writer
    var buffer bytes.Buffer

    buffer.WriteString("Hello")
    buffer.WriteString(", ")
    buffer.WriteString("World!")
    fmt.Println("Buffer:", buffer.String())

    fmt.Println()

    // ========== io.Copy：连接 Reader 和 Writer ==========
    src := strings.NewReader("通过 io.Copy 传输的数据")
    var dst bytes.Buffer
    n, _ := io.Copy(&dst, src)
    fmt.Printf("复制了 %d 字节: %s\n", n, dst.String())

    fmt.Println()

    // ========== 自定义 Writer ==========
    counter := &CountWriter{}
    fmt.Fprint(counter, "Hello, ")
    fmt.Fprint(counter, "World!")
    fmt.Printf("总共写入 %d 字节\n", counter.Count)
}

// 自定义 Writer：统计写入字节数
type CountWriter struct {
    Count int
}

func (cw *CountWriter) Write(p []byte) (int, error) {
    n := len(p)
    cw.Count += n
    return n, nil
}
```

```
// 输出:
逐块读取:
  读取 5 字节: "Hello"
  读取 5 字节: ", Go!"

Buffer: Hello, World!

复制了 33 字节: 通过 io.Copy 传输的数据

总共写入 13 字节
```

## 4. 错误处理

### 4.1 error 接口

`error` 是 Go 内置的接口，只有一个方法：

```go
type error interface {
    Error() string
}
```

```go
package main

import (
    "errors"
    "fmt"
    "math"
)

// 函数返回 error
func sqrt(x float64) (float64, error) {
    if x < 0 {
        return 0, fmt.Errorf("不能对负数 %.2f 求平方根", x)
    }
    return math.Sqrt(x), nil
}

// 使用 errors.New 创建简单错误
func divide(a, b float64) (float64, error) {
    if b == 0 {
        return 0, errors.New("除数不能为零")
    }
    return a / b, nil
}

func main() {
    // 正常调用
    result, err := sqrt(16)
    if err != nil {
        fmt.Println("错误:", err)
    } else {
        fmt.Printf("sqrt(16) = %.2f\n", result)
    }

    // 错误调用
    result, err = sqrt(-4)
    if err != nil {
        fmt.Println("错误:", err)
    }

    // 除法
    result, err = divide(10, 0)
    if err != nil {
        fmt.Println("错误:", err)
    }

    result, err = divide(10, 3)
    if err != nil {
        fmt.Println("错误:", err)
    } else {
        fmt.Printf("10 / 3 = %.4f\n", result)
    }
}
```

```
// 输出:
sqrt(16) = 4.00
错误: 不能对负数 -4.00 求平方根
错误: 除数不能为零
10 / 3 = 3.3333
```

### 4.2 自定义错误类型

```go
package main

import (
    "fmt"
    "time"
)

// 自定义错误类型
type ValidationError struct {
    Field   string
    Message string
}

func (e *ValidationError) Error() string {
    return fmt.Sprintf("验证失败 [%s]: %s", e.Field, e.Message)
}

// 带更多上下文的错误
type APIError struct {
    Code      int
    Message   string
    Timestamp time.Time
}

func (e *APIError) Error() string {
    return fmt.Sprintf("[%d] %s (at %s)",
        e.Code, e.Message, e.Timestamp.Format("15:04:05"))
}

// 模拟用户注册
func registerUser(name, email string) error {
    if name == "" {
        return &ValidationError{Field: "name", Message: "不能为空"}
    }
    if len(name) < 2 {
        return &ValidationError{Field: "name", Message: "至少2个字符"}
    }
    if email == "" {
        return &ValidationError{Field: "email", Message: "不能为空"}
    }
    return nil
}

// 模拟 API 调用
func callAPI(url string) error {
    if url == "" {
        return &APIError{
            Code:      400,
            Message:   "URL 不能为空",
            Timestamp: time.Now(),
        }
    }
    return nil
}

func main() {
    // 注册验证
    err := registerUser("", "test@example.com")
    if err != nil {
        fmt.Println(err)

        // 类型断言获取详细信息
        if ve, ok := err.(*ValidationError); ok {
            fmt.Printf("  字段: %s, 原因: %s\n", ve.Field, ve.Message)
        }
    }

    err = registerUser("A", "test@example.com")
    if err != nil {
        fmt.Println(err)
    }

    fmt.Println()

    // API 错误
    err = callAPI("")
    if err != nil {
        fmt.Println(err)
        if ae, ok := err.(*APIError); ok {
            fmt.Printf("  错误码: %d\n", ae.Code)
        }
    }
}
```

```
  字段: name, 原因: 不能为空
验证失败 [name]: 至少2个字符

[400] URL 不能为空 (at 10:30:15)
  错误码: 400
```

### 4.3 错误包装与解包（Go 1.13+）

```go
package main

import (
    "errors"
    "fmt"
    "os"
)

// 哨兵错误（Sentinel Error）
var (
    ErrNotFound     = errors.New("资源未找到")
    ErrUnauthorized = errors.New("未授权")
    ErrForbidden    = errors.New("禁止访问")
)

// 自定义错误类型
type QueryError struct {
    Query   string
    Message string
}

func (e *QueryError) Error() string {
    return fmt.Sprintf("查询错误 [%s]: %s", e.Query, e.Message)
}

// 模拟数据库查询
func findUser(id int) error {
    if id <= 0 {
        return fmt.Errorf("findUser: %w", ErrNotFound) // 包装错误
    }
    return nil
}

// 多层包装
func getUserProfile(id int) error {
    err := findUser(id)
    if err != nil {
        return fmt.Errorf("getUserProfile(id=%d): %w", id, err)
    }
    return nil
}

func getPageData(userID int) error {
    err := getUserProfile(userID)
    if err != nil {
        return fmt.Errorf("getPageData: %w", err)
    }
    return nil
}

func main() {
    // ========== errors.Is：判断错误链中是否包含特定错误 ==========
    err := getPageData(0)
    fmt.Println("错误信息:", err)
    fmt.Println("是 ErrNotFound?", errors.Is(err, ErrNotFound))       // true
    fmt.Println("是 ErrUnauthorized?", errors.Is(err, ErrUnauthorized)) // false

    fmt.Println()

    // ========== errors.As：提取错误链中特定类型的错误 ==========
    err2 := processQuery("SELECT * FROM users")
    fmt.Println("错误信息:", err2)

    var qe *QueryError
    if errors.As(err2, &qe) {
        fmt.Printf("查询语句: %s\n", qe.Query)
        fmt.Printf("错误描述: %s\n", qe.Message)
    }

    fmt.Println()

    // ========== 实际场景：文件操作错误 ==========
    err3 := readConfig("/nonexistent/config.yaml")
    fmt.Println("错误:", err3)

    // 检查是否是"文件不存在"错误
    if errors.Is(err3, os.ErrNotExist) {
        fmt.Println("配置文件不存在，使用默认配置")
    }
}

func processQuery(sql string) error {
    err := &QueryError{Query: sql, Message: "表不存在"}
    return fmt.Errorf("processQuery: %w", err)
}

func readConfig(path string) error {
    _, err := os.Open(path)
    if err != nil {
        return fmt.Errorf("读取配置文件 %s: %w", path, err)
    }
    return nil
}
```

**错误处理三件套：**

| 函数 | 用途 | 示例 |
|------|------|------|
| `fmt.Errorf("%w", err)` | 包装错误，保留错误链 | `fmt.Errorf("open db: %w", err)` |
| `errors.Is(err, target)` | 判断错误链中是否包含目标错误 | `errors.Is(err, os.ErrNotExist)` |
| `errors.As(err, &target)` | 从错误链中提取特定类型的错误 | `errors.As(err, &myErr)` |

### 4.4 错误处理最佳实践

```go
package main

import (
    "errors"
    "fmt"
    "log"
)

var ErrInsufficientBalance = errors.New("余额不足")

type Account struct {
    Name    string
    Balance float64
}

func (a *Account) Withdraw(amount float64) error {
    // 实践1：在最底层产生错误
    if amount <= 0 {
        return fmt.Errorf("取款金额必须大于0, 当前: %.2f", amount)
    }
    if amount > a.Balance {
        return fmt.Errorf("账户 %s: %w (需要 %.2f, 余额 %.2f)",
            a.Name, ErrInsufficientBalance, amount, a.Balance)
    }
    a.Balance -= amount
    return nil
}

func Transfer(from, to *Account, amount float64) error {
    // 实践2：中间层包装并传递错误
    if err := from.Withdraw(amount); err != nil {
        return fmt.Errorf("转账失败 (%s -> %s): %w", from.Name, to.Name, err)
    }
    to.Balance += amount
    return nil
}

func main() {
    alice := &Account{Name: "Alice", Balance: 100}
    bob := &Account{Name: "Bob", Balance: 50}

    // 实践3：在最顶层处理错误（记录日志、返回用户提示等）
    err := Transfer(alice, bob, 30)
    if err != nil {
        log.Println("ERROR:", err)
    } else {
        fmt.Printf("转账成功! Alice: %.2f, Bob: %.2f\n", alice.Balance, bob.Balance)
    }

    err = Transfer(alice, bob, 200)
    if err != nil {
        // 根据错误类型做不同处理
        if errors.Is(err, ErrInsufficientBalance) {
            fmt.Println("提示用户: 账户余额不足，请充值后再试")
        } else {
            fmt.Println("系统错误:", err)
        }
    }

    fmt.Println()

    // 实践4：不要忽略错误！
    // 错误的做法：
    // result, _ := someFunction()   // 忽略了错误

    // 正确的做法：
    // result, err := someFunction()
    // if err != nil {
    //     return fmt.Errorf("someFunction: %w", err)
    // }

    // 实践5：错误只处理一次（不要既记录日志又返回错误）
    fmt.Println("错误处理原则:")
    fmt.Println("  1. 错误应尽早返回（Guard Clause）")
    fmt.Println("  2. 每层添加上下文后向上传递")
    fmt.Println("  3. 只在最顶层处理错误（日志、用户提示）")
    fmt.Println("  4. 永远不要忽略错误")
    fmt.Println("  5. 错误只处理一次")
}
```

```go
package main

import (
    "fmt"
    "strings"
)

// 策略接口：不同的文本格式化策略
type Formatter interface {
    Format(text string) string
}

// 具体策略
type UpperFormatter struct{}

func (f UpperFormatter) Format(text string) string {
    return strings.ToUpper(text)
}

type TitleFormatter struct{}

func (f TitleFormatter) Format(text string) string {
    words := strings.Fields(text)
    for i, w := range words {
        if len(w) > 0 {
            words[i] = strings.ToUpper(w[:1]) + strings.ToLower(w[1:])
        }
    }
    return strings.Join(words, " ")
}

type SnakeFormatter struct{}

func (f SnakeFormatter) Format(text string) string {
    return strings.ReplaceAll(strings.ToLower(text), " ", "_")
}

// 使用策略
type TextProcessor struct {
    formatter Formatter
}

func (tp *TextProcessor) SetFormatter(f Formatter) {
    tp.formatter = f
}

func (tp *TextProcessor) Process(text string) string {
    if tp.formatter == nil {
        return text
    }
    return tp.formatter.Format(text)
}

func main() {
    processor := &TextProcessor{}
    text := "hello world go language"

    processor.SetFormatter(UpperFormatter{})
    fmt.Println("大写:", processor.Process(text))

    processor.SetFormatter(TitleFormatter{})
    fmt.Println("标题:", processor.Process(text))

    processor.SetFormatter(SnakeFormatter{})
    fmt.Println("蛇形:", processor.Process(text))
}
```

### 5.2 中间件模式

```go
package main

import (
    "fmt"
    "strings"
    "time"
)

// 处理器接口
type Handler interface {
    Handle(request string) string
}

// 处理器函数适配器（类似 http.HandlerFunc）
type HandlerFunc func(string) string

func (f HandlerFunc) Handle(request string) string {
    return f(request)
}

// 中间件类型
type Middleware func(Handler) Handler

// 日志中间件
func LoggingMiddleware() Middleware {
    return func(next Handler) Handler {
        return HandlerFunc(func(request string) string {
            fmt.Printf("[LOG] 收到请求: %s\n", request)
            start := time.Now()
            response := next.Handle(request)
            fmt.Printf("[LOG] 处理耗时: %v\n", time.Since(start))
            return response
        })
    }
}

// 认证中间件
func AuthMiddleware(token string) Middleware {
    return func(next Handler) Handler {
        return HandlerFunc(func(request string) string {
            if !strings.Contains(request, "token="+token) {
                return "401 Unauthorized"
            }
            fmt.Println("[AUTH] 认证通过")
            return next.Handle(request)
        })
    }
}

// 应用中间件链
func ApplyMiddleware(handler Handler, middlewares ...Middleware) Handler {
    // 从后往前包装，保证执行顺序
    for i := len(middlewares) - 1; i >= 0; i-- {
        handler = middlewares[i](handler)
    }
    return handler
}

func main() {
    // 最终处理器
    finalHandler := HandlerFunc(func(request string) string {
        return fmt.Sprintf("处理完成: %s -> OK", request)
    })

    // 组装中间件链
    handler := ApplyMiddleware(finalHandler,
        LoggingMiddleware(),
        AuthMiddleware("abc123"),
    )

    // 测试1：无 token
    fmt.Println("--- 请求1 ---")
    result := handler.Handle("GET /api/data")
    fmt.Println("响应:", result)

    // 测试2：有 token
    fmt.Println("\n--- 请求2 ---")
    result = handler.Handle("GET /api/data?token=abc123")
    fmt.Println("响应:", result)
}
```

```
// 输出:

[LOG] 收到请求: GET /api/data
[LOG] 处理耗时: 0s
响应: 401 Unauthorized

--- 请求2 ---
[LOG] 收到请求: GET /api/data?token=abc123
[AUTH] 认证通过
[LOG] 处理耗时: 0s
```

### 5.3 接口检查（编译期断言）

```go
package main

import "fmt"

type Saver interface {
    Save() error
}

type Loader interface {
    Load() error
}

type Database struct {
    Name string
}

func (db *Database) Save() error {
    fmt.Printf("[%s] 数据已保存\n", db.Name)
    return nil
}

func (db *Database) Load() error {
    fmt.Printf("[%s] 数据已加载\n", db.Name)
    return nil
}

// 编译期接口检查：确保 *Database 实现了 Saver 和 Loader
// 如果没有实现，编译时就会报错
var _ Saver = (*Database)(nil)
var _ Loader = (*Database)(nil)

func main() {
    db := &Database{Name: "MySQL"}
    db.Save()
    db.Load()
}
```

```
// 输出:
[MySQL] 数据已保存
[MySQL] 数据已加载
```

**编译期接口检查的作用：**

- `var _ Saver = (*Database)(nil)` 不会分配内存
- 如果 `Database` 缺少 `Save()` 方法，**编译时**就会报错
- 这是 Go 社区推荐的最佳实践，常见于库和框架代码中

### 练习 1：简易日志系统

```go
package main

import (
    "fmt"
    "strings"
    "time"
)

// 日志级别
type Level int

const (
    DEBUG Level = iota
    INFO
    WARN
    ERROR
)

func (l Level) String() string {
    return [...]string{"DEBUG", "INFO", "WARN", "ERROR"}[l]
}

// 日志输出接口
type LogOutput interface {
    Write(level Level, message string)
}

// 控制台输出
type ConsoleOutput struct{}

func (c *ConsoleOutput) Write(level Level, message string) {
    timestamp := time.Now().Format("15:04:05")
    fmt.Printf("[%s] %-5s %s\n", timestamp, level, message)
}

// 内存输出（用于测试）
type MemoryOutput struct {
    Logs []string
}

func (m *MemoryOutput) Write(level Level, message string) {
    entry := fmt.Sprintf("%-5s %s", level, message)
    m.Logs = append(m.Logs, entry)
}

// 多输出（同时写入多个目标）
type MultiOutput struct {
    outputs []LogOutput
}

func (m *MultiOutput) Write(level Level, message string) {
    for _, out := range m.outputs {
        out.Write(level, message)
    }
}

// 日志过滤（装饰器模式）
type FilteredOutput struct {
    MinLevel Level
    Output   LogOutput
}

func (f *FilteredOutput) Write(level Level, message string) {
    if level >= f.MinLevel {
        f.Output.Write(level, message)
    }
}

// 日志器
type Logger struct {
    output LogOutput
}

func NewLogger(output LogOutput) *Logger {
    return &Logger{output: output}
}

func (l *Logger) Debug(msg string) { l.output.Write(DEBUG, msg) }
func (l *Logger) Info(msg string)  { l.output.Write(INFO, msg) }
func (l *Logger) Warn(msg string)  { l.output.Write(WARN, msg) }
func (l *Logger) Error(msg string) { l.output.Write(ERROR, msg) }

func main() {
    // 场景1：控制台输出 + 级别过滤
    fmt.Println("=== 只显示 WARN 及以上 ===")
    filtered := &FilteredOutput{
        MinLevel: WARN,
        Output:   &ConsoleOutput{},
    }
    logger := NewLogger(filtered)
    logger.Debug("调试信息")        // 被过滤
    logger.Info("普通信息")         // 被过滤
    logger.Warn("警告信息")         // 显示
    logger.Error("错误信息")        // 显示

    fmt.Println()

    // 场景2：同时输出到控制台和内存
    fmt.Println("=== 多目标输出 ===")
    mem := &MemoryOutput{}
    multi := &MultiOutput{
        outputs: []LogOutput{
            &FilteredOutput{MinLevel: INFO, Output: &ConsoleOutput{}},
            mem, // 内存中记录所有级别
        },
    }
    logger2 := NewLogger(multi)
    logger2.Debug("debug msg")
    logger2.Info("info msg")
    logger2.Error("error msg")

    fmt.Println()
    fmt.Println("内存中的日志:")
    for _, entry := range mem.Logs {
        fmt.Println("  " + entry)
    }

    fmt.Println()
    fmt.Println("=== 关键设计点 ===")
    fmt.Println("1. LogOutput 接口只有一个方法 -> 易于实现")
    fmt.Println("2. FilteredOutput 装饰器 -> 不修改原有类型")
    fmt.Println("3. MultiOutput 组合多个输出 -> 灵活扩展")
    _ = strings.Builder{} // 避免 unused import
}
```

### 练习 2：简易计算器（接口 + 错误处理）

```go
package main

import (
    "errors"
    "fmt"
    "math"
)

var ErrDivisionByZero = errors.New("除数不能为零")
var ErrNegativeSqrt = errors.New("不能对负数求平方根")
var ErrUnknownOp = errors.New("未知运算")

// 运算接口
type Operation interface {
    Execute(a, b float64) (float64, error)
    Symbol() string
}

type Add struct{}
func (Add) Execute(a, b float64) (float64, error) { return a + b, nil }
func (Add) Symbol() string                         { return "+" }

type Sub struct{}
func (Sub) Execute(a, b float64) (float64, error) { return a - b, nil }
func (Sub) Symbol() string                         { return "-" }

type Mul struct{}
func (Mul) Execute(a, b float64) (float64, error) { return a * b, nil }
func (Mul) Symbol() string                         { return "×" }

type Div struct{}
func (Div) Execute(a, b float64) (float64, error) {
    if b == 0 {
        return 0, ErrDivisionByZero
    }
    return a / b, nil
}
func (Div) Symbol() string { return "÷" }

type Pow struct{}
func (Pow) Execute(a, b float64) (float64, error) { return math.Pow(a, b), nil }
func (Pow) Symbol() string                         { return "^" }

// 计算器
type Calculator struct {
    ops     map[string]Operation
    history []string
}

func NewCalculator() *Calculator {
    calc := &Calculator{
        ops: make(map[string]Operation),
    }
    // 注册运算
    calc.Register(Add{})
    calc.Register(Sub{})
    calc.Register(Mul{})
    calc.Register(Div{})
    calc.Register(Pow{})
    return calc
}

func (c *Calculator) Register(op Operation) {
    c.ops[op.Symbol()] = op
}

func (c *Calculator) Calculate(a float64, symbol string, b float64) (float64, error) {
    op, exists := c.ops[symbol]
    if !exists {
        return 0, fmt.Errorf("%w: %s", ErrUnknownOp, symbol)
    }

    result, err := op.Execute(a, b)
    if err != nil {
        return 0, fmt.Errorf("计算 %.2f %s %.2f: %w", a, symbol, b, err)
    }

    entry := fmt.Sprintf("%.4g %s %.4g = %.4g", a, symbol, b, result)
    c.history = append(c.history, entry)
    return result, nil
}

func (c *Calculator) History() []string {
    return c.history
}

func main() {
    calc := NewCalculator()

    // 执行一系列计算
    expressions := []struct {
        a, b   float64
        symbol string
    }{
        {10, 3, "+"},
        {10, 3, "-"},
        {10, 3, "×"},
        {10, 3, "÷"},
        {2, 10, "^"},
        {10, 0, "÷"},
        {10, 3, "%"},
    }

    for _, expr := range expressions {
        result, err := calc.Calculate(expr.a, expr.symbol, expr.b)
        if err != nil {
            if errors.Is(err, ErrDivisionByZero) {
                fmt.Printf("  %.4g %s %.4g = 错误(除零)\n", expr.a, expr.symbol, expr.b)
            } else if errors.Is(err, ErrUnknownOp) {
                fmt.Printf("  %.4g %s %.4g = 错误(未知运算)\n", expr.a, expr.symbol, expr.b)
            } else {
                fmt.Printf("  错误: %v\n", err)
            }
        } else {
            fmt.Printf("  %.4g %s %.4g = %.4g\n", expr.a, expr.symbol, expr.b, result)
        }
    }

    fmt.Println("\n计算历史:")
    for i, entry := range calc.History() {
        fmt.Printf("  %d. %s\n", i+1, entry)
    }
}
```

```
  10 × 3 = 30
  10 ÷ 3 = 3.333
```

## 7. 本章小结

| 知识点 | 要点 |
|--------|------|
| 接口定义 | 方法签名的集合，隐式实现，无需 `implements` |
| 小接口原则 | 1-2 个方法的接口最灵活（`io.Reader`、`fmt.Stringer`） |
| 接口组合 | 通过嵌入小接口组合大接口（`io.ReadWriteCloser`） |
| 空接口 `any` | 可接收任意类型，是 `interface{}` 的别名 |
| 类型断言 | `val.(Type)` 提取具体类型，带 `ok` 检查更安全 |
| 类型选择 | `switch v := val.(type)` 优雅处理多种类型 |
| `error` 接口 | 只有 `Error() string` 一个方法 |
| 自定义错误 | 实现 `error` 接口，携带更多上下文信息 |
| 错误包装 | `fmt.Errorf("%w", err)` 保留错误链 |
| `errors.Is` | 判断错误链中是否包含目标错误（值比较） |
| `errors.As` | 从错误链中提取特定类型的错误（类型匹配） |
| 编译期断言 | `var _ Interface = (*Type)(nil)` 确保类型实现接口 |
| 策略模式 | 通过接口切换不同算法实现 |
| 中间件模式 | 接口 + 函数适配器实现请求处理链 |
