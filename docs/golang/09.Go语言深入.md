# Go 语言深入

## 1. 内存管理

### 1.1 栈与堆

```go
package main

import "fmt"

// 栈分配：局部变量，函数返回后自动回收
func stackAlloc() int {
    x := 42 // x 在栈上分配
    return x // 返回值的拷贝，x 随函数返回被回收
}

// 堆分配：返回指针，变量"逃逸"到堆上
func heapAlloc() *int {
    x := 42  // x 逃逸到堆上，因为返回了它的指针
    return &x // 指针被返回，x 的生命周期超出函数作用域
}

func main() {
    a := stackAlloc()
    b := heapAlloc()

    fmt.Println("栈分配:", a)   // 42
    fmt.Println("堆分配:", *b)  // 42

    fmt.Println("\n栈 vs 堆:")
    fmt.Println("  栈: 分配/回收极快（移动指针），无 GC 开销")
    fmt.Println("  堆: 需要 GC 回收，分配相对较慢")
    fmt.Println("  Go 编译器自动决定变量放在栈还是堆（逃逸分析）")
}
```

```
// 输出:
栈分配: 42
堆分配: 42

栈 vs 堆:
  栈: 分配/回收极快（移动指针），无 GC 开销
  堆: 需要 GC 回收，分配相对较慢
  Go 编译器自动决定变量放在栈还是堆（逃逸分析）
```

### 1.2 逃逸分析

编译器通过逃逸分析（Escape Analysis）决定变量分配在栈还是堆。

```go
package main

import "fmt"

// 不逃逸：值返回
func noEscape() int {
    x := 100
    return x
}

// 逃逸：返回指针
func escapePointer() *int {
    x := 100
    return &x // x 逃逸到堆
}

// 逃逸：赋值给接口
func escapeInterface() {
    x := 42
    fmt.Println(x) // x 逃逸，因为 Println 参数是 interface{}
}

// 逃逸：闭包引用
func escapeClosure() func() int {
    x := 42
    return func() int { return x } // x 逃逸，被闭包捕获
}

// 逃逸：切片容量过大或动态大小
func escapeSlice() {
    s := make([]int, 10000) // 大切片逃逸到堆
    s[0] = 1
    _ = s
}

// 不逃逸：小切片
func noEscapeSlice() {
    s := make([]int, 3) // 小切片可能留在栈上
    s[0] = 1
    _ = s
}

func main() {
    _ = noEscape()
    _ = escapePointer()
    escapeInterface()
    _ = escapeClosure()
    escapeSlice()
    noEscapeSlice()
}
```

```bash
# 查看逃逸分析结果
go build -gcflags="-m" main.go

# 输出（简化）:
./main.go:9:2:  x escapes to heap          # escapePointer
./main.go:14:13: x escapes to heap          # escapeInterface (fmt.Println)
./main.go:19:2:  x escapes to heap          # escapeClosure
./main.go:25:15: make([]int, 10000) escapes to heap  # 大切片
./main.go:31:15: make([]int, 3) does not escape       # 小切片

# 更详细的输出
go build -gcflags="-m -m" main.go
```

**常见逃逸场景：**

| 场景 | 是否逃逸 | 原因 |
|------|----------|------|
| 返回局部变量值 | 否 | 值拷贝，原变量可安全回收 |
| 返回局部变量指针 | 是 | 指针被外部持有，变量不能回收 |
| 赋值给 `interface{}` | 是 | 接口内部存储指针 |
| 闭包捕获变量 | 是 | 闭包生命周期可能超过函数 |
| `make([]T, n)` n 太大 | 是 | 栈空间有限 |
| 被 slice/map 的元素引用 | 是 | 容器可能逃逸 |

### 1.3 内存对齐与结构体布局

```go
package main

import (
    "fmt"
    "unsafe"
)

// 不良布局：字段间有大量填充
type BadLayout struct {
    a bool    // 1 字节 + 7 填充
    b float64 // 8 字节
    c bool    // 1 字节 + 3 填充
    d int32   // 4 字节
}   // 总计: 24 字节

// 优化布局：大字段在前，小字段聚合
type GoodLayout struct {
    b float64 // 8 字节
    d int32   // 4 字节
    a bool    // 1 字节
    c bool    // 1 字节 + 2 填充
}   // 总计: 16 字节

func main() {
    fmt.Println("=== 内存对齐 ===")
    fmt.Printf("BadLayout  大小: %d 字节, 对齐: %d\n",
        unsafe.Sizeof(BadLayout{}), unsafe.Alignof(BadLayout{}))
    fmt.Printf("GoodLayout 大小: %d 字节, 对齐: %d\n",
        unsafe.Sizeof(GoodLayout{}), unsafe.Alignof(GoodLayout{}))
    fmt.Printf("节省: %d 字节 (%.0f%%)\n",
        unsafe.Sizeof(BadLayout{})-unsafe.Sizeof(GoodLayout{}),
        float64(unsafe.Sizeof(BadLayout{})-unsafe.Sizeof(GoodLayout{}))/float64(unsafe.Sizeof(BadLayout{}))*100)

    fmt.Println()

    // 详细查看每个字段的偏移量
    fmt.Println("BadLayout 字段偏移:")
    var bad BadLayout
    fmt.Printf("  a (bool):    偏移=%d, 大小=%d\n", unsafe.Offsetof(bad.a), unsafe.Sizeof(bad.a))
    fmt.Printf("  b (float64): 偏移=%d, 大小=%d\n", unsafe.Offsetof(bad.b), unsafe.Sizeof(bad.b))
    fmt.Printf("  c (bool):    偏移=%d, 大小=%d\n", unsafe.Offsetof(bad.c), unsafe.Sizeof(bad.c))
    fmt.Printf("  d (int32):   偏移=%d, 大小=%d\n", unsafe.Offsetof(bad.d), unsafe.Sizeof(bad.d))

    fmt.Println("\nGoodLayout 字段偏移:")
    var good GoodLayout
    fmt.Printf("  b (float64): 偏移=%d, 大小=%d\n", unsafe.Offsetof(good.b), unsafe.Sizeof(good.b))
    fmt.Printf("  d (int32):   偏移=%d, 大小=%d\n", unsafe.Offsetof(good.d), unsafe.Sizeof(good.d))
    fmt.Printf("  a (bool):    偏移=%d, 大小=%d\n", unsafe.Offsetof(good.a), unsafe.Sizeof(good.a))
    fmt.Printf("  c (bool):    偏移=%d, 大小=%d\n", unsafe.Offsetof(good.c), unsafe.Sizeof(good.c))
}
```

### 2.2 观察 GC 行为

```go
package main

import (
    "fmt"
    "runtime"
    "time"
)

func printGCStats(label string) {
    var stats runtime.MemStats
    runtime.ReadMemStats(&stats)
    fmt.Printf("[%s]\n", label)
    fmt.Printf("  堆分配: %.2f MB\n", float64(stats.HeapAlloc)/1024/1024)
    fmt.Printf("  堆对象: %d\n", stats.HeapObjects)
    fmt.Printf("  GC 次数: %d\n", stats.NumGC)
    fmt.Printf("  GC 总暂停: %v\n", time.Duration(stats.PauseTotalNs))
    if stats.NumGC > 0 {
        fmt.Printf("  上次 GC 暂停: %v\n", time.Duration(stats.PauseNs[(stats.NumGC+255)%256]))
    }
    fmt.Println()
}

func main() {
    printGCStats("初始状态")

    // 分配大量内存
    data := make([][]byte, 0, 100000)
    for i := 0; i < 100000; i++ {
        data = append(data, make([]byte, 100))
    }
    printGCStats("分配 10MB 后")

    // 释放引用
    data = nil
    runtime.GC() // 手动触发 GC
    printGCStats("释放引用 + GC 后")

    // 再次分配
    for i := 0; i < 50000; i++ {
        _ = make([]byte, 200)
    }
    printGCStats("再次分配后")
}
```

### 2.3 GC 调优

```go
package main

import (
    "fmt"
    "runtime"
    "runtime/debug"
)

func main() {
    // ========== GOGC ==========
    fmt.Println("=== GOGC ===")
    fmt.Println("  GOGC=100（默认）: 堆翻倍时触发 GC")
    fmt.Println("  GOGC=50:  堆增长 50% 时触发（更频繁，延迟更低）")
    fmt.Println("  GOGC=200: 堆增长 200% 时触发（更少 GC，吞吐更高）")
    fmt.Println("  GOGC=off: 关闭 GC（特殊场景慎用）")

    // 代码中设置 GOGC
    old := debug.SetGCPercent(200)
    fmt.Printf("  旧 GOGC: %d, 新 GOGC: 200\n", old)
    debug.SetGCPercent(old) // 恢复

    fmt.Println()

    // ========== GOMEMLIMIT（Go 1.19+）==========
    fmt.Println("=== GOMEMLIMIT ===")
    fmt.Println("  设置软内存上限，GC 会更积极地回收以不超过限制")
    fmt.Println("  适用于容器环境，防止 OOM")

    debug.SetMemoryLimit(100 * 1024 * 1024) // 100MB
    fmt.Println("  已设置 GOMEMLIMIT: 100MB")

    fmt.Println()

    // ========== 运行时查看 GC 详情 ==========
    fmt.Println("=== 命令行诊断 ===")
    fmt.Println("  GODEBUG=gctrace=1 go run main.go")
    fmt.Println("  输出: gc 1 @0.001s 2%: 0.015+0.30+0.003 ms clock, ...")
    fmt.Println("  含义: GC次数 @发生时间 CPU占比: STW+并发标记+STW")

    fmt.Println()
    fmt.Printf("GOMAXPROCS: %d\n", runtime.GOMAXPROCS(0))
    fmt.Printf("Go Version: %s\n", runtime.Version())
```

## 3. GMP 调度器详解

### 3.1 调度模型

```go
package main

import (
    "fmt"
    "runtime"
    "sync"
    "time"
)

func main() {
    fmt.Println("=== GMP 调度模型 ===\n")
    fmt.Printf("G (Goroutine): 用户态协程，初始栈约 2 KB\n")
    fmt.Printf("M (Machine):   OS 线程，执行 G 的载体\n")
    fmt.Printf("P (Processor): 逻辑处理器，数量 = GOMAXPROCS = %d\n\n", runtime.GOMAXPROCS(0))

    fmt.Println("调度策略:")
    fmt.Println("  1. M 从绑定的 P 的本地队列获取 G 执行")
    fmt.Println("  2. 本地队列空 → 从全局队列获取")
    fmt.Println("  3. 全局队列空 → 从其他 P 偷取一半（Work Stealing）")
    fmt.Println("  4. G 发生系统调用 → M 阻塞，P 交给空闲 M（Hand Off）")
    fmt.Println()

    /*
        ┌─────────────────────────────────────┐
        │           全局队列 (Global Queue)     │
        └────────────────┬────────────────────┘
                         │ 偷取
        ┌────────────────┼────────────────┐
        ▼                ▼                ▼
    ┌───────┐       ┌───────┐       ┌───────┐
    │  P0   │       │  P1   │       │  P2   │
    │[G1,G2]│       │[G3]   │       │[ ]    │──→ 偷取 P0 的 G2
    └───┬───┘       └───┬───┘       └───┬───┘
        ▼               ▼               ▼
    ┌───────┐       ┌───────┐       ┌───────┐
    │  M0   │       │  M1   │       │  M2   │
    └───────┘       └───────┘       └───────┘
    */

    // 演示并行调度
    fmt.Println("=== 调度观察 ===")
    var wg sync.WaitGroup
    for i := 0; i < 4; i++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            start := time.Now()
            sum := 0
            for j := 0; j < 10_000_000; j++ {
                sum += j
            }
            fmt.Printf("  G%d 耗时: %v, 当前 goroutine 总数: %d\n",
                id, time.Since(start).Round(time.Millisecond), runtime.NumGoroutine())
        }(i)
    }
    wg.Wait()
}
```

```
// 输出:

G (Goroutine): 用户态协程，初始栈约 2 KB
M (Machine):   OS 线程，执行 G 的载体

  4. G 发生系统调用 → M 阻塞，P 交给空闲 M（Hand Off）

=== 调度观察 ===
```

### 3.2 抢占式调度

```go
package main

import (
    "fmt"
    "runtime"
    "time"
)

func main() {
    fmt.Println("=== 抢占式调度 ===\n")

    runtime.GOMAXPROCS(1) // 只用 1 个 P，便于观察

    // 死循环 goroutine
    go func() {
        for {
            // 纯计算，无函数调用
            // Go 1.14+ 基于信号（SIGURG）异步抢占，不会饿死其他 G
        }
    }()

    done := make(chan bool)
    go func() {
        time.Sleep(100 * time.Millisecond)
        fmt.Println("  其他 goroutine 成功执行（异步抢占生效）")
        done <- true
    }()

    select {
    case <-done:
        fmt.Println("  测试通过!")
    case <-time.After(2 * time.Second):
        fmt.Println("  超时")
    }

    fmt.Println()
    fmt.Println("演进历史:")
    fmt.Println("  Go 1.13-: 协作式抢占（函数调用时检查标志）")
    fmt.Println("  Go 1.14+: 异步抢占（SIGURG 信号），消除饿死问题")
}
```

```go
func FibRecursive(n int) int {
    if n <= 1 {
        return n
    }
    return FibRecursive(n-1) + FibRecursive(n-2)
}

func FibIterative(n int) int {
    if n <= 1 {

    }
    a, b := 0, 1
    for i := 2; i <= n; i++ {

    }
    return b
}
```

## 5. 常见性能优化技巧

### 5.1 字符串拼接优化

```go
package main

import (
    "fmt"
    "strings"
    "time"
)

func concatPlus(n int) string {
    s := ""
    for i := 0; i < n; i++ {
        s += "x"
    }
    return s
}

func concatBuilder(n int) string {
    var b strings.Builder
    b.Grow(n)
    for i := 0; i < n; i++ {
        b.WriteString("x")
    }
    return b.String()
}

func concatJoin(n int) string {
    parts := make([]string, n)
    for i := 0; i < n; i++ {
        parts[i] = "x"
    }
    return strings.Join(parts, "")
}

func main() {
    n := 50000

    start := time.Now()
    _ = concatPlus(n)
    fmt.Printf("+ 拼接:    %v\n", time.Since(start))

    start = time.Now()
    _ = concatBuilder(n)
    fmt.Printf("Builder:   %v\n", time.Since(start))

    start = time.Now()
    _ = concatJoin(n)
    fmt.Printf("Join:      %v\n", time.Since(start))
}
```

```
// 输出:
+ 拼接:    580ms
Builder:   42µs
Join:      120µs
```

### 5.3 sync.Pool 复用对象

```go
package main

import (
    "bytes"
    "fmt"
    "sync"
    "time"
)

func withoutPool(n int) {
    for i := 0; i < n; i++ {
        buf := new(bytes.Buffer)
        buf.WriteString("hello world")
        _ = buf.String()
    }
}

func withPool(n int) {
    pool := &sync.Pool{
        New: func() any { return new(bytes.Buffer) },
    }
    for i := 0; i < n; i++ {
        buf := pool.Get().(*bytes.Buffer)
        buf.Reset()
        buf.WriteString("hello world")
        _ = buf.String()
        pool.Put(buf)
    }
}

func main() {
    n := 5_000_000

    start := time.Now()
    withoutPool(n)
    fmt.Printf("不用 Pool: %v\n", time.Since(start))

    start = time.Now()
    withPool(n)
    fmt.Printf("用 Pool:   %v\n", time.Since(start))
}
```

```
// 输出:
不用 Pool: 850ms
用 Pool:   320ms
```

### 5.4 减少锁竞争

```go
package main

import (
    "fmt"
    "sync"
    "sync/atomic"
    "time"
)

func benchMutex(goroutines, iterations int) time.Duration {
    var mu sync.Mutex
    var count int
    var wg sync.WaitGroup

    start := time.Now()
    for g := 0; g < goroutines; g++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for i := 0; i < iterations; i++ {
                mu.Lock()
                count++
                mu.Unlock()
            }
        }()
    }
    wg.Wait()
    return time.Since(start)
}

func benchAtomic(goroutines, iterations int) time.Duration {
    var count atomic.Int64
    var wg sync.WaitGroup

    start := time.Now()
    for g := 0; g < goroutines; g++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for i := 0; i < iterations; i++ {
                count.Add(1)
            }
        }()
    }
    wg.Wait()
    return time.Since(start)
}

// 分片锁：将一把锁拆分成多把
type ShardedCounter struct {
    shards [16]struct {
        mu    sync.Mutex
        count int
        _     [56]byte // 避免 false sharing
    }
}

func (c *ShardedCounter) Inc(id int) {
    shard := &c.shards[id%16]
    shard.mu.Lock()
    shard.count++
    shard.mu.Unlock()
}

func benchSharded(goroutines, iterations int) time.Duration {
    counter := &ShardedCounter{}
    var wg sync.WaitGroup

    start := time.Now()
    for g := 0; g < goroutines; g++ {
        wg.Add(1)
        go func(id int) {
            defer wg.Done()
            for i := 0; i < iterations; i++ {
                counter.Inc(id)
            }
        }(g)
    }
    wg.Wait()
    return time.Since(start)
}

func main() {
    goroutines := 100
    iterations := 100_000

    fmt.Printf("并发: %d goroutines × %d 次\n\n", goroutines, iterations)

    d1 := benchMutex(goroutines, iterations)
    fmt.Printf("Mutex:   %v\n", d1)

    d2 := benchAtomic(goroutines, iterations)
    fmt.Printf("Atomic:  %v\n", d2)

    d3 := benchSharded(goroutines, iterations)
    fmt.Printf("Sharded: %v\n", d3)
}
```

```
// 输出:
并发: 100 goroutines × 100000 次

Mutex:   280ms
Atomic:  85ms
Sharded: 95ms
```

## 7. 综合练习：性能优化对比

```go
package main

import (
    "fmt"
    "math/rand"
    "runtime"
    "strings"
    "time"
)

// ==================== 优化前 ====================

func badConcat(words []string) string {
    result := ""
    for _, w := range words {
        result += w + " "
    }
    return result
}

func badSlice(n int) []int {
    var result []int
    for i := 0; i < n; i++ {
        result = append(result, i*i)
    }
    return result
}

func badSearch(data []string, target string) int {
    for i, s := range data {
        if strings.ToLower(s) == strings.ToLower(target) {
            return i
        }
    }
    return -1
}

// ==================== 优化后 ====================

func goodConcat(words []string) string {
    var b strings.Builder
    b.Grow(len(words) * 10)
    for i, w := range words {
        if i > 0 {
            b.WriteByte(' ')
        }
        b.WriteString(w)
    }
    return b.String()
}

func goodSlice(n int) []int {
    result := make([]int, n)
    for i := 0; i < n; i++ {
        result[i] = i * i
    }
    return result
}

func goodSearch(data []string, target string) int {
    lowerTarget := strings.ToLower(target)
    for i, s := range data {
        if strings.EqualFold(s, lowerTarget) {
            return i
        }
    }
    return -1
}

// ==================== 基准 ====================

func bench(name string, fn func()) time.Duration {
    fn() // 预热
    start := time.Now()
    for i := 0; i < 100; i++ {
        fn()
    }
    return time.Since(start) / 100
}

func main() {
    words := make([]string, 10000)
    for i := range words {
        words[i] = fmt.Sprintf("word_%d", rand.IntN(10000))
    }

    searchData := make([]string, 100000)
    for i := range searchData {
        searchData[i] = fmt.Sprintf("Item_%d", i)
    }

    n := 1_000_000

    fmt.Println("=== 字符串拼接 (10000 词) ===")
    d1 := bench("bad", func() { _ = badConcat(words) })
    d2 := bench("good", func() { _ = goodConcat(words) })
    fmt.Printf("  优化前: %v\n  优化后: %v\n  提升: %.0fx\n\n", d1, d2, float64(d1)/float64(d2))

    fmt.Println("=== 切片分配 (1000000 元素) ===")
    d1 = bench("bad", func() { _ = badSlice(n) })
    d2 = bench("good", func() { _ = goodSlice(n) })
    fmt.Printf("  优化前: %v\n  优化后: %v\n  提升: %.1fx\n\n", d1, d2, float64(d1)/float64(d2))

    fmt.Println("=== 字符串搜索 (100000 条) ===")
    d1 = bench("bad", func() { _ = badSearch(searchData, "Item_99999") })
    d2 = bench("good", func() { _ = goodSearch(searchData, "Item_99999") })
    fmt.Printf("  优化前: %v\n  优化后: %v\n  提升: %.1fx\n\n", d1, d2, float64(d1)/float64(d2))

    var stats runtime.MemStats
    runtime.ReadMemStats(&stats)
    fmt.Printf("=== 内存统计 ===\n  总分配: %.2f MB\n  GC 次数: %d\n",
        float64(stats.TotalAlloc)/1024/1024, stats.NumGC)
}
```

## 8. 本章小结

| 知识点 | 要点 |
|--------|------|
| 栈 vs 堆 | 栈快、自动回收；堆慢、需 GC。编译器自动决定 |
| 逃逸分析 | `go build -gcflags="-m"` 查看；返回指针/接口赋值/闭包/大对象会逃逸 |
| 内存对齐 | 大字段在前、小字段聚合；`fieldalignment` 工具自动检测 |
| 三色标记 | 白灰黑三色，并发标记，写屏障保证正确性 |
| GOGC | 默认 100（翻倍触发 GC），调高减少 GC 频率 |
| GOMEMLIMIT | Go 1.19+ 软内存限制，容器环境防 OOM |
| GMP 模型 | G(协程) M(线程) P(逻辑处理器)，本地队列 + 全局队列 + 偷取 |
| Work Stealing | P 本地队列空时从其他 P 偷取一半 G |
| 抢占调度 | Go 1.14+ 基于信号（SIGURG），消除长运行 G 饿死问题 |
| pprof | CPU/堆/goroutine/阻塞/锁竞争分析；`-http` 火焰图 |
| trace | `go tool trace` 可视化调度、GC、网络事件 |
| 字符串优化 | `strings.Builder` 比 `+` 快 100x+ |
| 切片预分配 | `make([]T, 0, n)` 避免频繁扩容，快 3-4x |
| sync.Pool | 复用临时对象，降低 GC 压力，快 2-3x |
| 锁优化 | atomic > 分片锁 > RWMutex > Mutex |
| 编译注入 | `-ldflags="-X main.Version=..."` 注入版本信息 |
| 减小体积 | `-ldflags="-s -w"` 减小约 30% |
| 交叉编译 | `GOOS=linux GOARCH=amd64 go build` |
| Build Tags | `//go:build dev` 条件编译 |
