# Go 语言微服务与分布式

## 1. 微服务架构概述

### 1.1 单体 vs 微服务

```
单体架构:
┌─────────────────────────────┐
│         单体应用             │
│  ┌─────┐ ┌─────┐ ┌──────┐  │
│  │用户 │ │商品 │ │订单  │  │
│  │模块 │ │模块 │ │模块  │  │
│  └─────┘ └─────┘ └──────┘  │
│         共享数据库           │
└─────────────────────────────┘

微服务架构:
┌────────┐  ┌────────┐  ┌────────┐
│用户服务│  │商品服务│  │订单服务│
│  :8081 │  │  :8082 │  │  :8083 │
└───┬────┘  └───┬────┘  └───┬────┘
    │           │           │
 ┌──┴──┐    ┌──┴──┐    ┌──┴──┐
 │MySQL│    │MySQL│    │MySQL│
 └─────┘    └─────┘    └─────┘
```

**微服务的核心特征：**

| 特征 | 说明 |
|------|------|
| 独立部署 | 每个服务可以独立构建、测试、部署 |
| 技术异构 | 不同服务可以使用不同语言和数据库 |
| 去中心化 | 每个服务拥有自己的数据，不共享数据库 |
| 按业务拆分 | 围绕业务能力而非技术层拆分 |
| 弹性扩展 | 按需对高负载服务进行水平扩展 |

### 1.2 微服务间通信方式

```
同步通信:
  ┌────────┐  HTTP/gRPC  ┌────────┐
  │服务 A  │────────────→│服务 B  │
  │        │←────────────│        │
  └────────┘   响应       └────────┘

异步通信:
  ┌────────┐            ┌────────┐            ┌────────┐
  │服务 A  │──发布──→   │消息队列│  ──消费──→ │服务 B  │
  └────────┘            │(Kafka) │            └────────┘
                        └────────┘
```

### 2.1 使用 etcd 实现服务注册与发现

```go
package main

import (
    "context"
    "fmt"
    "log"
    "sync"
    "time"

    clientv3 "go.etcd.io/etcd/client/v3"
)

// ==================== 服务注册 ====================

type ServiceRegistry struct {
    client  *clientv3.Client
    leaseID clientv3.LeaseID
}

func NewServiceRegistry(endpoints []string) (*ServiceRegistry, error) {
    client, err := clientv3.New(clientv3.Config{
        Endpoints:   endpoints,
        DialTimeout: 5 * time.Second,
    })
    if err != nil {
        return nil, err
    }
    return &ServiceRegistry{client: client}, nil
}

// 注册服务（带 TTL 租约，服务挂了会自动移除）
func (r *ServiceRegistry) Register(ctx context.Context, serviceName, addr string, ttl int64) error {
    // 创建租约
    lease, err := r.client.Grant(ctx, ttl)
    if err != nil {
        return err
    }
    r.leaseID = lease.ID

    // 注册服务
    key := fmt.Sprintf("/services/%s/%s", serviceName, addr)
    _, err = r.client.Put(ctx, key, addr, clientv3.WithLease(lease.ID))
    if err != nil {
        return err
    }

    // 自动续租
    ch, err := r.client.KeepAlive(ctx, lease.ID)
    if err != nil {
        return err
    }

    // 消费续租响应（必须消费，否则会阻塞）
    go func() {
        for range ch {
            // 续租成功
        }
        log.Println("续租通道关闭")
    }()

    fmt.Printf("[注册] %s -> %s (TTL=%ds)\n", serviceName, addr, ttl)
    return nil
}

// 注销服务
func (r *ServiceRegistry) Deregister(ctx context.Context, serviceName, addr string) error {
    key := fmt.Sprintf("/services/%s/%s", serviceName, addr)
    _, err := r.client.Delete(ctx, key)
    if err != nil {
        return err
    }
    // 撤销租约
    r.client.Revoke(ctx, r.leaseID)
    fmt.Printf("[注销] %s -> %s\n", serviceName, addr)
    return nil
}

func (r *ServiceRegistry) Close() {
    r.client.Close()
}

// ==================== 服务发现 ====================

type ServiceDiscovery struct {
    client   *clientv3.Client
    mu       sync.RWMutex
    services map[string][]string // serviceName -> []addr
}

func NewServiceDiscovery(endpoints []string) (*ServiceDiscovery, error) {
    client, err := clientv3.New(clientv3.Config{
        Endpoints:   endpoints,
        DialTimeout: 5 * time.Second,
    })
    if err != nil {
        return nil, err
    }
    return &ServiceDiscovery{
        client:   client,
        services: make(map[string][]string),
    }, nil
}

// 获取服务所有实例
func (d *ServiceDiscovery) GetService(ctx context.Context, serviceName string) ([]string, error) {
    prefix := fmt.Sprintf("/services/%s/", serviceName)
    resp, err := d.client.Get(ctx, prefix, clientv3.WithPrefix())
    if err != nil {
        return nil, err
    }

    var addrs []string
    for _, kv := range resp.Kvs {
        addrs = append(addrs, string(kv.Value))
    }
    return addrs, nil
}

// 监听服务变更
func (d *ServiceDiscovery) Watch(ctx context.Context, serviceName string) {
    prefix := fmt.Sprintf("/services/%s/", serviceName)
    watchCh := d.client.Watch(ctx, prefix, clientv3.WithPrefix())

    go func() {
        for resp := range watchCh {
            for _, ev := range resp.Events {
                switch ev.Type {
                case clientv3.EventTypePut:
                    fmt.Printf("[发现] 新实例上线: %s\n", string(ev.Kv.Value))
                case clientv3.EventTypeDelete:
                    fmt.Printf("[发现] 实例下线: %s\n", string(ev.Kv.Key))
                }
            }
        }
    }()
}

func (d *ServiceDiscovery) Close() {
    d.client.Close()
}

func main() {
    etcdEndpoints := []string{"localhost:2379"}
    ctx := context.Background()

    // 模拟服务注册
    reg, err := NewServiceRegistry(etcdEndpoints)
    if err != nil {
        log.Fatal("注册中心连接失败:", err)
    }
    defer reg.Close()

    reg.Register(ctx, "user-service", "192.168.1.10:8081", 10)
    reg.Register(ctx, "user-service", "192.168.1.11:8081", 10)
    reg.Register(ctx, "order-service", "192.168.1.20:8082", 10)

    // 模拟服务发现
    disc, err := NewServiceDiscovery(etcdEndpoints)
    if err != nil {
        log.Fatal("发现中心连接失败:", err)
    }
    defer disc.Close()

    addrs, _ := disc.GetService(ctx, "user-service")
    fmt.Println("\nuser-service 实例:", addrs)

    // 监听变更
    disc.Watch(ctx, "user-service")

    // 模拟实例下线
    time.Sleep(2 * time.Second)
    reg.Deregister(ctx, "user-service", "192.168.1.11:8081")

    time.Sleep(time.Second)
}
```

### 2.2 简易客户端负载均衡

```go
package main

import (
    "fmt"
    "math/rand"
    "sync/atomic"
)

// 负载均衡策略接口
type Balancer interface {
    Pick(instances []string) string
}

// 随机
type RandomBalancer struct{}

func (b *RandomBalancer) Pick(instances []string) string {
    return instances[rand.IntN(len(instances))]
}

// 轮询
type RoundRobinBalancer struct {
    counter atomic.Uint64
}

func (b *RoundRobinBalancer) Pick(instances []string) string {
    idx := b.counter.Add(1) - 1
    return instances[idx%uint64(len(instances))]
}

// 加权轮询
type WeightedInstance struct {
    Addr   string
    Weight int
}

type WeightedBalancer struct {
    counter atomic.Uint64
}

func (b *WeightedBalancer) Pick(instances []WeightedInstance) string {
    // 展开为加权列表
    var expanded []string
    for _, inst := range instances {
        for i := 0; i < inst.Weight; i++ {
            expanded = append(expanded, inst.Addr)
        }
    }
    idx := b.counter.Add(1) - 1
    return expanded[idx%uint64(len(expanded))]
}

func main() {
    instances := []string{"10.0.0.1:8080", "10.0.0.2:8080", "10.0.0.3:8080"}

    // 随机
    random := &RandomBalancer{}
    fmt.Println("=== 随机 ===")
    for i := 0; i < 6; i++ {
        fmt.Println(" ", random.Pick(instances))
    }

    // 轮询
    rr := &RoundRobinBalancer{}
    fmt.Println("\n=== 轮询 ===")
    for i := 0; i < 6; i++ {
        fmt.Println(" ", rr.Pick(instances))
    }

    // 加权
    weighted := []WeightedInstance{
        {"10.0.0.1:8080", 3}, // 权重3
        {"10.0.0.2:8080", 2}, // 权重2
        {"10.0.0.3:8080", 1}, // 权重1
    }
    wb := &WeightedBalancer{}
    fmt.Println("\n=== 加权轮询 (3:2:1) ===")
    counts := make(map[string]int)
    for i := 0; i < 12; i++ {
        addr := wb.Pick(weighted)
        counts[addr]++
    }
    for addr, count := range counts {
        fmt.Printf("  %s: %d 次\n", addr, count)
    }
}
```

## 3. API 网关模式

### 3.1 简易反向代理网关

```go
package main

import (
    "fmt"
    "log"
    "net/http"
    "net/http/httputil"
    "net/url"
    "strings"
    "time"
)

// 服务路由配置
type RouteConfig struct {
    PathPrefix string
    Target     string // 后端服务地址
}

// API 网关
type Gateway struct {
    routes []RouteConfig
}

func NewGateway(routes []RouteConfig) *Gateway {
    return &Gateway{routes: routes}
}

func (g *Gateway) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    for _, route := range g.routes {
        if strings.HasPrefix(r.URL.Path, route.PathPrefix) {
            g.proxy(w, r, route)
            return
        }
    }
    http.Error(w, "Service not found", http.StatusNotFound)
}

func (g *Gateway) proxy(w http.ResponseWriter, r *http.Request, route RouteConfig) {
    target, _ := url.Parse(route.Target)
    proxy := httputil.NewSingleHostReverseProxy(target)

    // 修改请求路径（去掉前缀）
    originalPath := r.URL.Path
    r.URL.Path = strings.TrimPrefix(r.URL.Path, route.PathPrefix)
    if r.URL.Path == "" {
        r.URL.Path = "/"
    }
    r.Host = target.Host

    // 自定义错误处理
    proxy.ErrorHandler = func(w http.ResponseWriter, r *http.Request, err error) {
        log.Printf("[网关] 代理错误: %s -> %s: %v", originalPath, route.Target, err)
        http.Error(w, "Service Unavailable", http.StatusServiceUnavailable)
    }

    log.Printf("[网关] %s %s -> %s%s", r.Method, originalPath, route.Target, r.URL.Path)
    proxy.ServeHTTP(w, r)
}

// 网关中间件：限流
func RateLimitMiddleware(rps int, next http.Handler) http.Handler {
    ticker := time.NewTicker(time.Second / time.Duration(rps))
    tokens := make(chan struct{}, rps)

    go func() {
        for range ticker.C {
            select {
            case tokens <- struct{}{}:
            default:
            }
        }
    }()
    // 初始填充
    for i := 0; i < rps; i++ {
        tokens <- struct{}{}
    }

    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        select {
        case <-tokens:
            next.ServeHTTP(w, r)
        default:
            http.Error(w, `{"error":"rate limit exceeded"}`, http.StatusTooManyRequests)
        }
    })
}

func main() {
    gateway := NewGateway([]RouteConfig{
        {PathPrefix: "/api/users", Target: "http://localhost:8081"},
        {PathPrefix: "/api/orders", Target: "http://localhost:8082"},
        {PathPrefix: "/api/products", Target: "http://localhost:8083"},
    })

    // 组装网关中间件
    handler := RateLimitMiddleware(100, gateway)

    fmt.Println("API 网关启动: http://localhost:8080")
    fmt.Println("路由:")
    fmt.Println("  /api/users/*    -> localhost:8081")
    fmt.Println("  /api/orders/*   -> localhost:8082")
    fmt.Println("  /api/products/* -> localhost:8083")
    log.Fatal(http.ListenAndServe(":8080", handler))
}
```

## 4. 服务间 HTTP 调用

### 4.1 带重试与超时的 HTTP 客户端

```go
package main

import (
    "context"
    "encoding/json"
    "fmt"
    "math"
    "math/rand"
    "net/http"
    "time"
)

// 可配置的 HTTP 客户端
type ServiceClient struct {
    BaseURL    string
    HTTPClient *http.Client
    MaxRetries int
}

func NewServiceClient(baseURL string, timeout time.Duration, maxRetries int) *ServiceClient {
    return &ServiceClient{
        BaseURL: baseURL,
        HTTPClient: &http.Client{
            Timeout: timeout,
        },
        MaxRetries: maxRetries,
    }
}

// 带重试的 GET 请求
func (c *ServiceClient) Get(ctx context.Context, path string, result any) error {
    var lastErr error

    for attempt := 0; attempt <= c.MaxRetries; attempt++ {
        if attempt > 0 {
            delay := c.backoff(attempt)
            fmt.Printf("  [重试] 第 %d 次，等待 %v\n", attempt, delay.Round(time.Millisecond))
            select {
            case <-time.After(delay):
            case <-ctx.Done():
                return ctx.Err()
            }
        }

        req, err := http.NewRequestWithContext(ctx, "GET", c.BaseURL+path, nil)
        if err != nil {
            return err
        }

        resp, err := c.HTTPClient.Do(req)
        if err != nil {
            lastErr = err
            fmt.Printf("  [错误] %v\n", err)
            continue // 网络错误，重试
        }
        defer resp.Body.Close()

        if resp.StatusCode >= 500 {
            lastErr = fmt.Errorf("服务端错误: %d", resp.StatusCode)
            fmt.Printf("  [错误] %v\n", lastErr)
            continue // 5xx 错误，重试
        }

        if resp.StatusCode >= 400 {
            return fmt.Errorf("客户端错误: %d", resp.StatusCode)
        }

        return json.NewDecoder(resp.Body).Decode(result)
    }

    return fmt.Errorf("重试 %d 次后仍失败: %w", c.MaxRetries, lastErr)
}

// 指数退避 + 抖动
func (c *ServiceClient) backoff(attempt int) time.Duration {
    base := math.Pow(2, float64(attempt)) * 100 // 200ms, 400ms, 800ms...
    jitter := rand.Float64() * 100               // 0-100ms 随机抖动
    return time.Duration(base+jitter) * time.Millisecond
}

func main() {
    client := NewServiceClient("http://localhost:8081", 5*time.Second, 3)

    ctx, cancel := context.WithTimeout(context.Background(), 15*time.Second)
    defer cancel()

    var user map[string]any
    fmt.Println("调用 user-service:")
    err := client.Get(ctx, "/api/users/1", &user)
    if err != nil {
        fmt.Println("最终失败:", err)
    } else {
        fmt.Println("成功:", user)
    }
}
```

## 5. 消息队列

### 5.1 使用 Kafka

```bash
go get github.com/segmentio/kafka-go
```

```go
package main

import (
    "context"
    "encoding/json"
    "fmt"
    "log"
    "sync"
    "time"

    "github.com/segmentio/kafka-go"
)

// 事件定义
type OrderEvent struct {
    OrderID   string    `json:"order_id"`
    UserID    string    `json:"user_id"`
    Amount    float64   `json:"amount"`
    Action    string    `json:"action"` // created, paid, shipped, completed
    Timestamp time.Time `json:"timestamp"`
}

// ==================== 生产者 ====================

type EventPublisher struct {
    writer *kafka.Writer
}

func NewEventPublisher(brokers []string, topic string) *EventPublisher {
    return &EventPublisher{
        writer: &kafka.Writer{
            Addr:         kafka.TCP(brokers...),
            Topic:        topic,
            Balancer:     &kafka.LeastBytes{},
            BatchTimeout: 10 * time.Millisecond,
        },
    }
}

func (p *EventPublisher) Publish(ctx context.Context, event OrderEvent) error {
    data, _ := json.Marshal(event)

    err := p.writer.WriteMessages(ctx, kafka.Message{
        Key:   []byte(event.OrderID), // 相同订单发到同一分区
        Value: data,
    })
    if err != nil {
        return fmt.Errorf("发送消息失败: %w", err)
    }

    fmt.Printf("[生产者] 发送: %s -> %s ¥%.2f\n", event.OrderID, event.Action, event.Amount)
    return nil
}

func (p *EventPublisher) Close() error {
    return p.writer.Close()
}

// ==================== 消费者 ====================

type EventConsumer struct {
    reader  *kafka.Reader
    handler func(OrderEvent) error
}

func NewEventConsumer(brokers []string, topic, groupID string, handler func(OrderEvent) error) *EventConsumer {
    return &EventConsumer{
        reader: kafka.NewReader(kafka.ReaderConfig{
            Brokers:     brokers,
            Topic:       topic,
            GroupID:     groupID,
            StartOffset: kafka.FirstOffset,
            MaxWait:     500 * time.Millisecond,
        }),
        handler: handler,
    }
}

func (c *EventConsumer) Start(ctx context.Context) {
    fmt.Printf("[消费者] 开始监听...\n")
    for {
        msg, err := c.reader.ReadMessage(ctx)
        if err != nil {
            if ctx.Err() != nil {
                return // 上下文取消
            }
            log.Printf("[消费者] 读取错误: %v", err)
            continue
        }

        var event OrderEvent
        if err := json.Unmarshal(msg.Value, &event); err != nil {
            log.Printf("[消费者] 解析错误: %v", err)
            continue
        }

        if err := c.handler(event); err != nil {
            log.Printf("[消费者] 处理错误: %v", err)
            // 实际场景可放入死信队列
        }
    }
}

func (c *EventConsumer) Close() error {
    return c.reader.Close()
}

func main() {
    brokers := []string{"localhost:9092"}
    topic := "order-events"
    ctx, cancel := context.WithTimeout(context.Background(), 30*time.Second)
    defer cancel()

    // 生产者
    publisher := NewEventPublisher(brokers, topic)
    defer publisher.Close()

    // 消费者1：通知服务
    notifyConsumer := NewEventConsumer(brokers, topic, "notify-group", func(e OrderEvent) error {
        fmt.Printf("  [通知服务] 订单 %s %s, 发送通知给用户 %s\n", e.OrderID, e.Action, e.UserID)
        return nil
    })
    defer notifyConsumer.Close()

    // 消费者2：统计服务
    statsConsumer := NewEventConsumer(brokers, topic, "stats-group", func(e OrderEvent) error {
        fmt.Printf("  [统计服务] 记录: 订单 %s 金额 ¥%.2f\n", e.OrderID, e.Amount)
        return nil
    })
    defer statsConsumer.Close()

    // 启动消费者
    var wg sync.WaitGroup
    wg.Add(2)
    go func() { defer wg.Done(); notifyConsumer.Start(ctx) }()
    go func() { defer wg.Done(); statsConsumer.Start(ctx) }()

    // 发送事件
    time.Sleep(time.Second)
    events := []OrderEvent{
        {OrderID: "ORD-001", UserID: "U100", Amount: 99.9, Action: "created", Timestamp: time.Now()},
        {OrderID: "ORD-001", UserID: "U100", Amount: 99.9, Action: "paid", Timestamp: time.Now()},
        {OrderID: "ORD-002", UserID: "U200", Amount: 299.0, Action: "created", Timestamp: time.Now()},
    }
    for _, e := range events {
        publisher.Publish(ctx, e)
        time.Sleep(100 * time.Millisecond)
    }

    time.Sleep(2 * time.Second)
    cancel()
    wg.Wait()
}
```

### 6.1 令牌桶限流器

```go
package main

import (
    "context"
    "fmt"
    "sync"
    "time"

    "golang.org/x/time/rate"

func main() {
    // ========== 基本令牌桶 ==========
    // 每秒 5 个令牌，桶容量 10（允许短时间突发）
    limiter := rate.NewLimiter(5, 10)

    fmt.Println("=== 令牌桶限流 (5/s, burst=10) ===")

    // Allow：非阻塞，有令牌返回 true
    for i := 0; i < 15; i++ {
        if limiter.Allow() {
            fmt.Printf("  请求 %d: 通过\n", i+1)
        } else {
            fmt.Printf("  请求 %d: 被限流\n", i+1)
        }
    }

    fmt.Println()

    // Wait：阻塞等待令牌
    fmt.Println("=== Wait 模式（阻塞等待）===")
    limiter2 := rate.NewLimiter(2, 1) // 每秒 2 个，桶容量 1
    ctx, cancel := context.WithTimeout(context.Background(), 3*time.Second)
    defer cancel()

    for i := 0; i < 5; i++ {
        start := time.Now()
        err := limiter2.Wait(ctx)
        if err != nil {
            fmt.Printf("  请求 %d: 超时 (%v)\n", i+1, err)
            break
        }
        fmt.Printf("  请求 %d: 通过 (等待 %v)\n", i+1, time.Since(start).Round(time.Millisecond))
    }

    fmt.Println()

    // ========== 按 Key 限流（如按 IP / 用户 ID）==========
    fmt.Println("=== 按 IP 限流 ===")
    ipLimiter := NewIPLimiter(rate.Limit(2), 5)

    ips := []string{"192.168.1.1", "192.168.1.1", "192.168.1.1",
        "10.0.0.1", "192.168.1.1", "192.168.1.1", "192.168.1.1"}

    for i, ip := range ips {
        allowed := ipLimiter.Allow(ip)
        fmt.Printf("  [%s] 请求 %d: %v\n", ip, i+1, map[bool]string{true: "通过", false: "限流"}[allowed])
    }
}

// 按 Key 限流器
type IPLimiter struct {
    mu       sync.RWMutex
    limiters map[string]*rate.Limiter
    rate     rate.Limit
    burst    int
}

func NewIPLimiter(r rate.Limit, burst int) *IPLimiter {
    return &IPLimiter{
        limiters: make(map[string]*rate.Limiter),
        rate:     r,
        burst:    burst,
    }
}

func (l *IPLimiter) getLimiter(key string) *rate.Limiter {
    l.mu.RLock()
    limiter, exists := l.limiters[key]
    l.mu.RUnlock()
    if exists {
        return limiter
    }

    l.mu.Lock()
    defer l.mu.Unlock()
    limiter = rate.NewLimiter(l.rate, l.burst)
    l.limiters[key] = limiter
    return limiter
}

func (l *IPLimiter) Allow(key string) bool {
    return l.getLimiter(key).Allow()
}
```

### 6.2 熔断器

```go
package main

import (
    "errors"
    "fmt"
    "sync"
    "time"
)

// 熔断器状态
type State int

const (
    StateClosed   State = iota // 关闭（正常通过）
    StateOpen                   // 打开（直接拒绝）
    StateHalfOpen               // 半开（允许少量试探）
)

func (s State) String() string {
    return [...]string{"Closed", "Open", "HalfOpen"}[s]
}

type CircuitBreaker struct {
    mu             sync.Mutex
    state          State
    failureCount   int
    successCount   int
    maxFailures    int           // 最大失败次数，超过则打开
    timeout        time.Duration // 打开后多久尝试半开
    halfOpenMax    int           // 半开状态最大试探次数
    lastFailedTime time.Time
}

func NewCircuitBreaker(maxFailures int, timeout time.Duration) *CircuitBreaker {
    return &CircuitBreaker{
        state:       StateClosed,
        maxFailures: maxFailures,
        timeout:     timeout,
        halfOpenMax: 3,
    }
}

func (cb *CircuitBreaker) Execute(fn func() error) error {
    cb.mu.Lock()

    // 检查是否应该从 Open 转为 HalfOpen
    if cb.state == StateOpen {
        if time.Since(cb.lastFailedTime) > cb.timeout {
            fmt.Printf("  [熔断器] Open -> HalfOpen\n")
            cb.state = StateHalfOpen
            cb.successCount = 0
            cb.failureCount = 0
        } else {
            cb.mu.Unlock()
            return errors.New("熔断器已打开，请求被拒绝")
        }
    }
    cb.mu.Unlock()

    // 执行函数
    err := fn()

    cb.mu.Lock()
    defer cb.mu.Unlock()

    if err != nil {
        cb.failureCount++
        cb.lastFailedTime = time.Now()

        switch cb.state {
        case StateClosed:
            if cb.failureCount >= cb.maxFailures {
                fmt.Printf("  [熔断器] Closed -> Open (失败 %d 次)\n", cb.failureCount)
                cb.state = StateOpen
            }
        case StateHalfOpen:
            fmt.Printf("  [熔断器] HalfOpen -> Open (试探失败)\n")
            cb.state = StateOpen
        }
        return err
    }

    // 成功
    switch cb.state {
    case StateHalfOpen:
        cb.successCount++
        if cb.successCount >= cb.halfOpenMax {
            fmt.Printf("  [熔断器] HalfOpen -> Closed (连续成功 %d 次)\n", cb.successCount)
            cb.state = StateClosed
            cb.failureCount = 0
        }
    case StateClosed:
        cb.failureCount = 0 // 成功则重置计数
    }

    return nil
}

func (cb *CircuitBreaker) State() State {
    cb.mu.Lock()
    defer cb.mu.Unlock()
    return cb.state
}

func main() {
    cb := NewCircuitBreaker(3, 2*time.Second) // 3次失败后熔断，2秒后试探

    callCount := 0
    mockService := func() error {
        callCount++
        if callCount <= 5 || callCount == 8 {
            return errors.New("服务不可用")
        }
        return nil
    }

    fmt.Println("=== 熔断器演示 ===")
    for i := 1; i <= 15; i++ {
        err := cb.Execute(mockService)
        status := "成功"
        if err != nil {
            status = err.Error()
        }
        fmt.Printf("请求 %2d: [%-9s] %s\n", i, cb.State(), status)

        if cb.State() == StateOpen {
            fmt.Println("  等待熔断恢复...")
            time.Sleep(2200 * time.Millisecond)
        }
    }
}
```

## 7. 分布式锁

### 7.1 基于 Redis 的分布式锁

```go
package main

import (
    "context"
    "errors"
    "fmt"
    "sync"
    "time"

    "github.com/redis/go-redis/v9"
)

var (
    ErrLockFailed  = errors.New("获取锁失败")
    ErrLockExpired = errors.New("锁已过期")
)

type RedisLock struct {
    client *redis.Client
    key    string
    value  string        // 唯一标识，防止误解锁
    expiry time.Duration
}

func NewRedisLock(client *redis.Client, key, value string, expiry time.Duration) *RedisLock {
    return &RedisLock{
        client: client,
        key:    "lock:" + key,
        value:  value,
        expiry: expiry,
    }
}

// 尝试获取锁
func (l *RedisLock) Lock(ctx context.Context) error {
    ok, err := l.client.SetNX(ctx, l.key, l.value, l.expiry).Result()
    if err != nil {
        return err
    }
    if !ok {
        return ErrLockFailed
    }
    return nil
}

// 带重试的获取锁
func (l *RedisLock) LockWithRetry(ctx context.Context, retries int, interval time.Duration) error {
    for i := 0; i < retries; i++ {
        err := l.Lock(ctx)
        if err == nil {
            return nil
        }
        if !errors.Is(err, ErrLockFailed) {
            return err
        }
        select {
        case <-time.After(interval):
        case <-ctx.Done():
            return ctx.Err()
        }
    }
    return ErrLockFailed
}

// 释放锁（Lua 脚本保证原子性：只释放自己的锁）
func (l *RedisLock) Unlock(ctx context.Context) error {
    script := `
        if redis.call("GET", KEYS[1]) == ARGV[1] then
            return redis.call("DEL", KEYS[1])
        else
            return 0
        end`

    result, err := l.client.Eval(ctx, script, []string{l.key}, l.value).Int64()
    if err != nil {
        return err
    }
    if result == 0 {
        return ErrLockExpired // 锁已被其他持有者获取
    }
    return nil
}

func main() {
    rdb := redis.NewClient(&redis.Options{Addr: "localhost:6379"})
    defer rdb.Close()

    ctx := context.Background()

    // 模拟秒杀：10 个 goroutine 抢同一个锁
    stock := 5
    var mu sync.WaitGroup

    fmt.Println("=== 分布式锁秒杀模拟 ===")
    fmt.Printf("库存: %d\n\n", stock)

    for i := 1; i <= 10; i++ {
        mu.Add(1)
        go func(userID int) {
            defer mu.Done()

            lock := NewRedisLock(rdb,
                "seckill:product:100",
                fmt.Sprintf("user_%d", userID),
                5*time.Second,
            )

            // 尝试获取锁
            err := lock.LockWithRetry(ctx, 3, 200*time.Millisecond)
            if err != nil {
                fmt.Printf("  用户%d: 获取锁失败，放弃\n", userID)
                return
            }
            defer lock.Unlock(ctx)

            // 模拟业务：检查库存 → 扣减
            if stock <= 0 {
                fmt.Printf("  用户%d: 库存不足\n", userID)
                return
            }

            stock--
            fmt.Printf("  用户%d: 抢购成功! 剩余库存: %d\n", userID, stock)
            time.Sleep(100 * time.Millisecond) // 模拟处理耗时
        }(i)
    }

    mu.Wait()
    fmt.Printf("\n最终库存: %d\n", stock)
}
```

```go
package main

import (
    "fmt"
    "log"
```

```go
    "time"

    "github.com/prometheus/client_golang/prometheus"
    "github.com/prometheus/client_golang/prometheus/promhttp"

// 定义指标
var (
    // 计数器：只增不减
    httpRequestsTotal = prometheus.NewCounterVec(
        prometheus.CounterOpts{
            Name: "http_requests_total",
            Help: "HTTP 请求总数",
        },
        []string{"method", "path", "status"},
    )

    // 直方图：请求耗时分布
    httpRequestDuration = prometheus.NewHistogramVec(
        prometheus.HistogramOpts{
            Name:    "http_request_duration_seconds",
            Help:    "HTTP 请求耗时（秒）",
            Buckets: []float64{0.01, 0.05, 0.1, 0.25, 0.5, 1, 2.5, 5},
        },
        []string{"method", "path"},
    )

    // 仪表盘：当前连接数
    activeConnections = prometheus.NewGauge(
        prometheus.GaugeOpts{
            Name: "active_connections",
            Help: "当前活跃连接数",
        },
    )

    // 摘要：响应大小
    responseSize = prometheus.NewSummary(
        prometheus.SummaryOpts{
            Name:       "http_response_size_bytes",
            Help:       "HTTP 响应大小（字节）",

    prometheus.MustRegister(httpRequestsTotal)
    prometheus.MustRegister(httpRequestDuration)
    prometheus.MustRegister(activeConnections)
    prometheus.MustRegister(responseSize)
}

// 指标中间件
func MetricsMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        activeConnections.Inc()
        defer activeConnections.Dec()

        next.ServeHTTP(w, r)

        duration := time.Since(start).Seconds()
        httpRequestsTotal.WithLabelValues(r.Method, r.URL.Path, "200").Inc()
        httpRequestDuration.WithLabelValues(r.Method, r.URL.Path).Observe(duration)
        responseSize.Observe(float64(rand.IntN(1000) + 100))
    })
}

    mux := http.NewServeMux()

    mux.HandleFunc("/api/users", func(w http.ResponseWriter, r *http.Request) {
        time.Sleep(time.Duration(rand.IntN(100)) * time.Millisecond)
        fmt.Fprintln(w, `{"users": []}`)
    })

    mux.HandleFunc("/api/orders", func(w http.ResponseWriter, r *http.Request) {
        time.Sleep(time.Duration(rand.IntN(200)) * time.Millisecond)
        fmt.Fprintln(w, `{"orders": []}`)
    })

    // Prometheus 指标端点
    mux.Handle("/metrics", promhttp.Handler())

    handler := MetricsMiddleware(mux)

    fmt.Println("服务启动: http://localhost:8080")
    fmt.Println("指标端点: http://localhost:8080/metrics")
    log.Fatal(http.ListenAndServe(":8080", handler))
}
```

**Prometheus 指标类型：**

| 类型 | 说明 | 场景 |
|------|------|------|
| Counter | 只增不减的计数器 | 请求数、错误数、处理量 |
| Gauge | 可增可减的仪表盘 | 连接数、队列长度、温度 |
| Histogram | 值的分布（桶） | 请求耗时、响应大小 |
| Summary | 值的分位数 | 类似 Histogram，客户端计算分位数 |

**常用 PromQL 查询：**

```promql
# 每秒请求数（QPS）
rate(http_requests_total[5m])

# P99 延迟
histogram_quantile(0.99, rate(http_request_duration_seconds_bucket[5m]))

# 错误率
rate(http_requests_total{status=~"5.."}[5m]) / rate(http_requests_total[5m])
```

```go
package main

import (
    "context"
    "encoding/json"
    "fmt"
    "math/rand"
    "net/http"
    "sync"
    "time"

// ==================== 通用组件 ====================

// 简易服务注册表（内存版，演示用）
type Registry struct {
    mu       sync.RWMutex
    services map[string][]string
}

func NewRegistry() *Registry {
    return &Registry{services: make(map[string][]string)}
}

func (r *Registry) Register(name, addr string) {
    r.mu.Lock()
    defer r.mu.Unlock()
    r.services[name] = append(r.services[name], addr)
    fmt.Printf("[注册] %s -> %s\n", name, addr)
}

func (r *Registry) Discover(name string) string {
    r.mu.RLock()
    defer r.mu.RUnlock()
    addrs := r.services[name]
    if len(addrs) == 0 {
        return ""
    }
    return addrs[rand.IntN(len(addrs))] // 随机负载均衡
}

// ==================== 用户服务 ====================

type UserService struct {
    users map[int]map[string]any
}

func NewUserService() *UserService {
    return &UserService{
        users: map[int]map[string]any{
            1: {"id": 1, "name": "Alice", "email": "alice@test.com"},
            2: {"id": 2, "name": "Bob", "email": "bob@test.com"},
        },
    }
}

func (s *UserService) Handler() http.Handler {
    mux := http.NewServeMux()
    mux.HandleFunc("GET /users/{id}", func(w http.ResponseWriter, r *http.Request) {
        id := 0
        fmt.Sscanf(r.PathValue("id"), "%d", &id)

        time.Sleep(time.Duration(rand.IntN(50)) * time.Millisecond) // 模拟延迟

        user, ok := s.users[id]
        if !ok {
            w.WriteHeader(404)
            json.NewEncoder(w).Encode(map[string]string{"error": "用户不存在"})
            return
        }
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(user)
    })
    return mux
}

// ==================== 订单服务 ====================

type Order struct {
    ID       string  `json:"id"`
    UserID   int     `json:"user_id"`
    UserName string  `json:"user_name"`
    Product  string  `json:"product"`
    Amount   float64 `json:"amount"`
    Status   string  `json:"status"`
}

type OrderService struct {
    registry *Registry
    orders   []Order
    mu       sync.Mutex
}

func NewOrderService(registry *Registry) *OrderService {
    return &OrderService{registry: registry}
}

func (s *OrderService) Handler() http.Handler {
    mux := http.NewServeMux()

    mux.HandleFunc("POST /orders", func(w http.ResponseWriter, r *http.Request) {
        var input struct {
            UserID  int     `json:"user_id"`
            Product string  `json:"product"`
            Amount  float64 `json:"amount"`
        }
        json.NewDecoder(r.Body).Decode(&input)

        // 调用用户服务获取用户名
        userName, err := s.getUserName(r.Context(), input.UserID)
        if err != nil {
            w.WriteHeader(500)
            json.NewEncoder(w).Encode(map[string]string{"error": "获取用户信息失败: " + err.Error()})
            return
        }

        // 创建订单
        order := Order{
            ID:       fmt.Sprintf("ORD-%04d", rand.IntN(10000)),
            UserID:   input.UserID,
            UserName: userName,
            Product:  input.Product,
            Amount:   input.Amount,
            Status:   "created",
        }

        s.mu.Lock()
        s.orders = append(s.orders, order)
        s.mu.Unlock()

        w.Header().Set("Content-Type", "application/json")
        w.WriteHeader(201)
        json.NewEncoder(w).Encode(order)
    })

    mux.HandleFunc("GET /orders", func(w http.ResponseWriter, r *http.Request) {
        s.mu.Lock()
        orders := make([]Order, len(s.orders))
        copy(orders, s.orders)
        s.mu.Unlock()

        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(map[string]any{"orders": orders, "total": len(orders)})
    })

    return mux
}

// 服务间调用：订单服务 → 用户服务
func (s *OrderService) getUserName(ctx context.Context, userID int) (string, error) {
    addr := s.registry.Discover("user-service")
    if addr == "" {
        return "", fmt.Errorf("user-service 不可用")
    }

    url := fmt.Sprintf("http://%s/users/%d", addr, userID)
    req, _ := http.NewRequestWithContext(ctx, "GET", url, nil)
    resp, err := http.DefaultClient.Do(req)
    if err != nil {
        return "", err
    }
    defer resp.Body.Close()

    if resp.StatusCode != 200 {
        return "", fmt.Errorf("用户不存在 (status=%d)", resp.StatusCode)
    }

    var user map[string]any
    json.NewDecoder(resp.Body).Decode(&user)
    name, _ := user["name"].(string)
    return name, nil
}

// ==================== 启动 ====================

func main() {
    registry := NewRegistry()

    // 启动用户服务
    userSvc := NewUserService()
    go func() {
        http.ListenAndServe(":8081", userSvc.Handler())
    }()
    registry.Register("user-service", "localhost:8081")

    // 启动订单服务
    orderSvc := NewOrderService(registry)
    go func() {
        http.ListenAndServe(":8082", orderSvc.Handler())
    }()
    registry.Register("order-service", "localhost:8082")

    time.Sleep(500 * time.Millisecond) // 等待服务启动

    fmt.Println("\n=== 模拟请求 ===\n")

    // 创建订单（订单服务会调用用户服务获取用户名）
    createOrder := func(userID int, product string, amount float64) {
        body := fmt.Sprintf(`{"user_id":%d,"product":"%s","amount":%.2f}`, userID, product, amount)
        resp, err := http.Post("http://localhost:8082/orders", "application/json",
            strings.NewReader(body))
        if err != nil {
            fmt.Println("请求失败:", err)
            return
        }
        defer resp.Body.Close()

        var order Order
        json.NewDecoder(resp.Body).Decode(&order)
        fmt.Printf("创建订单: %s | 用户: %s | 商品: %s | ¥%.2f\n",
            order.ID, order.UserName, order.Product, order.Amount)
    }

    createOrder(1, "Go 语言圣经", 79.00)
    createOrder(2, "机械键盘", 599.00)
    createOrder(1, "显示器", 1999.00)

    // 查询订单列表
    fmt.Println("\n--- 订单列表 ---")
    resp, _ := http.Get("http://localhost:8082/orders")
    var result map[string]any
    json.NewDecoder(resp.Body).Decode(&result)
    resp.Body.Close()
    data, _ := json.MarshalIndent(result, "", "  ")
    fmt.Println(string(data))
}
```
