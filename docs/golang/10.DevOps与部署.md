# Go 语言 DevOps 与部署

## 1. Docker 容器化

### 1.1 基础 Dockerfile

```dockerfile
# 最简单的 Dockerfile
FROM golang:1.22

WORKDIR /app
COPY . .
RUN go build -o server .

EXPOSE 8080
CMD ["./server"]
```

这个镜像约 **800MB+**，因为包含了完整的 Go 工具链。下面用多阶段构建优化。

### 1.2 多阶段构建（推荐）

```dockerfile
# ==================== 构建阶段 ====================
FROM golang:1.22-alpine AS builder

# 设置 Go 环境
ENV CGO_ENABLED=0 \
    GOOS=linux \
    GOARCH=amd64

WORKDIR /build

# 先复制依赖文件，利用 Docker 缓存层
COPY go.mod go.sum ./
RUN go mod download

# 再复制源码并编译
COPY . .
RUN go build \
    -ldflags="-s -w -X main.Version=1.0.0 -X main.BuildTime=$(date +%Y%m%d%H%M%S)" \
    -o server ./cmd/server

# ==================== 运行阶段 ====================
FROM alpine:3.19

# 安装必要的运行时依赖
RUN apk --no-cache add ca-certificates tzdata

# 创建非 root 用户
RUN adduser -D -g '' appuser

WORKDIR /app

# 从构建阶段复制编译好的二进制
COPY --from=builder /build/server .
COPY --from=builder /build/config ./config

# 切换到非 root 用户
USER appuser

EXPOSE 8080

# 健康检查
HEALTHCHECK --interval=30s --timeout=3s --start-period=5s --retries=3 \
    CMD wget --no-verbose --tries=1 --spider http://localhost:8080/health || exit 1

ENTRYPOINT ["./server"]
```

```bash
# 构建镜像
docker build -t myapp:1.0.0 .

# 查看镜像大小
docker images myapp
```


**镜像大小对比：**

| 基础镜像 | 最终大小 | 说明 |
|----------|----------|------|
| `golang:1.22` | ~850MB | 包含完整工具链 |
| `golang:1.22-alpine` + `alpine` | ~18MB | 多阶段构建 |
| `scratch` | ~8MB | 最小镜像，无 shell |
| `distroless` | ~12MB | Google 提供，安全 |

### 1.3 使用 scratch 极简镜像

```dockerfile
FROM golang:1.22-alpine AS builder
ENV CGO_ENABLED=0
WORKDIR /build
COPY go.mod go.sum ./
RUN go mod download
COPY . .
RUN go build -ldflags="-s -w" -o server .

# scratch：空镜像，不包含任何东西
FROM scratch
# 需要手动复制 CA 证书（HTTPS 请求需要）
COPY --from=builder /etc/ssl/certs/ca-certificates.crt /etc/ssl/certs/
COPY --from=builder /build/server /server
EXPOSE 8080
ENTRYPOINT ["/server"]
```

### 2.2 Makefile 完整示例

```makefile
# 项目信息
APP_NAME    := myserver
VERSION     := $(shell git describe --tags --always --dirty 2>/dev/null || echo "dev")
BUILD_TIME  := $(shell date +%Y%m%d%H%M%S)
GIT_COMMIT  := $(shell git rev-parse --short HEAD 2>/dev/null || echo "unknown")
GO          := go
LDFLAGS     := -s -w \
    -X main.Version=$(VERSION) \
    -X main.BuildTime=$(BUILD_TIME) \
    -X main.GitCommit=$(GIT_COMMIT)

# 构建目录
BUILD_DIR   := ./build
CMD_DIR     := ./cmd/server

.PHONY: all build run test lint clean docker help

all: lint test build

## 构建
build:
	@echo "==> 构建 $(APP_NAME) $(VERSION)..."
	$(GO) build -ldflags="$(LDFLAGS)" -o $(BUILD_DIR)/$(APP_NAME) $(CMD_DIR)
	@echo "==> 完成: $(BUILD_DIR)/$(APP_NAME)"

## 跨平台构建
build-all:
	@echo "==> 构建所有平台..."
	GOOS=linux   GOARCH=amd64 $(GO) build -ldflags="$(LDFLAGS)" -o $(BUILD_DIR)/$(APP_NAME)-linux-amd64 $(CMD_DIR)
	GOOS=linux   GOARCH=arm64 $(GO) build -ldflags="$(LDFLAGS)" -o $(BUILD_DIR)/$(APP_NAME)-linux-arm64 $(CMD_DIR)
	GOOS=darwin  GOARCH=arm64 $(GO) build -ldflags="$(LDFLAGS)" -o $(BUILD_DIR)/$(APP_NAME)-darwin-arm64 $(CMD_DIR)
	GOOS=windows GOARCH=amd64 $(GO) build -ldflags="$(LDFLAGS)" -o $(BUILD_DIR)/$(APP_NAME)-windows-amd64.exe $(CMD_DIR)
	@ls -lh $(BUILD_DIR)/

## 运行
run:
	$(GO) run $(CMD_DIR)

## 测试
test:
	$(GO) test -v -race -cover ./...

## 基准测试
bench:
	$(GO) test -bench=. -benchmem ./...

## 代码检查
lint:
	$(GO) vet ./...
	@which golangci-lint > /dev/null 2>&1 && golangci-lint run || echo "golangci-lint 未安装，跳过"

## 格式化
fmt:
	$(GO) fmt ./...
	@which goimports > /dev/null 2>&1 && goimports -w . || true

## Docker 构建
docker:
	docker build -t $(APP_NAME):$(VERSION) .
	docker tag $(APP_NAME):$(VERSION) $(APP_NAME):latest

## Docker 推送
docker-push: docker
	docker push $(APP_NAME):$(VERSION)
	docker push $(APP_NAME):latest

## 清理
clean:
	rm -rf $(BUILD_DIR)
	$(GO) clean -cache

## 依赖整理
tidy:
	$(GO) mod tidy
	$(GO) mod verify

## 帮助
help:
	@echo "可用命令:"
	@echo "  make build       编译项目"
	@echo "  make build-all   交叉编译所有平台"
	@echo "  make run         运行项目"
	@echo "  make test        运行测试"
	@echo "  make bench       基准测试"
	@echo "  make lint        代码检查"
	@echo "  make fmt         格式化代码"
	@echo "  make docker      构建 Docker 镜像"
	@echo "  make clean       清理构建产物"
	@echo "  make tidy        整理依赖"
```

```yaml
          go-version: ${{ matrix.go-version }}

      - name: 运行测试

          DB_PORT: 5432

          DB_PASSWORD: test

        if: matrix.go-version == '1.22'

          file: ./coverage.out

  build:
    needs: test
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4

      - name: 安装 Go
        uses: actions/setup-go@v5
        with:
          go-version: "1.22"

      - name: 构建
        run: |
          CGO_ENABLED=0 go build \
            -ldflags="-s -w -X main.Version=${{ github.sha }}" \
            -o build/server ./cmd/server

        uses: actions/upload-artifact@v4
        with:
          name: server-binary
          path: build/server

    needs: build
    runs-on: ubuntu-latest
    if: github.ref == 'refs/heads/main'
    steps:
      - uses: actions/checkout@v4

      - name: 登录 Docker Hub

        with:
          username: ${{ secrets.DOCKER_USERNAME }}

      - name: 构建并推送
```

### 4.1 结构化日志（slog）

```go
package main

import (
    "context"
    "log/slog"
    "os"
    "time"
)

// 自定义 contextKey
type contextKey string

const requestIDKey contextKey = "request_id"

func main() {
    // JSON 格式日志（生产环境推荐）
    logger := slog.New(slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{
        Level: slog.LevelInfo,
    }))
    slog.SetDefault(logger)

    // 基本日志
    slog.Info("服务启动",
        "addr", ":8080",
        "version", "1.2.0",
    )

    // 带上下文的日志
    ctx := context.WithValue(context.Background(), requestIDKey, "req-abc-123")
    handleRequest(ctx, "/api/users", "GET")

    // 错误日志
    slog.Error("数据库连接失败",
        "host", "db.example.com",
        "port", 5432,
        "error", "connection refused",
        "retry_in", 5*time.Second,
    )

    // 带固定字段的子 logger
    userLogger := slog.With("service", "user-service", "env", "production")
    userLogger.Info("用户登录", "user_id", 12345, "ip", "192.168.1.1")
    userLogger.Warn("登录尝试过多", "user_id", 12345, "attempts", 5)

    // 分组
    slog.Info("HTTP 请求",
        slog.Group("request",
            slog.String("method", "POST"),
            slog.String("path", "/api/orders"),
        ),
        slog.Group("response",
            slog.Int("status", 201),
            slog.Duration("duration", 42*time.Millisecond),
        ),
    )
}

func handleRequest(ctx context.Context, path, method string) {
    reqID, _ := ctx.Value(requestIDKey).(string)
    start := time.Now()

    slog.InfoContext(ctx, "处理请求",
        "request_id", reqID,
        "method", method,
        "path", path,
    )

    // 模拟处理
    time.Sleep(50 * time.Millisecond)

    slog.InfoContext(ctx, "请求完成",
        "request_id", reqID,
        "method", method,
        "path", path,
        "duration_ms", time.Since(start).Milliseconds(),
        "status", 200,
    )
}
```

### 4.2 Prometheus 指标暴露

```go
package main

import (
    "fmt"
    "log"
    "math/rand"
    "net/http"
    "time"

    "github.com/prometheus/client_golang/prometheus"
    "github.com/prometheus/client_golang/prometheus/promauto"
    "github.com/prometheus/client_golang/prometheus/promhttp"
)

var (
    requestsTotal = promauto.NewCounterVec(
        prometheus.CounterOpts{
            Name: "http_requests_total",
            Help: "HTTP 请求总数",
        },
        []string{"method", "path", "status"},
    )

    requestDuration = promauto.NewHistogramVec(
        prometheus.HistogramOpts{
            Name:    "http_request_duration_seconds",
            Help:    "HTTP 请求耗时",
            Buckets: []float64{0.01, 0.05, 0.1, 0.25, 0.5, 1, 2.5},
        },
        []string{"method", "path"},
    )

    activeRequests = promauto.NewGauge(
        prometheus.GaugeOpts{
            Name: "http_active_requests",
            Help: "当前活跃请求数",
        },
    )

    dbPoolSize = promauto.NewGauge(
        prometheus.GaugeOpts{
            Name: "db_pool_active_connections",
            Help: "数据库连接池活跃连接数",
        },
    )
)

func metricsMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        activeRequests.Inc()
        defer activeRequests.Dec()

        next.ServeHTTP(w, r)

        duration := time.Since(start).Seconds()
        requestsTotal.WithLabelValues(r.Method, r.URL.Path, "200").Inc()
        requestDuration.WithLabelValues(r.Method, r.URL.Path).Observe(duration)
    })
}

func main() {
    mux := http.NewServeMux()

    mux.HandleFunc("GET /api/users", func(w http.ResponseWriter, r *http.Request) {
        time.Sleep(time.Duration(rand.IntN(100)) * time.Millisecond)
        fmt.Fprintln(w, `{"users":[]}`)
    })

    mux.HandleFunc("GET /health", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, `{"status":"ok"}`)
    })

    // Prometheus 指标端点
    mux.Handle("GET /metrics", promhttp.Handler())

    // 模拟数据库连接池指标
    go func() {
        for {
            dbPoolSize.Set(float64(rand.IntN(20) + 5))
            time.Sleep(5 * time.Second)
        }
    }()

    handler := metricsMiddleware(mux)
    fmt.Println("服务启动: :8080")
    fmt.Println("指标: http://localhost:8080/metrics")
    log.Fatal(http.ListenAndServe(":8080", handler))
}
```

### 4.4 健康检查与就绪探针

```go
package main

import (
    "context"
    "database/sql"
    "encoding/json"
    "fmt"
    "log"
    "net/http"
    "sync/atomic"
    "time"
)

type HealthChecker struct {
    db    *sql.DB
    ready atomic.Bool
}

type HealthStatus struct {
    Status    string            `json:"status"`
    Timestamp string            `json:"timestamp"`
    Checks    map[string]string `json:"checks"`
    Uptime    string            `json:"uptime,omitempty"`
}

var startTime = time.Now()

func (h *HealthChecker) LivenessHandler(w http.ResponseWriter, r *http.Request) {
    // 存活检查：服务进程是否正常
    status := HealthStatus{
        Status:    "ok",
        Timestamp: time.Now().Format(time.RFC3339),
        Uptime:    time.Since(startTime).Round(time.Second).String(),
    }
    w.Header().Set("Content-Type", "application/json")
    json.NewEncoder(w).Encode(status)
}

func (h *HealthChecker) ReadinessHandler(w http.ResponseWriter, r *http.Request) {
    // 就绪检查：服务是否可以接受流量
    checks := make(map[string]string)
    allHealthy := true

    // 检查数据库
    if h.db != nil {
        ctx, cancel := context.WithTimeout(r.Context(), 2*time.Second)
        defer cancel()
        if err := h.db.PingContext(ctx); err != nil {
            checks["database"] = "unhealthy: " + err.Error()
            allHealthy = false
        } else {
            checks["database"] = "healthy"
        }
    }

    // 检查是否就绪
    if !h.ready.Load() {
        checks["app"] = "not ready"
        allHealthy = false
    } else {
        checks["app"] = "ready"
    }

    status := HealthStatus{
        Timestamp: time.Now().Format(time.RFC3339),
        Checks:    checks,
    }

    if allHealthy {
        status.Status = "ok"
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(status)
    } else {
        status.Status = "unhealthy"
        w.Header().Set("Content-Type", "application/json")
        w.WriteHeader(http.StatusServiceUnavailable)
        json.NewEncoder(w).Encode(status)
    }
}

func main() {
    checker := &HealthChecker{}

    mux := http.NewServeMux()
    mux.HandleFunc("GET /healthz", checker.LivenessHandler)    // 存活
    mux.HandleFunc("GET /readyz", checker.ReadinessHandler)     // 就绪

    mux.HandleFunc("GET /", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, "Hello!")
    })

    // 模拟初始化过程
    go func() {
        log.Println("初始化中...")
        time.Sleep(3 * time.Second) // 模拟加载配置、预热缓存等
        checker.ready.Store(true)
        log.Println("就绪!")
    }()

    fmt.Println("服务启动: :8080")
    fmt.Println("  存活检查: GET /healthz")
    fmt.Println("  就绪检查: GET /readyz")
    log.Fatal(http.ListenAndServe(":8080", mux))
}
```

### 6.1 环境变量 + 配置文件

```go
package main

import (
    "encoding/json"
    "fmt"
    "os"
    "strconv"
    "time"
)

type Config struct {
    Server   ServerConfig   `json:"server"`
    Database DatabaseConfig `json:"database"`
    Redis    RedisConfig    `json:"redis"`
    Log      LogConfig      `json:"log"`
}

type ServerConfig struct {
    Host         string        `json:"host"`
    Port         int           `json:"port"`
    ReadTimeout  time.Duration `json:"read_timeout"`
    WriteTimeout time.Duration `json:"write_timeout"`
}

type DatabaseConfig struct {
    Host     string `json:"host"`
    Port     int    `json:"port"`
    User     string `json:"user"`
    Password string `json:"password"`
    Name     string `json:"name"`
    MaxConns int    `json:"max_conns"`
}

type RedisConfig struct {
    Addr     string `json:"addr"`
    Password string `json:"password"`
    DB       int    `json:"db"`
}

type LogConfig struct {
    Level  string `json:"level"`
    Format string `json:"format"` // json, text
}

// 默认配置
func DefaultConfig() *Config {
    return &Config{
        Server: ServerConfig{
            Host:         "0.0.0.0",
            Port:         8080,
            ReadTimeout:  10 * time.Second,
            WriteTimeout: 15 * time.Second,
        },
        Database: DatabaseConfig{
            Host:     "localhost",
            Port:     5432,
            User:     "app",
            Name:     "myapp",
            MaxConns: 25,
        },
        Redis: RedisConfig{
            Addr: "localhost:6379",
            DB:   0,
        },
        Log: LogConfig{
            Level:  "info",
            Format: "json",
        },
    }
}

// 从环境变量覆盖（环境变量优先级最高）
func (c *Config) LoadFromEnv() {
    if v := os.Getenv("SERVER_PORT"); v != "" {
        if port, err := strconv.Atoi(v); err == nil {
            c.Server.Port = port
        }
    }
    if v := os.Getenv("DB_HOST"); v != "" {
        c.Database.Host = v
    }
    if v := os.Getenv("DB_PORT"); v != "" {
        if port, err := strconv.Atoi(v); err == nil {
            c.Database.Port = port
        }
    }
    if v := os.Getenv("DB_USER"); v != "" {
        c.Database.User = v
    }
    if v := os.Getenv("DB_PASSWORD"); v != "" {
        c.Database.Password = v
    }
    if v := os.Getenv("DB_NAME"); v != "" {
        c.Database.Name = v
    }
    if v := os.Getenv("REDIS_ADDR"); v != "" {
        c.Redis.Addr = v
    }
    if v := os.Getenv("LOG_LEVEL"); v != "" {
        c.Log.Level = v
    }
}

// 从 JSON 文件加载
func (c *Config) LoadFromFile(path string) error {
    data, err := os.ReadFile(path)
    if err != nil {
        return err
    }
    return json.Unmarshal(data, c)
}

func main() {
    cfg := DefaultConfig()

    // 尝试加载配置文件
    if err := cfg.LoadFromFile("config.json"); err != nil {
        fmt.Println("配置文件未找到，使用默认值")
    }

    // 环境变量覆盖
    cfg.LoadFromEnv()

    // 打印最终配置（隐藏密码）
    fmt.Println("=== 最终配置 ===")
    fmt.Printf("  服务: %s:%d\n", cfg.Server.Host, cfg.Server.Port)
    fmt.Printf("  数据库: %s@%s:%d/%s (最大连接: %d)\n",
        cfg.Database.User, cfg.Database.Host, cfg.Database.Port,
        cfg.Database.Name, cfg.Database.MaxConns)
    fmt.Printf("  Redis: %s (DB: %d)\n", cfg.Redis.Addr, cfg.Redis.DB)
    fmt.Printf("  日志: level=%s, format=%s\n", cfg.Log.Level, cfg.Log.Format)
}
```

## 8. 本章小结

| 知识点 | 要点 |
|--------|------|
| 多阶段构建 | 构建阶段用 `golang:alpine`，运行阶段用 `alpine`/`scratch` |
| 镜像优化 | 多阶段构建 ~18MB，scratch ~8MB；非 root 用户；HEALTHCHECK |
| .dockerignore | 排除 `.git`、`*_test.go`、文档等不需要的文件 |
| Docker Compose | 本地开发环境编排，`depends_on` + `healthcheck` 保证启动顺序 |
| 交叉编译 | `GOOS`/`GOARCH` 指定目标平台，`CGO_ENABLED=0` 静态编译 |
| Makefile | 统一构建入口，`-ldflags` 注入版本信息 |
| GitHub Actions | 矩阵测试多版本、构建产物上传、Docker 镜像推送 |
| slog | JSON 结构化日志，`slog.With` 固定字段，`slog.Group` 分组 |
| Prometheus | Counter/Gauge/Histogram 指标，`/metrics` 端点 |
| PromQL | `rate()` 计算 QPS，`histogram_quantile()` 计算分位延迟 |
| 健康检查 | `/healthz` 存活探针，`/readyz` 就绪探针 |
| K8s Deployment | 滚动更新、资源限制、探针配置 |
| K8s Service | ClusterIP 内部负载均衡 |
| K8s Ingress | 外部流量入口，域名路由 |
| 配置管理 | 优先级：环境变量 > 配置文件 > 默认值 |
| 部署流程 | 开发 → CI → 构建镜像 → K8s 滚动更新 → 监控 → 回滚 |
