## 1. 后端功能集成

### 1.1 SQLite 数据库集成

桌面应用最常用的本地数据库是 SQLite。推荐使用 `modernc.org/sqlite`（纯 Go 实现，无需 CGO）。

**安装依赖**：

```bash
go get modernc.org/sqlite
go get github.com/jmoiron/sqlx  # 对 database/sql 的增强封装
```

**数据库服务封装**：

```go
package main

import (
    "context"
    "fmt"
    "time"

    "github.com/jmoiron/sqlx"
    _ "modernc.org/sqlite"
)

// 数据模型
type Note struct {
    ID        int    `db:"id"        json:"id"`
    Title     string `db:"title"     json:"title"`
    Content   string `db:"content"   json:"content"`
    CreatedAt string `db:"created_at" json:"createdAt"`
    UpdatedAt string `db:"updated_at" json:"updatedAt"`
}

type DBService struct {
    ctx context.Context
    db  *sqlx.DB
}

func NewDBService() *DBService {
    return &DBService{}
}

func (s *DBService) startup(ctx context.Context) {
    s.ctx = ctx

    // 打开（或创建）数据库文件
    db, err := sqlx.Open("sqlite", "app_data.db")
    if err != nil {
        fmt.Printf("数据库连接失败: %v\n", err)
        // 输出: 数据库连接失败: ...（仅在出错时）
        return
    }
    s.db = db

    // 创建表
    s.db.MustExec(`
        CREATE TABLE IF NOT EXISTS notes (
            id         INTEGER PRIMARY KEY AUTOINCREMENT,
            title      TEXT NOT NULL,
            content    TEXT DEFAULT '',
            created_at TEXT NOT NULL,
            updated_at TEXT NOT NULL
        )
    `)

    fmt.Println("数据库初始化完成")
    // 输出: 数据库初始化完成
}

func (s *DBService) shutdown(ctx context.Context) {
    if s.db != nil {
        s.db.Close()
        fmt.Println("数据库连接已关闭")
        // 输出: 数据库连接已关闭
    }
}

// 获取所有笔记
func (s *DBService) GetAllNotes() ([]Note, error) {
    var notes []Note
    err := s.db.Select(&notes, "SELECT * FROM notes ORDER BY updated_at DESC")
    if err != nil {
        return nil, fmt.Errorf("查询失败: %w", err)
    }
    fmt.Printf("查询到 %d 条笔记\n", len(notes))
    // 输出: 查询到 5 条笔记
    return notes, nil
}

// 创建笔记
func (s *DBService) CreateNote(title, content string) (Note, error) {
    now := time.Now().Format("2006-01-02 15:04:05")
    result, err := s.db.Exec(
        "INSERT INTO notes (title, content, created_at, updated_at) VALUES (?, ?, ?, ?)",
        title, content, now, now,
    )
    if err != nil {
        return Note{}, fmt.Errorf("创建失败: %w", err)
    }

    id, _ := result.LastInsertId()
    note := Note{
        ID:        int(id),
        Title:     title,
        Content:   content,
        CreatedAt: now,
        UpdatedAt: now,
    }
    fmt.Printf("创建笔记: [%d] %s\n", note.ID, note.Title)
    // 输出: 创建笔记: [1] Go学习笔记
    return note, nil
}

// 更新笔记
func (s *DBService) UpdateNote(id int, title, content string) error {
    now := time.Now().Format("2006-01-02 15:04:05")
    result, err := s.db.Exec(
        "UPDATE notes SET title = ?, content = ?, updated_at = ? WHERE id = ?",
        title, content, now, id,
    )
    if err != nil {
        return fmt.Errorf("更新失败: %w", err)
    }

    rows, _ := result.RowsAffected()
    if rows == 0 {
        return fmt.Errorf("笔记 #%d 不存在", id)
    }
    fmt.Printf("更新笔记: [%d] %s\n", id, title)
    // 输出: 更新笔记: [1] Go学习笔记（修订版）
    return nil
}

// 删除笔记
func (s *DBService) DeleteNote(id int) error {
    result, err := s.db.Exec("DELETE FROM notes WHERE id = ?", id)
    if err != nil {
        return fmt.Errorf("删除失败: %w", err)
    }

    rows, _ := result.RowsAffected()
    if rows == 0 {
        return fmt.Errorf("笔记 #%d 不存在", id)
    }
    fmt.Printf("删除笔记: #%d\n", id)
    // 输出: 删除笔记: #1
    return nil
}

// 搜索笔记
func (s *DBService) SearchNotes(keyword string) ([]Note, error) {
    var notes []Note
    query := "SELECT * FROM notes WHERE title LIKE ? OR content LIKE ? ORDER BY updated_at DESC"
    pattern := "%" + keyword + "%"
    err := s.db.Select(&notes, query, pattern, pattern)
    if err != nil {
        return nil, fmt.Errorf("搜索失败: %w", err)
    }
    fmt.Printf("搜索 \"%s\" 找到 %d 条结果\n", keyword, len(notes))
    // 输出: 搜索 "Go" 找到 3 条结果
    return notes, nil
}
```

### 1.2 文件系统操作

```go
package main

import (
    "context"
    "fmt"
    "os"
    "path/filepath"
    "strings"
)

type FileInfo struct {
    Name    string `json:"name"`
    Path    string `json:"path"`
    Size    int64  `json:"size"`
    IsDir   bool   `json:"isDir"`
    ModTime string `json:"modTime"`
}

type FileService struct {
    ctx context.Context
}

func NewFileService() *FileService {
    return &FileService{}
}

func (f *FileService) startup(ctx context.Context) {
    f.ctx = ctx
}

// 读取文件内容
func (f *FileService) ReadFile(path string) (string, error) {
    data, err := os.ReadFile(path)
    if err != nil {
        return "", fmt.Errorf("读取失败: %w", err)
    }
    fmt.Printf("读取文件: %s (%d 字节)\n", path, len(data))
    // 输出: 读取文件: C:\Users\docs\note.txt (256 字节)
    return string(data), nil
}

// 写入文件
func (f *FileService) WriteFile(path, content string) error {
    // 确保父目录存在
    dir := filepath.Dir(path)
    if err := os.MkdirAll(dir, 0755); err != nil {
        return fmt.Errorf("创建目录失败: %w", err)
    }

    err := os.WriteFile(path, []byte(content), 0644)
    if err != nil {
        return fmt.Errorf("写入失败: %w", err)
    }
    fmt.Printf("写入文件: %s (%d 字节)\n", path, len(content))
    // 输出: 写入文件: C:\Users\docs\note.txt (128 字节)
    return nil
}

// 列出目录内容
func (f *FileService) ListDirectory(dirPath string) ([]FileInfo, error) {
    entries, err := os.ReadDir(dirPath)
    if err != nil {
        return nil, fmt.Errorf("读取目录失败: %w", err)
    }

    var files []FileInfo
    for _, entry := range entries {
        // 跳过隐藏文件
        if strings.HasPrefix(entry.Name(), ".") {
            continue
        }

        info, err := entry.Info()
        if err != nil {
            continue
        }

        files = append(files, FileInfo{
            Name:    entry.Name(),
            Path:    filepath.Join(dirPath, entry.Name()),
            Size:    info.Size(),
            IsDir:   entry.IsDir(),
            ModTime: info.ModTime().Format("2006-01-02 15:04:05"),
        })
    }

    fmt.Printf("目录 %s 下有 %d 个文件/文件夹\n", dirPath, len(files))
    // 输出: 目录 C:\Users\docs 下有 12 个文件/文件夹
    return files, nil
}

// 获取文件/目录大小（递归）
func (f *FileService) GetSize(path string) (int64, error) {
    var totalSize int64
    err := filepath.Walk(path, func(_ string, info os.FileInfo, err error) error {
        if err != nil {
            return err
        }
        if !info.IsDir() {
            totalSize += info.Size()
        }
        return nil
    })
    if err != nil {
        return 0, err
    }
    fmt.Printf("路径 %s 总大小: %s\n", path, formatSize(totalSize))
    // 输出: 路径 C:\Users\project 总大小: 15.3 MB
    return totalSize, nil
}

// 检查文件/目录是否存在
func (f *FileService) Exists(path string) bool {
    _, err := os.Stat(path)
    exists := !os.IsNotExist(err)
    fmt.Printf("路径 %s 存在: %v\n", path, exists)
    // 输出: 路径 C:\Users\docs\note.txt 存在: true
    return exists
}

func formatSize(bytes int64) string {
    const (
        KB = 1024
        MB = KB * 1024
        GB = MB * 1024
    )
    switch {
    case bytes >= GB:
        return fmt.Sprintf("%.1f GB", float64(bytes)/float64(GB))
    case bytes >= MB:
        return fmt.Sprintf("%.1f MB", float64(bytes)/float64(MB))
    case bytes >= KB:
        return fmt.Sprintf("%.1f KB", float64(bytes)/float64(KB))
    default:
        return fmt.Sprintf("%d B", bytes)
    }
}
```

### 1.3 HTTP 客户端：调用外部 API

```go
package main

import (
    "context"
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "time"
)

type WeatherData struct {
    City    string  `json:"city"`
    Temp    float64 `json:"temp"`
    Desc    string  `json:"description"`
    Wind    float64 `json:"wind"`
}

type APIService struct {
    ctx    context.Context
    client *http.Client
}

func NewAPIService() *APIService {
    return &APIService{
        client: &http.Client{
            Timeout: 10 * time.Second, // 请求超时
        },
    }
}

func (a *APIService) startup(ctx context.Context) {
    a.ctx = ctx
}

// 通用 GET 请求
func (a *APIService) Get(url string) (string, error) {
    fmt.Printf("GET %s\n", url)
    // 输出: GET https://api.example.com/data

    resp, err := a.client.Get(url)
    if err != nil {
        return "", fmt.Errorf("请求失败: %w", err)
    }
    defer resp.Body.Close()

    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return "", fmt.Errorf("读取响应失败: %w", err)
    }

    fmt.Printf("响应状态: %d, 大小: %d 字节\n", resp.StatusCode, len(body))
    // 输出: 响应状态: 200, 大小: 1024 字节

    if resp.StatusCode != http.StatusOK {
        return "", fmt.Errorf("HTTP %d: %s", resp.StatusCode, string(body))
    }

    return string(body), nil
}

// 带请求头的 GET（如 API Key 鉴权）
func (a *APIService) GetWithAuth(url, apiKey string) (string, error) {
    req, err := http.NewRequest("GET", url, nil)
    if err != nil {
        return "", err
    }

    req.Header.Set("Authorization", "Bearer "+apiKey)
    req.Header.Set("Content-Type", "application/json")

    resp, err := a.client.Do(req)
    if err != nil {
        return "", fmt.Errorf("请求失败: %w", err)
    }
    defer resp.Body.Close()

    body, err := io.ReadAll(resp.Body)
    if err != nil {
        return "", err
    }

    fmt.Printf("API 响应: %s\n", string(body[:min(100, len(body))]))
    // 输出: API 响应: {"status":"ok","data":[...]}
    return string(body), nil
}

// 模拟获取天气
func (a *APIService) FetchWeather(city string) (WeatherData, error) {
    url := fmt.Sprintf("https://api.example.com/weather?city=%s", city)

    resp, err := a.client.Get(url)
    if err != nil {
        // 模拟返回示例数据（实际项目中应返回错误）
        fmt.Printf("API 请求失败，使用模拟数据: %v\n", err)
        return WeatherData{
            City: city,
            Temp: 22.5,
            Desc: "晴天",
            Wind: 3.2,
        }, nil
    }
    defer resp.Body.Close()

    var data WeatherData
    if err := json.NewDecoder(resp.Body).Decode(&data); err != nil {
        return WeatherData{}, fmt.Errorf("解析响应失败: %w", err)
    }

    fmt.Printf("天气数据: %s %.1f°C %s\n", data.City, data.Temp, data.Desc)
    // 输出: 天气数据: 北京 22.5°C 晴天
    return data, nil
}
```

### 1.4 配置管理

使用 Go 标准库实现轻量级配置管理（适合桌面应用场景）：

```go
package main

import (
    "context"
    "encoding/json"
    "fmt"
    "os"
    "path/filepath"
    "sync"
)

type AppConfig struct {
    Theme     string `json:"theme"`      // "light" | "dark" | "system"
    Language  string `json:"language"`    // "zh-CN" | "en-US"
    FontSize  int    `json:"fontSize"`
    AutoSave  bool   `json:"autoSave"`
    LastOpenDir string `json:"lastOpenDir"`
    WindowWidth  int  `json:"windowWidth"`
    WindowHeight int  `json:"windowHeight"`
}

type ConfigService struct {
    ctx      context.Context
    config   AppConfig
    filePath string
    mu       sync.RWMutex
}

func NewConfigService() *ConfigService {
    // 配置文件存储在用户目录下
    homeDir, _ := os.UserHomeDir()
    return &ConfigService{
        filePath: filepath.Join(homeDir, ".myapp", "config.json"),
        config: AppConfig{
            Theme:        "system",
            Language:     "zh-CN",
            FontSize:     14,
            AutoSave:     true,
            WindowWidth:  1024,
            WindowHeight: 768,
        },
    }
}

func (c *ConfigService) startup(ctx context.Context) {
    c.ctx = ctx
    c.load()
    fmt.Printf("配置加载完成: 主题=%s, 语言=%s, 字体=%d\n",
        c.config.Theme, c.config.Language, c.config.FontSize)
    // 输出: 配置加载完成: 主题=dark, 语言=zh-CN, 字体=16
}

// 获取全部配置
func (c *ConfigService) GetConfig() AppConfig {
    c.mu.RLock()
    defer c.mu.RUnlock()
    return c.config
}

// 更新配置
func (c *ConfigService) SetConfig(config AppConfig) error {
    c.mu.Lock()
    defer c.mu.Unlock()

    c.config = config
    if err := c.save(); err != nil {
        return fmt.Errorf("保存配置失败: %w", err)
    }

    fmt.Printf("配置已更新: %+v\n", config)
    // 输出: 配置已更新: {Theme:dark Language:zh-CN FontSize:16 AutoSave:true ...}
    return nil
}

// 获取单个配置项
func (c *ConfigService) GetTheme() string {
    c.mu.RLock()
    defer c.mu.RUnlock()
    return c.config.Theme
}

// 设置单个配置项
func (c *ConfigService) SetTheme(theme string) error {
    c.mu.Lock()
    defer c.mu.Unlock()

    c.config.Theme = theme
    fmt.Printf("主题已切换为: %s\n", theme)
    // 输出: 主题已切换为: dark
    return c.save()
}

func (c *ConfigService) load() {
    data, err := os.ReadFile(c.filePath)
    if err != nil {
        fmt.Println("配置文件不存在，使用默认配置")
        // 输出: 配置文件不存在，使用默认配置
        return
    }
    json.Unmarshal(data, &c.config)
}

func (c *ConfigService) save() error {
    // 确保目录存在
    dir := filepath.Dir(c.filePath)
    if err := os.MkdirAll(dir, 0755); err != nil {
        return err
    }

    data, err := json.MarshalIndent(c.config, "", "  ")
    if err != nil {
        return err
    }
    return os.WriteFile(c.filePath, data, 0644)
}
```

**创建 Store** — `frontend/src/stores/noteStore.ts`：

```typescript
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { GetAllNotes, CreateNote, UpdateNote, DeleteNote } from '../../wailsjs/go/main/DBService'
import { EventsOn } from '../../wailsjs/runtime/runtime'

export interface Note {
    id: number
    title: string
    content: string
    createdAt: string
    updatedAt: string
}

export const useNoteStore = defineStore('notes', () => {
    const notes = ref<Note[]>([])
    const loading = ref(false)
    const currentId = ref<number | null>(null)

    // 计算属性
    const currentNote = computed(() =>
        notes.value.find(n => n.id === currentId.value) || null
    )
    const noteCount = computed(() => notes.value.length)

    // 加载笔记列表
    async function fetchNotes() {
        loading.value = true
        try {
            notes.value = await GetAllNotes()
            console.log(`加载了 ${notes.value.length} 条笔记`)
            // 输出: 加载了 5 条笔记
        } catch (err) {
            console.error('加载失败:', err)
            // 输出: 加载失败: 查询失败: ...（仅在出错时）
        } finally {
            loading.value = false
        }
    }

    // 创建笔记
    async function addNote(title: string, content: string) {
        const note = await CreateNote(title, content)
        notes.value.unshift(note)
        currentId.value = note.id
        console.log(`新建笔记: ${note.title}`)
        // 输出: 新建笔记: 我的新笔记
        return note
    }

    // 保存笔记
    async function saveNote(id: number, title: string, content: string) {
        await UpdateNote(id, title, content)
        const idx = notes.value.findIndex(n => n.id === id)
        if (idx !== -1) {
            notes.value[idx] = { ...notes.value[idx], title, content }
        }
        console.log(`保存笔记: [${id}] ${title}`)
        // 输出: 保存笔记: [1] 我的新笔记
    }

    // 删除笔记
    async function removeNote(id: number) {
        await DeleteNote(id)
        notes.value = notes.value.filter(n => n.id !== id)
        if (currentId.value === id) {
            currentId.value = notes.value[0]?.id || null
        }
        console.log(`删除笔记: #${id}`)
        // 输出: 删除笔记: #1
    }

    return {
        notes, loading, currentId,
        currentNote, noteCount,
        fetchNotes, addNote, saveNote, removeNote
    }
})
```

### 2.3 暗色/亮色主题切换

**Go 端获取系统主题**（Windows）：

```go
import "github.com/wailsapp/wails/v2/pkg/options/windows"

// 在 options.App 中配置
Windows: &windows.Options{

    // Theme: windows.Dark,       // 强制深色

}

// 运行时切换（仅 Windows）
```

### 3.2 资源加载流程

```
前端请求资源（如 /index.html、/assets/main.js）
       │
       ▼
┌─────────────────────────────────────────────┐
│              AssetServer                     │
│                                              │
│  1. 先在 embed.FS (Assets) 中查找            │
│     ├── 找到 → 返回文件内容                   │

│                                              │
│  3. Middleware 可在整个链路中拦截             │
└─────────────────────────────────────────────┘
```

### 3.3 自定义 AssetServer Handler

当需要从磁盘动态加载文件（如用户上传的图片）时，可以自定义 Handler：

```go
package main

import (
    "embed"
    "fmt"
    "net/http"
    "os"
    "path/filepath"
    "strings"
)

//go:embed all:frontend/dist
var assets embed.FS

// 自定义文件加载器 — 从磁盘读取文件
type DiskFileLoader struct {
    basePath string // 磁盘上的基础路径
}

func NewDiskFileLoader(basePath string) *DiskFileLoader {
    return &DiskFileLoader{basePath: basePath}
}

func (h *DiskFileLoader) ServeHTTP(w http.ResponseWriter, r *http.Request) {
    // 只处理以 /local/ 开头的请求
    if !strings.HasPrefix(r.URL.Path, "/local/") {
        w.WriteHeader(http.StatusNotFound)
        return
    }

    // 构造磁盘文件路径
    relativePath := strings.TrimPrefix(r.URL.Path, "/local/")
    filePath := filepath.Join(h.basePath, relativePath)

    fmt.Printf("从磁盘加载文件: %s\n", filePath)
    // 输出: 从磁盘加载文件: C:\Users\myapp\data\image.png

    data, err := os.ReadFile(filePath)
    if err != nil {
        w.WriteHeader(http.StatusNotFound)
        w.Write([]byte(fmt.Sprintf("文件不存在: %s", relativePath)))
        fmt.Printf("文件未找到: %s\n", filePath)
        // 输出: 文件未找到: C:\Users\myapp\data\missing.png
        return
    }

    // 根据扩展名设置 Content-Type
    ext := filepath.Ext(filePath)
    contentTypes := map[string]string{
        ".html": "text/html",
        ".css":  "text/css",
        ".js":   "application/javascript",
        ".json": "application/json",
        ".png":  "image/png",
        ".jpg":  "image/jpeg",
        ".svg":  "image/svg+xml",
        ".pdf":  "application/pdf",
    }
    if ct, ok := contentTypes[ext]; ok {
        w.Header().Set("Content-Type", ct)
    }

    w.Write(data)
    fmt.Printf("已返回文件: %s (%d 字节)\n", relativePath, len(data))
    // 输出: 已返回文件: image.png (45678 字节)
}
```

## 4. 后台任务与并发

### 4.1 Goroutine 后台任务 + 事件进度通知

```go
package main

import (
    "context"
    "fmt"
    "sync"
    "time"

    "github.com/wailsapp/wails/v2/pkg/runtime"
)

type TaskStatus struct {
    ID       string  `json:"id"`
    Name     string  `json:"name"`
    Progress float64 `json:"progress"` // 0-100
    Status   string  `json:"status"`   // "running" | "completed" | "cancelled" | "error"
    Message  string  `json:"message"`
}

type TaskManager struct {
    ctx     context.Context
    tasks   map[string]context.CancelFunc // 任务 ID → 取消函数
    mu      sync.Mutex
}

func NewTaskManager() *TaskManager {
    return &TaskManager{
        tasks: make(map[string]context.CancelFunc),
    }
}

func (t *TaskManager) startup(ctx context.Context) {
    t.ctx = ctx
}

// 启动后台任务
func (t *TaskManager) StartTask(taskId, taskName string, totalSteps int) {
    // 创建可取消的 context
    taskCtx, cancel := context.WithCancel(t.ctx)

    t.mu.Lock()
    t.tasks[taskId] = cancel
    t.mu.Unlock()

    fmt.Printf("启动任务: [%s] %s (%d 步)\n", taskId, taskName, totalSteps)
    // 输出: 启动任务: [task-1] 数据导入 (100 步)

    // 在 goroutine 中执行后台任务
    go func() {
        defer func() {
            t.mu.Lock()
            delete(t.tasks, taskId)
            t.mu.Unlock()
        }()

        for i := 1; i <= totalSteps; i++ {
            // 检查是否被取消
            select {
            case <-taskCtx.Done():
                fmt.Printf("任务 [%s] 已取消，进度: %d/%d\n", taskId, i, totalSteps)
                // 输出: 任务 [task-1] 已取消，进度: 35/100
                runtime.EventsEmit(t.ctx, "task:update", TaskStatus{
                    ID:      taskId,
                    Name:    taskName,
                    Progress: float64(i) / float64(totalSteps) * 100,
                    Status:  "cancelled",
                    Message: "任务已被用户取消",
                })
                return
            default:
            }

            // 模拟耗时操作
            time.Sleep(100 * time.Millisecond)

            // 推送进度
            progress := float64(i) / float64(totalSteps) * 100
            runtime.EventsEmit(t.ctx, "task:update", TaskStatus{
                ID:       taskId,
                Name:     taskName,
                Progress: progress,
                Status:   "running",
                Message:  fmt.Sprintf("正在处理 %d/%d", i, totalSteps),
            })

            if i%20 == 0 {
                fmt.Printf("任务 [%s] 进度: %.0f%%\n", taskId, progress)
                // 输出: 任务 [task-1] 进度: 20%
                // 输出: 任务 [task-1] 进度: 40%
                // 输出: 任务 [task-1] 进度: 60%
                // 输出: 任务 [task-1] 进度: 80%
                // 输出: 任务 [task-1] 进度: 100%
            }
        }

        // 任务完成
        runtime.EventsEmit(t.ctx, "task:update", TaskStatus{
            ID:       taskId,
            Name:     taskName,
            Progress: 100,
            Status:   "completed",
            Message:  "任务完成",
        })

        fmt.Printf("任务 [%s] 完成!\n", taskId)
        // 输出: 任务 [task-1] 完成!
    }()
}

// 取消任务
func (t *TaskManager) CancelTask(taskId string) {
    t.mu.Lock()
    defer t.mu.Unlock()

    if cancel, ok := t.tasks[taskId]; ok {
        cancel()
        fmt.Printf("正在取消任务: [%s]\n", taskId)
        // 输出: 正在取消任务: [task-1]
    } else {
        fmt.Printf("任务不存在或已完成: [%s]\n", taskId)
        // 输出: 任务不存在或已完成: [task-1]
    }
}

// 获取正在运行的任务数
func (t *TaskManager) GetRunningCount() int {
    t.mu.Lock()
    defer t.mu.Unlock()
    count := len(t.tasks)
    fmt.Printf("正在运行的任务数: %d\n", count)
    // 输出: 正在运行的任务数: 2
    return count
}
```

### 4.2 并发安全：保护共享状态

当多个 goroutine 访问同一数据时，必须用互斥锁保护：

```go
type Counter struct {
    ctx   context.Context
    value int
    mu    sync.Mutex // 互斥锁
}

func (c *Counter) startup(ctx context.Context) {
    c.ctx = ctx
}

// ✅ 正确 — 使用锁保护
func (c *Counter) Increment() int {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.value++
    fmt.Printf("计数器: %d\n", c.value)
    // 输出: 计数器: 1
    // 输出: 计数器: 2（并发调用时，每次递增 1，不会跳数）
    return c.value
}

// ❌ 错误 — 没有锁，并发不安全
// func (c *Counter) IncrementUnsafe() int {
//     c.value++ // 多个 goroutine 同时执行时会产生竞态条件
//     return c.value
// }

func (c *Counter) GetValue() int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.value
}
```

**使用 `sync.RWMutex` 优化读多写少的场景**：

```go
type Cache struct {
    data map[string]string
    mu   sync.RWMutex // 读写锁
}

// 读操作 — 使用读锁（允许多个并发读）
func (c *Cache) Get(key string) (string, bool) {
    c.mu.RLock()
    defer c.mu.RUnlock()
    val, ok := c.data[key]
    return val, ok
}

// 写操作 — 使用写锁（独占）
func (c *Cache) Set(key, value string) {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.data[key] = value
    fmt.Printf("缓存设置: %s = %s\n", key, value)
    // 输出: 缓存设置: user:name = 张三
}
```

### 5.2 自定义日志 — 集成 zerolog 输出到文件

```go
package main

import (
    "fmt"
    "io"
    "os"
    "path/filepath"
    "time"

    "github.com/rs/zerolog"
    "github.com/rs/zerolog/log"
    wailslogger "github.com/wailsapp/wails/v2/pkg/logger"
)

// 实现 Wails 的 logger.Logger 接口
type ZerologAdapter struct {
    logger zerolog.Logger
}

func NewZerologAdapter(logDir string) *ZerologAdapter {
    // 确保日志目录存在
    os.MkdirAll(logDir, 0755)

    // 日志文件（按日期命名）
    logFile := filepath.Join(logDir, time.Now().Format("2006-01-02")+".log")
    file, err := os.OpenFile(logFile, os.O_CREATE|os.O_WRONLY|os.O_APPEND, 0644)
    if err != nil {
        fmt.Printf("无法创建日志文件: %v\n", err)
        file = os.Stderr
    }

    // 同时输出到终端和文件
    multi := io.MultiWriter(
        zerolog.ConsoleWriter{Out: os.Stdout, TimeFormat: "15:04:05"}, // 终端（带颜色）
        file, // 文件
    )

    logger := zerolog.New(multi).With().Timestamp().Logger()

    fmt.Printf("日志文件: %s\n", logFile)
    // 输出: 日志文件: C:\Users\myapp\logs\2026-03-23.log

    return &ZerologAdapter{logger: logger}
```

### 5.3 应用内日志查看器

将日志同时推送到前端，方便用户查看：

```go
type LogService struct {
    ctx context.Context
}

func (l *LogService) startup(ctx context.Context) {
    l.ctx = ctx
}

// 自定义日志方法 — 同时写日志和推送到前端
func (l *LogService) Log(level, message string) {
    entry := map[string]interface{}{
        "level":   level,
        "message": message,
        "time":    time.Now().Format("15:04:05"),
    }

    // 写入日志
    switch level {
    case "info":
        runtime.LogInfo(l.ctx, message)
    case "warn":
        runtime.LogWarning(l.ctx, message)
    case "error":
        runtime.LogError(l.ctx, message)
    }

    // 推送到前端
    runtime.EventsEmit(l.ctx, "log:new", entry)

    fmt.Printf("[%s] %s: %s\n", entry["time"], level, message)
    // 输出: [10:30:15] info: 应用启动完成
}
```

前端日志面板：

```typescript
import { EventsOn } from '../../wailsjs/runtime/runtime'

const logs = ref<Array<{ level: string; message: string; time: string }>>([])

EventsOn("log:new", (entry) => {
    logs.value.push(entry)
    // 只保留最近 500 条
    if (logs.value.length > 500) {
        logs.value = logs.value.slice(-500)
    }
    console.log(`[${entry.time}] ${entry.level}: ${entry.message}`)
    // 输出: [10:30:15] info: 应用启动完成
})
```

## 6. 综合实战：笔记应用后端架构

将以上所有模块整合到 `main.go`：

```go
package main

import (
    "context"
    "embed"
    "log"
    "os"
    "path/filepath"

    "github.com/wailsapp/wails/v2"
    "github.com/wailsapp/wails/v2/pkg/options"
    "github.com/wailsapp/wails/v2/pkg/options/assetserver"
    wailslogger "github.com/wailsapp/wails/v2/pkg/logger"
)

//go:embed all:frontend/dist
var assets embed.FS

func main() {
    // 初始化各服务
    app := NewApp()
    dbService := NewDBService()
    fileService := NewFileService()
    configService := NewConfigService()
    taskManager := NewTaskManager()

    // 应用数据目录
    homeDir, _ := os.UserHomeDir()
    dataDir := filepath.Join(homeDir, ".myapp", "data")

    err := wails.Run(&options.App{
        Title:  "智能笔记",
        Width:  1200,
        Height: 800,
        AssetServer: &assetserver.Options{
            Assets:  assets,
            Handler: NewDiskFileLoader(dataDir),
        },
        LogLevel:           wailslogger.DEBUG,
        LogLevelProduction: wailslogger.ERROR,
        BackgroundColour:   &options.RGBA{R: 27, G: 38, B: 54, A: 1},
        OnStartup: func(ctx context.Context) {
            app.startup(ctx)
            dbService.startup(ctx)
            fileService.startup(ctx)
            configService.startup(ctx)
            taskManager.startup(ctx)
        },
        OnShutdown: func(ctx context.Context) {
            dbService.shutdown(ctx)
        },
        Bind: []interface{}{
            app,
            dbService,
            fileService,
            configService,
            taskManager,
        },
    })

    if err != nil {
        log.Fatal("应用启动失败:", err)
    }
}
```

## 小结

| 模块 | 核心库/技术 | 关键要点 |
|------|------------|----------|
| 数据库 | `modernc.org/sqlite` + `sqlx` | 纯 Go 无 CGO，使用 `db` tag 映射字段 |
| 文件操作 | `os` + `filepath` | `ReadFile`/`WriteFile`/`ReadDir`/`Walk` |
| HTTP 客户端 | `net/http` | 设置超时、自定义请求头、JSON 解析 |
| 配置管理 | `encoding/json` | 存储在用户目录、读写锁保护 |
| 前端状态管理 | Pinia | `defineStore` + composable 模式 |
| 前端路由 | Vue Router | **必须使用 Hash 模式** |
| 主题切换 | CSS 变量 + `data-theme` | Go 端 `WindowSetDarkTheme` / 前端 CSS 变量 |
| 资源嵌入 | `embed.FS` | `//go:embed all:frontend/dist` |
| 自定义 AssetServer | `http.Handler` | 处理磁盘文件加载、API 请求 |
| 后台任务 | Goroutine + context | `context.WithCancel` 取消、事件推送进度 |
| 并发安全 | `sync.Mutex` / `sync.RWMutex` | 读多写少用 `RWMutex` |
| 日志 | Wails 内置 / zerolog | 实现 `logger.Logger` 接口替换内置日志 |
