## 1. 事件系统（Events）

事件系统是 Wails 中 Go 后端与前端**双向通信**的核心机制。与绑定（Binding）不同，事件是异步的、发布-订阅模式的，适合推送通知、进度更新、广播消息等场景。

```
┌──────────────┐                     ┌──────────────────┐
│   Go 后端     │   EventsEmit ────► │   前端（WebView） │
│              │                     │                  │
│              │ ◄──── EventsEmit    │                  │
│              │                     │                  │
│  EventsOn    │   双向事件通信       │  EventsOn        │
│  EventsOff   │ ◄────────────────► │  EventsOff       │
│  EventsOnce  │                     │  EventsOnce      │
└──────────────┘                     └──────────────────┘
```

### 1.1 Go 端事件 API

所有事件 API 都在 `github.com/wailsapp/wails/v2/pkg/runtime` 包中。

```go
import "github.com/wailsapp/wails/v2/pkg/runtime"
```

#### EventsEmit — 发送事件

```go
// 发送不带数据的事件
runtime.EventsEmit(ctx, "app:started")

// 发送带单个数据的事件
runtime.EventsEmit(ctx, "user:login", "张三")

// 发送带多个数据的事件
runtime.EventsEmit(ctx, "file:saved", "/path/to/file.txt", 1024)

// 发送带结构化数据的事件
runtime.EventsEmit(ctx, "task:progress", map[string]interface{}{
    "taskId":   1,
    "progress": 75,
    "status":   "正在处理...",
})
```

#### EventsOn — 持续监听事件

```go
// 监听前端发来的事件（一直监听，直到调用 EventsOff）
runtime.EventsOn(ctx, "frontend:search", func(optionalData ...interface{}) {
    keyword := optionalData[0].(string)
    fmt.Printf("收到搜索请求: %s\n", keyword)
    // 输出: 收到搜索请求: Wails教程
})
```

#### EventsOnce — 只监听一次

```go
// 只触发一次，之后自动取消监听
runtime.EventsOnce(ctx, "app:firstLoad", func(optionalData ...interface{}) {
    fmt.Println("应用首次加载完成！")
    // 输出: 应用首次加载完成！
    // 后续再触发 "app:firstLoad" 不会再执行此回调
})
```

#### EventsOnMultiple — 监听指定次数

```go
// 最多触发 3 次后自动取消监听
runtime.EventsOnMultiple(ctx, "retry:attempt", func(optionalData ...interface{}) {
    attempt := optionalData[0].(float64) // JSON 数字默认为 float64
    fmt.Printf("第 %.0f 次重试\n", attempt)
}, 3)
// 输出（最多 3 次）:
// 第 1 次重试
// 第 2 次重试
// 第 3 次重试
```

#### EventsOff — 取消监听

```go
// 取消单个事件的监听
runtime.EventsOff(ctx, "frontend:search")

// 同时取消多个事件的监听
runtime.EventsOff(ctx, "event1", "event2", "event3")
```

### 1.2 前端事件 API

```typescript
import {
    EventsOn,
    EventsOnce,
    EventsOnMultiple,
    EventsOff,
    EventsEmit
} from '../wailsjs/runtime/runtime'
```

#### 前端监听 Go 事件

```typescript
// 持续监听
EventsOn("task:progress", (data: { taskId: number; progress: number; status: string }) => {
    console.log(`任务 ${data.taskId}: ${data.progress}% - ${data.status}`)
    // 输出: 任务 1: 75% - 正在处理...
})

// 只监听一次
EventsOnce("app:firstLoad", () => {
    console.log("首次加载完成")
    // 输出: 首次加载完成
})

// 监听指定次数
EventsOnMultiple("retry:attempt", (attempt: number) => {
    console.log(`第 ${attempt} 次重试`)
}, 3)

// 取消监听
EventsOff("task:progress")
```

#### 前端发送事件到 Go

```typescript
// 发送搜索事件给 Go
EventsEmit("frontend:search", "Wails教程")

// 发送带结构化数据的事件
EventsEmit("frontend:saveSettings", {
    theme: "dark",
    language: "zh-CN",
    fontSize: 14
})
```

### 1.3 完整案例：文件下载进度通知

**Go 端（模拟下载任务）**：

```go
package main

import (
    "context"
    "fmt"
    "time"

    "github.com/wailsapp/wails/v2/pkg/runtime"
)

type Downloader struct {
    ctx context.Context
}

func NewDownloader() *Downloader {
    return &Downloader{}
}

func (d *Downloader) startup(ctx context.Context) {
    d.ctx = ctx

    // 监听前端的下载请求
    runtime.EventsOn(ctx, "download:start", func(optionalData ...interface{}) {
        url := optionalData[0].(string)
        fmt.Printf("开始下载: %s\n", url)
        // 输出: 开始下载: https://example.com/file.zip
        go d.simulateDownload(url)
    })

    // 监听取消请求（只需要处理一次当前下载）
    runtime.EventsOn(ctx, "download:cancel", func(optionalData ...interface{}) {
        fmt.Println("下载已取消")
        // 输出: 下载已取消
    })
}

func (d *Downloader) simulateDownload(url string) {
    for i := 0; i <= 100; i += 10 {
        time.Sleep(500 * time.Millisecond)

        // 向前端推送进度
        runtime.EventsEmit(d.ctx, "download:progress", map[string]interface{}{
            "url":      url,
            "progress": i,
            "speed":    "2.5 MB/s",
        })
        fmt.Printf("下载进度: %d%%\n", i)
        // 输出:
        // 下载进度: 0%
        // 下载进度: 10%
        // 下载进度: 20%
        // ... (每 500ms 一次)
        // 下载进度: 100%
    }

    // 下载完成
    runtime.EventsEmit(d.ctx, "download:complete", map[string]interface{}{
        "url":      url,
        "fileSize": "128 MB",
        "savePath": "C:/Downloads/file.zip",
    })
    fmt.Println("下载完成!")
    // 输出: 下载完成!
}
```

**前端（Vue 3）**：

```vue
<script setup lang="ts">
import { ref, onMounted, onUnmounted } from 'vue'
import { EventsOn, EventsOff, EventsEmit } from '../wailsjs/runtime/runtime'

const progress = ref(0)
const speed = ref('')
const isDownloading = ref(false)
const message = ref('')

onMounted(() => {
    // 监听下载进度
    EventsOn("download:progress", (data: any) => {
        progress.value = data.progress
        speed.value = data.speed
        console.log(`进度: ${data.progress}%, 速度: ${data.speed}`)
        // 输出: 进度: 10%, 速度: 2.5 MB/s
        // 输出: 进度: 20%, 速度: 2.5 MB/s
        // ...
    })

    // 监听下载完成（只监听一次）
    EventsOnce("download:complete", (data: any) => {
        isDownloading.value = false
        message.value = `下载完成！保存到: ${data.savePath}`
        console.log(`文件大小: ${data.fileSize}`)
        // 输出: 文件大小: 128 MB
    })
})

onUnmounted(() => {
    // 组件销毁时取消监听
    EventsOff("download:progress")
    EventsOff("download:complete")
})

function startDownload() {
    isDownloading.value = true
    progress.value = 0
    EventsEmit("download:start", "https://example.com/file.zip")
}

function cancelDownload() {
    EventsEmit("download:cancel")
    isDownloading.value = false
}
</script>

<template>
    <div>
        <button @click="startDownload" :disabled="isDownloading">开始下载</button>
        <button @click="cancelDownload" :disabled="!isDownloading">取消</button>
        <div v-if="isDownloading">
            <progress :value="progress" max="100"></progress>
            <span>{{ progress }}% - {{ speed }}</span>
        </div>
        <p v-if="message">{{ message }}</p>
    </div>
</template>
```

### 1.4 事件 vs 绑定：何时用哪个？

| 场景 | 推荐方式 | 原因 |
|------|----------|------|
| 前端请求数据、调用后端方法 | **绑定** | 有明确的请求-响应，返回 Promise |
| 后端主动推送通知给前端 | **事件** | 绑定是前端发起的，后端不能主动调用前端 |
| 后台任务的进度更新 | **事件** | 持续推送多次，不是一次性的请求响应 |
| 广播消息（多个监听者） | **事件** | 事件支持多个监听者，绑定是一对一的 |
| 获取用户输入 / 表单提交 | **绑定** | 需要返回值（处理结果） |
| 系统状态变更通知 | **事件** | 后端检测到变化后主动推送 |

## 2. 窗口管理

Wails 提供了一套完整的运行时 API 来控制应用窗口。

### 2.1 窗口 API 一览

```go
import "github.com/wailsapp/wails/v2/pkg/runtime"
```

| API | 说明 |
|-----|------|
| `WindowSetTitle(ctx, title)` | 设置窗口标题 |
| `WindowSetSize(ctx, w, h)` | 设置窗口大小 |
| `WindowGetSize(ctx)` | 获取窗口大小，返回 `(int, int)` |
| `WindowSetMinSize(ctx, w, h)` | 设置最小尺寸 |
| `WindowSetMaxSize(ctx, w, h)` | 设置最大尺寸 |
| `WindowSetPosition(ctx, x, y)` | 设置窗口位置 |
| `WindowGetPosition(ctx)` | 获取窗口位置，返回 `(int, int)` |
| `WindowCenter(ctx)` | 窗口居中 |
| `WindowMaximise(ctx)` | 最大化 |
| `WindowUnmaximise(ctx)` | 取消最大化 |
| `WindowToggleMaximise(ctx)` | 切换最大化 |
| `WindowMinimise(ctx)` | 最小化 |
| `WindowUnminimise(ctx)` | 取消最小化 |
| `WindowFullscreen(ctx)` | 全屏 |
| `WindowUnfullscreen(ctx)` | 取消全屏 |
| `WindowIsMaximised(ctx)` | 是否最大化，返回 `bool` |
| `WindowIsMinimised(ctx)` | 是否最小化，返回 `bool` |
| `WindowIsNormal(ctx)` | 是否正常状态，返回 `bool` |
| `WindowIsFullscreen(ctx)` | 是否全屏，返回 `bool` |
| `WindowShow(ctx)` | 显示窗口 |
| `WindowHide(ctx)` | 隐藏窗口 |
| `WindowSetAlwaysOnTop(ctx, bool)` | 设置置顶 |
| `WindowSetDarkTheme(ctx)` | 深色主题（仅 Windows） |
| `WindowSetLightTheme(ctx)` | 浅色主题（仅 Windows） |
| `WindowReload(ctx)` | 刷新页面 |
| `WindowReloadApp(ctx)` | 重载应用前端 |
| `WindowExecJS(ctx, js)` | 执行 JS 代码 |

### 2.2 案例：窗口控制面板

**Go 端**：

```go
type WindowManager struct {
    ctx context.Context
}

func NewWindowManager() *WindowManager {
    return &WindowManager{}
}

func (w *WindowManager) startup(ctx context.Context) {
    w.ctx = ctx
}

// 获取窗口信息
func (w *WindowManager) GetWindowInfo() map[string]interface{} {
    width, height := runtime.WindowGetSize(w.ctx)
    x, y := runtime.WindowGetPosition(w.ctx)
    return map[string]interface{}{
        "width":        width,
        "height":       height,
        "x":            x,
        "y":            y,
        "isMaximised":  runtime.WindowIsMaximised(w.ctx),
        "isMinimised":  runtime.WindowIsMinimised(w.ctx),
        "isFullscreen": runtime.WindowIsFullscreen(w.ctx),
        "isNormal":     runtime.WindowIsNormal(w.ctx),
    }
    // 返回示例:
    // {
    //     "width": 1024, "height": 768,
    //     "x": 448, "y": 156,
    //     "isMaximised": false, "isMinimised": false,
    //     "isFullscreen": false, "isNormal": true
    // }
}

// 设置窗口大小
func (w *WindowManager) SetSize(width, height int) {
    runtime.WindowSetSize(w.ctx, width, height)
    runtime.WindowCenter(w.ctx) // 改变大小后居中
    fmt.Printf("窗口大小已设置为 %dx%d\n", width, height)
    // 输出: 窗口大小已设置为 800x600
}

// 设置窗口标题
func (w *WindowManager) SetTitle(title string) {
    runtime.WindowSetTitle(w.ctx, title)
    fmt.Printf("窗口标题已设置为: %s\n", title)
    // 输出: 窗口标题已设置为: 我的应用 - v2.0
}

// 切换全屏
func (w *WindowManager) ToggleFullscreen() bool {
    if runtime.WindowIsFullscreen(w.ctx) {
        runtime.WindowUnfullscreen(w.ctx)
        fmt.Println("退出全屏")
        // 输出: 退出全屏
        return false
    }
    runtime.WindowFullscreen(w.ctx)
    fmt.Println("进入全屏")
    // 输出: 进入全屏
    return true
}

// 切换置顶
func (w *WindowManager) SetAlwaysOnTop(onTop bool) {
    runtime.WindowSetAlwaysOnTop(w.ctx, onTop)
    fmt.Printf("窗口置顶: %v\n", onTop)
    // 输出: 窗口置顶: true
}

// 最小化
func (w *WindowManager) Minimise() {
    runtime.WindowMinimise(w.ctx)
}

// 切换最大化
func (w *WindowManager) ToggleMaximise() {
    runtime.WindowToggleMaximise(w.ctx)
}
```

**前端调用**：

```typescript
import {
    GetWindowInfo, SetSize, SetTitle,
    ToggleFullscreen, SetAlwaysOnTop
} from '../wailsjs/go/main/WindowManager'

// 获取窗口信息

console.log(info)
// 输出: { width: 1024, height: 768, x: 448, y: 156, isMaximised: false, ... }

// 调整窗口大小
await SetSize(800, 600)

// 修改标题
await SetTitle("我的应用 - v2.0")

// 切换全屏
const isFullscreen = await ToggleFullscreen()
console.log("全屏状态:", isFullscreen)
// 输出: 全屏状态: true
```

## 3. 原生对话框（Dialogs）

Wails 封装了操作系统的原生对话框，包括消息提示、文件选择、目录选择和文件保存。

### 3.1 消息对话框

```go
import "github.com/wailsapp/wails/v2/pkg/runtime"
```

**四种对话框类型**：

| 类型 | 常量 | 图标 |
|------|------|------|
| 信息 | `runtime.InfoDialog` | ℹ️ |
| 警告 | `runtime.WarningDialog` | ⚠️ |
| 错误 | `runtime.ErrorDialog` | ❌ |
| 问答 | `runtime.QuestionDialog` | ❓ |

**基本用法**：

```go
func (a *App) ShowInfo() {
    result, err := runtime.MessageDialog(a.ctx, runtime.MessageDialogOptions{
        Type:    runtime.InfoDialog,
        Title:   "提示",
        Message: "文件已保存成功！",
    })
    if err != nil {
        fmt.Println("对话框错误:", err)
        return
    }
    fmt.Println("用户点击了:", result)
    // 输出: 用户点击了: Ok
}

func (a *App) ShowWarning() {
    result, _ := runtime.MessageDialog(a.ctx, runtime.MessageDialogOptions{
        Type:    runtime.WarningDialog,
        Title:   "警告",
        Message: "磁盘空间不足，仅剩 500MB！",
    })
    fmt.Println("用户点击了:", result)
    // 输出: 用户点击了: Ok
}

func (a *App) ShowError() {
    result, _ := runtime.MessageDialog(a.ctx, runtime.MessageDialogOptions{
        Type:    runtime.ErrorDialog,
        Title:   "错误",
        Message: "无法连接到服务器，请检查网络连接。",
    })
    fmt.Println("用户点击了:", result)
    // 输出: 用户点击了: Ok
}
```

**确认对话框（带自定义按钮）**：

```go
func (a *App) ConfirmDelete(fileName string) bool {
    result, err := runtime.MessageDialog(a.ctx, runtime.MessageDialogOptions{
        Type:          runtime.QuestionDialog,
        Title:         "确认删除",
        Message:       fmt.Sprintf("确定要删除 \"%s\" 吗？此操作不可撤销。", fileName),
        Buttons:       []string{"删除", "取消"},
        DefaultButton: "取消",       // 默认焦点在「取消」上
        CancelButton:  "取消",       // ESC 键等同于「取消」
    })
    if err != nil {
        return false
    }

    fmt.Println("用户选择:", result)
    // 用户点击「删除」→ 输出: 用户选择: 删除
    // 用户点击「取消」→ 输出: 用户选择: 取消

    return result == "删除"
}
```

前端调用：

```typescript

if (confirmed) {

    // 输出: 执行删除

```

### 3.2 文件打开对话框

```go
// 打开单个文件
func (a *App) OpenFile() (string, error) {
    selection, err := runtime.OpenFileDialog(a.ctx, runtime.OpenDialogOptions{
        Title:            "选择文件",
        DefaultDirectory: "C:/Users",  // 默认打开的目录
        Filters: []runtime.FileFilter{
            {DisplayName: "文本文件 (*.txt, *.md)", Pattern: "*.txt;*.md"},
            {DisplayName: "图片 (*.png, *.jpg)", Pattern: "*.png;*.jpg;*.jpeg"},
            {DisplayName: "所有文件", Pattern: "*.*"},
        },
    })
    if err != nil {
        return "", err
    }

    if selection == "" {
        fmt.Println("用户取消了选择")
        // 输出: 用户取消了选择
        return "", nil
    }

    fmt.Println("选中的文件:", selection)
    // 输出: 选中的文件: C:\Users\Documents\readme.txt
    return selection, nil
}
```

```go

}
```

### 3.5 完整案例：简易文本编辑器的文件操作

```go
type Editor struct {
    ctx         context.Context
    currentFile string
    content     string
}

func NewEditor() *Editor {
    return &Editor{}
}

func (e *Editor) startup(ctx context.Context) {
    e.ctx = ctx
}

// 打开文件
func (e *Editor) Open() (string, error) {
    filePath, err := runtime.OpenFileDialog(e.ctx, runtime.OpenDialogOptions{
        Title: "打开文件",
        Filters: []runtime.FileFilter{
            {DisplayName: "文本文件", Pattern: "*.txt;*.md;*.go;*.js"},
            {DisplayName: "所有文件", Pattern: "*.*"},
        },
    })
    if err != nil {
        return "", err
    }
    if filePath == "" {
        return "", nil // 用户取消
    }

    data, err := os.ReadFile(filePath)
    if err != nil {
        return "", fmt.Errorf("读取文件失败: %w", err)
    }

    e.currentFile = filePath
    e.content = string(data)

    // 更新窗口标题
    runtime.WindowSetTitle(e.ctx, fmt.Sprintf("%s - 文本编辑器", filepath.Base(filePath)))

    fmt.Printf("已打开: %s (%d 字节)\n", filePath, len(data))
    // 输出: 已打开: C:\Users\Documents\notes.txt (1234 字节)

    return e.content, nil
}

// 保存文件
func (e *Editor) Save(content string) error {
    if e.currentFile == "" {
        return e.SaveAs(content)
    }

    err := os.WriteFile(e.currentFile, []byte(content), 0644)
    if err != nil {
        return fmt.Errorf("保存失败: %w", err)
    }

    e.content = content
    fmt.Printf("已保存: %s\n", e.currentFile)
    // 输出: 已保存: C:\Users\Documents\notes.txt

    runtime.EventsEmit(e.ctx, "file:saved", e.currentFile)
    return nil
}

// 另存为
func (e *Editor) SaveAs(content string) error {
    savePath, err := runtime.SaveFileDialog(e.ctx, runtime.SaveDialogOptions{
        Title:           "另存为",
        DefaultFilename: "untitled.txt",
        Filters: []runtime.FileFilter{
            {DisplayName: "文本文件", Pattern: "*.txt"},
            {DisplayName: "所有文件", Pattern: "*.*"},
        },
    })
    if err != nil {
        return err
    }
    if savePath == "" {
        return nil // 用户取消
    }

    err = os.WriteFile(savePath, []byte(content), 0644)
    if err != nil {
        return fmt.Errorf("保存失败: %w", err)
    }

    e.currentFile = savePath
    e.content = content
    runtime.WindowSetTitle(e.ctx, fmt.Sprintf("%s - 文本编辑器", filepath.Base(savePath)))

    fmt.Printf("另存为: %s\n", savePath)
    // 输出: 另存为: C:\Users\Documents\backup.txt

    return nil
}
```

### 4.2 完整菜单示例

```go
package main

import (
    "fmt"
    goruntime "runtime"

    "github.com/wailsapp/wails/v2/pkg/keys"
    "github.com/wailsapp/wails/v2/pkg/menu"
    wailsruntime "github.com/wailsapp/wails/v2/pkg/runtime"
)

func (a *App) createMenu() *menu.Menu {
    appMenu := menu.NewMenu()

    // macOS 必须在最前面添加应用菜单
    if goruntime.GOOS == "darwin" {
        appMenu.Append(menu.AppMenu())
    }

    // ===== 文件菜单 =====
    fileMenu := appMenu.AddSubmenu("文件")

    fileMenu.AddText("新建", keys.CmdOrCtrl("n"), func(cd *menu.CallbackData) {
        fmt.Println("菜单点击: 新建")
        // 输出: 菜单点击: 新建
        wailsruntime.EventsEmit(a.ctx, "menu:new")
    })

    fileMenu.AddText("打开", keys.CmdOrCtrl("o"), func(cd *menu.CallbackData) {
        fmt.Println("菜单点击: 打开")
        // 输出: 菜单点击: 打开
        wailsruntime.EventsEmit(a.ctx, "menu:open")
    })

    fileMenu.AddText("保存", keys.CmdOrCtrl("s"), func(cd *menu.CallbackData) {
        fmt.Println("菜单点击: 保存")
        // 输出: 菜单点击: 保存
        wailsruntime.EventsEmit(a.ctx, "menu:save")
    })

    fileMenu.AddSeparator()

    fileMenu.AddText("退出", keys.CmdOrCtrl("q"), func(cd *menu.CallbackData) {
        fmt.Println("菜单点击: 退出")
        // 输出: 菜单点击: 退出
        wailsruntime.Quit(a.ctx)
    })

    // ===== 编辑菜单 =====
    editMenu := appMenu.AddSubmenu("编辑")

    editMenu.AddText("撤销", keys.CmdOrCtrl("z"), func(cd *menu.CallbackData) {
        fmt.Println("菜单点击: 撤销")
        wailsruntime.EventsEmit(a.ctx, "menu:undo")
    })

    editMenu.AddText("重做", keys.Combo("z", keys.CmdOrCtrlKey, keys.ShiftKey), func(cd *menu.CallbackData) {
        fmt.Println("菜单点击: 重做")
        wailsruntime.EventsEmit(a.ctx, "menu:redo")
    })

    // macOS 上追加标准编辑菜单（Cmd+C/V/X）
    if goruntime.GOOS == "darwin" {
        appMenu.Append(menu.EditMenu())
    }

    // ===== 视图菜单 =====
    viewMenu := appMenu.AddSubmenu("视图")

    // 复选框菜单项
    sidebarItem := viewMenu.AddCheckbox("显示侧边栏", true, keys.CmdOrCtrl("b"),
        func(cd *menu.CallbackData) {
            fmt.Printf("侧边栏: %v\n", cd.MenuItem.Checked)
            // 第一次点击 → 输出: 侧边栏: false（取消勾选）
            // 再次点击 → 输出: 侧边栏: true（重新勾选）
            wailsruntime.EventsEmit(a.ctx, "menu:toggleSidebar", cd.MenuItem.Checked)
        })
    _ = sidebarItem // 可以保存引用，后续动态修改

    viewMenu.AddSeparator()

    // 单选按钮组（同一组中只能选一个）
    viewMenu.AddRadio("紧凑模式", false, nil, func(cd *menu.CallbackData) {
        fmt.Println("视图模式: 紧凑")
        // 输出: 视图模式: 紧凑
    })
    viewMenu.AddRadio("标准模式", true, nil, func(cd *menu.CallbackData) {
        fmt.Println("视图模式: 标准")
        // 输出: 视图模式: 标准
    })
    viewMenu.AddRadio("宽松模式", false, nil, func(cd *menu.CallbackData) {
        fmt.Println("视图模式: 宽松")
        // 输出: 视图模式: 宽松
    })

    // ===== 帮助菜单 =====
    helpMenu := appMenu.AddSubmenu("帮助")

    helpMenu.AddText("在线文档", nil, func(cd *menu.CallbackData) {
        wailsruntime.BrowserOpenURL(a.ctx, "https://wails.io/docs/introduction")
        fmt.Println("已在浏览器中打开文档")
        // 输出: 已在浏览器中打开文档
    })

    helpMenu.AddText("关于", nil, func(cd *menu.CallbackData) {
        wailsruntime.MessageDialog(a.ctx, wailsruntime.MessageDialogOptions{
            Type:    wailsruntime.InfoDialog,
            Title:   "关于",
            Message: "我的应用 v1.0.0\n基于 Wails 构建",
        })
    })

    return appMenu
}
```

### 4.4 动态更新菜单

通过保存菜单项引用，可以在运行时动态修改菜单：

```go
type App struct {
    ctx      context.Context
    saveItem *menu.MenuItem // 保存菜单项引用
}

func (a *App) createMenu() *menu.Menu {
    appMenu := menu.NewMenu()
    fileMenu := appMenu.AddSubmenu("文件")

    // 保存引用
    a.saveItem = fileMenu.AddText("保存", keys.CmdOrCtrl("s"), func(cd *menu.CallbackData) {
        // ...
    })
    a.saveItem.Disabled = true // 初始状态禁用

    return appMenu
}

// 当有内容修改时，启用保存按钮
func (a *App) OnContentChanged() {
    a.saveItem.Disabled = false
    wailsruntime.MenuUpdateApplicationMenu(a.ctx) // ⭐ 必须调用才能刷新菜单
    fmt.Println("保存按钮已启用")
    // 输出: 保存按钮已启用
}

// 保存完成后，禁用保存按钮
func (a *App) OnSaveComplete() {
    a.saveItem.Disabled = true
    wailsruntime.MenuUpdateApplicationMenu(a.ctx)
    fmt.Println("保存按钮已禁用")
    // 输出: 保存按钮已禁用
}
```

> **注意**：修改菜单项属性后，必须调用 `runtime.MenuUpdateApplicationMenu(ctx)` 才能使更改生效。

## 7. 综合实战：待办事项应用

下面是一个结合所有核心功能的完整案例。

### 7.1 Go 后端 `todo.go`

```go
package main

import (
    "context"
    "encoding/json"
    "fmt"
    "os"
    "time"

    "github.com/wailsapp/wails/v2/pkg/runtime"
)

type Todo struct {
    ID        int    `json:"id"`
    Title     string `json:"title"`
    Completed bool   `json:"completed"`
    CreatedAt string `json:"createdAt"`
}

type TodoService struct {
    ctx      context.Context
    todos    []Todo
    nextID   int
    filePath string
}

func NewTodoService() *TodoService {
    return &TodoService{
        todos:    []Todo{},
        nextID:   1,
        filePath: "todos.json",
    }
}

func (t *TodoService) startup(ctx context.Context) {
    t.ctx = ctx
    t.loadFromFile()
    fmt.Printf("已加载 %d 条待办事项\n", len(t.todos))
    // 输出: 已加载 3 条待办事项
}

func (t *TodoService) shutdown(ctx context.Context) {
    t.saveToFile()
    fmt.Println("待办事项已保存")
    // 输出: 待办事项已保存
}

// 获取所有待办
func (t *TodoService) GetAll() []Todo {
    fmt.Printf("返回 %d 条待办事项\n", len(t.todos))
    // 输出: 返回 3 条待办事项
    return t.todos
}

// 添加待办
func (t *TodoService) Add(title string) Todo {
    todo := Todo{
        ID:        t.nextID,
        Title:     title,
        Completed: false,
        CreatedAt: time.Now().Format("2006-01-02 15:04:05"),
    }
    t.nextID++
    t.todos = append(t.todos, todo)
    t.saveToFile()

    fmt.Printf("新增待办: [%d] %s\n", todo.ID, todo.Title)
    // 输出: 新增待办: [1] 学习 Wails 事件系统

    // 通过事件通知前端（如果有其他窗口也需要更新）
    runtime.EventsEmit(t.ctx, "todo:added", todo)
    return todo
}

// 切换完成状态
func (t *TodoService) Toggle(id int) (Todo, error) {
    for i := range t.todos {
        if t.todos[i].ID == id {
            t.todos[i].Completed = !t.todos[i].Completed
            t.saveToFile()
            status := "未完成"
            if t.todos[i].Completed {
                status = "已完成"
            }
            fmt.Printf("待办 [%d] %s → %s\n", id, t.todos[i].Title, status)
            // 输出: 待办 [1] 学习 Wails 事件系统 → 已完成
            return t.todos[i], nil
        }
    }
    return Todo{}, fmt.Errorf("待办 #%d 不存在", id)
}

// 删除待办（带确认对话框）
func (t *TodoService) Delete(id int) error {
    var title string
    for _, todo := range t.todos {
        if todo.ID == id {
            title = todo.Title
            break
        }
    }
    if title == "" {
        return fmt.Errorf("待办 #%d 不存在", id)
    }

    // 弹出确认对话框
    result, _ := runtime.MessageDialog(t.ctx, runtime.MessageDialogOptions{
        Type:          runtime.QuestionDialog,
        Title:         "确认删除",
        Message:       fmt.Sprintf("确定要删除「%s」吗？", title),
        Buttons:       []string{"删除", "取消"},
        DefaultButton: "取消",
    })

    if result != "删除" {
        fmt.Println("用户取消了删除")
        // 输出: 用户取消了删除
        return nil
    }

    // 执行删除
    newTodos := make([]Todo, 0, len(t.todos)-1)
    for _, todo := range t.todos {
        if todo.ID != id {
            newTodos = append(newTodos, todo)
        }
    }
    t.todos = newTodos
    t.saveToFile()

    fmt.Printf("已删除待办: [%d] %s\n", id, title)
    // 输出: 已删除待办: [1] 学习 Wails 事件系统

    runtime.EventsEmit(t.ctx, "todo:deleted", id)
    return nil
}

// 导出为文件
func (t *TodoService) Export() error {
    savePath, err := runtime.SaveFileDialog(t.ctx, runtime.SaveDialogOptions{
        Title:           "导出待办事项",
        DefaultFilename: "todos.json",
        Filters: []runtime.FileFilter{
            {DisplayName: "JSON 文件", Pattern: "*.json"},
        },
    })
    if err != nil {
        return err
    }
    if savePath == "" {
        return nil
    }

    data, _ := json.MarshalIndent(t.todos, "", "  ")
    err = os.WriteFile(savePath, data, 0644)
    if err != nil {
        return fmt.Errorf("导出失败: %w", err)
    }

    fmt.Printf("已导出到: %s\n", savePath)
    // 输出: 已导出到: C:\Users\Documents\todos.json

    runtime.MessageDialog(t.ctx, runtime.MessageDialogOptions{
        Type:    runtime.InfoDialog,
        Title:   "导出成功",
        Message: fmt.Sprintf("已导出 %d 条待办事项到:\n%s", len(t.todos), savePath),
    })
    return nil
}

// 复制待办列表到剪贴板
func (t *TodoService) CopyToClipboard() error {
    text := ""
    for _, todo := range t.todos {
        mark := "☐"
        if todo.Completed {
            mark = "☑"
        }
        text += fmt.Sprintf("%s %s\n", mark, todo.Title)
    }

    err := runtime.ClipboardSetText(t.ctx, text)
    if err != nil {
        return err
    }

    fmt.Printf("已复制 %d 条待办到剪贴板\n", len(t.todos))
    // 输出: 已复制 3 条待办到剪贴板
    // 剪贴板内容:
    // ☑ 学习 Wails 事件系统
    // ☐ 学习 Wails 窗口管理
    // ☐ 学习 Wails 对话框
    return nil
}

// 从文件加载
func (t *TodoService) loadFromFile() {
    data, err := os.ReadFile(t.filePath)
    if err != nil {
        return // 文件不存在，使用空列表
    }
    json.Unmarshal(data, &t.todos)
    for _, todo := range t.todos {
        if todo.ID >= t.nextID {
            t.nextID = todo.ID + 1
        }
    }
}

// 保存到文件
func (t *TodoService) saveToFile() {
    data, _ := json.MarshalIndent(t.todos, "", "  ")
    os.WriteFile(t.filePath, data, 0644)
}
```
