## 1. 文件与 I/O 操作

### 1.1 os 包：文件基础操作

```go
package main

import (
    "fmt"
    "os"
)

func main() {
    // ========== 创建并写入文件 ==========
    file, err := os.Create("demo.txt")
    if err != nil {
        fmt.Println("创建失败:", err)
        return
    }

    file.WriteString("第一行: Hello, Go!\n")
    file.WriteString("第二行: 文件操作示例\n")
    file.WriteString("第三行: 2024年\n")
    file.Close()
    fmt.Println("文件已创建")

    // ========== 读取整个文件 ==========
    data, err := os.ReadFile("demo.txt")
    if err != nil {
        fmt.Println("读取失败:", err)
        return
    }
    fmt.Println("文件内容:")
    fmt.Println(string(data))

    // ========== 文件信息 ==========
    info, _ := os.Stat("demo.txt")
    fmt.Println("文件名:", info.Name())
    fmt.Println("大小:", info.Size(), "字节")
    fmt.Println("修改时间:", info.ModTime().Format("2006-01-02 15:04:05"))
    fmt.Println("是否为目录:", info.IsDir())

    // ========== 判断文件是否存在 ==========
    if _, err := os.Stat("不存在.txt"); os.IsNotExist(err) {
        fmt.Println("\n文件不存在.txt 不存在")
    }

    // ========== 追加写入 ==========
    f, _ := os.OpenFile("demo.txt", os.O_APPEND|os.O_WRONLY, 0644)
    f.WriteString("第四行: 追加的内容\n")
    f.Close()

    data, _ = os.ReadFile("demo.txt")
    fmt.Println("追加后:")
    fmt.Println(string(data))

    // ========== 重命名与删除 ==========
    os.Rename("demo.txt", "demo_renamed.txt")
    fmt.Println("文件已重命名")

    os.Remove("demo_renamed.txt")
    fmt.Println("文件已删除")
}
```

### 1.2 bufio 包：带缓冲的 I/O

```go
package main

import (
    "bufio"
    "fmt"
    "os"
    "strings"
)

func main() {
    // 写入文件
    os.WriteFile("lines.txt", []byte(
        "Go 语言\nPython\nJavaScript\nRust\nTypeScript\n",
    ), 0644)

    // ========== 逐行读取 ==========
    file, _ := os.Open("lines.txt")
    defer file.Close()

    scanner := bufio.NewScanner(file)
    lineNum := 0
    fmt.Println("逐行读取:")
    for scanner.Scan() {
        lineNum++
        fmt.Printf("  %d: %s\n", lineNum, scanner.Text())
    }

    // ========== 按单词扫描 ==========
    text := "Hello World Go Language"
    wordScanner := bufio.NewScanner(strings.NewReader(text))
    wordScanner.Split(bufio.ScanWords)

    fmt.Print("\n单词: ")
    for wordScanner.Scan() {
        fmt.Printf("[%s] ", wordScanner.Text())
    }
    fmt.Println()

    // ========== bufio.Writer 带缓冲写入 ==========
    outFile, _ := os.Create("buffered.txt")
    writer := bufio.NewWriter(outFile)

    writer.WriteString("缓冲写入第一行\n")
    writer.WriteString("缓冲写入第二行\n")
    writer.Flush() // 必须刷新缓冲区
    outFile.Close()

    data, _ := os.ReadFile("buffered.txt")
    fmt.Println("\n缓冲写入结果:")
    fmt.Println(string(data))

    // 清理
    os.Remove("lines.txt")
    os.Remove("buffered.txt")
}
```

```
  5: TypeScript
```

### 1.3 目录操作

```go
package main

import (
    "fmt"
    "os"
    "path/filepath"
)

func main() {
    // 创建目录
    os.MkdirAll("testdir/sub1/deep", 0755)
    os.MkdirAll("testdir/sub2", 0755)
    os.WriteFile("testdir/a.txt", []byte("file a"), 0644)
    os.WriteFile("testdir/b.go", []byte("package main"), 0644)
    os.WriteFile("testdir/sub1/c.txt", []byte("file c"), 0644)
    os.WriteFile("testdir/sub1/deep/d.go", []byte("package deep"), 0644)

    // ========== 读取目录内容 ==========
    entries, _ := os.ReadDir("testdir")
    fmt.Println("testdir 目录内容:")
    for _, e := range entries {
        kind := "文件"
        if e.IsDir() {
            kind = "目录"
        }
        fmt.Printf("  %-10s [%s]\n", e.Name(), kind)
    }

    // ========== 递归遍历 ==========
    fmt.Println("\n递归遍历所有文件:")
    filepath.Walk("testdir", func(path string, info os.FileInfo, err error) error {
        if err != nil {
            return err
        }
        indent := ""
        if info.IsDir() {
            fmt.Printf("%s📁 %s/\n", indent, path)
        } else {
            fmt.Printf("%s  📄 %s (%d 字节)\n", indent, path, info.Size())
        }
        return nil
    })

    // ========== Glob 模式匹配 ==========
    goFiles, _ := filepath.Glob("testdir/**/*.go")
    fmt.Println("\n.go 文件:", goFiles)

    // ========== 路径操作 ==========
    p := "/home/user/project/main.go"
    fmt.Println("\n路径操作:")
    fmt.Println("  目录:", filepath.Dir(p))
    fmt.Println("  文件名:", filepath.Base(p))
    fmt.Println("  扩展名:", filepath.Ext(p))
    fmt.Println("  拼接:", filepath.Join("home", "user", "file.txt"))

    // 清理
    os.RemoveAll("testdir")
}
```

## 2. JSON 处理

### 2.1 序列化与反序列化

```go
package main

import (
    "encoding/json"
    "fmt"
)

type Address struct {
    City    string `json:"city"`
    ZipCode string `json:"zip_code"`
}

type User struct {
    ID       int      `json:"id"`
    Name     string   `json:"name"`
    Email    string   `json:"email,omitempty"`
    Age      int      `json:"age"`
    Active   bool     `json:"active"`
    Tags     []string `json:"tags,omitempty"`
    Address  Address  `json:"address"`
    Password string   `json:"-"` // 不参与序列化
}

func main() {
    // ========== 结构体 → JSON ==========
    user := User{
        ID:     1,
        Name:   "Alice",
        Email:  "alice@example.com",
        Age:    25,
        Active: true,
        Tags:   []string{"admin", "dev"},
        Address: Address{
            City:    "Beijing",
            ZipCode: "100000",
        },
        Password: "secret123",
    }

    // 紧凑 JSON
    data, _ := json.Marshal(user)
    fmt.Println("紧凑JSON:")
    fmt.Println(string(data))

    // 格式化 JSON
    pretty, _ := json.MarshalIndent(user, "", "  ")
    fmt.Println("\n格式化JSON:")
    fmt.Println(string(pretty))

    // omitempty：空值字段不输出
    emptyUser := User{ID: 2, Name: "Bob", Active: false}
    data2, _ := json.MarshalIndent(emptyUser, "", "  ")
    fmt.Println("\nomitempty 效果:")
    fmt.Println(string(data2))

    // ========== JSON → 结构体 ==========
    jsonStr := `{
        "id": 3,
        "name": "Carol",
        "age": 30,
        "active": true,
        "address": {"city": "Shanghai", "zip_code": "200000"}
    }`

    var parsed User
    json.Unmarshal([]byte(jsonStr), &parsed)
    fmt.Println("\n反序列化:")
    fmt.Printf("  %+v\n", parsed)
}
```

### 2.2 处理动态 JSON

```go
package main

import (
    "encoding/json"
    "fmt"
)

func main() {
    // ========== 解析到 map（未知结构）==========
    jsonStr := `{
        "name": "Go",
        "year": 2009,
        "features": ["并发", "简洁", "高效"],
        "typed": true
    }`

    var result map[string]any
    json.Unmarshal([]byte(jsonStr), &result)

    fmt.Println("动态解析:")
    fmt.Println("  name:", result["name"])
    fmt.Println("  year:", result["year"]) // 数字默认解析为 float64
    fmt.Println("  features:", result["features"])
    fmt.Println("  typed:", result["typed"])

    // 类型断言获取具体值
    if year, ok := result["year"].(float64); ok {
        fmt.Println("  year (int):", int(year))
    }

    fmt.Println()

    // ========== json.RawMessage 延迟解析 ==========
    apiResp := `{
        "type": "user",
        "data": {"id": 1, "name": "Alice"}
    }`

    var resp struct {
        Type string          `json:"type"`
        Data json.RawMessage `json:"data"` // 先不解析
    }
    json.Unmarshal([]byte(apiResp), &resp)

    fmt.Println("类型:", resp.Type)
    fmt.Println("原始 data:", string(resp.Data))

    // 根据 type 决定如何解析 data
    if resp.Type == "user" {
        var user struct {
            ID   int    `json:"id"`
            Name string `json:"name"`
        }
        json.Unmarshal(resp.Data, &user)
        fmt.Printf("解析后: %+v\n", user)
    }

    fmt.Println()

    // ========== 切片/Map 序列化 ==========
    items := []map[string]any{
        {"name": "Go", "score": 95},
        {"name": "Rust", "score": 90},
    }
    data, _ := json.MarshalIndent(items, "", "  ")
    fmt.Println("切片序列化:")
    fmt.Println(string(data))
}
```

### 3.1 HTTP 客户端

```go
package main

import (
    "encoding/json"
    "fmt"
    "io"
    "net/http"
    "net/url"
    "strings"
    "time"
)

func main() {
    // ========== 简单 GET 请求 ==========
    fmt.Println("=== GET 请求 ===")
    resp, err := http.Get("https://httpbin.org/get?name=Go")
    if err != nil {
        fmt.Println("请求失败:", err)
        return
    }
    defer resp.Body.Close()

    fmt.Println("状态码:", resp.StatusCode)
    fmt.Println("Content-Type:", resp.Header.Get("Content-Type"))

    body, _ := io.ReadAll(resp.Body)
    fmt.Println("响应体(前100字节):", string(body[:min(100, len(body))]))

    fmt.Println()

    // ========== 带超时的客户端 ==========
    client := &http.Client{
        Timeout: 5 * time.Second,
    }

    // ========== POST JSON 请求 ==========
    fmt.Println("=== POST 请求 ===")
    payload := `{"name": "Alice", "age": 25}`
    resp2, err := client.Post(
        "https://httpbin.org/post",
        "application/json",
        strings.NewReader(payload),
    )
    if err != nil {
        fmt.Println("POST 失败:", err)
        return
    }
    defer resp2.Body.Close()
    fmt.Println("POST 状态码:", resp2.StatusCode)

    fmt.Println()

    // ========== 自定义请求（带 Header）==========
    fmt.Println("=== 自定义请求 ===")
    req, _ := http.NewRequest("GET", "https://httpbin.org/headers", nil)
    req.Header.Set("Authorization", "Bearer my-token")
    req.Header.Set("User-Agent", "GoClient/1.0")
    req.Header.Set("Accept", "application/json")

    resp3, err := client.Do(req)
    if err != nil {
        fmt.Println("请求失败:", err)
        return
    }
    defer resp3.Body.Close()

    var result map[string]any
    json.NewDecoder(resp3.Body).Decode(&result)
    fmt.Println("响应:", result)

    // ========== URL 构建 ==========
    fmt.Println("\n=== URL 构建 ===")
    u, _ := url.Parse("https://api.example.com/search")
    q := u.Query()
    q.Set("q", "Go 语言")
    q.Set("page", "1")
    q.Set("limit", "10")
    u.RawQuery = q.Encode()
    fmt.Println("完整URL:", u.String())
}
```

### 3.2 HTTP 服务端

```go
package main

import (
    "encoding/json"
    "fmt"
    "log"
    "net/http"
    "time"
)

type APIResponse struct {
    Code    int    `json:"code"`
    Message string `json:"message"`
    Data    any    `json:"data,omitempty"`
}

func main() {
    mux := http.NewServeMux()

    // 基本路由
    mux.HandleFunc("GET /", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, "Hello, Go HTTP Server!")
    })

    // 返回 JSON
    mux.HandleFunc("GET /api/time", func(w http.ResponseWriter, r *http.Request) {
        resp := APIResponse{
            Code:    200,
            Message: "success",
            Data: map[string]string{
                "time": time.Now().Format("2006-01-02 15:04:05"),
            },
        }
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(resp)
    })

    // 路径参数（Go 1.22+）
    mux.HandleFunc("GET /api/users/{id}", func(w http.ResponseWriter, r *http.Request) {
        id := r.PathValue("id")
        resp := APIResponse{
            Code:    200,
            Message: "success",
            Data:    map[string]string{"id": id, "name": "User_" + id},
        }
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(resp)
    })

    // 查询参数
    mux.HandleFunc("GET /api/search", func(w http.ResponseWriter, r *http.Request) {
        q := r.URL.Query().Get("q")
        page := r.URL.Query().Get("page")
        if q == "" {
            w.WriteHeader(http.StatusBadRequest)
            json.NewEncoder(w).Encode(APIResponse{Code: 400, Message: "缺少参数 q"})
            return
        }
        resp := APIResponse{
            Code:    200,
            Message: "success",
            Data:    map[string]string{"query": q, "page": page},
        }
        w.Header().Set("Content-Type", "application/json")
        json.NewEncoder(w).Encode(resp)
    })

    // POST 接收 JSON
    mux.HandleFunc("POST /api/users", func(w http.ResponseWriter, r *http.Request) {
        var input struct {
            Name  string `json:"name"`
            Email string `json:"email"`
        }
        if err := json.NewDecoder(r.Body).Decode(&input); err != nil {
            w.WriteHeader(http.StatusBadRequest)
            json.NewEncoder(w).Encode(APIResponse{Code: 400, Message: "无效的 JSON"})
            return
        }
        resp := APIResponse{
            Code:    201,
            Message: "用户已创建",
            Data:    input,
        }
        w.Header().Set("Content-Type", "application/json")
        w.WriteHeader(http.StatusCreated)
        json.NewEncoder(w).Encode(resp)
    })

    addr := ":8080"
    fmt.Printf("服务启动: http://localhost%s\n", addr)
    log.Fatal(http.ListenAndServe(addr, mux))
}
```

### 3.3 中间件

```go
package main

import (
    "fmt"
    "log"
    "net/http"
    "time"
)

// 中间件类型
type Middleware func(http.Handler) http.Handler

// 日志中间件
func LoggingMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        next.ServeHTTP(w, r)
        log.Printf("%s %s %v", r.Method, r.URL.Path, time.Since(start))
    })
}

// CORS 中间件
func CORSMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        w.Header().Set("Access-Control-Allow-Origin", "*")
        w.Header().Set("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE")
        w.Header().Set("Access-Control-Allow-Headers", "Content-Type, Authorization")

        if r.Method == "OPTIONS" {
            w.WriteHeader(http.StatusOK)
            return
        }
        next.ServeHTTP(w, r)
    })
}

// Recovery 中间件（捕获 panic）
func RecoveryMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        defer func() {
            if err := recover(); err != nil {
                log.Printf("PANIC: %v", err)
                http.Error(w, "Internal Server Error", http.StatusInternalServerError)
            }
        }()
        next.ServeHTTP(w, r)
    })
}

// 串联多个中间件
func Chain(handler http.Handler, middlewares ...Middleware) http.Handler {
    for i := len(middlewares) - 1; i >= 0; i-- {
        handler = middlewares[i](handler)
    }
    return handler
}

func main() {
    mux := http.NewServeMux()

    mux.HandleFunc("GET /", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintln(w, "Hello!")
    })

    mux.HandleFunc("GET /panic", func(w http.ResponseWriter, r *http.Request) {
        panic("模拟崩溃!")
    })

    // 应用中间件链
    handler := Chain(mux,
        RecoveryMiddleware,
        LoggingMiddleware,
        CORSMiddleware,
    )

    fmt.Println("服务启动: http://localhost:8080")
    log.Fatal(http.ListenAndServe(":8080", handler))
}
```

```
// 日志输出:

// 2024/01/15 10:30:05 GET /panic 120.3µs
// 2024/01/15 10:30:05 PANIC: 模拟崩溃!
```

## 4. time 包

### 4.1 时间操作

```go
package main

import (
    "fmt"
    "time"
)

func main() {
    // ========== 获取当前时间 ==========
    now := time.Now()
    fmt.Println("当前时间:", now)
    fmt.Println("年:", now.Year())
    fmt.Println("月:", now.Month())
    fmt.Println("日:", now.Day())
    fmt.Println("时:分:秒:", now.Hour(), now.Minute(), now.Second())
    fmt.Println("星期:", now.Weekday())
    fmt.Println("Unix 时间戳:", now.Unix())
    fmt.Println("毫秒时间戳:", now.UnixMilli())

    fmt.Println()

    // ========== 时间格式化（Go 使用参考时间 2006-01-02 15:04:05）==========
    fmt.Println("格式化示例:")
    fmt.Println("  标准:", now.Format("2006-01-02 15:04:05"))
    fmt.Println("  日期:", now.Format("2006/01/02"))
    fmt.Println("  时间:", now.Format("15:04:05"))
    fmt.Println("  RFC3339:", now.Format(time.RFC3339))
    fmt.Println("  自定义:", now.Format("2006年01月02日 15时04分"))

    fmt.Println()

    // ========== 时间解析 ==========
    t1, _ := time.Parse("2006-01-02", "2024-06-15")
    fmt.Println("解析日期:", t1)

    t2, _ := time.Parse("2006-01-02 15:04:05", "2024-12-25 18:30:00")
    fmt.Println("解析日期时间:", t2)

    // 解析带时区的时间
    loc, _ := time.LoadLocation("Asia/Shanghai")
    t3, _ := time.ParseInLocation("2006-01-02 15:04:05", "2024-12-25 18:30:00", loc)
    fmt.Println("上海时间:", t3)

    fmt.Println()

    // ========== 时间运算 ==========
    future := now.Add(72 * time.Hour) // 3天后
    fmt.Println("3天后:", future.Format("2006-01-02"))

    past := now.Add(-24 * time.Hour) // 1天前
    fmt.Println("1天前:", past.Format("2006-01-02"))

    // 时间差
    diff := future.Sub(past)
    fmt.Printf("时间差: %.0f 小时 = %.0f 天\n", diff.Hours(), diff.Hours()/24)

    // 比较
    fmt.Println("future 在 past 之后:", future.After(past))
    fmt.Println("past 在 future 之前:", past.Before(future))

    fmt.Println()

    // ========== 定时器 ==========
    fmt.Println("=== 定时器 ===")

    // time.After：一次性定时
    fmt.Println("等待 100ms...")
    <-time.After(100 * time.Millisecond)
    fmt.Println("100ms 已过")

    // time.Tick：周期性定时
    ticker := time.NewTicker(100 * time.Millisecond)
    defer ticker.Stop()

    count := 0
    for range ticker.C {
        count++
        fmt.Printf("  tick %d\n", count)
        if count >= 3 {
            break
        }
    }
}
```

```go
package main

import (
    "fmt"

func main() {
    // 基本匹配

    fmt.Println("包含数字:", matched) // true

    // 编译正则（推荐复用）
    re := regexp.MustCompile(`(\d{4})-(\d{2})-(\d{2})`)

    // 查找
    fmt.Println("查找:", re.FindString("日期是 2024-01-15 和 2024-06-20"))

    // 查找所有
    fmt.Println("查找所有:", re.FindAllString("2024-01-15 到 2024-06-20", -1))

    // 捕获组
    match := re.FindStringSubmatch("日期: 2024-01-15")
    if match != nil {
        fmt.Printf("完整: %s, 年: %s, 月: %s, 日: %s\n",
            match[0], match[1], match[2], match[3])
    }

    // 替换
    emailRe := regexp.MustCompile(`[\w.]+@[\w.]+`)
    masked := emailRe.ReplaceAllString(
        "联系: alice@example.com 或 bob@test.com",
        "***@***",
    )
    fmt.Println("替换:", masked)

    // 分割
    splitRe := regexp.MustCompile(`[,;\s]+`)
    parts := splitRe.Split("go, rust; python java", -1)
    fmt.Println("分割:", parts)
}
```

### 5.2 sort 排序

```go
package main

import (
    "fmt"
    "sort"
    "slices"
)

func main() {
    // 基本排序
    nums := []int{5, 2, 8, 1, 9, 3}
    sort.Ints(nums)
    fmt.Println("升序:", nums) // [1 2 3 5 8 9]

    strs := []string{"banana", "apple", "cherry"}
    sort.Strings(strs)
    fmt.Println("字符串排序:", strs) // [apple banana cherry]

    // 自定义排序
    type Item struct {
        Name  string
        Price float64
    }

    items := []Item{
        {"键盘", 299},
        {"鼠标", 99},
        {"显示器", 1999},
        {"耳机", 599},
    }

    // 按价格升序
    sort.Slice(items, func(i, j int) bool {
        return items[i].Price < items[j].Price
    })
    fmt.Println("\n按价格升序:")
    for _, item := range items {
        fmt.Printf("  %-6s ¥%.0f\n", item.Name, item.Price)
    }

    // 稳定排序（相等元素保持原始顺序）
    sort.SliceStable(items, func(i, j int) bool {
        return items[i].Price > items[j].Price
    })
    fmt.Println("按价格降序:")
    for _, item := range items {
        fmt.Printf("  %-6s ¥%.0f\n", item.Name, item.Price)
    }

    // 二分查找（必须已排序）
    sorted := []int{1, 3, 5, 7, 9, 11, 13}
    idx := sort.SearchInts(sorted, 7)
    fmt.Printf("\n在 %v 中找 7: 索引 %d\n", sorted, idx)

    // slices 包（Go 1.21+，泛型版本）
    nums2 := []int{5, 2, 8, 1, 9}
    slices.Sort(nums2)
    fmt.Println("slices.Sort:", nums2)

    fmt.Println("最小值:", slices.Min(nums2))
    fmt.Println("最大值:", slices.Max(nums2))
    fmt.Println("包含3:", slices.Contains(nums2, 3))
}
```

### 6.3 httptest 测试 HTTP 处理器

```go
package main

import (
    "encoding/json"
    "net/http"
    "net/http/httptest"
    "strings"
    "testing"

// 待测试的处理器
func HealthHandler(w http.ResponseWriter, r *http.Request) {
    w.Header().Set("Content-Type", "application/json")
    json.NewEncoder(w).Encode(map[string]string{"status": "ok"})
}

func EchoHandler(w http.ResponseWriter, r *http.Request) {
    var body map[string]any
    json.NewDecoder(r.Body).Decode(&body)
    w.Header().Set("Content-Type", "application/json")
    json.NewEncoder(w).Encode(body)
}

// 测试 GET
func TestHealthHandler(t *testing.T) {
    req := httptest.NewRequest("GET", "/health", nil)
    rec := httptest.NewRecorder()

    HealthHandler(rec, req)

    // 检查状态码
    if rec.Code != http.StatusOK {
        t.Errorf("状态码 = %d; want 200", rec.Code)
    }

    // 检查响应体
    var resp map[string]string
    json.NewDecoder(rec.Body).Decode(&resp)
    if resp["status"] != "ok" {
        t.Errorf("status = %q; want \"ok\"", resp["status"])
    }
}

// 测试 POST
func TestEchoHandler(t *testing.T) {
    body := strings.NewReader(`{"message": "hello"}`)
    req := httptest.NewRequest("POST", "/echo", body)
    req.Header.Set("Content-Type", "application/json")
    rec := httptest.NewRecorder()

    EchoHandler(rec, req)

    var resp map[string]any
    json.NewDecoder(rec.Body).Decode(&resp)
    if resp["message"] != "hello" {
        t.Errorf("message = %v; want \"hello\"", resp["message"])
    }
}

// 使用 httptest.Server 测试完整的 HTTP 服务
func TestWithServer(t *testing.T) {
    mux := http.NewServeMux()
    mux.HandleFunc("/health", HealthHandler)

    server := httptest.NewServer(mux)
    defer server.Close()

    // 发送真实 HTTP 请求
    resp, err := http.Get(server.URL + "/health")
    if err != nil {
        t.Fatalf("请求失败: %v", err)
    }
    defer resp.Body.Close()

    if resp.StatusCode != 200 {
        t.Errorf("状态码 = %d; want 200", resp.StatusCode)
    }
}
```

### model.go

```go
package main

import (
    "errors"
    "sync"
    "sync/atomic"
)

var (
    ErrNotFound = errors.New("todo not found")
)

type Todo struct {
    ID        int    `json:"id"`
    Title     string `json:"title"`
    Completed bool   `json:"completed"`
}

// 线程安全的内存存储
type TodoStore struct {
    mu     sync.RWMutex
    todos  map[int]*Todo
    nextID atomic.Int32
}

func NewTodoStore() *TodoStore {
    s := &TodoStore{todos: make(map[int]*Todo)}
    s.nextID.Store(1)
    return s
}

func (s *TodoStore) Create(title string) *Todo {
    id := int(s.nextID.Add(1) - 1)
    todo := &Todo{ID: id, Title: title, Completed: false}
    s.mu.Lock()
    s.todos[id] = todo
    s.mu.Unlock()
    return todo
}

func (s *TodoStore) GetAll() []*Todo {
    s.mu.RLock()
    defer s.mu.RUnlock()
    list := make([]*Todo, 0, len(s.todos))
    for _, t := range s.todos {
        list = append(list, t)
    }
    return list
}

func (s *TodoStore) GetByID(id int) (*Todo, error) {
    s.mu.RLock()
    defer s.mu.RUnlock()
    t, ok := s.todos[id]
    if !ok {
        return nil, ErrNotFound
    }
    return t, nil
}

func (s *TodoStore) Toggle(id int) (*Todo, error) {
    s.mu.Lock()
    defer s.mu.Unlock()
    t, ok := s.todos[id]
    if !ok {
        return nil, ErrNotFound
    }
    t.Completed = !t.Completed
    return t, nil
}

func (s *TodoStore) Delete(id int) error {
    s.mu.Lock()
    defer s.mu.Unlock()
    if _, ok := s.todos[id]; !ok {
        return ErrNotFound
    }
    delete(s.todos, id)
    return nil
}
```

### handler.go

```go
package main

import (
    "encoding/json"
    "errors"
    "net/http"
    "strconv"
)

type Response struct {
    Code    int    `json:"code"`
    Message string `json:"message"`
    Data    any    `json:"data,omitempty"`
}

func writeJSON(w http.ResponseWriter, status int, resp Response) {
    w.Header().Set("Content-Type", "application/json")
    w.WriteHeader(status)
    json.NewEncoder(w).Encode(resp)
}

func SetupRoutes(store *TodoStore) http.Handler {
    mux := http.NewServeMux()

    // 获取所有
    mux.HandleFunc("GET /api/todos", func(w http.ResponseWriter, r *http.Request) {
        todos := store.GetAll()
        writeJSON(w, http.StatusOK, Response{Code: 200, Message: "ok", Data: todos})
    })

    // 创建
    mux.HandleFunc("POST /api/todos", func(w http.ResponseWriter, r *http.Request) {
        var input struct {
            Title string `json:"title"`
        }
        if err := json.NewDecoder(r.Body).Decode(&input); err != nil || input.Title == "" {
            writeJSON(w, http.StatusBadRequest, Response{Code: 400, Message: "title 不能为空"})
            return
        }
        todo := store.Create(input.Title)
        writeJSON(w, http.StatusCreated, Response{Code: 201, Message: "created", Data: todo})
    })

    // 切换完成状态
    mux.HandleFunc("PATCH /api/todos/{id}/toggle", func(w http.ResponseWriter, r *http.Request) {
        id, err := strconv.Atoi(r.PathValue("id"))
        if err != nil {
            writeJSON(w, http.StatusBadRequest, Response{Code: 400, Message: "无效的 ID"})
            return
        }
        todo, err := store.Toggle(id)
        if err != nil {
            if errors.Is(err, ErrNotFound) {
                writeJSON(w, http.StatusNotFound, Response{Code: 404, Message: "未找到"})
                return
            }
            writeJSON(w, http.StatusInternalServerError, Response{Code: 500, Message: err.Error()})
            return
        }
        writeJSON(w, http.StatusOK, Response{Code: 200, Message: "toggled", Data: todo})
    })

    // 删除
    mux.HandleFunc("DELETE /api/todos/{id}", func(w http.ResponseWriter, r *http.Request) {
        id, _ := strconv.Atoi(r.PathValue("id"))
        if err := store.Delete(id); err != nil {
            if errors.Is(err, ErrNotFound) {
                writeJSON(w, http.StatusNotFound, Response{Code: 404, Message: "未找到"})
                return
            }
        }
        writeJSON(w, http.StatusOK, Response{Code: 200, Message: "deleted"})
    })

    return mux
}
```
